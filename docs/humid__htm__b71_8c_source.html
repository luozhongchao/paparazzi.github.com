00070 void humid_htm_read( void ) {
00071   if (htm_status == HTM_MR_OK) {
00072     /* read humid and temp*/
00073     htm_status = HTM_READ_DATA;
00074     I2CReceive(HTM_I2C_DEV, htm_trans, HTM_SLAVE_ADDR, 4);

00078 void humid_htm_event( void ) {
00079   if (htm_trans.status == I2CTransSuccess) {
00080     switch (htm_status) {
00081 
00082     case HTM_MR:
00083       htm_status = HTM_MR_OK;
00084       htm_trans.status = I2CTransDone;
00085       break;
00086 
00087     case HTM_READ_DATA:
00088       /* check stale status */
00089       if (((htm_trans.buf[0] >> 6) & 0x3) == 0) {
00090         /* humidity */
00091         humidhtm = ((htm_trans.buf[0] & 0x3F) << 8) | htm_trans.buf[1];
00092         fhumidhtm = humidhtm / 163.83;
00093         /* temperature */
00094         temphtm = (htm_trans.buf[2] << 6) | (htm_trans.buf[3] >> 2);
00095         ftemphtm = -40.00 + 0.01 * temphtm;
00096         DOWNLINK_SEND_HTM_STATUS(DefaultChannel, &humidhtm, &temphtm, &fhumidhtm, &ftemphtm);
00097       }
00098       htm_trans.status = I2CTransDone;
00099       break;
00100 
00101     default:
00102       htm_trans.status = I2CTransDone;
00103       break;
00104     }
00105   }
