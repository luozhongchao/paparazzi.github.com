00120 __attribute__ ((always_inline)) static inline void gv_update_ref_from_z_sp(int32_t z_sp) {
00121 
00122   gv_z_ref  += gv_zd_ref;
00123   gv_zd_ref += gv_zdd_ref;
00124 
00125   // compute the "speed part" of zdd = -2*zeta*omega*zd -omega^2(z_sp - z)
00126   int32_t zd_zdd_res = gv_zd_ref>>(GV_ZD_REF_FRAC - GV_ZDD_REF_FRAC);
00127   int32_t zdd_speed = ((int32_t)(-2*GV_ZETA_OMEGA)*zd_zdd_res)>>(GV_ZETA_OMEGA_FRAC);
00128   // compute z error in z_sp resolution
00129   int32_t z_err_sp = z_sp - (int32_t)(gv_z_ref>>(GV_Z_REF_FRAC-INT32_POS_FRAC));
00130   // convert to accel resolution
00131   int32_t z_err_accel = z_err_sp>>(INT32_POS_FRAC-GV_ZDD_REF_FRAC);
00132   int32_t zdd_pos = ((int32_t)(GV_OMEGA_2)*z_err_accel)>>GV_OMEGA_2_FRAC;
00133   gv_zdd_ref = zdd_speed + zdd_pos;
00134 
00135   /* Saturate accelerations */
00136   Bound(gv_zdd_ref, GV_MIN_ZDD, GV_MAX_ZDD);
00137 
00138   /* Saturate speed and adjust acceleration accordingly */
00139   if (gv_zd_ref <= GV_MIN_ZD) {
00140     gv_zd_ref = GV_MIN_ZD;
00141     if (gv_zdd_ref < 0)
00142       gv_zdd_ref = 0;
00143   }
00144   else if (gv_zd_ref >= GV_MAX_ZD) {
00145     gv_zd_ref = GV_MAX_ZD;
00146     if (gv_zdd_ref > 0)
00147       gv_zdd_ref = 0;
00148   }
00149 }
