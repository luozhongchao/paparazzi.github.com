00088 void imu_umarim_event( void )
00089 {
00090 
00091   // If the itg3200 I2C transaction has succeeded: convert the data
00092   itg3200_event();
00093   if (itg3200_data_available) {
00094     RATES_ASSIGN(imu.gyro_unscaled, itg3200_data.p, itg3200_data.q, itg3200_data.r);
00095     itg3200_data_available = FALSE;
00096     gyr_valid = TRUE;
00097   }
00098 
00099   // If the adxl345 I2C transaction has succeeded: convert the data
00100   adxl345_event();
00101   if (adxl345_data_available) {
00102     // Be careful with orientation of the ADXL (ITG axes are taken as default reference)
00103     VECT3_ASSIGN(imu.accel_unscaled, adxl345_data.y, -adxl345_data.x, adxl345_data.z);
00104     adxl345_data_available = FALSE;
00105     acc_valid = TRUE;
00106   }
