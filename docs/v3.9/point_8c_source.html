00110 /*******************************************************************
00111 ; function name:   vSubtractVectors
00112 ; description:     subtracts two vectors a = b - c
00113 ; parameters:
00114 ;*******************************************************************/
00115 void vSubtractVectors(VECTOR* svA, VECTOR svB, VECTOR svC)
00116 {
00117   svA->fx = svB.fx - svC.fx;
00118   svA->fy = svB.fy - svC.fy;
00119   svA->fz = svB.fz - svC.fz;
00120 }

00122 /*******************************************************************
00123 ; function name:   vMultiplyMatrixByVector
00124 ; description:     multiplies matrix by vector svA = smB * svC
00125 ; parameters:
00126 ;*******************************************************************/
00127 void vMultiplyMatrixByVector(VECTOR* svA, MATRIX smB, VECTOR svC)
00128 {
00129   svA->fx = smB.fx1 * svC.fx  +  smB.fx2 * svC.fy  +  smB.fx3 * svC.fz;
00130   svA->fy = smB.fy1 * svC.fx  +  smB.fy2 * svC.fy  +  smB.fy3 * svC.fz;
00131   svA->fz = smB.fz1 * svC.fx  +  smB.fz2 * svC.fy  +  smB.fz3 * svC.fz;
00132 }
