00142 void ins_propagate() {
00143   /* untilt accels */
00144   struct Int32Vect3 accel_meas_body;
00145   INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
00146   struct Int32Vect3 accel_meas_ltp;
00147   INT32_RMAT_TRANSP_VMULT(accel_meas_ltp, ahrs.ltp_to_body_rmat, accel_meas_body);
00148 
00149 #if USE_VFF
00150   float z_accel_meas_float = ACCEL_FLOAT_OF_BFP(accel_meas_ltp.z);
00151   if (baro.status == BS_RUNNING && ins_baro_initialised) {
00152     vff_propagate(z_accel_meas_float);
00153     ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
00154     ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
00155     ins_ltp_pos.z   = POS_BFP_OF_REAL(vff_z);
00156   }
00157   else { // feed accel from the sensors
00158     // subtract -9.81m/s2 (acceleration measured due to gravity, but vehivle not accelerating in ltp)
00159     ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
00160   }
00161 #else
00162   ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
00163 #endif /* USE_VFF */
00164 
00165 #if USE_HFF
00166   /* propagate horizontal filter */
00167   b2_hff_propagate();
00168 #else
00169   ins_ltp_accel.x = accel_meas_ltp.x;
00170   ins_ltp_accel.y = accel_meas_ltp.y;
00171 #endif /* USE_HFF */
00172 
00173   INT32_VECT3_ENU_OF_NED(ins_enu_pos, ins_ltp_pos);
00174   INT32_VECT3_ENU_OF_NED(ins_enu_speed, ins_ltp_speed);
00175   INT32_VECT3_ENU_OF_NED(ins_enu_accel, ins_ltp_accel);
00176 }

00178 void ins_update_baro() {
00179 #if USE_VFF
00180   if (baro.status == BS_RUNNING) {
00181     if (!ins_baro_initialised) {
00182       ins_qfe = baro.absolute;
00183       ins_baro_initialised = TRUE;
00184     }
00185     if (ins_vf_realign) {
00186       ins_vf_realign = FALSE;
00187       ins_qfe = baro.absolute;
00188 #if USE_SONAR
00189       ins_sonar_offset = sonar_meas;
00190 #endif
00191       vff_realign(0.);
00192       ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
00193       ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
00194       ins_ltp_pos.z   = POS_BFP_OF_REAL(vff_z);
00195       ins_enu_pos.z = -ins_ltp_pos.z;
00196       ins_enu_speed.z = -ins_ltp_speed.z;
00197       ins_enu_accel.z = -ins_ltp_accel.z;
00198     }
00199     else { /* not realigning, so normal update with baro measurement */
00200       ins_baro_alt = ((baro.absolute - ins_qfe) * INS_BARO_SENS_NUM)/INS_BARO_SENS_DEN;
00201       float alt_float = POS_FLOAT_OF_BFP(ins_baro_alt);
00202       vff_update(alt_float);
00203     }
00204   }
00205 #endif
00206 }

00268 void ins_update_sonar() {
00269 #if USE_SONAR && USE_VFF
00270   static int32_t sonar_filtered = 0;
00271   sonar_filtered = (sonar_meas + 2*sonar_filtered) / 3;
00272   /* update baro_qfe assuming a flat ground */
00273   if (ins_update_on_agl && baro.status == BS_RUNNING) {
00274     int32_t d_sonar = (((int32_t)sonar_filtered - ins_sonar_offset) * INS_SONAR_SENS_NUM) / INS_SONAR_SENS_DEN;
00275     ins_qfe = baro.absolute + (d_sonar * (INS_BARO_SENS_DEN))/INS_BARO_SENS_NUM;
