00062 void ads1114_read( struct ads1114_periph * p ) {
00063   // Config done with success
00064   // start new reading when previous is done (and read if success)
00065   if (p->config_done && p->trans.status == I2CTransDone) {
00066     p->trans.buf[0] = ADS1114_POINTER_CONV_REG;
00067     I2CTransceive(ADS1114_I2C_DEVICE, p->trans, p->i2c_addr, 1, 2);
00068   }
00069 }
