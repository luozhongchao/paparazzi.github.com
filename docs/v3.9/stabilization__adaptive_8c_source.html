00204 void h_ctl_course_loop ( void ) {
00205   static float last_err;
00206 
00207   // Ground path error
00208   float err = h_ctl_course_setpoint - estimator_hspeed_dir;
00209   NormRadAngle(err);
00210 
00211   float d_err = err - last_err;
00212   last_err = err;
00213   NormRadAngle(d_err);
00214 
00215   float speed_depend_nav = estimator_hspeed_mod/NOMINAL_AIRSPEED;
00216   Bound(speed_depend_nav, 0.66, 1.5);
00217 
00218   h_ctl_roll_setpoint = h_ctl_course_pre_bank_correction * h_ctl_course_pre_bank
00219     + h_ctl_course_pgain * speed_depend_nav * err
00220     + h_ctl_course_dgain * d_err;
00221 
00222   BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
00223 }

00255 inline static void h_ctl_roll_loop( void ) {
00256 
00257   static float cmd_fb = 0.;
00258 
00259 #if USE_ANGLE_REF
00260   // Update reference setpoints for roll
00261   h_ctl_ref_roll_angle += h_ctl_ref_roll_rate * H_CTL_REF_DT;
00262   h_ctl_ref_roll_rate += h_ctl_ref_roll_accel * H_CTL_REF_DT;
00263   h_ctl_ref_roll_accel = H_CTL_REF_W*H_CTL_REF_W * (h_ctl_roll_setpoint - h_ctl_ref_roll_angle) - 2*H_CTL_REF_XI*H_CTL_REF_W * h_ctl_ref_roll_rate;
00264   // Saturation on references
00265   BoundAbs(h_ctl_ref_roll_accel, H_CTL_REF_MAX_P_DOT);
00266   if (h_ctl_ref_roll_rate > H_CTL_REF_MAX_P) {
00267     h_ctl_ref_roll_rate = H_CTL_REF_MAX_P;
00268     if (h_ctl_ref_roll_accel > 0.) h_ctl_ref_roll_accel = 0.;
00269   }
00270   else if (h_ctl_ref_roll_rate < - H_CTL_REF_MAX_P) {
00271     h_ctl_ref_roll_rate = - H_CTL_REF_MAX_P;
00272     if (h_ctl_ref_roll_accel < 0.) h_ctl_ref_roll_accel = 0.;
00273   }
00274 #else
00275   h_ctl_ref_roll_angle = h_ctl_roll_setpoint;
00276   h_ctl_ref_roll_rate = 0.;
00277   h_ctl_ref_roll_accel = 0.;
00278 #endif
00279 
00280 #ifdef USE_KFF_UPDATE
00281   // update Kff gains
00282   h_ctl_roll_Kffa += KFFA_UPDATE * h_ctl_ref_roll_accel * cmd_fb / (H_CTL_REF_MAX_P_DOT*H_CTL_REF_MAX_P_DOT);
00283   h_ctl_roll_Kffd += KFFD_UPDATE * h_ctl_ref_roll_rate  * cmd_fb / (H_CTL_REF_MAX_P*H_CTL_REF_MAX_P);
00284 #ifdef SITL
00285   printf("%f %f %f\n", h_ctl_roll_Kffa, h_ctl_roll_Kffd, cmd_fb);
00286 #endif
00287   h_ctl_roll_Kffa = Max(h_ctl_roll_Kffa, 0);
00288   h_ctl_roll_Kffd = Max(h_ctl_roll_Kffd, 0);
00289 #endif
00290 
00291   // Compute errors
00292   float err = h_ctl_ref_roll_angle - estimator_phi;
00293 #ifdef SITL
00294   static float last_err = 0;
00295   estimator_p = (err - last_err)/(1/60.);
00296   last_err = err;
00297 #endif
00298   float d_err = h_ctl_ref_roll_rate - estimator_p;
00299 
00300   if (pprz_mode == PPRZ_MODE_MANUAL || launch == 0) {
00301     h_ctl_roll_sum_err = 0.;
00302   }
00303   else {
00304     if (h_ctl_roll_igain > 0.) {
00305       h_ctl_roll_sum_err += err * H_CTL_REF_DT;
00306       BoundAbs(h_ctl_roll_sum_err, H_CTL_ROLL_SUM_ERR_MAX / h_ctl_roll_igain);
00307     } else {
00308       h_ctl_roll_sum_err = 0.;
00309     }
00310   }
00311 
00312   cmd_fb = h_ctl_roll_attitude_gain * err;
00313   float cmd = - h_ctl_roll_Kffa * h_ctl_ref_roll_accel
00314     - h_ctl_roll_Kffd * h_ctl_ref_roll_rate
00315     - cmd_fb
00316     - h_ctl_roll_rate_gain * d_err
00317     - h_ctl_roll_igain * h_ctl_roll_sum_err
00318     + v_ctl_throttle_setpoint * h_ctl_aileron_of_throttle;
00319 
00320   cmd /= airspeed_ratio2;
00321 
00322   // Set aileron commands
00323   h_ctl_aileron_setpoint = TRIM_PPRZ(cmd);
00324 }

00387 
00388   // Compute errors
00389   float err =  h_ctl_ref_pitch_angle - estimator_theta;
00390 #if USE_GYRO_PITCH_RATE

00396 
00397   if (pprz_mode == PPRZ_MODE_MANUAL || launch == 0) {
00398     h_ctl_pitch_sum_err = 0.;
00399   }
00400   else {

00402       h_ctl_pitch_sum_err += err * H_CTL_REF_DT;

00404     } else {
00405       h_ctl_pitch_sum_err = 0.;
00406     }
00407   }
00408 
00409   float cmd = - h_ctl_pitch_Kffa * h_ctl_ref_pitch_accel
00410     - h_ctl_pitch_Kffd * h_ctl_ref_pitch_rate
00411     + h_ctl_pitch_pgain * err
00412     + h_ctl_pitch_dgain * d_err
00413     + h_ctl_pitch_igain * h_ctl_pitch_sum_err;
00414 
00415   cmd /= airspeed_ratio2;
00416 
00417   h_ctl_elevator_setpoint = TRIM_PPRZ(cmd);
