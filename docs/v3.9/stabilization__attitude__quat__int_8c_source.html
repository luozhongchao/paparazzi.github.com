00095 static void attitude_run_ff(int32_t ff_commands[], struct Int32AttitudeGains *gains, struct Int32Rates *ref_accel)
00096 {
00097   /* Compute feedforward based on reference acceleration */
00098 
00099   ff_commands[COMMAND_ROLL]  = GAIN_PRESCALER_FF * gains->dd.x * RATE_FLOAT_OF_BFP(ref_accel->p) / (1 << 7);
00100   ff_commands[COMMAND_PITCH] = GAIN_PRESCALER_FF * gains->dd.y * RATE_FLOAT_OF_BFP(ref_accel->q) / (1 << 7);
00101   ff_commands[COMMAND_YAW]   = GAIN_PRESCALER_FF * gains->dd.z * RATE_FLOAT_OF_BFP(ref_accel->r) / (1 << 7);
00102 }

00125 void stabilization_attitude_run(bool_t enable_integrator) {
00126 
00127   /*
00128    * Update reference
00129    */
00130   stabilization_attitude_ref_update();
00131 
00132   /*
00133    * Compute errors for feedback
00134    */
00135 
00136   /* attitude error                          */
00137   struct Int32Quat att_err;
00138   INT32_QUAT_INV_COMP(att_err, ahrs.ltp_to_body_quat, stab_att_ref_quat);
00139   /* wrap it in the shortest direction       */
00140   INT32_QUAT_WRAP_SHORTEST(att_err);
00141   INT32_QUAT_NORMALIZE(att_err);
00142 
00143   /*  rate error                */
00144   struct Int32Rates rate_err;
00145   RATES_DIFF(rate_err, stab_att_ref_rate, ahrs.body_rate);
00146 
00147   /* integrated error */
00148   if (enable_integrator) {
00149     struct Int32Quat new_sum_err, scaled_att_err;
00150     /* update accumulator */
00151     scaled_att_err.qi = att_err.qi;
00152     scaled_att_err.qx = att_err.qx / IERROR_SCALE;
00153     scaled_att_err.qy = att_err.qy / IERROR_SCALE;
00154     scaled_att_err.qz = att_err.qz / IERROR_SCALE;
00155     INT32_QUAT_COMP(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
00156     INT32_QUAT_NORMALIZE(new_sum_err);
00157     QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
00158     INT32_EULERS_OF_QUAT(stabilization_att_sum_err, stabilization_att_sum_err_quat);
00159   } else {
00160     /* reset accumulator */
00161     INT32_QUAT_ZERO( stabilization_att_sum_err_quat );
00162     INT_EULERS_ZERO( stabilization_att_sum_err );
00163   }
00164 
00165   /* compute the feed forward command */
00166   attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains, &stab_att_ref_accel);
00167 
00168   /* compute the feed back command */
00169   attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains, &att_err, &rate_err, &stabilization_att_sum_err_quat);
00170 
00171   /* sum feedforward and feedback */
00172   stabilization_cmd[COMMAND_ROLL] = stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL];
00173   stabilization_cmd[COMMAND_PITCH] = stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH];
00174   stabilization_cmd[COMMAND_YAW] = stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW];
00175 
00176   /* bound the result */
00177   BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
00178   BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
00179   BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
00180 }
