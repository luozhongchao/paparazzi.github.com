00158 void dma1_c4_irq_handler(void) {
00159 
00160   ADS8344_values[channel] = (buf_in[1] << 8 | buf_in[2]) << 1 | buf_in[3] >> 7;
00161   channel++;
00162   if (channel > 6) {
00163     ADS8344_available = TRUE;
00164     ADS8344Unselect();
00165     DMA_ITConfig(DMA1_Channel4, DMA_IT_TC, DISABLE);
00166     /* Disable SPI_2 Rx and TX request */
00167     SPI_I2S_DMACmd(SPI2, SPI_I2S_DMAReq_Rx, DISABLE);
00168     SPI_I2S_DMACmd(SPI2, SPI_I2S_DMAReq_Tx, DISABLE);
00169     /* Disable DMA1 Channel4 and 5 */
00170     DMA_Cmd(DMA1_Channel4, DISABLE);
00171     DMA_Cmd(DMA1_Channel5, DISABLE);
00172   }
00173   else {
00174     ADS8344_read_channel();
00175   }
