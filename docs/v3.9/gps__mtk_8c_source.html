00117 static void gps_mtk_time2itow(uint32_t  gps_date, uint32_t  gps_time,
00118                            int16_t* gps_week, uint32_t* gps_itow) {
00119   /* convert UTC date/time to GPS week/itow, we have no idea about GPS
00120      leap seconds for now */
00121   uint16_t gps_msecond = gps_time % 1000;
00122   uint8_t  gps_second  = (gps_time / 1000) % 100;
00123   uint8_t  gps_minute  = (gps_time / 100000) % 100;
00124   uint8_t  gps_hour    = (gps_time / 10000000) % 100;
00125   uint16_t gps_year    = 2000 + (gps_date % 100);
00126   uint8_t  gps_month   = (gps_date / 100) % 100;
00127   uint8_t  gps_day     = (gps_date / 10000) % 100;
00128   int32_t  i, days;
00129 
00130   *gps_week = 0;
00131   *gps_itow = 0;
00132 
00133   /* sanity checks */
00134   if (gps_month > 12) return;
00135   if (gps_day > (DAYS_MONTH[gps_month] +
00136     ((gps_month == 1) ? isleap(gps_year):0))) return;
00137   if (gps_hour > 23) return;
00138   if (gps_minute > 59) return;
00139   if (gps_second > 59) return;
00140 
00141   /* days since 6-JAN-1980 */
00142   days = -6;
00143   for (i = 1980; i < gps_year; i++) days += (365 + isleap(i));
00144 
00145   /* add days in gps_year */
00146   for (i = 0; i < gps_month-1; i++) {
00147       days += DAYS_MONTH[i] + ((i == 1) ? isleap(gps_year):0);
00148   }
00149   days += gps_day;
00150 
00151   /* convert */
00152   *gps_week = (uint16_t) (days / 7);
00153   *gps_itow = ((days % 7) * SECS_DAY +
00154     gps_hour * SECS_HOUR +
00155     gps_minute * SECS_MINUTE +
00156     gps_second) * 1000 +
00157     gps_msecond;
00158 }

00285 void gps_mtk_parse( uint8_t c ) {
00286   if (gps_mtk.status < GOT_PAYLOAD) {
00287     gps_mtk.ck_a += c;
00288     gps_mtk.ck_b += gps_mtk.ck_a;
00289   }
00290   switch (gps_mtk.status) {
00291   case UNINIT:
00292     if (c == MTK_DIY14_SYNC1)
00293       gps_mtk.status = GOT_SYNC1_14;
00294     if (c == MTK_DIY16_ID)
00295       gps_mtk.msg_class = c;
00296       gps_mtk.status = GOT_SYNC1_16;
00297     break;
00298   /* MTK_DIY_VER_14 */
00299   case GOT_SYNC1_14:
00300     if (c != MTK_DIY14_SYNC2) {
00301       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00302       goto error;
00303     }
00304     if (gps_mtk.msg_available) {
00305       /* Previous message has not yet been parsed: discard this one */
00306       gps_mtk.error_last = GPS_MTK_ERR_OVERRUN;
00307       goto error;
00308     }
00309     gps_mtk.ck_a = 0;
00310     gps_mtk.ck_b = 0;
00311     gps_mtk.status++;
00312     gps_mtk.len = MTK_DIY14_NAV_LENGTH;
00313     break;
00314   case GOT_SYNC2_14:
00315     if (c != MTK_DIY14_ID) {
00316       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00317       goto error;
00318     }
00319     gps_mtk.msg_class = c;
00320     gps_mtk.msg_idx = 0;
00321     gps_mtk.status++;
00322     break;
00323   case GOT_CLASS_14:
00324     if (c != MTK_DIY14_NAV_ID) {
00325       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00326       goto error;
00327     }
00328     gps_mtk.msg_id = c;
00329     gps_mtk.status = GOT_ID;
00330     break;
00331   /* MTK_DIY_VER_16 */
00332   case GOT_SYNC1_16:
00333     if (c != MTK_DIY16_NAV_ID) {
00334       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00335       goto error;
00336     }
00337     if (gps_mtk.msg_available) {
00338       /* Previous message has not yet been parsed: discard this one */
00339       gps_mtk.error_last = GPS_MTK_ERR_OVERRUN;
00340       goto error;
00341     }
00342     gps_mtk.msg_id = c;
00343     gps_mtk.ck_a = 0;
00344     gps_mtk.ck_b = 0;
00345     gps_mtk.status++;
00346     break;
00347   case GOT_SYNC2_16:
00348     gps_mtk.len = c;
00349     gps_mtk.msg_idx = 0;
00350     gps_mtk.status = GOT_ID;
00351     break;
00352   case GOT_ID:
00353     gps_mtk.msg_buf[gps_mtk.msg_idx] = c;
00354     gps_mtk.msg_idx++;
00355     if (gps_mtk.msg_idx >= gps_mtk.len) {
00356       gps_mtk.status++;
00357     }
00358     break;
00359   case GOT_PAYLOAD:
00360     if (c != gps_mtk.ck_a) {
00361       gps_mtk.error_last = GPS_MTK_ERR_CHECKSUM;
00362       goto error;
00363     }
00364     gps_mtk.status++;
00365     break;
00366   case GOT_CHECKSUM1:
00367     if (c != gps_mtk.ck_b) {
00368       gps_mtk.error_last = GPS_MTK_ERR_CHECKSUM;
00369       goto error;
00370     }
00371     gps_mtk.msg_available = TRUE;
00372     goto restart;
00373     break;
00374   default:
00375     gps_mtk.error_last = GPS_MTK_ERR_UNEXPECTED;
00376     goto error;
00377   }
00378   return;
00379  error:
00380   gps_mtk.error_cnt++;
00381  restart:
00382   gps_mtk.status = UNINIT;
00383   return;
00384 }
