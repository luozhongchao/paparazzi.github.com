00078 /* Gps callback is called when receiving a VELNED or a SOL message
00079  * All position/speed messages are sent in one shot and VELNED is the last one on fixedwing
00080  * For rotorcraft, only SOL message is needed for pos/speed data
00081  */
00082 #define GpsEvent(_sol_available_callback) {        \
00083     if (GpsBuffer()) {                             \
00084       ReadGpsBuffer();                             \
00085     }                                              \
00086     if (gps_ubx.msg_available) {                   \
00087       gps_ubx_read_message();                      \
00088       gps_ubx_ucenter_event();                     \
00089       if (gps_ubx.msg_class == UBX_NAV_ID &&       \
00090           (gps_ubx.msg_id == UBX_NAV_VELNED_ID ||  \
00091            (gps_ubx.msg_id == UBX_NAV_SOL_ID &&    \
00092             gps_ubx.have_velned == 0))) {          \
00093         if (gps.fix == GPS_FIX_3D) {               \
00094           gps.last_fix_ticks = sys_time.nb_sec_rem; \
00095           gps.last_fix_time = sys_time.nb_sec;      \
00096         }                                          \
00097         _sol_available_callback();                 \
00098       }                                            \
00099       gps_ubx.msg_available = FALSE;               \
00100     }                                              \
00101   }
