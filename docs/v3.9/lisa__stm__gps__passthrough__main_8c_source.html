00241   static uint8_t uart2_sent = 0;
00242   static uint8_t uart3_sent = 0;
00243   static uint8_t uart1_has_data = 0;
00244   static uint8_t uart2_has_data = 0;

00248   static uint8_t uart1_enabled = 1;
00249   static uint8_t uart2_enabled = 1;
00250   static uint8_t uart3_enabled = 1;
00251 
00252   // We cache data availability, so it doesn't change between checks:
00253   uart1_has_data = Uart1ChAvailable();
00254   uart2_has_data = Uart2ChAvailable();
00255   uart3_has_data = Uart3ChAvailable();
00256 
00257   // Fill stage: Read data from UARTs into buffers, or increment
00258   // their timeouts if no data is available:
00259   if(!uart1_sent && uart1_has_data) {
00260     spistream_uart1_msg.uart_data[uart1_num_rx_bytes] = Uart1Getch();
00261     timeout_uart1 = 0;
00262     if(uart1_num_rx_bytes < SPISTREAM_MAX_MESSAGE_LENGTH)
00263       { uart1_num_rx_bytes++; }
00264   } else { if(timeout_uart1 < timeout_trig) { timeout_uart1++; } }
00265 
00266   if(!uart2_sent && uart2_has_data) {
00267     spistream_uart2_msg.uart_data[uart2_num_rx_bytes] = Uart2Getch();
00268     timeout_uart2 = 0;
00269     if(uart2_num_rx_bytes < SPISTREAM_MAX_MESSAGE_LENGTH)
00270       { uart2_num_rx_bytes++; }
00271   } else { if(timeout_uart2 < timeout_trig) { timeout_uart2++; } }
00272 
00273   if(!uart3_sent && uart3_has_data) {
00274     spistream_uart3_msg.uart_data[uart3_num_rx_bytes] = Uart3Getch();
00275     timeout_uart3 = 0;
00276     if(uart3_num_rx_bytes < SPISTREAM_MAX_MESSAGE_LENGTH)
00277       { uart3_num_rx_bytes++; }
00278   } else { if(timeout_uart3 < timeout_trig) { timeout_uart3++; } }
00279 
00280   trigger_send = ((!uart1_enabled ||
00281                   (timeout_uart1 >= timeout_trig)) &&
00282                  (!uart2_enabled ||
00283                   (timeout_uart2 >= timeout_trig)) &&
00284                  (!uart3_enabled ||
00285                   (timeout_uart3 >= timeout_trig)));
00286 
00287   // Send stage: If all UART timeouts reach the timeout
00288   // trigger value and have accumulated data to send
00289   if(trigger_send) {
00290 
00291     // If there was no new data on any UART for some time
00292     // and there is data in every rx buffer:
00293 
00294     if(spistream_wait_for_num_transfers > 0)
00295     {
00296       // Warning LED: Could not finish all transactions
00297       // from last call.
00298       LED_ON(6);
00299     }
00300     else
00301     {
00302       LED_OFF(6);
00303 
00304       uart1_sent = !uart1_enabled; // If we set uartX_sent to 1 here, it
00305       uart2_sent = !uart2_enabled; // is just ignored for every read poll
00306       uart3_sent = !uart3_enabled; // as it seems to have been read already.
00307 
00308       if(!uart1_sent && uart1_num_rx_bytes > 0) {
00309         if(spistream_send_msg((uint8_t *)&spistream_uart1_msg,
00310                               uart1_num_rx_bytes+1, // +1 for UART id
00311                               SPISTREAM_WAIT_FOR_READ)) {
00312            uart1_sent = 1;
00313            spistream_wait_for_num_transfers++;
00314          }
00315       }
00316 
00317       if(!uart2_sent && uart1_num_rx_bytes > 0) {
00318         if(spistream_send_msg((uint8_t *)&spistream_uart2_msg,
00319                               uart1_num_rx_bytes+1, // +1 for UART id
00320                               SPISTREAM_WAIT_FOR_READ)) {
00321            uart2_sent = 1;
00322            spistream_wait_for_num_transfers++;
00323         }
00324       }
00325 
00326       if(!uart3_sent && uart3_num_rx_bytes > 0) {
00327         if(spistream_send_msg((uint8_t *)&spistream_uart3_msg,
00328                               uart3_num_rx_bytes+1, // +1 for UART id
00329                               SPISTREAM_WAIT_FOR_READ)) {
00330           uart3_sent = 1;
00331           spistream_wait_for_num_transfers++;
00332         }
00333       }
00334 
00335       // Transaction completed, reset state.
00336       // Note: Only reset when all uart buffers have been transmitted,
00337       // otherwise the timeout would start from the beginning and the
00338       // loop phase shifts (aka "you're in the deep").
00339       uart1_num_rx_bytes = 0;
00340       uart2_num_rx_bytes = 0;
00341       uart3_num_rx_bytes = 0;
00342       timeout_uart1 = 0;
00343       timeout_uart2 = 0;
00344       timeout_uart3 = 0;
00345       uart1_sent = 0;
00346       uart2_sent = 0;
00347       uart3_sent = 0;
00348     }
00349   }
