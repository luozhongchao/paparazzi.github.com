00150  * -----------
00151  *

00153  *  bit 23  0=quaternion, 1=euler, 2=DCM

00243 void ins_periodic_task( void ) {
00244   if (xsens_configured > 0)
00245   {
00246      switch (xsens_configured)
00247      {
00248   case 20:
00249     /* send mode and settings to MT */
00250     XSENS_GoToConfig();
00251     XSENS_SetOutputMode(xsens_output_mode);
00252     XSENS_SetOutputSettings(xsens_output_settings);
00253     break;
00254   case 18:
00255     // Give pulses on SyncOut
00256     XSENS_SetSyncOutSettings(0,0x0002);
00257     break;
00258   case 17:
00259     // 1 pulse every 100 samples
00260     XSENS_SetSyncOutSettings(1,100);
00261     break;
00262   case 2:
00263     XSENS_ReqLeverArmGps();
00264     break;
00265 
00266   case 6:
00267     XSENS_ReqMagneticDeclination();
00268     break;
00269 
00270   case 13:
00271     #ifdef AHRS_H_X
00272     #pragma message "Sending XSens Magnetic Declination."
00273     xsens_declination = atan2(AHRS_H_Y, AHRS_H_X);
00274     XSENS_SetMagneticDeclination(xsens_declination);
00275     #endif
00276     break;
00277   case 12:
00278     #ifdef GPS_IMU_LEVER_ARM_X
00279     #pragma message "Sending XSens GPS Arm."
00280     XSENS_SetLeverArmGps(GPS_IMU_LEVER_ARM_X,GPS_IMU_LEVER_ARM_Y,GPS_IMU_LEVER_ARM_Z);
00281     #endif
00282     break;
00283   case 10:
00284     {
00285     uint8_t baud = 1;
00286     XSENS_SetBaudrate(baud);
00287     }
00288     break;
00289 
00290   case 1:
00291     XSENS_GoToMeasurment();
00292     break;
00293      }
00294      xsens_configured--;
00295      return;
00296   }
00297 
00298   RunOnceEvery(100,XSENS_ReqGPSStatus());
00299 }

00303 void handle_ins_msg( void) {
00304 
00305 
00306   // Send to Estimator (Control)
00307 #ifdef XSENS_BACKWARDS
00308   EstimatorSetAtt((-ins_phi+ins_roll_neutral), (ins_psi + RadOfDeg(180)), (-ins_theta+ins_pitch_neutral));
00309   EstimatorSetRate(-ins_p,-ins_q, ins_r);
00310 #else
00311   EstimatorSetAtt(ins_phi+ins_roll_neutral, ins_psi, ins_theta+ins_pitch_neutral);
00312   EstimatorSetRate(ins_p, ins_q, ins_r);
00313 #endif
00314 
00315   // Position
00316   float gps_east = gps.utm_pos.east / 100.;
00317   float gps_north = gps.utm_pos.north / 100.;
00318   gps_east -= nav_utm_east0;
00319   gps_north -= nav_utm_north0;
00320   EstimatorSetPosXY(gps_east, gps_north);
00321 
00322   // Altitude and vertical speed
00323   float hmsl = gps.hmsl;
00324   hmsl /= 1000.0f;
00325   EstimatorSetAlt(hmsl);
00326 
00327   #ifndef ALT_KALMAN
00328   #warning NO_VZ
00329   #endif
00330 
00331   // Horizontal speed
00332   float fspeed = sqrt(ins_vx*ins_vx + ins_vy*ins_vy);
00333   if (gps.fix != GPS_FIX_3D)
00334   {
00335     fspeed = 0;
00336   }
00337   float fclimb = -ins_vz;
00338   float fcourse = atan2f((float)ins_vy, (float)ins_vx);
00339   EstimatorSetSpeedPol(fspeed, fcourse, fclimb);
00340 
00341   // Now also finish filling the gps struct for telemetry purposes
00342   gps.gspeed = fspeed * 100.;
00343   gps.speed_3d = (uint16_t)(sqrt(ins_vx*ins_vx + ins_vy*ins_vy + ins_vz*ins_vz) * 100);
00344   gps.course = fcourse * 1e7;
00345 
00346 }
