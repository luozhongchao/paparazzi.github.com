00142 STATIC_INLINE void handle_periodic_tasks( void ) {
00143   if (sys_time_check_and_ack_timer(main_periodic_tid))
00144     main_periodic();
00145   if (sys_time_check_and_ack_timer(radio_control_tid))
00146     radio_control_periodic_task();
00147   if (sys_time_check_and_ack_timer(failsafe_tid))
00148     failsafe_check();
00149   if (sys_time_check_and_ack_timer(electrical_tid))
00150     electrical_periodic();
00151   if (sys_time_check_and_ack_timer(baro_tid))
00152     baro_periodic();
00153   if (sys_time_check_and_ack_timer(telemetry_tid))
00154     telemetry_periodic();
00155 }
