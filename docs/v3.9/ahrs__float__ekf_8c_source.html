00096 /*
00097  * Propagate our dynamic system
00098  *
00099  *      quat_dot = Wxq(pqr) * quat
00100  *      bias_dot = 0
00101  *
00102  * Wxq is the quaternion omega matrix:
00103  *
00104  *              [ 0, -p, -q, -r ]
00105  *      1/2 *   [ p,  0,  r, -q ]
00106  *              [ q, -r,  0,  p ]
00107  *              [ r,  q, -p,  0 ]
00108  *
00109  *
00110  *
00111  *
00112  *                 [ 0  -p  -q  -r   q1  q2  q3]
00113  *   F =   1/2 *   [ p   0   r  -q  -q0  q3 -q2]
00114  *                 [ q  -r   0   p  -q3 -q0  q1]
00115  *                 [ r   q  -p   0   q2 -q1 -q0]
00116  *                 [ 0   0   0   0    0   0   0]
00117  *                 [ 0   0   0   0    0   0   0]
00118  *                 [ 0   0   0   0    0   0   0]
00119  *
00120  */

00122 void ahrs_propagate(void) {
00123   /* compute unbiased rates */
00124   RATES_FLOAT_OF_BFP(bafe_rates, imu.gyro);
00125   RATES_SUB(bafe_rates, bafe_bias);
00126 
00127   /* compute F
00128      F is only needed later on to update the state covariance P.
00129      However, its [0:3][0:3] region is actually the Wxq(pqr) which is needed to
00130      compute the time derivative of the quaternion, so we compute F now
00131   */
00132 
00133   /* Fill in Wxq(pqr) into F */
00134   bafe_F[0][0] = bafe_F[1][1] = bafe_F[2][2] = bafe_F[3][3] = 0;
00135   bafe_F[1][0] = bafe_F[2][3] = bafe_rates.p * 0.5;
00136   bafe_F[2][0] = bafe_F[3][1] = bafe_rates.q * 0.5;
00137   bafe_F[3][0] = bafe_F[1][2] = bafe_rates.r * 0.5;
00138 
00139   bafe_F[0][1] = bafe_F[3][2] = -bafe_F[1][0];
00140   bafe_F[0][2] = bafe_F[1][3] = -bafe_F[2][0];
00141   bafe_F[0][3] = bafe_F[2][1] = -bafe_F[3][0];
00142   /* Fill in [4:6][0:3] region */
00143   bafe_F[0][4] = bafe_F[2][6] = bafe_quat.qx * 0.5;
00144   bafe_F[0][5] = bafe_F[3][4] = bafe_quat.qy * 0.5;
00145   bafe_F[0][6] = bafe_F[1][5] = bafe_quat.qz * 0.5;
00146 
00147   bafe_F[1][4] =  bafe_F[2][5] = bafe_F[3][6] = bafe_quat.qi * -0.5;
00148   bafe_F[3][5] = -bafe_F[0][4];
00149   bafe_F[1][6] = -bafe_F[0][5];
00150   bafe_F[2][4] = -bafe_F[0][6];
00151   /* quat_dot = Wxq(pqr) * quat */
00152   bafe_qdot.qi=                           bafe_F[0][1]*bafe_quat.qx+bafe_F[0][2]*bafe_quat.qy+bafe_F[0][3] * bafe_quat.qz;
00153   bafe_qdot.qx= bafe_F[1][0]*bafe_quat.qi                          +bafe_F[1][2]*bafe_quat.qy+bafe_F[1][3] * bafe_quat.qz;
00154   bafe_qdot.qy= bafe_F[2][0]*bafe_quat.qi+bafe_F[2][1]*bafe_quat.qx                          +bafe_F[2][3] * bafe_quat.qz;
00155   bafe_qdot.qz= bafe_F[3][0]*bafe_quat.qi+bafe_F[3][1]*bafe_quat.qx+bafe_F[3][2]*bafe_quat.qy                            ;
00156   /* propagate quaternion */
00157   bafe_quat.qi += bafe_qdot.qi * BAFE_DT;
00158   bafe_quat.qx += bafe_qdot.qx * BAFE_DT;
00159   bafe_quat.qy += bafe_qdot.qy * BAFE_DT;
00160   bafe_quat.qz += bafe_qdot.qz * BAFE_DT;
00161 
00162 
00163 }
