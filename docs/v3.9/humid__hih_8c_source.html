00062 void humid_hih_periodic( void ) {
00063   float fvout, fhih_temp;
00064 
00065   /* get temperature from external source */
00066   fhih_temp = ftempsht;
00067   /****************************************/
00068 
00069   adc_humid_hih = buf_humid_hih.sum / buf_humid_hih.av_nb_sample;
00070 
00071   /* 36k/68k voltage divider, 3.3V full sweep, 10 bits adc */
00072   fvout = (adc_humid_hih / 1024.) * 3.3 * (104./68.);
00073 
00074   /* 5V supply, 1st order curve fit */
00075   fhih_humid = ((fvout / 5.0)-0.16)/0.0062;
00076 
00077   /* temperature compensation */
00078   fhih_humid = fhih_humid/(1.0546 - (0.00216 * fhih_temp));
00079 
00080   DOWNLINK_SEND_HIH_STATUS(DefaultChannel, DefaultDevice, &adc_humid_hih, &fhih_humid, &fhih_temp);
