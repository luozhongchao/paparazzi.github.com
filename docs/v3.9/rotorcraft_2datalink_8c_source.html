00049 
00050   datalink_time = 0;

00052   uint8_t msg_id = IdOfMsg(dl_buffer);

00054

00056     {
00057       DOWNLINK_SEND_PONG(DefaultChannel, DefaultDevice);
00058     }
00059     break;
00060 
00061   case DL_SETTING :
00062     {
00063       if (DL_SETTING_ac_id(dl_buffer) != AC_ID) break;

00065       float var = DL_SETTING_value(dl_buffer);

00068     }
00069     break;
00070 
00071   case DL_GET_SETTING :
00072     {
00073       if (DL_GET_SETTING_ac_id(dl_buffer) != AC_ID) break;

00075       float val = settings_get_value(i);
00076       DOWNLINK_SEND_DL_VALUE(DefaultChannel, DefaultDevice, &i, &val);
00077     }
00078     break;
00079 
00080 #if defined USE_NAVIGATION
00081   case DL_BLOCK :
00082     {
00083       if (DL_BLOCK_ac_id(dl_buffer) != AC_ID) break;
00084       nav_goto_block(DL_BLOCK_block_id(dl_buffer));
00085     }
00086     break;
00087 
00088   case DL_MOVE_WP :
00089     {
00090       uint8_t ac_id = DL_MOVE_WP_ac_id(dl_buffer);
00091       if (ac_id != AC_ID) break;

00097       /* WP_alt is in cm, lla.alt in mm */

00099       enu_of_lla_point_i(&enu,&ins_ltp_def,&lla);
00100       enu.x = POS_BFP_OF_REAL(enu.x)/100;

00105     }
00106     break;
00107 #endif /* USE_NAVIGATION */
00108   default:
00109     break;
00110   }
00111   /* Parse modules datalink */
00112   modules_parse_datalink(msg_id);
00113 }
