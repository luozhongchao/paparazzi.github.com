00093 void booz_cam_periodic(void) {
00094 
00095   switch (booz_cam_mode) {
00096     case BOOZ_CAM_MODE_NONE:
00097 #ifdef BOOZ_CAM_USE_TILT
00098       booz_cam_tilt_pwm = BOOZ_CAM_TILT_NEUTRAL;
00099 #endif
00100 #ifdef BOOZ_CAM_USE_PAN
00101       booz_cam_pan = ahrs.ltp_to_body_euler.psi;
00102 #endif
00103       break;
00104     case BOOZ_CAM_MODE_MANUAL:
00105 #ifdef BOOZ_CAM_USE_TILT
00106       Bound(booz_cam_tilt_pwm, CT_MIN, CT_MAX);
00107 #endif
00108       break;
00109     case BOOZ_CAM_MODE_HEADING:
00110 #ifdef BOOZ_CAM_USE_TILT_ANGLES
00111       Bound(booz_cam_tilt,CAM_TA_MIN,CAM_TA_MAX);
00112       booz_cam_tilt_pwm = BOOZ_CAM_TILT_MIN + D_TILT * (booz_cam_tilt - CAM_TA_MIN) / (CAM_TA_MAX - CAM_TA_MIN);
00113       Bound(booz_cam_tilt_pwm, CT_MIN, CT_MAX);
00114 #endif
00115 #ifdef BOOZ_CAM_USE_PAN
00116       Bound(booz_cam_pan, CP_MIN, CP_MAX);
00117       nav_heading = booz_cam_pan;
00118 #endif
00119       break;
00120     case BOOZ_CAM_MODE_WP:
00121 #ifdef WP_CAM
00122       {
00123         struct Int32Vect2 diff;
00124         VECT2_DIFF(diff, waypoints[WP_CAM], ins_enu_pos);
00125         INT32_VECT2_RSHIFT(diff,diff,INT32_POS_FRAC);
00126         INT32_ATAN2(booz_cam_pan,diff.x,diff.y);
00127         nav_heading = booz_cam_pan;
00128 #ifdef BOOZ_CAM_USE_TILT_ANGLES
00129         int32_t dist, height;
00130         INT32_VECT2_NORM(dist, diff);
00131         height = (waypoints[WP_CAM].z - ins_enu_pos.z) >> INT32_POS_FRAC;
00132         INT32_ATAN2(booz_cam_tilt, height, dist);
00133         Bound(booz_cam_tilt, CAM_TA_MIN, CAM_TA_MAX);
00134         booz_cam_tilt_pwm = BOOZ_CAM_TILT_MIN + D_TILT * (booz_cam_tilt - CAM_TA_MIN) / (CAM_TA_MAX - CAM_TA_MIN);
00135         Bound(booz_cam_tilt_pwm, CT_MIN, CT_MAX);
00136 #endif
00137       }
00138 #endif
00139       break;
00140   }
00141 #ifdef BOOZ_CAM_USE_TILT
00142   BOOZ_CAM_SetPwm(booz_cam_tilt_pwm);
00143 #endif
00144 }
