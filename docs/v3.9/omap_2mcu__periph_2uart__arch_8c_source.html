00053 void uart_transmit(struct uart_periph* p, uint8_t data ) {
00054   uint16_t temp = (p->tx_insert_idx + 1) % UART_TX_BUFFER_SIZE;
00055 
00056   if (temp == p->tx_extract_idx)
00057     return;                          // no room
00058 
00059   // check if in process of sending data
00060   if (p->tx_running) { // yes, add to queue
00061     p->tx_buf[p->tx_insert_idx] = data;
00062     p->tx_insert_idx = temp;
00063   }
00064   else { // no, set running flag and write to output register
00065     p->tx_running = TRUE;
00066     struct FmsSerialPort* fmssp = (struct FmsSerialPort*)(p->reg_addr);
00067     write((int)(fmssp->fd),&data,1);
00068     //printf("w %x\n",data);
00069   }
00070 }

00072 static inline void uart_handler(struct uart_periph* p) {
00073   unsigned char c='D';
00074 
00075   if (p->reg_addr == NULL) return; // device not initialized ?
00076 
00077   struct FmsSerialPort* fmssp = (struct FmsSerialPort*)(p->reg_addr);
00078   int fd = fmssp->fd;
00079 
00080   // check if more data to send
00081   if (p->tx_insert_idx != p->tx_extract_idx) {
00082     write(fd,&(p->tx_buf[p->tx_extract_idx]),1);
00083     //printf("w %x\n",p->tx_buf[p->tx_extract_idx]);
00084     p->tx_extract_idx++;
00085     p->tx_extract_idx %= UART_TX_BUFFER_SIZE;
00086   }
00087   else {
00088     p->tx_running = FALSE;   // clear running flag
00089   }
00090 
00091   if(read(fd,&c,1) > 0){
00092     //printf("r %x %c\n",c,c);
00093     uint16_t temp = (p->rx_insert_idx + 1) % UART_RX_BUFFER_SIZE;
00094     p->rx_buf[p->rx_insert_idx] = c;
00095     // check for more room in queue
00096     if (temp != p->rx_extract_idx)
00097       p->rx_insert_idx = temp; // update insert index
00098   }
00099 
00100 }
