00171   if (status == STATUS_RESET) {
00172     uint8_t i;

00199 static void calibration( void ) {
00200   /* End of init, configuration (page 11) */
00201   c1 = words[0] >> 1;
00202   c2 = ((words[2] & 0x3f) << 6) | (words[3] & 0x3f);

00218 void baro_MS5534A_event_task( void ) {
00219   if (status_read_data) {
00220     switch (status) {
00221     case STATUS_MEASURE_PRESSURE:
00222       d1 = Uint16(buf_input);
00223       break;
00224     case STATUS_MEASURE_TEMPERATURE:
00225       d2 = Uint16(buf_input);
00226       /* Compute pressure and temp (page 10) */
00227       int16_t dT = d2 - ut1;
00228       baro_MS5534A_temp = 200 + (dT*(c6+50)) / (1 << 10);
00229       int16_t off = c2*4 + ((c4-512)*dT)/(1 << 12);
00230       uint32_t sens = c1 + (c3*dT)/(1<<10) + 24576;
00231       uint16_t x = (sens*(d1-7168))/(1<<14) - off;
00232       // baro_MS5534A = ((x*10)>>5) + 250*10;
00233       baro_MS5534A_pressure = ((x*100)>>5) + 250*100;
00234       baro_MS5534A_available = TRUE;
00235 
00236       break;
00237     case STATUS_RESET:
00238       break;
00239     default: /* Init status */
00240       words[status] = Uint16(buf_input);
00241       if (status == STATUS_INIT4) {
00242         calibration();
00243       }
00244     }
00245   } /* else nothing to read */
00246 
00247   NextStatus();
00248   if (!status_read_data) {
00249     /* Ask next conversion now */
00250     baro_MS5534A_send();
00251   }
00252 }
