00192 /* UBX parsing */
00193 void gps_ubx_parse( uint8_t c ) {
00194   if (gps_ubx.status < GOT_PAYLOAD) {
00195     gps_ubx.ck_a += c;
00196     gps_ubx.ck_b += gps_ubx.ck_a;
00197   }
00198   switch (gps_ubx.status) {
00199   case UNINIT:
00200     if (c == UBX_SYNC1)
00201       gps_ubx.status++;
00202     break;
00203   case GOT_SYNC1:
00204     if (c != UBX_SYNC2) {
00205       gps_ubx.error_last = GPS_UBX_ERR_OUT_OF_SYNC;
00206       goto error;
00207     }
00208     gps_ubx.ck_a = 0;
00209     gps_ubx.ck_b = 0;
00210     gps_ubx.status++;
00211     break;
00212   case GOT_SYNC2:
00213     if (gps_ubx.msg_available) {
00214       /* Previous message has not yet been parsed: discard this one */
00215       gps_ubx.error_last = GPS_UBX_ERR_OVERRUN;
00216       goto error;
00217     }
00218     gps_ubx.msg_class = c;
00219     gps_ubx.status++;
00220     break;
00221   case GOT_CLASS:
00222     gps_ubx.msg_id = c;
00223     gps_ubx.status++;
00224     break;
00225   case GOT_ID:
00226     gps_ubx.len = c;
00227     gps_ubx.status++;
00228     break;
00229   case GOT_LEN1:
00230     gps_ubx.len |= (c<<8);
00231     if (gps_ubx.len > GPS_UBX_MAX_PAYLOAD) {
00232       gps_ubx.error_last = GPS_UBX_ERR_MSG_TOO_LONG;
00233       goto error;
00234     }
00235     gps_ubx.msg_idx = 0;
00236     gps_ubx.status++;
00237     break;
00238   case GOT_LEN2:
00239     gps_ubx.msg_buf[gps_ubx.msg_idx] = c;
00240     gps_ubx.msg_idx++;
00241     if (gps_ubx.msg_idx >= gps_ubx.len) {
00242       gps_ubx.status++;
00243     }
00244     break;
00245   case GOT_PAYLOAD:
00246     if (c != gps_ubx.ck_a) {
00247       gps_ubx.error_last = GPS_UBX_ERR_CHECKSUM;
00248       goto error;
00249     }
00250     gps_ubx.status++;
00251     break;
00252   case GOT_CHECKSUM1:
00253     if (c != gps_ubx.ck_b) {
00254       gps_ubx.error_last = GPS_UBX_ERR_CHECKSUM;
00255       goto error;
00256     }
00257     gps_ubx.msg_available = TRUE;
00258     goto restart;
00259     break;
00260   default:
00261     gps_ubx.error_last = GPS_UBX_ERR_UNEXPECTED;
00262     goto error;
00263   }
00264   return;
00265  error:
00266   gps_ubx.error_cnt++;
00267  restart:
00268   gps_ubx.status = UNINIT;
00269   return;
00270 }
