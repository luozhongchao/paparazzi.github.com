52 void rotorcraft_cam_init(void) {
53  rotorcraft_cam_SetCamMode(ROTORCRAFT_CAM_DEFAULT_MODE);
54 #if ROTORCRAFT_CAM_USE_TILT
55  rotorcraft_cam_tilt_pwm = ROTORCRAFT_CAM_TILT_NEUTRAL;
56  ActuatorSet(ROTORCRAFT_CAM_TILT_SERVO, rotorcraft_cam_tilt_pwm);
57 #else
58  rotorcraft_cam_tilt_pwm = 1500;
59 #endif
60  rotorcraft_cam_tilt = 0;
61  rotorcraft_cam_pan = 0;
62 }

64 void rotorcraft_cam_periodic(void) {
65 
66  switch (rotorcraft_cam_mode) {
67  case ROTORCRAFT_CAM_MODE_NONE:
68 #if ROTORCRAFT_CAM_USE_TILT
69  rotorcraft_cam_tilt_pwm = ROTORCRAFT_CAM_TILT_NEUTRAL;
70 #endif
71 #if ROTORCRAFT_CAM_USE_PAN
72  rotorcraft_cam_pan = stateGetNedToBodyEulers_i()->psi;
73 #endif
74  break;
75  case ROTORCRAFT_CAM_MODE_MANUAL:
76  // nothing to do here, just apply tilt pwm at the end
77  break;
78  case ROTORCRAFT_CAM_MODE_HEADING:
79 #if ROTORCRAFT_CAM_USE_TILT_ANGLES
80  Bound(rotorcraft_cam_tilt,CT_MIN,CT_MAX);
81  rotorcraft_cam_tilt_pwm = ROTORCRAFT_CAM_TILT_MIN + D_TILT * (rotorcraft_cam_tilt - CAM_TA_MIN) / (CAM_TA_MAX - CAM_TA_MIN);
82 #endif
83 #if ROTORCRAFT_CAM_USE_PAN
84  INT32_COURSE_NORMALIZE(rotorcraft_cam_pan);
85  nav_heading = rotorcraft_cam_pan;
86 #endif
87  break;
88  case ROTORCRAFT_CAM_MODE_WP:
89 #ifdef ROTORCRAFT_CAM_TRACK_WP
90  {
91  struct Int32Vect2 diff;
92  VECT2_DIFF(diff, waypoints[ROTORCRAFT_CAM_TRACK_WP], *stateGetPositionEnu_i());
93  INT32_VECT2_RSHIFT(diff,diff,INT32_POS_FRAC);
94  INT32_ATAN2(rotorcraft_cam_pan,diff.x,diff.y);
95  nav_heading = rotorcraft_cam_pan;
96 #if ROTORCRAFT_CAM_USE_TILT_ANGLES
97  int32_t dist, height;
98  INT32_VECT2_NORM(dist, diff);
99  height = (waypoints[ROTORCRAFT_CAM_TRACK_WP].z - stateGetPositionEnu_i()->z) >> INT32_POS_FRAC;
100  INT32_ATAN2(rotorcraft_cam_tilt, height, dist);
101  Bound(rotorcraft_cam_tilt, CAM_TA_MIN, CAM_TA_MAX);
102  rotorcraft_cam_tilt_pwm = ROTORCRAFT_CAM_TILT_MIN + D_TILT * (rotorcraft_cam_tilt - CAM_TA_MIN) / (CAM_TA_MAX - CAM_TA_MIN);
103 #endif
104  }
105 #endif
106  break;
107  default:
108  break;
109  }
110 #if ROTORCRAFT_CAM_USE_TILT
111  ActuatorSet(ROTORCRAFT_CAM_TILT_SERVO, rotorcraft_cam_tilt_pwm);
112 #endif
113 }
