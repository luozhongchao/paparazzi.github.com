00088 __attribute__ ((always_inline)) static inline void I2cAutomaton(int32_t state, struct i2c_periph* p) {
00089   struct i2c_transaction* trans = p->trans[p->trans_extract_idx];
00090   switch (state) {
00091     case I2C_START:
00092     case I2C_RESTART:
00093       // Set R/W flag
00094       switch (trans->type) {
00095         case I2CTransRx :
00096           SetBit(trans->slave_addr,0);
00097           break;
00098         case I2CTransTx:
00099         case I2CTransTxRx:
00100           ClearBit(trans->slave_addr,0);
00101           break;
00102       }
00103       I2cSendByte(p->reg_addr,trans->slave_addr);
00104       I2cClearStart(p->reg_addr);
00105       p->idx_buf = 0;
00106       break;
00107     case I2C_MR_DATA_ACK:
00108       if (p->idx_buf < trans->len_r) {
00109         trans->buf[p->idx_buf] = ((i2cRegs_t *)(p->reg_addr))->dat;
00110         p->idx_buf++;
00111         I2cReceive(p->reg_addr,p->idx_buf < trans->len_r - 1);
00112       }
00113       else {
00114         /* error , we should have got NACK */
00115         I2cFail(p,trans);
00116       }
00117       break;
00118     case I2C_MR_DATA_NACK:
00119       if (p->idx_buf < trans->len_r) {
00120         trans->buf[p->idx_buf] = ((i2cRegs_t *)(p->reg_addr))->dat;
00121       }
00122       I2cSendStop(p,trans);
00123       break;
00124     case I2C_MR_SLA_ACK: /* At least one char */
00125       /* Wait and reply with ACK or NACK */
00126       I2cReceive(p->reg_addr,p->idx_buf < trans->len_r - 1);
00127       break;
00128     case I2C_MR_SLA_NACK:
00129     case I2C_MT_SLA_NACK:
00130       /* Slave is not responding, transaction is failed */
00131       I2cFail(p,trans);
00132       break;
00133     case I2C_MT_SLA_ACK:
00134     case I2C_MT_DATA_ACK:
00135       if (p->idx_buf < trans->len_w) {
00136         I2cSendByte(p->reg_addr,trans->buf[p->idx_buf]);
00137         p->idx_buf++;
00138       } else {
00139         if (trans->type == I2CTransTxRx) {
00140           trans->type = I2CTransRx; /* FIXME should not change type */
00141           p->idx_buf = 0;
00142           trans->slave_addr |= 1;
00143           I2cSendStart(p);
00144         } else {
00145           I2cSendStop(p,trans);
00146         }
00147       }
00148       break;
00149     default:
00150       I2cFail(p,trans);
00151       /* FIXME log error */
00152       break;
00153   }
00154 }

00326 bool_t i2c_submit(struct i2c_periph* p, struct i2c_transaction* t) {
00327   unsigned cpsr;
00328 
00329   uint8_t idx;
00330   idx = p->trans_insert_idx + 1;
00331   if (idx >= I2C_TRANSACTION_QUEUE_LEN) idx = 0;
00332   if (idx == p->trans_extract_idx) {
00333     t->status = I2CTransFailed;
00334     return FALSE;  /* queue full */
00335   }
00336   t->status = I2CTransPending;
00337 
00338   uint8_t* vic = (uint8_t*)(p->init_struct);
00339   cpsr = disableIRQ();                                // disable global interrupts
00340   VICIntEnClear = VIC_BIT(*vic);
00341   restoreIRQ(cpsr);                                   // restore global interrupts
00342 
00343   p->trans[p->trans_insert_idx] = t;
00344   p->trans_insert_idx = idx;
00345   /* if peripheral is idle, start the transaction */
00346   if (p->status == I2CIdle)
00347     I2cSendStart(p);
00348   /* else it will be started by the interrupt handler */
00349   /* when the previous transactions completes         */
00350 
00351   //int_enable();
00352   cpsr = disableIRQ();                                // disable global interrupts
00353   VICIntEnable = VIC_BIT(*vic);
00354   restoreIRQ(cpsr);                                   // restore global interrupts
00355 
00356   return TRUE;
00357 }
