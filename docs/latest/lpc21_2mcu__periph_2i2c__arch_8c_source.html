00044 __attribute__ ((always_inline)) static inline void I2cEndOfTransaction(struct i2c_periph* p) {
00045   // handle fifo here
00046   p->trans_extract_idx++;
00047   if (p->trans_extract_idx >= I2C_TRANSACTION_QUEUE_LEN)
00048     p->trans_extract_idx = 0;
00049   // if no more transaction to process, stop here, else start next transaction
00050   if (p->trans_extract_idx == p->trans_insert_idx) {
00051     p->status = I2CIdle;
00052   }
00053   else {
00054     I2cSendStart(p);
00055   }

00090 __attribute__ ((always_inline)) static inline void I2cAutomaton(int32_t state, struct i2c_periph* p) {
00091   struct i2c_transaction* trans = p->trans[p->trans_extract_idx];
00092   switch (state) {
00093     case I2C_START:
00094     case I2C_RESTART:
00095       // Set R/W flag
00096       switch (trans->type) {
00097         case I2CTransRx :
00098           SetBit(trans->slave_addr,0);
00099           break;
00100         case I2CTransTx:
00101         case I2CTransTxRx:
00102           ClearBit(trans->slave_addr,0);
00103           break;
00104       }
00105       I2cSendByte(p->reg_addr,trans->slave_addr);
00106       I2cClearStart(p->reg_addr);
00107       p->idx_buf = 0;
00108       break;
00109     case I2C_MR_DATA_ACK:
00110       if (p->idx_buf < trans->len_r) {
00111         trans->buf[p->idx_buf] = ((i2cRegs_t *)(p->reg_addr))->dat;
00112         p->idx_buf++;
00113         I2cReceive(p->reg_addr,p->idx_buf < trans->len_r - 1);
00114       }
00115       else {
00116         /* error , we should have got NACK */
00117         I2cFail(p,trans);
00118       }
00119       break;
00120     case I2C_MR_DATA_NACK:
00121       if (p->idx_buf < trans->len_r) {
00122         trans->buf[p->idx_buf] = ((i2cRegs_t *)(p->reg_addr))->dat;
00123       }
00124       I2cSendStop(p,trans);
00125       break;
00126     case I2C_MR_SLA_ACK: /* At least one char */
00127       /* Wait and reply with ACK or NACK */
00128       I2cReceive(p->reg_addr,p->idx_buf < trans->len_r - 1);
00129       break;
00130     case I2C_MR_SLA_NACK:
00131     case I2C_MT_SLA_NACK:
00132       /* Slave is not responding, transaction is failed */
00133       I2cFail(p,trans);
00134       break;
00135     case I2C_MT_SLA_ACK:
00136     case I2C_MT_DATA_ACK:
00137       if (p->idx_buf < trans->len_w) {
00138         I2cSendByte(p->reg_addr,trans->buf[p->idx_buf]);
00139         p->idx_buf++;
00140       } else {
00141         if (trans->type == I2CTransTxRx) {
00142           trans->type = I2CTransRx; /* FIXME should not change type */
00143           p->idx_buf = 0;
00144           trans->slave_addr |= 1;
00145           I2cSendStart(p);
00146         } else {
00147           I2cSendStop(p,trans);
00148         }
00149       }
00150       break;
00151     default:
00152       I2cFail(p,trans);
00153       /* FIXME log error */
00154       break;
00155   }
00156 }

00321 bool_t i2c_submit(struct i2c_periph* p, struct i2c_transaction* t) {
00322 
00323   uint8_t idx;
00324   idx = p->trans_insert_idx + 1;
00325   if (idx >= I2C_TRANSACTION_QUEUE_LEN) idx = 0;
00326   if (idx == p->trans_extract_idx) {
00327     t->status = I2CTransFailed;
00328     return FALSE;  /* queue full */
00329   }
00330   t->status = I2CTransPending;
00331   int_disable();
00332   p->trans[p->trans_insert_idx] = t;
00333   p->trans_insert_idx = idx;
00334   /* if peripheral is idle, start the transaction */
00335   if (p->status == I2CIdle)
00336     I2cSendStart(p);
00337   /* else it will be started by the interrupt handler */
00338   /* when the previous transactions completes         */
00339   int_enable();
00340 
00341   return TRUE;
00342 }
