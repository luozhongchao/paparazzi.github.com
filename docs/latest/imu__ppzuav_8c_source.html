00074   /* set gyro range to 2000deg/s and low pass at 20Hz (< 60Hz/2) internal sampling at 1kHz */

00082 #  else
00083 #  error PERIODIC_FREQUENCY should be either 60Hz or 120Hz. Otherwise manually fix the sensor rates
00084 #  endif
00085 #endif
00086   ppzuavimu_itg3200.len_w = 2;
00087   i2c_submit(&PPZUAVIMU_I2C_DEVICE,&ppzuavimu_itg3200);
00088     while(ppzuavimu_itg3200.status == I2CTransPending);
00089 
00090   /* set sample rate to 66Hz: so at 60Hz there is always a new sample ready and you loose little */
00091   ppzuavimu_itg3200.buf[0] = ITG3200_REG_SMPLRT_DIV;
00092 #if PERIODIC_FREQUENCY == 60
00093   ppzuavimu_itg3200.buf[1] = 19;  // 50Hz
00094 #else
00095   ppzuavimu_itg3200.buf[1] = 9;  // 100Hz
00096 #endif

00098     while(ppzuavimu_itg3200.status == I2CTransPending);

00102   ppzuavimu_itg3200.buf[1] = 0x01;
00103   i2c_submit(&PPZUAVIMU_I2C_DEVICE,&ppzuavimu_itg3200);
00104     while(ppzuavimu_itg3200.status == I2CTransPending);
00105 
00106   /* no interrupts for now, but set data ready interrupt to enable reading status bits */
00107   ppzuavimu_itg3200.buf[0] = ITG3200_REG_INT_CFG;

00207 void ppzuavimu_module_event( void )
00208 {
00209   int32_t x, y, z;
00210 
00211   // If the itg3200 I2C transaction has succeeded: convert the data
00212   if (ppzuavimu_itg3200.status == I2CTransSuccess)
00213   {
00214 #define ITG_STA_DAT_OFFSET 3
00215     x = (int16_t) ((ppzuavimu_itg3200.buf[0+ITG_STA_DAT_OFFSET] << 8) | ppzuavimu_itg3200.buf[1+ITG_STA_DAT_OFFSET]);
00216     y = (int16_t) ((ppzuavimu_itg3200.buf[2+ITG_STA_DAT_OFFSET] << 8) | ppzuavimu_itg3200.buf[3+ITG_STA_DAT_OFFSET]);
00217     z = (int16_t) ((ppzuavimu_itg3200.buf[4+ITG_STA_DAT_OFFSET] << 8) | ppzuavimu_itg3200.buf[5+ITG_STA_DAT_OFFSET]);
00218 
00219     // Is this is new data
00220     if (ppzuavimu_itg3200.buf[0] & 0x01)
00221     {
00222       //LED_ON(3);
00223       gyr_valid = TRUE;
00224       //LED_OFF(3);
00225     }
00226     else
00227     {
00228     }
00229 
00230     // Signs depend on the way sensors are soldered on the board: so they are hardcoded
00231 #ifdef ASPIRIN_IMU
00232     RATES_ASSIGN(imu.gyro_unscaled, x, -y, -z);
00233 #else // PPZIMU
00234     RATES_ASSIGN(imu.gyro_unscaled, -x, y, -z);
00235 #endif
00236 
00237     ppzuavimu_itg3200.status = I2CTransDone;  // remove the I2CTransSuccess status, otherwise data ready will be triggered again without new data
00238   }
00239 
00240   // If the adxl345 I2C transaction has succeeded: convert the data
00241   if (ppzuavimu_adxl345.status == I2CTransSuccess)
00242   {
00243     x = (int16_t) ((ppzuavimu_adxl345.buf[1] << 8) | ppzuavimu_adxl345.buf[0]);
00244     y = (int16_t) ((ppzuavimu_adxl345.buf[3] << 8) | ppzuavimu_adxl345.buf[2]);
00245     z = (int16_t) ((ppzuavimu_adxl345.buf[5] << 8) | ppzuavimu_adxl345.buf[4]);
00246 
00247 #ifdef ASPIRIN_IMU
00248     VECT3_ASSIGN(imu.accel_unscaled, x, -y, -z);
00249 #else // PPZIMU
00250     VECT3_ASSIGN(imu.accel_unscaled, -x, y, -z);
00251 #endif
00252 
00253     acc_valid = TRUE;
00254     ppzuavimu_adxl345.status = I2CTransDone;
00255   }
00256 
00257   // If the hmc5843 I2C transaction has succeeded: convert the data
00258   if (ppzuavimu_hmc5843.status == I2CTransSuccess)
00259   {
00260     x = (int16_t) ((ppzuavimu_hmc5843.buf[0] << 8) | ppzuavimu_hmc5843.buf[1]);
00261     y = (int16_t) ((ppzuavimu_hmc5843.buf[2] << 8) | ppzuavimu_hmc5843.buf[3]);
00262     z = (int16_t) ((ppzuavimu_hmc5843.buf[4] << 8) | ppzuavimu_hmc5843.buf[5]);
00263 
00264 #ifdef ASPIRIN_IMU
00265     VECT3_ASSIGN(imu.mag_unscaled, x, -y, -z);
00266 #else // PPZIMU
00267     VECT3_ASSIGN(imu.mag_unscaled, -y, -x, -z);
00268 #endif
00269 
00270     mag_valid = TRUE;
00271     ppzuavimu_hmc5843.status = I2CTransDone;
00272   }
