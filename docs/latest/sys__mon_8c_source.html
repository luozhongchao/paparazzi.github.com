00091 void periodic_sysmon(void) {
00093   uint32_t periodic_usec = SysTimeTimer(periodic_timer);
00094   SysTimeTimerStart(periodic_timer);
00095   sum_time_periodic += periodic_usec;
00096 
00097   /* only periodic cycle : periodic_cycle = periodic_usec - sum_time_event; */
00098   periodic_cycle = periodic_usec - n_event * min_time_event;
00099   sum_cycle_periodic += periodic_cycle;
00100 
00101   /* remember min and max periodic cycle times */
00102   if (periodic_cycle < sys_mon.periodic_cycle_min)
00103     sys_mon.periodic_cycle_min = periodic_cycle;
00104   if (periodic_cycle > sys_mon.periodic_cycle_max)
00105     sys_mon.periodic_cycle_max = periodic_cycle;
00106 
00107   n_periodic++;
00108   sum_n_event += n_event;
00109   n_event = 0;
00110   sum_time_event = 0;
00111 }

00113 void event_sysmon(void) {

00116     uint32_t t = SysTimeTimer(event_timer);
00117     if (t < min_time_event)
00118       min_time_event = t;

00121   SysTimeTimerStart(event_timer);
00122   n_event++;
