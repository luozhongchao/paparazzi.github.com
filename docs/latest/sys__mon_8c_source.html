00070 void periodic_report_sysmon(void) {
00072   if (n_periodic > 0) {
00073     sys_mon.periodic_time = Max(sum_time_periodic / n_periodic, 1);
00074     sys_mon.periodic_cycle = sum_cycle_periodic / n_periodic;
00075     sys_mon.cpu_load = 100 * sys_mon.periodic_cycle / sys_mon.periodic_time;
00076     sys_mon.event_number = sum_n_event / n_periodic;

00078     DOWNLINK_SEND_SYS_MON(DefaultChannel, DefaultDevice, &sys_mon.periodic_time, &sys_mon.periodic_cycle, &sys_mon.periodic_cycle_min, &sys_mon.periodic_cycle_max, &sys_mon.event_number, &sys_mon.cpu_load);
00079   }

00089 void periodic_sysmon(void) {
00091   uint32_t periodic_usec = SysTimeTimer(periodic_timer);
00092   SysTimeTimerStart(periodic_timer);
00093   sum_time_periodic += periodic_usec;
00094 
00095   /* only periodic cycle : periodic_cycle = periodic_usec - sum_time_event; */
00096   periodic_cycle = periodic_usec - n_event * min_time_event;
00097   sum_cycle_periodic += periodic_cycle;
00098 
00099   /* remember min and max periodic cycle times */
00100   if (periodic_cycle < sys_mon.periodic_cycle_min)
00101     sys_mon.periodic_cycle_min = periodic_cycle;
00102   if (periodic_cycle > sys_mon.periodic_cycle_max)
00103     sys_mon.periodic_cycle_max = periodic_cycle;
00104 
00105   n_periodic++;
00106   sum_n_event += n_event;
00107   n_event = 0;
00108   sum_time_event = 0;
00109 }

00111 void event_sysmon(void) {
00113   if (n_event > 0) {
00114     uint32_t t = SysTimeTimer(event_timer);
00115     if (t < min_time_event)
00116       min_time_event = t;
00117     sum_time_event += t;
00118   }
00119   SysTimeTimerStart(event_timer);
00120   n_event++;
00121 }
