00197 void v_ctl_altitude_loop( void ) {
00198   float altitude_pgain_boost = 1.0;
00199 
00200 #if USE_AIRSPEED && defined(AGR_CLIMB)
00201   // Aggressive climb mode (boost gain of altitude loop)
00202   if ( v_ctl_climb_mode == V_CTL_CLIMB_MODE_AUTO_THROTTLE) {
00203     float dist = fabs(v_ctl_altitude_error);
00204     altitude_pgain_boost = 1.0 + (V_CTL_AUTO_AGR_CLIMB_GAIN-1.0)*(dist-AGR_BLEND_END)/(AGR_BLEND_START-AGR_BLEND_END);
00205     Bound(altitude_pgain_boost, 1.0, V_CTL_AUTO_AGR_CLIMB_GAIN);
00206   }
00207 #endif
00208 
00209   v_ctl_altitude_error = v_ctl_altitude_setpoint - stateGetPositionUtm_f()->alt;
00210   v_ctl_climb_setpoint = altitude_pgain_boost * v_ctl_altitude_pgain * v_ctl_altitude_error
00211     + v_ctl_altitude_pre_climb * v_ctl_altitude_pre_climb_correction;
00212   BoundAbs(v_ctl_climb_setpoint, v_ctl_altitude_max_climb);
00213 
00214 #ifdef AGR_CLIMB
00215   if ( v_ctl_climb_mode == V_CTL_CLIMB_MODE_AUTO_THROTTLE) {
00216     float dist = fabs(v_ctl_altitude_error);
00217     if (dist < AGR_BLEND_END) {
00218       v_ctl_auto_throttle_submode = V_CTL_AUTO_THROTTLE_STANDARD;
00219     }
00220     else if (dist > AGR_BLEND_START) {
00221       v_ctl_auto_throttle_submode = V_CTL_AUTO_THROTTLE_AGRESSIVE;
00222     }
00223     else {
00224       v_ctl_auto_throttle_submode = V_CTL_AUTO_THROTTLE_BLENDED;
00225     }
00226   }
00227 #endif
00228 }

00252 inline static void v_ctl_climb_auto_throttle_loop(void) {
00253   static float last_err;
00254 
00255   float f_throttle = 0;
00256   float err  = stateGetSpeedEnu_f()->z - v_ctl_climb_setpoint;
00257   float d_err = err - last_err;
00258   last_err = err;
00259   float controlled_throttle = v_ctl_auto_throttle_cruise_throttle
00260     + v_ctl_auto_throttle_climb_throttle_increment * v_ctl_climb_setpoint
00261     - v_ctl_auto_throttle_pgain *
00262     (err + v_ctl_auto_throttle_igain * v_ctl_auto_throttle_sum_err
00263      + v_ctl_auto_throttle_dgain * d_err);
00264 
00265   /* pitch pre-command */
00266   float v_ctl_pitch_of_vz = (v_ctl_climb_setpoint + d_err * v_ctl_auto_throttle_pitch_of_vz_dgain) * v_ctl_auto_throttle_pitch_of_vz_pgain;
00267 
00268 #if defined AGR_CLIMB
00269   switch (v_ctl_auto_throttle_submode) {
00270   case V_CTL_AUTO_THROTTLE_AGRESSIVE:
00271     if (v_ctl_climb_setpoint > 0) { /* Climbing */
00272       f_throttle =  AGR_CLIMB_THROTTLE;
00273       v_ctl_pitch_setpoint = AGR_CLIMB_PITCH;
00274     }
00275     else { /* Going down */
00276       f_throttle =  AGR_DESCENT_THROTTLE;
00277       v_ctl_pitch_setpoint = AGR_DESCENT_PITCH;
00278     }
00279     break;
00280 
00281   case V_CTL_AUTO_THROTTLE_BLENDED: {
00282     float ratio = (fabs(v_ctl_altitude_error) - AGR_BLEND_END)
00283       / (AGR_BLEND_START - AGR_BLEND_END);
00284     f_throttle = (1-ratio) * controlled_throttle;
00285     v_ctl_pitch_setpoint = (1-ratio) * v_ctl_pitch_of_vz;
00286     v_ctl_auto_throttle_sum_err += (1-ratio) * err;
00287     BoundAbs(v_ctl_auto_throttle_sum_err, V_CTL_AUTO_THROTTLE_MAX_SUM_ERR);
00288     /* positive error -> too low */
00289     if (v_ctl_altitude_error > 0) {
00290       f_throttle +=  ratio * AGR_CLIMB_THROTTLE;
00291       v_ctl_pitch_setpoint += ratio * AGR_CLIMB_PITCH;
00292     } else {
00293       f_throttle += ratio * AGR_DESCENT_THROTTLE;
00294       v_ctl_pitch_setpoint += ratio * AGR_DESCENT_PITCH;
00295     }
00296     break;
00297   }
00298 
00299   case V_CTL_AUTO_THROTTLE_STANDARD:
00300   default:
00301 #endif
00302     f_throttle = controlled_throttle;
00303     v_ctl_auto_throttle_sum_err += err;
00304     BoundAbs(v_ctl_auto_throttle_sum_err, V_CTL_AUTO_THROTTLE_MAX_SUM_ERR);
00305     v_ctl_pitch_setpoint = v_ctl_pitch_of_vz + nav_pitch;
00306 #if defined AGR_CLIMB
00307     break;
00308   } /* switch submode */
00309 #endif
00310 
00311   v_ctl_throttle_setpoint = TRIM_UPPRZ(f_throttle * MAX_PPRZ);
00312 }

00316 inline static void v_ctl_climb_auto_throttle_loop(void) {
00317   float f_throttle = 0;
00318   float controlled_throttle;
00319   float v_ctl_pitch_of_vz;
00320 
00321   // Limit rate of change of climb setpoint (to ensure that airspeed loop can catch-up)
00322   static float v_ctl_climb_setpoint_last = 0;
00323   float diff_climb = v_ctl_climb_setpoint - v_ctl_climb_setpoint_last;
00324   Bound(diff_climb, -V_CTL_AUTO_CLIMB_LIMIT, V_CTL_AUTO_CLIMB_LIMIT);
00325   v_ctl_climb_setpoint = v_ctl_climb_setpoint_last + diff_climb;
00326   v_ctl_climb_setpoint_last = v_ctl_climb_setpoint;
00327 
00328   // Pitch control (input: rate of climb error, output: pitch setpoint)
00329   float err  = stateGetSpeedEnu_f()->z - v_ctl_climb_setpoint;
00330   v_ctl_auto_pitch_sum_err += err;
00331   BoundAbs(v_ctl_auto_pitch_sum_err, V_CTL_AUTO_PITCH_MAX_SUM_ERR);
00332   v_ctl_pitch_of_vz = -v_ctl_auto_pitch_pgain *
00333     (err + v_ctl_auto_pitch_igain * v_ctl_auto_pitch_sum_err);
00334 
00335   // Ground speed control loop (input: groundspeed error, output: airspeed controlled)
00336   float err_groundspeed = (v_ctl_auto_groundspeed_setpoint - *stateGetHorizontalSpeedNorm_f());
00337   v_ctl_auto_groundspeed_sum_err += err_groundspeed;
00338   BoundAbs(v_ctl_auto_groundspeed_sum_err, V_CTL_AUTO_GROUNDSPEED_MAX_SUM_ERR);
00339   v_ctl_auto_airspeed_controlled = (err_groundspeed + v_ctl_auto_groundspeed_sum_err * v_ctl_auto_groundspeed_igain) * v_ctl_auto_groundspeed_pgain;
00340 
00341   // Do not allow controlled airspeed below the setpoint
00342   if (v_ctl_auto_airspeed_controlled < v_ctl_auto_airspeed_setpoint) {
00343     v_ctl_auto_airspeed_controlled = v_ctl_auto_airspeed_setpoint;
00344     v_ctl_auto_groundspeed_sum_err = v_ctl_auto_airspeed_controlled/(v_ctl_auto_groundspeed_pgain*v_ctl_auto_groundspeed_igain); // reset integrator of ground speed loop
00345   }
00346 
00347   // Airspeed control loop (input: airspeed controlled, output: throttle controlled)
00348   float err_airspeed = (v_ctl_auto_airspeed_controlled - *stateGetAirspeed_f());
00349   v_ctl_auto_airspeed_sum_err += err_airspeed;
00350   BoundAbs(v_ctl_auto_airspeed_sum_err, V_CTL_AUTO_AIRSPEED_MAX_SUM_ERR);
00351   controlled_throttle = (err_airspeed + v_ctl_auto_airspeed_sum_err * v_ctl_auto_airspeed_igain) * v_ctl_auto_airspeed_pgain;
00352 
00353   // Done, set outputs
00354   Bound(controlled_throttle, 0, v_ctl_auto_throttle_max_cruise_throttle);
00355   f_throttle = controlled_throttle;
00356   v_ctl_pitch_setpoint = v_ctl_pitch_of_vz;
00357   v_ctl_throttle_setpoint = TRIM_UPPRZ(f_throttle * MAX_PPRZ);
00358   Bound(v_ctl_pitch_setpoint, V_CTL_AUTO_PITCH_MIN_PITCH, V_CTL_AUTO_PITCH_MAX_PITCH);
00359 }
