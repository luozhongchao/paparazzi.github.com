162 void alt_kalman(float z_meas) {
163  float DT;
164  float R;
165  float SIGMA2;
166 
167 #if USE_BAROMETER
168 #if USE_BARO_MS5534A
169  if (alt_baro_enabled) {
170  DT = BARO_DT;
171  R = baro_MS5534A_r;
172  SIGMA2 = baro_MS5534A_sigma2;
173  } else
174 #elif USE_BARO_ETS
175  if (baro_ets_enabled) {
176  DT = BARO_ETS_DT;
177  R = baro_ets_r;
178  SIGMA2 = baro_ets_sigma2;
179  } else
180 #elif USE_BARO_MS5611
181  if (baro_ms5611_enabled) {
182  DT = BARO_MS5611_DT;
183  R = baro_ms5611_r;
184  SIGMA2 = baro_ms5611_sigma2;
185  } else
186 #elif USE_BARO_BMP
187  if (baro_bmp_enabled) {
188  DT = BARO_BMP_DT;
189  R = baro_bmp_r;
190  SIGMA2 = baro_bmp_sigma2;
191  } else
192 #endif
193 #endif // USE_BAROMETER
194  {
195  DT = GPS_DT;
196  R = GPS_R;
197  SIGMA2 = GPS_SIGMA2;
198  }
199 
200  float q[2][2];
201  q[0][0] = DT*DT*DT*DT/4.;
202  q[0][1] = DT*DT*DT/2.;
203  q[1][0] = DT*DT*DT/2.;
204  q[1][1] = DT*DT;
205 
206 
207  /* predict */
208  estimator_z += estimator_z_dot * DT;
209  p[0][0] = p[0][0]+p[1][0]*DT+DT*(p[0][1]+p[1][1]*DT) + SIGMA2*q[0][0];
210  p[0][1] = p[0][1]+p[1][1]*DT + SIGMA2*q[0][1];
211  p[1][0] = p[1][0]+p[1][1]*DT + SIGMA2*q[1][0];
212  p[1][1] = p[1][1] + SIGMA2*q[1][1];
213 
214  /* error estimate */
215  float e = p[0][0] + R;
216 
217  if (fabs(e) > 1e-5) {
218  float k_0 = p[0][0] / e;
219  float k_1 = p[1][0] / e;
220  e = z_meas - estimator_z;
221 
222  /* correction */
223  estimator_z += k_0 * e;
224  estimator_z_dot += k_1 * e;
225 
226  p[1][0] = -p[0][0]*k_1+p[1][0];
227  p[1][1] = -p[0][1]*k_1+p[1][1];
228  p[0][0] = p[0][0] * (1-k_0);
229  p[0][1] = p[0][1] * (1-k_0);
230  }
231 
232 #ifdef DEBUG_ALT_KALMAN
233  DOWNLINK_SEND_ALT_KALMAN(DefaultChannel,DefaultDevice,&(p[0][0]),&(p[0][1]),&(p[1][0]), &(p[1][1]));
234 #endif
235 }
