00100 #define FillBufWith32bit(_buf, _index, _value) {  \
00101   _buf[_index] = (uint8_t) (_value);              \
00102   _buf[_index+1] = (uint8_t) ((_value) >> 8);     \
00103   _buf[_index+2] = (uint8_t) ((_value) >> 16);    \
00104   _buf[_index+3] = (uint8_t) ((_value) >> 24);    \

00107 void ArduIMU_periodicGPS( void ) {
00108 
00109   if (ardu_gps_trans.status != I2CTransDone) { return; }
00110 
00111 #if USE_HIGH_ACCEL_FLAG
00112   // Test for high acceleration:
00113   //  - low speed
00114   //  - high thrust
00115   float speed = *stateGetHorizontalSpeedNorm_f();
00116   if (speed < HIGH_ACCEL_LOW_SPEED && ap_state->commands[COMMAND_THROTTLE] > HIGH_ACCEL_HIGH_THRUST && !high_accel_done) {
00117     high_accel_flag = TRUE;
00118   } else {
00119     high_accel_flag = FALSE;
00120     if (speed > HIGH_ACCEL_LOW_SPEED && !high_accel_done) {
00121       high_accel_done = TRUE; // After takeoff, don't use high accel before landing (GS small, Throttle small)
00122     }
00123     if (speed < HIGH_ACCEL_HIGH_THRUST_RESUME && ap_state->commands[COMMAND_THROTTLE] < HIGH_ACCEL_HIGH_THRUST_RESUME) {
00124       high_accel_done = FALSE; // Activate high accel after landing
00125     }
00126   }
00127 #endif
00128 
00129   FillBufWith32bit(ardu_gps_trans.buf, 0, (int32_t)gps.speed_3d); // speed 3D
00130   FillBufWith32bit(ardu_gps_trans.buf, 4, (int32_t)gps.gspeed);   // ground speed
00131   FillBufWith32bit(ardu_gps_trans.buf, 8, (int32_t)gps.course);   // course
00132   ardu_gps_trans.buf[12] = gps.fix;                               // status gps fix
00133   ardu_gps_trans.buf[13] = (uint8_t)arduimu_calibrate_neutrals;   // calibration flag
00134   ardu_gps_trans.buf[14] = (uint8_t)high_accel_flag;              // high acceleration flag (disable accelerometers in the arduimu filter)
00135   I2CTransmit(ARDUIMU_I2C_DEV, ardu_gps_trans, ArduIMU_SLAVE_ADDR, 15);
00136 
00137   // Reset calibration flag
00138   if (arduimu_calibrate_neutrals) arduimu_calibrate_neutrals = FALSE;
00139 }

00163 void ArduIMU_event( void ) {
00164   // Handle INS I2C event
00165   if (ardu_ins_trans.status == I2CTransSuccess) {
00166     // received data from I2C transaction
00167     recievedData[0] = (ardu_ins_trans.buf[1]<<8) | ardu_ins_trans.buf[0];
00168     recievedData[1] = (ardu_ins_trans.buf[3]<<8) | ardu_ins_trans.buf[2];
00169     recievedData[2] = (ardu_ins_trans.buf[5]<<8) | ardu_ins_trans.buf[4];
00170     recievedData[3] = (ardu_ins_trans.buf[7]<<8) | ardu_ins_trans.buf[6];
00171     recievedData[4] = (ardu_ins_trans.buf[9]<<8) | ardu_ins_trans.buf[8];
00172     recievedData[5] = (ardu_ins_trans.buf[11]<<8) | ardu_ins_trans.buf[10];
00173     recievedData[6] = (ardu_ins_trans.buf[13]<<8) | ardu_ins_trans.buf[12];
00174     recievedData[7] = (ardu_ins_trans.buf[15]<<8) | ardu_ins_trans.buf[14];
00175     recievedData[8] = (ardu_ins_trans.buf[17]<<8) | ardu_ins_trans.buf[16];
00176 
00177     // Update ArduIMU data
00178     arduimu_eulers.phi = ANGLE_FLOAT_OF_BFP(recievedData[0]) - ins_roll_neutral;
00179     arduimu_eulers.theta = ANGLE_FLOAT_OF_BFP(recievedData[1]) - ins_pitch_neutral;
00180     arduimu_eulers.psi = ANGLE_FLOAT_OF_BFP(recievedData[2]);
00181     arduimu_rates.p = RATE_FLOAT_OF_BFP(recievedData[3]);
00182     arduimu_rates.q = RATE_FLOAT_OF_BFP(recievedData[4]);
00183     arduimu_rates.r = RATE_FLOAT_OF_BFP(recievedData[5]);
00184     arduimu_accel.x = ACCEL_FLOAT_OF_BFP(recievedData[6]);
00185     arduimu_accel.y = ACCEL_FLOAT_OF_BFP(recievedData[7]);
00186     arduimu_accel.z = ACCEL_FLOAT_OF_BFP(recievedData[8]);
00187 
00188     // Update estimator
00189     stateSetNedToBodyEulers_f(&arduimu_eulers);
00190     stateSetBodyRates_f(&arduimu_rates);
00191     stateSetAccelNed_f(&((struct NedCoor_f)arduimu_accel));
00192     ardu_ins_trans.status = I2CTransDone;
00193 
00194 #ifdef ARDUIMU_SYNC_SEND
00195     //RunOnceEvery(15, DOWNLINK_SEND_AHRS_EULER(DefaultChannel, DefaultDevice, &arduimu_eulers.phi, &arduimu_eulers.theta, &arduimu_eulers.psi));
00196     RunOnceEvery(15, DOWNLINK_SEND_IMU_GYRO(DefaultChannel, DefaultDevice, &arduimu_rates.p, &arduimu_rates.q, &arduimu_rates.r));
00197     RunOnceEvery(15, DOWNLINK_SEND_IMU_ACCEL(DefaultChannel, DefaultDevice, &arduimu_accel.x, &arduimu_accel.y, &arduimu_accel.z));
00198 #endif
00199   }
00200   else if (ardu_ins_trans.status == I2CTransFailed) {
00201     ardu_ins_trans.status = I2CTransDone;
00202   }
00203   // Handle GPS I2C event
00204   if (ardu_gps_trans.status == I2CTransSuccess || ardu_gps_trans.status == I2CTransFailed) {
00205     ardu_gps_trans.status = I2CTransDone;
00206   }
00207 }
