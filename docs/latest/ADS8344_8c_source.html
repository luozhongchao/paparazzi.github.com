88 void ADS8344_init( void ) {
89  channel = 0;
90  ADS8344_available = FALSE;
91 
92  /* setup pins for SSP (SCK, MISO, MOSI) */
93  PINSEL1 |= 2 << 2 | 2 << 4 | 2 << 6;
94 
95  /* setup SSP */
96  SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
97  SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
98  SSPCPSR = CPSDVSR; /* -> 50kHz */
99 
100  /* initialize interrupt vector */
101  VICIntSelect &= ~VIC_BIT(VIC_SPI1); // SPI1 selected as IRQ
102  VICIntEnable = VIC_BIT(VIC_SPI1); // SPI1 interrupt enabled
103  _VIC_CNTL(SPI1_VIC_SLOT) = VIC_ENABLE | VIC_SPI1;
104  _VIC_CNTL(SPI1_VIC_SLOT) = (uint32_t)SPI1_ISR; /* address of the ISR */
105 
106  /* setup slave select */
107  /* configure SS pin */
108  SetBit( ADS8344_SS_IODIR, ADS8344_SS_PIN); /* pin is output */
109  ADS8344Unselect(); /* pin low */
110 }

112 static inline void read_values( void ) {
113  uint8_t foo __attribute__ ((unused)) = SSPDR;
114  uint8_t msb = SSPDR;

137 void SPI1_ISR(void) {
138  ISR_ENTRY();
139  LED_TOGGLE(2);
