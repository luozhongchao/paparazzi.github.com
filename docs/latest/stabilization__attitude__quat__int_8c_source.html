89 static void attitude_run_ff(int32_t ff_commands[], struct Int32AttitudeGains *gains, struct Int32Rates *ref_accel)
90 {
91  /* Compute feedforward based on reference acceleration */
92 
93  ff_commands[COMMAND_ROLL] = GAIN_PRESCALER_FF * gains->dd.x * RATE_FLOAT_OF_BFP(ref_accel->p) / (1 << 7);
94  ff_commands[COMMAND_PITCH] = GAIN_PRESCALER_FF * gains->dd.y * RATE_FLOAT_OF_BFP(ref_accel->q) / (1 << 7);
95  ff_commands[COMMAND_YAW] = GAIN_PRESCALER_FF * gains->dd.z * RATE_FLOAT_OF_BFP(ref_accel->r) / (1 << 7);
96 }

98 static void attitude_run_fb(int32_t fb_commands[], struct Int32AttitudeGains *gains, struct Int32Quat *att_err,
99  struct Int32Rates *rate_err, struct Int32Quat *sum_err)
100 {
101  /* PID feedback */
102  fb_commands[COMMAND_ROLL] =
103  GAIN_PRESCALER_P * gains->p.x * QUAT1_FLOAT_OF_BFP(att_err->qx) / 4 +
104  GAIN_PRESCALER_D * gains->d.x * RATE_FLOAT_OF_BFP(rate_err->p) / 16 +
105  GAIN_PRESCALER_I * gains->i.x * QUAT1_FLOAT_OF_BFP(sum_err->qx) / 2;
106 
107  fb_commands[COMMAND_PITCH] =
108  GAIN_PRESCALER_P * gains->p.y * QUAT1_FLOAT_OF_BFP(att_err->qy) / 4 +
109  GAIN_PRESCALER_D * gains->d.y * RATE_FLOAT_OF_BFP(rate_err->q) / 16 +
110  GAIN_PRESCALER_I * gains->i.y * QUAT1_FLOAT_OF_BFP(sum_err->qy) / 2;
111 
112  fb_commands[COMMAND_YAW] =
113  GAIN_PRESCALER_P * gains->p.z * QUAT1_FLOAT_OF_BFP(att_err->qz) / 4 +
114  GAIN_PRESCALER_D * gains->d.z * RATE_FLOAT_OF_BFP(rate_err->r) / 16 +
115  GAIN_PRESCALER_I * gains->i.z * QUAT1_FLOAT_OF_BFP(sum_err->qz) / 2;

117 }

119 void stabilization_attitude_run(bool_t enable_integrator) {
120 
121  /*
122  * Update reference
123  */
124  stabilization_attitude_ref_update();
125 
126  /*
127  * Compute errors for feedback
128  */
129 
130  /* attitude error */
131  struct Int32Quat att_err;
132  struct Int32Quat* att_quat = stateGetNedToBodyQuat_i();
133  INT32_QUAT_INV_COMP(att_err, *att_quat, stab_att_ref_quat);
134  /* wrap it in the shortest direction */
135  INT32_QUAT_WRAP_SHORTEST(att_err);
136  INT32_QUAT_NORMALIZE(att_err);
137 
138  /* rate error */
139  struct Int32Rates rate_err;
140  struct Int32Rates* body_rate = stateGetBodyRates_i();
141  RATES_DIFF(rate_err, stab_att_ref_rate, *body_rate);
142 
143  /* integrated error */
144  if (enable_integrator) {
145  struct Int32Quat new_sum_err, scaled_att_err;
146  /* update accumulator */
147  scaled_att_err.qi = att_err.qi;
148  scaled_att_err.qx = att_err.qx / IERROR_SCALE;
149  scaled_att_err.qy = att_err.qy / IERROR_SCALE;
150  scaled_att_err.qz = att_err.qz / IERROR_SCALE;
151  INT32_QUAT_COMP(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
152  INT32_QUAT_NORMALIZE(new_sum_err);
153  QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
154  INT32_EULERS_OF_QUAT(stabilization_att_sum_err, stabilization_att_sum_err_quat);
155  } else {
156  /* reset accumulator */
157  INT32_QUAT_ZERO( stabilization_att_sum_err_quat );
158  INT_EULERS_ZERO( stabilization_att_sum_err );
159  }
160 
161  /* compute the feed forward command */
162  attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains, &stab_att_ref_accel);
163 
164  /* compute the feed back command */
165  attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains, &att_err, &rate_err, &stabilization_att_sum_err_quat);
166 
167  /* sum feedforward and feedback */
168  stabilization_cmd[COMMAND_ROLL] = stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL];
169  stabilization_cmd[COMMAND_PITCH] = stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH];
170  stabilization_cmd[COMMAND_YAW] = stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW];
171 
172  /* bound the result */
173  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
174  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
175  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
176 }
