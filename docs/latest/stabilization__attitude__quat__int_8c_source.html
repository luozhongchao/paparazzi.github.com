00089 static void attitude_run_ff(int32_t ff_commands[], struct Int32AttitudeGains *gains, struct Int32Rates *ref_accel)
00090 {
00091   /* Compute feedforward based on reference acceleration */
00092 
00093   ff_commands[COMMAND_ROLL]  = GAIN_PRESCALER_FF * gains->dd.x * RATE_FLOAT_OF_BFP(ref_accel->p) / (1 << 7);
00094   ff_commands[COMMAND_PITCH] = GAIN_PRESCALER_FF * gains->dd.y * RATE_FLOAT_OF_BFP(ref_accel->q) / (1 << 7);
00095   ff_commands[COMMAND_YAW]   = GAIN_PRESCALER_FF * gains->dd.z * RATE_FLOAT_OF_BFP(ref_accel->r) / (1 << 7);
00096 }

00119 void stabilization_attitude_run(bool_t enable_integrator) {
00120 
00121   /*
00122    * Update reference
00123    */
00124   stabilization_attitude_ref_update();
00125 
00126   /*
00127    * Compute errors for feedback
00128    */
00129 
00130   /* attitude error                          */
00131   struct Int32Quat att_err;
00132   struct Int32Quat* att_quat = stateGetNedToBodyQuat_i();
00133   INT32_QUAT_INV_COMP(att_err, *att_quat, stab_att_ref_quat);
00134   /* wrap it in the shortest direction       */
00135   INT32_QUAT_WRAP_SHORTEST(att_err);
00136   INT32_QUAT_NORMALIZE(att_err);
00137 
00138   /*  rate error                */
00139   struct Int32Rates rate_err;
00140   struct Int32Rates* body_rate = stateGetBodyRates_i();
00141   RATES_DIFF(rate_err, stab_att_ref_rate, *body_rate);
00142 
00143   /* integrated error */
00144   if (enable_integrator) {
00145     struct Int32Quat new_sum_err, scaled_att_err;
00146     /* update accumulator */
00147     scaled_att_err.qi = att_err.qi;
00148     scaled_att_err.qx = att_err.qx / IERROR_SCALE;
00149     scaled_att_err.qy = att_err.qy / IERROR_SCALE;
00150     scaled_att_err.qz = att_err.qz / IERROR_SCALE;
00151     INT32_QUAT_COMP(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
00152     INT32_QUAT_NORMALIZE(new_sum_err);
00153     QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
00154     INT32_EULERS_OF_QUAT(stabilization_att_sum_err, stabilization_att_sum_err_quat);
00155   } else {
00156     /* reset accumulator */
00157     INT32_QUAT_ZERO( stabilization_att_sum_err_quat );
00158     INT_EULERS_ZERO( stabilization_att_sum_err );
00159   }
00160 
00161   /* compute the feed forward command */
00162   attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains, &stab_att_ref_accel);
00163 
00164   /* compute the feed back command */
00165   attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains, &att_err, &rate_err, &stabilization_att_sum_err_quat);
00166 
00167   /* sum feedforward and feedback */
00168   stabilization_cmd[COMMAND_ROLL] = stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL];
00169   stabilization_cmd[COMMAND_PITCH] = stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH];
00170   stabilization_cmd[COMMAND_YAW] = stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW];
00171 
00172   /* bound the result */
00173   BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
00174   BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
00175   BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
00176 }
