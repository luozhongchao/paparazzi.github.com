00200 void stabilization_attitude_run(bool_t enable_integrator) {
00201 
00202   /*
00203    * Update reference
00204    */
00205   stabilization_attitude_ref_update();
00206 
00207   /*
00208    * Compute errors for feedback
00209    */
00210 
00211   /* attitude error                          */
00212   struct FloatQuat att_err;
00213   struct FloatQuat* att_quat = stateGetNedToBodyQuat_f();
00214   FLOAT_QUAT_INV_COMP(att_err, *att_quat, stab_att_ref_quat);
00215   /* wrap it in the shortest direction       */
00216   FLOAT_QUAT_WRAP_SHORTEST(att_err);
00217 
00218   /*  rate error                */
00219   struct FloatRates rate_err;
00220   struct FloatRates* body_rate = stateGetBodyRates_f();
00221   RATES_DIFF(rate_err, stab_att_ref_rate, *body_rate);
00222   /* rate_d error               */
00223   struct FloatRates body_rate_d;
00224   RATES_DIFF(body_rate_d, *body_rate, last_body_rate);
00225   RATES_COPY(last_body_rate, *body_rate);
00226 
00227   /* integrated error */
00228   if (enable_integrator) {
00229     struct FloatQuat new_sum_err, scaled_att_err;
00230     /* update accumulator */
00231     scaled_att_err.qi = att_err.qi;
00232     scaled_att_err.qx = att_err.qx / IERROR_SCALE;
00233     scaled_att_err.qy = att_err.qy / IERROR_SCALE;
00234     scaled_att_err.qz = att_err.qz / IERROR_SCALE;
00235     FLOAT_QUAT_COMP(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
00236     FLOAT_QUAT_NORMALIZE(new_sum_err);
00237     FLOAT_QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
00238     FLOAT_EULERS_OF_QUAT(stabilization_att_sum_err_eulers, stabilization_att_sum_err_quat);
00239   } else {
00240     /* reset accumulator */
00241     FLOAT_QUAT_ZERO( stabilization_att_sum_err_quat );
00242     FLOAT_EULERS_ZERO( stabilization_att_sum_err_eulers );
00243   }
00244 
00245   attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains[gain_idx], &stab_att_ref_accel);
00246 
00247   attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains[gain_idx], &att_err, &rate_err, &body_rate_d, &stabilization_att_sum_err_quat);
00248 
00249   stabilization_cmd[COMMAND_ROLL] = stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL];
00250   stabilization_cmd[COMMAND_PITCH] = stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH];
00251   stabilization_cmd[COMMAND_YAW] = stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW];
00252 
00253 #ifdef HAS_SURFACE_COMMANDS
00254   stabilization_cmd[COMMAND_ROLL_SURFACE] = stabilization_att_fb_cmd[COMMAND_ROLL_SURFACE] + stabilization_att_ff_cmd[COMMAND_ROLL_SURFACE];
00255   stabilization_cmd[COMMAND_PITCH_SURFACE] = stabilization_att_fb_cmd[COMMAND_PITCH_SURFACE] + stabilization_att_ff_cmd[COMMAND_PITCH_SURFACE];
00256   stabilization_cmd[COMMAND_YAW_SURFACE] = stabilization_att_fb_cmd[COMMAND_YAW_SURFACE] + stabilization_att_ff_cmd[COMMAND_YAW_SURFACE];
00257 #endif
00258 
00259   /* bound the result */
00260   BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
00261   BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
00262   BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
00263 }
