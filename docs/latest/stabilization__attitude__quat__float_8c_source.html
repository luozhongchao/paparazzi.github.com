200 void stabilization_attitude_run(bool_t enable_integrator) {
201 
202  /*
203  * Update reference
204  */
205  stabilization_attitude_ref_update();
206 
207  /*
208  * Compute errors for feedback
209  */
210 
211  /* attitude error */
212  struct FloatQuat att_err;
213  struct FloatQuat* att_quat = stateGetNedToBodyQuat_f();
214  FLOAT_QUAT_INV_COMP(att_err, *att_quat, stab_att_ref_quat);
215  /* wrap it in the shortest direction */
216  FLOAT_QUAT_WRAP_SHORTEST(att_err);
217 
218  /* rate error */
219  struct FloatRates rate_err;
220  struct FloatRates* body_rate = stateGetBodyRates_f();
221  RATES_DIFF(rate_err, stab_att_ref_rate, *body_rate);
222  /* rate_d error */
223  struct FloatRates body_rate_d;
224  RATES_DIFF(body_rate_d, *body_rate, last_body_rate);
225  RATES_COPY(last_body_rate, *body_rate);
226 
227  /* integrated error */
228  if (enable_integrator) {
229  struct FloatQuat new_sum_err, scaled_att_err;
230  /* update accumulator */
231  scaled_att_err.qi = att_err.qi;
232  scaled_att_err.qx = att_err.qx / IERROR_SCALE;
233  scaled_att_err.qy = att_err.qy / IERROR_SCALE;
234  scaled_att_err.qz = att_err.qz / IERROR_SCALE;
235  FLOAT_QUAT_COMP(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
236  FLOAT_QUAT_NORMALIZE(new_sum_err);
237  FLOAT_QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
238  FLOAT_EULERS_OF_QUAT(stabilization_att_sum_err_eulers, stabilization_att_sum_err_quat);
239  } else {
240  /* reset accumulator */
241  FLOAT_QUAT_ZERO( stabilization_att_sum_err_quat );
242  FLOAT_EULERS_ZERO( stabilization_att_sum_err_eulers );
243  }
244 
245  attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains[gain_idx], &stab_att_ref_accel);
246 
247  attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains[gain_idx], &att_err, &rate_err, &body_rate_d, &stabilization_att_sum_err_quat);
248 
249  stabilization_cmd[COMMAND_ROLL] = stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL];
250  stabilization_cmd[COMMAND_PITCH] = stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH];
251  stabilization_cmd[COMMAND_YAW] = stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW];
252 
253 #ifdef HAS_SURFACE_COMMANDS
254  stabilization_cmd[COMMAND_ROLL_SURFACE] = stabilization_att_fb_cmd[COMMAND_ROLL_SURFACE] + stabilization_att_ff_cmd[COMMAND_ROLL_SURFACE];
255  stabilization_cmd[COMMAND_PITCH_SURFACE] = stabilization_att_fb_cmd[COMMAND_PITCH_SURFACE] + stabilization_att_ff_cmd[COMMAND_PITCH_SURFACE];
256  stabilization_cmd[COMMAND_YAW_SURFACE] = stabilization_att_fb_cmd[COMMAND_YAW_SURFACE] + stabilization_att_ff_cmd[COMMAND_YAW_SURFACE];
257 #endif
258 
259  /* bound the result */
260  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
261  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
262  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
263 }
