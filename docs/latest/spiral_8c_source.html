51 bool_t InitializeSpiral(uint8_t CenterWP, uint8_t EdgeWP, float StartRad, float IncRad, float Segments, float ZKoord)
52 {
53  Center = CenterWP; // center of the helix
54  Edge = EdgeWP; // edge point on the maximaum radius
55  SRad = StartRad; // start radius of the helix
56  Segmente = Segments;
57  ZPoint = ZKoord;
58  nav_radius_min = MIN_CIRCLE_RADIUS;
59  if (SRad < nav_radius_min) SRad = nav_radius_min;
60  IRad = IncRad; // multiplier for increasing the spiral
61 
62  EdgeCurrentX = WaypointX(Edge) - WaypointX(Center);
63  EdgeCurrentY = WaypointY(Edge) - WaypointY(Center);
64 
65  Spiralradius = sqrt(EdgeCurrentX*EdgeCurrentX+EdgeCurrentY*EdgeCurrentY);
66 
67  TransCurrentX = stateGetPositionEnu_f()->x - WaypointX(Center);
68  TransCurrentY = stateGetPositionEnu_f()->y - WaypointY(Center);
69  TransCurrentZ = stateGetPositionEnu_f()->z - ZPoint;
70  DistanceFromCenter = sqrt(TransCurrentX*TransCurrentX+TransCurrentY*TransCurrentY);
71 
72  // SpiralTheta = atan2(TransCurrentY,TransCurrentX);
73  // Fly2X = Spiralradius*cos(SpiralTheta+M_PI)+WaypointX(Center);
74  // Fly2Y = Spiralradius*sin(SpiralTheta+M_PI)+WaypointY(Center);
75 
76  // Alphalimit denotes angle, where the radius will be increased
77  Alphalimit = 2*M_PI / Segments;
78  //current position
79  FlyFromX = stateGetPositionEnu_f()->x;
80  FlyFromY = stateGetPositionEnu_f()->y;
81 
82  if(DistanceFromCenter > Spiralradius)
83  CSpiralStatus = Outside;
84  return FALSE;
85 }

87 bool_t SpiralNav(void)
88 {
89  TransCurrentX = stateGetPositionEnu_f()->x - WaypointX(Center);
90  TransCurrentY = stateGetPositionEnu_f()->y - WaypointY(Center);
91  DistanceFromCenter = sqrt(TransCurrentX*TransCurrentX+TransCurrentY*TransCurrentY);
92 
93  float DistanceStartEstim;
94  float CircleAlpha;
95 
96  switch(CSpiralStatus)
97  {
98  case Outside:
99  //flys until center of the helix is reached an start helix
100  nav_route_xy(FlyFromX,FlyFromY,WaypointX(Center), WaypointY(Center));
101  // center reached?
102  if (nav_approaching_xy(WaypointX(Center), WaypointY(Center), FlyFromX, FlyFromY, 0)) {
103  // nadir image
104 #ifdef DIGITAL_CAM
105  dc_send_command(DC_SHOOT);
106 #endif
107  CSpiralStatus = StartCircle;
108  }
109  break;
110  case StartCircle:
111  // Starts helix
112  // storage of current coordinates
113  // calculation needed, State switch to Circle
114  nav_circle_XY(WaypointX(Center), WaypointY(Center), SRad);
115  if(DistanceFromCenter >= SRad){
116  LastCircleX = stateGetPositionEnu_f()->x;
117  LastCircleY = stateGetPositionEnu_f()->y;
118  CSpiralStatus = Circle;
119  // Start helix
120 #ifdef DIGITAL_CAM
121  dc_Circle(360/Segmente);
122 #endif
123  }
124  break;
125  case Circle: {
126  nav_circle_XY(WaypointX(Center), WaypointY(Center), SRad);
127  // Trigonometrische Berechnung des bereits geflogenen Winkels alpha
128  // equation:
129  // alpha = 2 * asin ( |Starting position angular - current positon| / (2* SRad)
130  // if alphamax already reached, increase radius.
131 
132  //DistanceStartEstim = |Starting position angular - current positon|
133  DistanceStartEstim = sqrt (((LastCircleX-stateGetPositionEnu_f()->x)*(LastCircleX-stateGetPositionEnu_f()->x))
134  + ((LastCircleY-stateGetPositionEnu_f()->y)*(LastCircleY-stateGetPositionEnu_f()->y)));
135  CircleAlpha = (2.0 * asin (DistanceStartEstim / (2 * SRad)));
136  if (CircleAlpha >= Alphalimit) {
137  LastCircleX = stateGetPositionEnu_f()->x;
138  LastCircleY = stateGetPositionEnu_f()->y;
139  CSpiralStatus = IncSpiral;
140  }
141  break;
142  }
143  case IncSpiral:
144  // increasing circle radius as long as it is smaller than max helix radius
145  if(SRad + IRad < Spiralradius)
146  {
147  SRad = SRad + IRad;
148 #ifdef DIGITAL_CAM
149  if (dc_cam_tracing) {
150  // calculating Cam angle for camera alignment
151  TransCurrentZ = stateGetPositionEnu_f()->z - ZPoint;
152  dc_cam_angle = atan(SRad/TransCurrentZ) * 180 / M_PI;
153  }
154 #endif
155  }
156  else {
157  SRad = Spiralradius;
158 #ifdef DIGITAL_CAM
159  // Stopps DC
160  dc_stop();
161 #endif
162  }
163  CSpiralStatus = Circle;
164  break;
165  default:
166  break;
167  }
168  return TRUE;
169 }
