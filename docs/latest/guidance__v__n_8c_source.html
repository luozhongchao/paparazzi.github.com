00207   static float last_err = 0.;
00208 
00209   if (pprz_mode == PPRZ_MODE_MANUAL || launch == 0)
00210     v_ctl_auto_pitch_sum_err = 0;
00211 
00212   // Compute errors
00213   float err = v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z;
