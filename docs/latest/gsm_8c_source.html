00201 // A line of length gsm_buf_len is available in the gsm_buf buffer
00202 static void gsm_got_line(void)
00203 {
00204   if (gsm_status == STATUS_WAITING_DATA) { // Currently receiving a SMS
00205     gsm_receive_content();
00206     Suppr_SMS(index_msg);
00207     gsm_status = STATUS_DELETE_SMS;
00208   } else if (gsm_status == STATUS_IDLE
00209       && strncmp(CMTI, gsm_buf, strlen(CMTI)) == 0) {
00210     /* A SMS is available */
00211     /* Extracting the index of the message */
00212     char * first_comma = indexn(gsm_buf, ',',MAXLEN_CMTI_ANSWER);
00213     if (first_comma) {
00214       index_msg = atoi(first_comma+1);
00215       request_for_msg();
00216       gsm_status = STATUS_REQUESTING_MESSAGE;
00217     }
00218   } else if (waiting_for_reply) { // Other cases
00219     // Do we get what we were expecting
00220 
00221     bool gsm_answer = strncmp(expected_ack, gsm_buf, strlen(expected_ack)) == 0;
00222     if (gsm_answer) {
00223       waiting_for_reply = false;
00224 
00225       switch(gsm_status) {
00226         case STATUS_CSQ :
00227           gsm_send_report_continue();
00228           gsm_status = STATUS_WAITING_PROMPT;
00229           break;
00230 
00231         case STATUS_REQUESTING_MESSAGE:
00232           parse_msg_header();
00233           gsm_status = STATUS_WAITING_DATA;
00234           break;
00235 
00236         case STATUS_SEND_AT :
00237           gsm_answer = false;
00238           Send_CMGF();
00239           gsm_status = STATUS_SEND_CMGF;
00240           break;
00241 
00242         case STATUS_SEND_CMGF :
00243           gsm_answer = false;
00244           Send_CNMI();
00245           gsm_status = STATUS_SEND_CNMI;
00246           break;
00247 
00248         case STATUS_SEND_CNMI :
00249           gsm_answer = false;
00250           Send_CPMS();
00251           gsm_status = STATUS_SEND_CPMS;
00252           break;
00253 
00254         case STATUS_SEND_CPMS :
00255           gsm_answer = false;
00256           gsm_status = STATUS_IDLE;
00257           gsm_gsm_send_report_status = MODULES_START; 
00258           break;
00259 
00260         case STATUS_DELETE_SMS :
00261           gsm_status = STATUS_IDLE;
00262           break;
00263 
00264         default:
00265           break;
00266       }

00477 static void gsm_parse(uint8_t c) {
00478   switch(c) {
00479     case GSM_CMD_LINE_TERMINATION:
00480       break;
00481     case '>':
00482       prompt_received = true;
00483       break;
00484     case GSM_RESPONSE_FORMATING:
00485       gsm_buf[gsm_buf_idx] = '\0';
00486       gsm_line_received = true;
00487       gsm_buf_len = gsm_buf_idx;
00488       gsm_buf_idx=0;
00489       break;
00490     default:
00491       if (gsm_buf_idx < GSM_MAX_PAYLOAD) {
00492         gsm_buf[gsm_buf_idx] = c;
00493         gsm_buf_idx++;
00494       } /* else extra characters are ignored */
00495       break;
00496   }
00497 }
