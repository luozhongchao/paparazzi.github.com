97 void infrared_i2c_update( void ) {
98 #if ! (defined SITL || defined HITL)
99  // IR horizontal
100  if (irh_trans.status == I2CTransDone && ir_i2c_hor_status == IR_I2C_IDLE) {
101  if (ValidConfWord(ir_i2c_conf_word) && !ir_i2c_conf_hor_done) {
102  irh_trans.buf[0] = ir_i2c_conf_word | IR_HOR_OC_BIT | IR_START_CONV ;
103  i2c_transmit(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 1);
104  ir_i2c_hor_status = IR_I2C_CONFIGURE_HOR;
105  } else {
106  // Read next values
107  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
108  ir_i2c_data_hor_available = FALSE;
109  ir_i2c_hor_status = IR_I2C_READ_IR1;
110  }
111  }
112  // IR vertical
113  if (irv_trans.status == I2CTransDone) {
114  if (ValidConfWord(ir_i2c_conf_word) && !ir_i2c_conf_ver_done) {
115  irv_trans.buf[0] = ir_i2c_conf_word | IR_VER_OC_BIT;
116  i2c_transmit(&i2c0, &irv_trans, IR_VER_I2C_ADDR, 1);
117  } else {
118  // Read next values
119  i2c_receive(&i2c0, &irv_trans, IR_VER_I2C_ADDR, 2);
120  ir_i2c_data_ver_available = FALSE;
121  }
122  }
123 #endif /* SITL || HITL */
124 }

128 void infrared_i2c_hor_event( void ) {
129 #if ! (defined SITL || defined HITL)
130  irh_trans.status = I2CTransDone;
131  switch (ir_i2c_hor_status) {
132  case IR_I2C_IDLE :
133  break;
134  case IR_I2C_READ_IR1 :
135  if (bit_is_set(irh_trans.buf[2],7)) {
136  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
137  break;
138  }
139  // Read IR1 value
140  int16_t ir1 = (irh_trans.buf[0]<<8) | irh_trans.buf[1];
141  ir1 = ir1 - (IR_I2C_IR1_NEUTRAL << ir_i2c_conf_word);
142  ir_i2c.ir1 = FilterIR(ir_i2c.ir1, ir1);
143  // Select IR2 channel
144  irh_trans.buf[0] = IR_HOR_I2C_SELECT_IR2 | IR_HOR_OC_BIT | ir_i2c_conf_word | IR_START_CONV;
145  i2c_transmit(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 1);
146  ir_i2c_hor_status = IR_I2C_IR2_SELECTED;
147  break;
148  case IR_I2C_IR2_SELECTED :
149  // IR2 selected, asking for IR2 value
150  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
151  ir_i2c_hor_status = IR_I2C_READ_IR2;
152  break;
153  case IR_I2C_READ_IR2 :
154  // Read IR2 value
155  if (bit_is_set(irh_trans.buf[2],7)) {
156  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
157  break;
158  }
159  int16_t ir2 = (irh_trans.buf[0]<<8) | irh_trans.buf[1];
160  ir2 = ir2 - (IR_I2C_IR2_NEUTRAL << ir_i2c_conf_word);
161  ir_i2c.ir2 = FilterIR(ir_i2c.ir2, ir2);
162  // Update estimator
163  ir_i2c_data_hor_available = TRUE;
164 #ifndef IR_I2C_READ_ONLY
165  if (ir_i2c_data_ver_available) {
166  ir_i2c_data_hor_available = FALSE;
167  ir_i2c_data_ver_available = FALSE;
168  UpdateIRValue(ir_i2c);
169  }
170 #endif
171  // Select IR1 channel
172  irh_trans.buf[0] = IR_HOR_I2C_SELECT_IR1 | IR_HOR_OC_BIT | ir_i2c_conf_word | IR_START_CONV;
173  i2c_transmit(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 1);
174  ir_i2c_hor_status = IR_I2C_IR1_SELECTED;
175  break;
176  case IR_I2C_IR1_SELECTED :
177  // End reading cycle
178  ir_i2c_hor_status = IR_I2C_IDLE;
179  break;
180  case IR_I2C_CONFIGURE_HOR :
181  // End conf cycle
182  ir_i2c_conf_hor_done = TRUE;
183  ir_i2c_hor_status = IR_I2C_IDLE;
184  break;
185  }
186 #endif /* !SITL && !HITL */
187 }

189 void infrared_i2c_ver_event( void ) {
190 #if ! (defined SITL || defined HITL)
191  irv_trans.status = I2CTransDone;
192  // Read TOP value
193  if (irv_trans.type == I2CTransRx) {
194  int16_t ir3 = (irv_trans.buf[0]<<8) | irv_trans.buf[1];
195  ir3 = ir3 - (IR_I2C_TOP_NEUTRAL << ir_i2c_conf_word);
196  ir_i2c.ir3 = FilterIR(ir_i2c.ir3, ir3);
197  ir_i2c_data_ver_available = TRUE;
198 #ifndef IR_I2C_READ_ONLY
199  if (ir_i2c_data_hor_available) {
200  ir_i2c_data_hor_available = FALSE;
201  ir_i2c_data_ver_available = FALSE;
202  UpdateIRValue(ir_i2c);
203  }
204 #endif
205  }
206  if (irv_trans.type == I2CTransTx) {
207  ir_i2c_conf_ver_done = TRUE;
208  }
209 #endif /* !SITL && !HITL */
210 }
