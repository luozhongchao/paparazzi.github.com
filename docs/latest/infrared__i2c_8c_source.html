00097 void infrared_i2c_update( void ) {
00098 #if ! (defined SITL || defined HITL)
00099   // IR horizontal
00100   if (irh_trans.status == I2CTransDone && ir_i2c_hor_status == IR_I2C_IDLE) {
00101     if (ValidConfWord(ir_i2c_conf_word) && !ir_i2c_conf_hor_done) {
00102       irh_trans.buf[0] = ir_i2c_conf_word | IR_HOR_OC_BIT | IR_START_CONV ;
00103       I2CTransmit(i2c0, irh_trans, IR_HOR_I2C_ADDR, 1);
00104       ir_i2c_hor_status = IR_I2C_CONFIGURE_HOR;
00105     } else {
00106       // Read next values
00107       I2CReceive(i2c0, irh_trans, IR_HOR_I2C_ADDR, 3);
00108       ir_i2c_data_hor_available = FALSE;
00109       ir_i2c_hor_status = IR_I2C_READ_IR1;
00110     }
00111   }
00112   // IR vertical
00113   if (irv_trans.status == I2CTransDone) {
00114     if (ValidConfWord(ir_i2c_conf_word) && !ir_i2c_conf_ver_done) {
00115       irv_trans.buf[0] = ir_i2c_conf_word | IR_VER_OC_BIT;
00116       I2CTransmit(i2c0, irv_trans, IR_VER_I2C_ADDR, 1);
00117     } else {
00118       // Read next values
00119       I2CReceive(i2c0, irv_trans, IR_VER_I2C_ADDR, 2);
00120       ir_i2c_data_ver_available = FALSE;
00121     }
00122   }
00123 #endif /* SITL || HITL */
00124 }

00128 void infrared_i2c_hor_event( void ) {
00129 #if ! (defined SITL || defined HITL)
00130   irh_trans.status = I2CTransDone;
00131   switch (ir_i2c_hor_status) {
00132     case IR_I2C_IDLE :
00133       break;
00134     case IR_I2C_READ_IR1 :
00135       if (bit_is_set(irh_trans.buf[2],7)) {
00136         I2CReceive(i2c0, irh_trans, IR_HOR_I2C_ADDR, 3);
00137         break;
00138       }
00139       // Read IR1 value
00140       int16_t ir1 = (irh_trans.buf[0]<<8) | irh_trans.buf[1];
00141       ir1 = ir1 - (IR_I2C_IR1_NEUTRAL << ir_i2c_conf_word);
00142       ir_i2c.ir1 = FilterIR(ir_i2c.ir1, ir1);
00143       // Select IR2 channel
00144       irh_trans.buf[0] = IR_HOR_I2C_SELECT_IR2 | IR_HOR_OC_BIT | ir_i2c_conf_word | IR_START_CONV;
00145       I2CTransmit(i2c0, irh_trans, IR_HOR_I2C_ADDR, 1);
00146       ir_i2c_hor_status = IR_I2C_IR2_SELECTED;
00147       break;
00148     case IR_I2C_IR2_SELECTED :
00149       // IR2 selected, asking for IR2 value
00150       I2CReceive(i2c0, irh_trans, IR_HOR_I2C_ADDR, 3);
00151       ir_i2c_hor_status = IR_I2C_READ_IR2;
00152       break;
00153     case IR_I2C_READ_IR2 :
00154       // Read IR2 value
00155       if (bit_is_set(irh_trans.buf[2],7)) {
00156         I2CReceive(i2c0, irh_trans, IR_HOR_I2C_ADDR, 3);
00157         break;
00158       }
00159       int16_t ir2 = (irh_trans.buf[0]<<8) | irh_trans.buf[1];
00160       ir2 = ir2 - (IR_I2C_IR2_NEUTRAL << ir_i2c_conf_word);
00161       ir_i2c.ir2 = FilterIR(ir_i2c.ir2, ir2);
00162       // Update estimator
00163       ir_i2c_data_hor_available = TRUE;
00164 #ifndef IR_I2C_READ_ONLY
00165       if (ir_i2c_data_ver_available) {
00166         ir_i2c_data_hor_available = FALSE;
00167         ir_i2c_data_ver_available = FALSE;
00168         UpdateIRValue(ir_i2c);
00169       }
00170 #endif
00171       // Select IR1 channel
00172       irh_trans.buf[0] = IR_HOR_I2C_SELECT_IR1 | IR_HOR_OC_BIT | ir_i2c_conf_word | IR_START_CONV;
00173       I2CTransmit(i2c0, irh_trans, IR_HOR_I2C_ADDR, 1);
00174       ir_i2c_hor_status = IR_I2C_IR1_SELECTED;
00175       break;
00176     case IR_I2C_IR1_SELECTED :
00177       // End reading cycle
00178       ir_i2c_hor_status = IR_I2C_IDLE;
00179       break;
00180     case IR_I2C_CONFIGURE_HOR :
00181       // End conf cycle
00182       ir_i2c_conf_hor_done = TRUE;
00183       ir_i2c_hor_status = IR_I2C_IDLE;
00184       break;
00185   }
00186 #endif /* !SITL && !HITL */
00187 }
