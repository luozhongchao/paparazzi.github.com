00163 void intermcu_parse( uint8_t c ) {
00164   if (intermcu_data.status < GOT_PAYLOAD) {
00165     intermcu_data.ck_a += c;
00166     intermcu_data.ck_b += intermcu_data.ck_a;
00167   }
00168   switch (intermcu_data.status) {
00169   case UNINIT:
00170     if (c == INTERMCU_SYNC1)
00171       intermcu_data.status++;
00172     break;
00173   case GOT_SYNC1:
00174     if (c != INTERMCU_SYNC2) {
00175       goto error;
00176     }
00177     intermcu_data.ck_a = 0;
00178     intermcu_data.ck_b = 0;
00179     intermcu_data.status++;
00180     break;
00181   case GOT_SYNC2:
00182     if (intermcu_data.msg_available) {
00183       /* Previous message has not yet been parsed: discard this one */
00184       goto error;
00185     }
00186     intermcu_data.msg_class = c;
00187     intermcu_data.status++;
00188     break;
00189   case GOT_CLASS:
00190     intermcu_data.msg_id = c;
00191     intermcu_data.status++;
00192     break;
00193   case GOT_ID:
00194     intermcu_data.len = c;
00195     intermcu_data.status++;
00196     break;
00197   case GOT_LEN1:
00198     intermcu_data.len |= (c<<8);
00199     if (intermcu_data.len > INTERMCU_MAX_PAYLOAD) {
00200       goto error;
00201     }
00202     intermcu_data.msg_idx = 0;
00203     intermcu_data.status++;
00204     break;
00205   case GOT_LEN2:
00206     intermcu_data.msg_buf[intermcu_data.msg_idx] = c;
00207     intermcu_data.msg_idx++;
00208     if (intermcu_data.msg_idx >= intermcu_data.len) {
00209       intermcu_data.status++;
00210     }
00211     break;
00212   case GOT_PAYLOAD:
00213     if (c != intermcu_data.ck_a) {
00214       goto error;
00215     }
00216     intermcu_data.status++;
00217     break;
00218   case GOT_CHECKSUM1:
00219     if (c != intermcu_data.ck_b) {
00220       goto error;
00221     }
00222     intermcu_data.msg_available = TRUE;
00223     goto restart;
00224     break;
00225   default:
00226     goto error;
00227   }
00228   return;
00229  error:
00230   intermcu_data.error_cnt++;
00231  restart:
00232   intermcu_data.status = UNINIT;
00233   return;
00234 }
