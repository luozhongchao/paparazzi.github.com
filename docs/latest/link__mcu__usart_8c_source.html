161 /* INTERMCU parsing */
162 void intermcu_parse( uint8_t c );
163 void intermcu_parse( uint8_t c ) {
164  if (intermcu_data.status < GOT_PAYLOAD) {
165  intermcu_data.ck_a += c;
166  intermcu_data.ck_b += intermcu_data.ck_a;
167  }
168  switch (intermcu_data.status) {
169  case UNINIT:
170  if (c == INTERMCU_SYNC1)
171  intermcu_data.status++;
172  break;
173  case GOT_SYNC1:
174  if (c != INTERMCU_SYNC2) {
175  goto error;
176  }
177  intermcu_data.ck_a = 0;
178  intermcu_data.ck_b = 0;
179  intermcu_data.status++;
180  break;
181  case GOT_SYNC2:
182  if (intermcu_data.msg_available) {
183  /* Previous message has not yet been parsed: discard this one */
184  goto error;
185  }
186  intermcu_data.msg_class = c;
187  intermcu_data.status++;
188  break;
189  case GOT_CLASS:
190  intermcu_data.msg_id = c;
191  intermcu_data.status++;
192  break;
193  case GOT_ID:
194  intermcu_data.len = c;
195  intermcu_data.status++;
196  break;
197  case GOT_LEN1:
198  intermcu_data.len |= (c<<8);
199  if (intermcu_data.len > INTERMCU_MAX_PAYLOAD) {
200  goto error;
201  }
202  intermcu_data.msg_idx = 0;
203  intermcu_data.status++;
204  break;
205  case GOT_LEN2:
206  intermcu_data.msg_buf[intermcu_data.msg_idx] = c;
207  intermcu_data.msg_idx++;
208  if (intermcu_data.msg_idx >= intermcu_data.len) {
209  intermcu_data.status++;
210  }
211  break;
212  case GOT_PAYLOAD:
213  if (c != intermcu_data.ck_a) {
214  goto error;
215  }
216  intermcu_data.status++;
217  break;
218  case GOT_CHECKSUM1:
219  if (c != intermcu_data.ck_b) {
220  goto error;
221  }
222  intermcu_data.msg_available = TRUE;
223  goto restart;
224  break;
225  default:
226  goto error;
227  }
228  return;
229  error:
230  intermcu_data.error_cnt++;
231  restart:
232  intermcu_data.status = UNINIT;
233  return;
234 }

254 void parse_mavpilot_msg( void )
255 {
256  if (intermcu_data.msg_class == MSG_INTERMCU_ID)
257  {
258  if (intermcu_data.msg_id == MSG_INTERMCU_COMMAND_ID)
259  {
260 #if COMMANDS_NB > 8
261 #error "INTERMCU UART CAN ONLY SEND 8 COMMANDS OR THE UART WILL BE OVERFILLED"
262 #endif
263 
264  for (int i=0; i< COMMANDS_NB; i++)
265  {
266  ap_state->commands[i] = ((pprz_t)MSG_INTERMCU_COMMAND(intermcu_data.msg_buf, i));
267  }
268 
269 #ifdef LINK_MCU_LED
270  LED_TOGGLE(LINK_MCU_LED);
271 #endif
272  inter_mcu_received_ap = TRUE;
273  }
274  else if (intermcu_data.msg_id == MSG_INTERMCU_RADIO_ID)
275  {
276 #if RADIO_CONTROL_NB_CHANNEL > 10
277 #error "INTERMCU UART CAN ONLY SEND 10 RADIO CHANNELS OR THE UART WILL BE OVERFILLED"
278 #endif
279 
280  for (int i=0; i< RADIO_CONTROL_NB_CHANNEL; i++)
281  {
282  fbw_state->channels[i] = ((pprz_t)MSG_INTERMCU_RADIO(intermcu_data.msg_buf, i));
283  }
284  }
285  else if (intermcu_data.msg_id == MSG_INTERMCU_TRIM_ID)
286  {
287  ap_state->command_roll_trim = ((pprz_t) MSG_INTERMCU_TRIM_ROLL(intermcu_data.msg_buf));
288  ap_state->command_pitch_trim = ((pprz_t) MSG_INTERMCU_TRIM_PITCH(intermcu_data.msg_buf));
289  }
290  else if (intermcu_data.msg_id == MSG_INTERMCU_FBW_ID)
291  {
292  fbw_state->ppm_cpt = MSG_INTERMCU_FBW_MOD(intermcu_data.msg_buf);
293  fbw_state->status = MSG_INTERMCU_FBW_STAT(intermcu_data.msg_buf);
294  fbw_state->nb_err = MSG_INTERMCU_FBW_ERR(intermcu_data.msg_buf);
295  fbw_state->vsupply = MSG_INTERMCU_FBW_VOLT(intermcu_data.msg_buf);
296  fbw_state->current = MSG_INTERMCU_FBW_CURRENT(intermcu_data.msg_buf);
297 
298  inter_mcu_received_fbw = TRUE;
299  }
300  }
301 }

342 void link_mcu_event_task( void ) {
343  /* A message has been received */
344  if (InterMcuBuffer()) {
345  while (InterMcuLink(ChAvailable())&&!intermcu_data.msg_available)
346  intermcu_parse(InterMcuLink(Getch()));
347  }
348 
349  if (intermcu_data.msg_available) {
350  parse_mavpilot_msg();
351  intermcu_data.msg_available = FALSE;
