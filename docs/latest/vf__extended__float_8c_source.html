00080 /*
00081 
00082  F = [ 1 dt -dt^2/2 0
00083        0  1 -dt     0
00084        0  0   1     0
00085        0  0   0     1 ];
00086 
00087  B = [ dt^2/2 dt 0 0]';
00088 
00089  Q = [ Qzz   0          0         0
00090        0     Qzdotzdot  0         0
00091        0     0          Qbiasbias 0
00092        0     0     0    0         Qoffoff ];
00093 
00094  Xk1 = F * Xk0 + B * accel;
00095 
00096  Pk1 = F * Pk0 * F' + Q;
00097 
00098 */
00099 void vff_propagate(float accel) {
00100   /* update state */
00101   vff_zdotdot = accel + 9.81 - vff_bias;
00102   vff_z = vff_z + DT_VFILTER * vff_zdot;
00103   vff_zdot = vff_zdot + DT_VFILTER * vff_zdotdot;
00104   /* update covariance */
00105   const float FPF00 = vff_P[0][0] + DT_VFILTER * ( vff_P[1][0] + vff_P[0][1] + DT_VFILTER * vff_P[1][1] );
00106   const float FPF01 = vff_P[0][1] + DT_VFILTER * ( vff_P[1][1] - vff_P[0][2] - DT_VFILTER * vff_P[1][2] );
00107   const float FPF02 = vff_P[0][2] + DT_VFILTER * ( vff_P[1][2] );
00108   const float FPF10 = vff_P[1][0] + DT_VFILTER * (-vff_P[2][0] + vff_P[1][1] - DT_VFILTER * vff_P[2][1] );
00109   const float FPF11 = vff_P[1][1] + DT_VFILTER * (-vff_P[2][1] - vff_P[1][2] + DT_VFILTER * vff_P[2][2] );
00110   const float FPF12 = vff_P[1][2] + DT_VFILTER * (-vff_P[2][2] );
00111   const float FPF20 = vff_P[2][0] + DT_VFILTER * ( vff_P[2][1] );
00112   const float FPF21 = vff_P[2][1] + DT_VFILTER * (-vff_P[2][2] );
00113   const float FPF22 = vff_P[2][2];
00114   const float FPF33 = vff_P[3][3];
00115 
00116   vff_P[0][0] = FPF00 + Qzz;
00117   vff_P[0][1] = FPF01;
00118   vff_P[0][2] = FPF02;
00119   vff_P[1][0] = FPF10;
00120   vff_P[1][1] = FPF11 + Qzdotzdot;
00121   vff_P[1][2] = FPF12;
00122   vff_P[2][0] = FPF20;
00123   vff_P[2][1] = FPF21;
00124   vff_P[2][2] = FPF22 + Qbiasbias;
00125   vff_P[3][3] = FPF33 + Qoffoff;

00132 /*
00133  * Update sensor "with" offset (baro)
00134   H = [1 0 0 -1];
00135   // state residual
00136   y = rangemeter - H * Xm;
00137   // covariance residual
00138   S = H*Pm*H' + R;
00139   // kalman gain
00140   K = Pm*H'*inv(S);
00141   // update state
00142   Xp = Xm + K*y;
00143   // update covariance
00144   Pp = Pm - K*H*Pm;
00145 */
00146 __attribute__ ((always_inline)) static inline void update_baro_conf(float z_meas, float conf) {
00147   vff_z_meas_baro = z_meas;
00148 
00149   const float y = z_meas - vff_z + vff_offset;
00150   const float S = vff_P[0][0] - vff_P[0][3] - vff_P[3][0] + vff_P[3][3] + conf;
00151   const float K0 = (vff_P[0][0] - vff_P[0][3]) * 1/S;
00152   const float K1 = (vff_P[1][0] - vff_P[1][3]) * 1/S;
00153   const float K2 = (vff_P[2][0] - vff_P[2][3]) * 1/S;
00154   const float K3 = (vff_P[3][0] - vff_P[3][3]) * 1/S;
00155 
00156   vff_z       = vff_z       + K0 * y;
00157   vff_zdot    = vff_zdot    + K1 * y;
00158   vff_bias    = vff_bias    + K2 * y;
00159   vff_offset  = vff_offset  + K3 * y;
00160 
00161   const float P0 = vff_P[0][0] - vff_P[3][0];
00162   const float P1 = vff_P[0][1] - vff_P[3][1];
00163   const float P2 = vff_P[0][2] - vff_P[3][2];
00164   const float P3 = vff_P[0][3] - vff_P[3][3];
00165 
00166   vff_P[0][0] -= K0 * P0;
00167   vff_P[0][1] -= K0 * P1;
00168   vff_P[0][2] -= K0 * P2;
00169   vff_P[0][3] -= K0 * P3;
00170   vff_P[1][0] -= K1 * P0;
00171   vff_P[1][1] -= K1 * P1;
00172   vff_P[1][2] -= K1 * P2;
00173   vff_P[1][3] -= K1 * P3;
00174   vff_P[2][0] -= K2 * P0;
00175   vff_P[2][1] -= K2 * P1;
00176   vff_P[2][2] -= K2 * P2;
00177   vff_P[2][3] -= K2 * P3;
00178   vff_P[3][0] -= K3 * P0;
00179   vff_P[3][1] -= K3 * P1;
00180   vff_P[3][2] -= K3 * P2;
00181   vff_P[3][3] -= K3 * P3;
00182 
00183 }
