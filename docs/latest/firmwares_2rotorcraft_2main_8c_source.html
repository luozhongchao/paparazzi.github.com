168 STATIC_INLINE void handle_periodic_tasks( void ) {
169  if (sys_time_check_and_ack_timer(main_periodic_tid))
170  main_periodic();
171  if (sys_time_check_and_ack_timer(modules_tid))
172  modules_periodic_task();
173  if (sys_time_check_and_ack_timer(radio_control_tid))
174  radio_control_periodic_task();
175  if (sys_time_check_and_ack_timer(failsafe_tid))
176  failsafe_check();
177  if (sys_time_check_and_ack_timer(electrical_tid))
178  electrical_periodic();
179  if (sys_time_check_and_ack_timer(baro_tid))
180  baro_periodic();
181  if (sys_time_check_and_ack_timer(telemetry_tid))
182  telemetry_periodic();
183 }

206 STATIC_INLINE void failsafe_check( void ) {
207  if (radio_control.status != RC_OK &&
208  autopilot_mode != AP_MODE_KILL &&
209  autopilot_mode != AP_MODE_NAV)
210  {
211  autopilot_set_mode(AP_MODE_FAILSAFE);
212  }
213 
214 #if USE_GPS
215  if (autopilot_mode == AP_MODE_NAV &&
216 #if NO_GPS_LOST_WITH_RC_VALID
217  radio_control.status != RC_OK &&
218 #endif
219  GpsIsLost())
220  {
221  autopilot_set_mode(AP_MODE_FAILSAFE);
222  }
223 #endif
224 }

260 static inline void on_gyro_event( void ) {
261 
262  ImuScaleGyro(imu);
263 
264  if (ahrs.status == AHRS_UNINIT) {
265  ahrs_aligner_run();
