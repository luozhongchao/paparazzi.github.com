72 void link_mcu_event_task( void ) {
73  if (link_mcu_trans.status == SPITransSuccess) {
74  /* Got a message on SPI. */
75  link_mcu_trans.status = SPITransDone;
76 
77  /* A message has been received */
78  ComputeChecksum(link_mcu_from_ap_msg);
79  link_mcu_received = TRUE;
80  if (link_mcu_from_ap_msg.checksum == crc)
81  inter_mcu_received_ap = TRUE;
82  else
83  fbw_state->nb_err++;
84  }
85  if (link_mcu_trans.status == SPITransFailed) {
86  link_mcu_trans.status = SPITransDone;
87  link_mcu_received = TRUE;
88  fbw_state->nb_err++;
89  }
