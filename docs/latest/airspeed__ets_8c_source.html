24 /**
25  * @file modules/sensors/airspeed_ets.c
26  *
27  * Driver for the EagleTree Systems Airspeed Sensor.
28  * Has only been tested with V3 of the sensor hardware.
29  *
30  * Notes:
31  * Connect directly to TWOG/Tiny I2C port. Multiple sensors can be chained together.
32  * Sensor should be in the proprietary mode (default) and not in 3rd party mode.
33  *
34  * Sensor module wire assignments:
35  * Red wire: 5V
36  * White wire: Ground
37  * Yellow wire: SDA
38  * Brown wire: SCL

120 void airspeed_ets_read_event( void ) {
121  int n;
122  float airspeed_tmp = 0.0;
123 
124  // Get raw airspeed from buffer
125  airspeed_ets_raw = ((uint16_t)(airspeed_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(airspeed_ets_i2c_trans.buf[0]);
126  // Check if this is valid airspeed
127  if (airspeed_ets_raw == 0)
128  airspeed_ets_valid = FALSE;
129  else
130  airspeed_ets_valid = TRUE;
131 
132  // Continue only if a new airspeed value was received
133  if (airspeed_ets_valid) {
134  // Calculate offset average if not done already
135  if (!airspeed_ets_offset_init) {
136  --airspeed_ets_cnt;
137  // Check if averaging completed
138  if (airspeed_ets_cnt == 0) {
139  // Calculate average
140  airspeed_ets_offset = (uint16_t)(airspeed_ets_offset_tmp / AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG);
141  // Limit offset
142  if (airspeed_ets_offset < AIRSPEED_ETS_OFFSET_MIN)
143  airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MIN;
144  if (airspeed_ets_offset > AIRSPEED_ETS_OFFSET_MAX)
145  airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MAX;
146  airspeed_ets_offset_init = TRUE;
147  }
148  // Check if averaging needs to continue
149  else if (airspeed_ets_cnt <= AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG)
150  airspeed_ets_offset_tmp += airspeed_ets_raw;
151  }
152  // Convert raw to m/s
153 #ifdef AIRSPEED_ETS_REVERSE
154  if (airspeed_ets_offset_init && airspeed_ets_raw < airspeed_ets_offset)
155  airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf( (float)(airspeed_ets_offset-airspeed_ets_raw) ) - AIRSPEED_ETS_OFFSET;
156 #else
157  if (airspeed_ets_offset_init && airspeed_ets_raw > airspeed_ets_offset)
158  airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf( (float)(airspeed_ets_raw-airspeed_ets_offset) ) - AIRSPEED_ETS_OFFSET;
159 #endif
160  else
161  airspeed_tmp = 0.0;
162  // Airspeed should always be positive
163  if (airspeed_tmp < 0.0)
164  airspeed_tmp = 0.0;
165  // Moving average
166  airspeed_ets_buffer[airspeed_ets_buffer_idx++] = airspeed_tmp;
167  if (airspeed_ets_buffer_idx >= AIRSPEED_ETS_NBSAMPLES_AVRG)
168  airspeed_ets_buffer_idx = 0;
169  airspeed_ets = 0.0;
170  for (n = 0; n < AIRSPEED_ETS_NBSAMPLES_AVRG; ++n)
171  airspeed_ets += airspeed_ets_buffer[n];
172  airspeed_ets = airspeed_ets / (float)AIRSPEED_ETS_NBSAMPLES_AVRG;
173 #if USE_AIRSPEED
174  stateSetAirspeed_f(&airspeed_ets);
175 #endif
176 #ifdef SENSOR_SYNC_SEND
177  DOWNLINK_SEND_AIRSPEED_ETS(DefaultChannel, DefaultDevice, &airspeed_ets_raw, &airspeed_ets_offset, &airspeed_ets);
178 #endif
179  } else {
180  airspeed_ets = 0.0;
181  }
182 
183  // Transaction has been read
184  airspeed_ets_i2c_trans.status = I2CTransDone;
185 }
