00124   // Get raw airspeed from buffer
00125   airspeed_ets_raw = ((uint16_t)(airspeed_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(airspeed_ets_i2c_trans.buf[0]);
00126   // Check if this is valid airspeed
00127   if (airspeed_ets_raw == 0)
00128     airspeed_ets_valid = FALSE;
00129   else
00130     airspeed_ets_valid = TRUE;
00131 
00132   // Continue only if a new airspeed value was received
00133   if (airspeed_ets_valid) {
00134     // Calculate offset average if not done already
00135     if (!airspeed_ets_offset_init) {
00136       --airspeed_ets_cnt;
00137       // Check if averaging completed
00138       if (airspeed_ets_cnt == 0) {
00139         // Calculate average
00140         airspeed_ets_offset = (uint16_t)(airspeed_ets_offset_tmp / AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG);
00141         // Limit offset
00142         if (airspeed_ets_offset < AIRSPEED_ETS_OFFSET_MIN)
00143           airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MIN;
00144         if (airspeed_ets_offset > AIRSPEED_ETS_OFFSET_MAX)
00145           airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MAX;
00146         airspeed_ets_offset_init = TRUE;
00147       }
00148       // Check if averaging needs to continue
00149       else if (airspeed_ets_cnt <= AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG)
00150         airspeed_ets_offset_tmp += airspeed_ets_raw;
00151     }
00152     // Convert raw to m/s
00153 #ifdef AIRSPEED_ETS_REVERSE
00154     if (airspeed_ets_offset_init && airspeed_ets_raw < airspeed_ets_offset)
00155       airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf( (float)(airspeed_ets_offset-airspeed_ets_raw) ) - AIRSPEED_ETS_OFFSET;
00156 #else

00158       airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf( (float)(airspeed_ets_raw-airspeed_ets_offset) ) - AIRSPEED_ETS_OFFSET;
00159 #endif
00160     else

00162     // Airspeed should always be positive
