00049 void orientationCalcQuat_i(struct OrientationReps* orientation) {
00050   if (bit_is_set(orientation->status, ORREP_QUAT_I))
00051     return;
00052 
00053   if (bit_is_set(orientation->status, ORREP_QUAT_F)) {
00054     QUAT_BFP_OF_REAL(orientation->quat_i, orientation->quat_f);
00055   }
00056   else if (bit_is_set(orientation->status, ORREP_RMAT_I)) {
00057     INT32_QUAT_OF_RMAT(orientation->quat_i, orientation->rmat_i);
00058   }
00059   else if (bit_is_set(orientation->status, ORREP_EULER_I)) {
00060     INT32_QUAT_OF_EULERS(orientation->quat_i, orientation->eulers_i);
00061   }
00062   else if (bit_is_set(orientation->status, ORREP_RMAT_F)) {
00063     RMAT_BFP_OF_REAL(orientation->rmat_i, orientation->rmat_f);
00064     SetBit(orientation->status, ORREP_RMAT_I);
00065     INT32_QUAT_OF_RMAT(orientation->quat_i, orientation->rmat_i);
00066   }
00067   else if (bit_is_set(orientation->status, ORREP_EULER_F)) {
00068     EULERS_BFP_OF_REAL(orientation->eulers_i, orientation->eulers_f);
00069     SetBit(orientation->status, ORREP_EULER_I);
00070     INT32_QUAT_OF_EULERS(orientation->quat_i, orientation->eulers_i);
00071   }
00072   /* set bit to indicate this representation is computed */
00073   SetBit(orientation->status, ORREP_QUAT_I);
