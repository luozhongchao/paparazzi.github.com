56 void baro_periodic(void) {
57 
58  baro.absolute = baro_board.buf.sum/baro_board.buf.av_nb_sample;
59  baro_board.value_filtered = (3*baro_board.value_filtered + baro.absolute)/4;
60  if (baro.status == BS_UNINITIALIZED) {
61  RunOnceEvery(10, { baro_board_calibrate();});
62  }
63  /* else */
64  baro_board.data_available = TRUE;
65 }
66 
67 /* decrement offset until adc reading is over a threshold */
68 void baro_board_calibrate(void) {
69  if (baro_board.value_filtered < BOOZ_ANALOG_BARO_THRESHOLD && baro_board.offset >= 1) {
70  if (baro_board.value_filtered == 0 && baro_board.offset > 15)
71  baro_board.offset -= 15;
72  else
73  baro_board.offset--;
74  DACSet(baro_board.offset);
75 #ifdef ROTORCRAFT_BARO_LED
76  LED_TOGGLE(ROTORCRAFT_BARO_LED);
77 #endif
78  }
79  else {
80  baro.status = BS_RUNNING;
81 #ifdef ROTORCRAFT_BARO_LED
82  LED_ON(ROTORCRAFT_BARO_LED);
83 #endif
84  }
85 }
