60 void airspeed_otf_parse(char c)
61 {
62  static unsigned char otf_status = OTF_UNINIT, otf_idx = 0, otf_crs_idx;
63  static char otf_inp[64];
64  static unsigned int counter;
65  static short course[3];
66  static unsigned int altitude;
67  static unsigned char checksum;
68 
69  switch (otf_status) {
70 
71  case OTF_WAIT_START:
72  if (c == OTF_START) {
73  otf_status++;
74  otf_idx = 0;
75  } else {
76  otf_status = OTF_UNINIT;
77  }
78  break;
79 
80  case OTF_WAIT_COUNTER:
81  if (isdigit((int)c)) {
82  if (otf_idx == 0) {
83 //FIXME otf_timestamp = getclock();
84  }
85  otf_inp[otf_idx++] = c;
86  } else {
87  if ((otf_idx == 5) && (c == OTF_LIMITER)) {
88  otf_inp[otf_idx] = 0;
89  counter = atoi(otf_inp);
90  otf_idx = 0;
91  otf_crs_idx = 0;
92  otf_status++;
93  } else {
94  otf_status = OTF_UNINIT;
95  }
96  }
97  break;
98 
99  case OTF_WAIT_ANGLES:
100  if (isdigit((int)c) || (c == '-') || (c == '.')) {
101  otf_inp[otf_idx++] = c;
102  } else {
103  if ((otf_idx > 1) && (otf_idx < 9) && (c == OTF_LIMITER)) {
104  otf_inp[otf_idx] = 0;
105  course[otf_crs_idx] = (int16_t) (100. * atof(otf_inp));
106  otf_idx = 0;
107  if (otf_crs_idx++ == 2) {
108  otf_status++;
109  }
110  } else {
111  otf_status = OTF_UNINIT;
112  }
113  }
114  break;
115 
116  case OTF_WAIT_ALTITUDE:
117  if (isdigit((int)c) || (c == '-') || (c == '.')) {
118  otf_inp[otf_idx++] = c;
119  } else {
120  if ((otf_idx > 1) && (otf_idx < 9) && (c == OTF_LIMITER)) {
121  otf_inp[otf_idx] = 0;
122  altitude = (int32_t) (100. * atof(otf_inp));
123  otf_idx = 0;
124  otf_status++;
125  } else {
126  otf_status = OTF_UNINIT;
127  }
128  }
129  break;
130 
131  case OTF_WAIT_CHECKSUM:
132  if (isxdigit((int)c)) {
133  otf_inp[otf_idx++] = c;
134  } else {
135  if ((otf_idx == 2) && (c == OTF_END)) {
136  otf_inp[otf_idx] = 0;
137  checksum = strtol(otf_inp, NULL, 16);
138  otf_idx = 0;
139  DOWNLINK_SEND_FLOW_AP_OTF(DefaultChannel, DefaultDevice, &counter, &course[0], &course[1], &course[2], &altitude, &checksum);
140  }
141  otf_status = OTF_UNINIT;
142  }
143  break;
144 
145  default:
146  otf_status = OTF_UNINIT;
147  break;
148  }
149 }
