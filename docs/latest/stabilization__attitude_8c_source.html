00358 
00359   /* I term calculation */
00360   static float roll_rate_sum_err = 0.;
00361   static uint8_t roll_rate_sum_idx = 0;
00362   static float roll_rate_sum_values[H_CTL_ROLL_RATE_SUM_NB_SAMPLES];
00363 
00364   roll_rate_sum_err -= roll_rate_sum_values[roll_rate_sum_idx];
00365   roll_rate_sum_values[roll_rate_sum_idx] = err;
00366   roll_rate_sum_err += err;
00367   roll_rate_sum_idx++;
00368   if (roll_rate_sum_idx >= H_CTL_ROLL_RATE_SUM_NB_SAMPLES) roll_rate_sum_idx = 0;
00369 
00370   /* D term calculations */
00371   static float last_err = 0;
00372   float d_err = err - last_err;
00373   last_err = err;

00375   float throttle_dep_pgain =
00376     Blend(h_ctl_hi_throttle_roll_rate_pgain, h_ctl_lo_throttle_roll_rate_pgain, v_ctl_throttle_setpoint/((float)MAX_PPRZ));
00377   float cmd = throttle_dep_pgain * ( err + h_ctl_roll_rate_igain * roll_rate_sum_err / H_CTL_ROLL_RATE_SUM_NB_SAMPLES + h_ctl_roll_rate_dgain * d_err);
