60 inline bool_t UM6_verify_chk(uint8_t packet_buffer[], uint8_t packet_length) {
61  chk_rec = (packet_buffer[packet_length-2] << 8) | packet_buffer[packet_length-1];
62  chk_calc = UM6_calculate_checksum(packet_buffer, packet_length-2);
63  return (chk_calc == chk_rec);
64 }

66 inline uint16_t UM6_calculate_checksum(uint8_t packet_buffer[], uint8_t packet_length) {
67  uint16_t chk = 0;
68  for (int i=0; i<packet_length; i++) {
69  chk += packet_buffer[i];
70  }
71  return chk;
72 }

286 void UM6_packet_parse( uint8_t c ) {
287  switch (UM6_packet.status) {
288  case UM6PacketWaiting:
289  UM6_packet.msg_idx = 0;
290  if (c == 's') {
291  UM6_packet.status = UM6PacketReadingS;
292  UM6_packet.msg_buf[UM6_packet.msg_idx] = c;
293  UM6_packet.msg_idx++;
294  } else {
295  UM6_packet.hdr_error++;
296  UM6_packet.status = UM6PacketWaiting;
297  }
298  break;
299  case UM6PacketReadingS:
300  if (c == 'n') {
301  UM6_packet.status = UM6PacketReadingN;
302  UM6_packet.msg_buf[UM6_packet.msg_idx] = c;
303  UM6_packet.msg_idx++;
304  } else {
305  UM6_packet.hdr_error++;
306  UM6_packet.status = UM6PacketWaiting;
307  }
308  break;
309  case UM6PacketReadingN:
310  if (c == 'p') {
311  UM6_packet.status = UM6PacketReadingPT;
312  UM6_packet.msg_buf[UM6_packet.msg_idx] = c;
313  UM6_packet.msg_idx++;
314  } else {
315  UM6_packet.hdr_error++;
316  UM6_packet.status = UM6PacketWaiting;
317  }
318  break;
319  case UM6PacketReadingPT:
320  PacketType = c;
321  UM6_packet.msg_buf[UM6_packet.msg_idx] = c;
322  UM6_packet.msg_idx++;
323  UM6_packet.status = UM6PacketReadingAddr;
324  if ((PacketType & 0xC0) == 0xC0) {
325  PacketLength = 4*((PacketType >>2) & 0xF) + 7; // Batch, has 4*BatchLength bytes of data
326  }
327  else if ((PacketType & 0xC0) == 0x80) {
328  PacketLength = 11; // Not batch, has 4 bytes of data
329  }
330  else if ((PacketType & 0xC0) == 0x00) {
331  PacketLength = 7; // Not batch, no data
332  }
333  break;
334  case UM6PacketReadingAddr:
335  PacketAddr = c;
336  UM6_packet.msg_buf[UM6_packet.msg_idx] = c;
337  UM6_packet.msg_idx++;
338  UM6_packet.status = UM6PacketReadingData;
339  break;
340  case UM6PacketReadingData:
341  UM6_packet.msg_buf[UM6_packet.msg_idx] = c;
342  UM6_packet.msg_idx++;
343  if (UM6_packet.msg_idx == PacketLength) {
344  if (UM6_verify_chk(UM6_packet.msg_buf, PacketLength)) {
345  UM6_packet.msg_available = TRUE;
346  } else {
347  UM6_packet.msg_available = FALSE;
348  UM6_packet.chksm_error++;
349  }
350  UM6_packet.status = UM6PacketWaiting;
351  }
352  break;
353  default:
354  UM6_packet.status = UM6PacketWaiting;
355  UM6_packet.msg_idx = 0;
356  break;
357  }
