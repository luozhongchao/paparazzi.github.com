00131 void ahrs_propagate(void) {
00132 
00133   /* converts gyro to floating point */
00134   struct FloatRates gyro_float;
00135   RATES_FLOAT_OF_BFP(gyro_float, imu.gyro_prev);
00136   /* unbias measurement */
00137   RATES_SUB(gyro_float, ahrs_impl.gyro_bias);
00138 
00139 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
00140   const float alpha = 0.1;
00141   FLOAT_RATES_LIN_CMB(ahrs_impl.imu_rate, ahrs_impl.imu_rate, (1.-alpha), gyro_float, alpha);
00142 #else
00143   RATES_COPY(ahrs_impl.imu_rate,gyro_float);
00144 #endif
00145 
00146   /* add correction     */
00147   struct FloatRates omega;
00148   RATES_SUM(omega, gyro_float, ahrs_impl.rate_correction);
00149   /* and zeros it */
00150   FLOAT_RATES_ZERO(ahrs_impl.rate_correction);
00151 
00152   const float dt = 1./AHRS_PROPAGATE_FREQUENCY;
00153 #if AHRS_PROPAGATE_RMAT
00154   FLOAT_RMAT_INTEGRATE_FI(ahrs_impl.ltp_to_imu_rmat, omega, dt);
00155   float_rmat_reorthogonalize(&ahrs_impl.ltp_to_imu_rmat);
00156   FLOAT_QUAT_OF_RMAT(ahrs_impl.ltp_to_imu_quat, ahrs_impl.ltp_to_imu_rmat);
00157 #endif
00158 #if AHRS_PROPAGATE_QUAT
00159   FLOAT_QUAT_INTEGRATE(ahrs_impl.ltp_to_imu_quat, omega, dt);
00160   FLOAT_QUAT_NORMALIZE(ahrs_impl.ltp_to_imu_quat);
00161   FLOAT_RMAT_OF_QUAT(ahrs_impl.ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);
00162 #endif
00163   compute_body_orientation_and_rates();

00167 void ahrs_update_accel(void) {
00168 
00169   /* last column of roation matrix = ltp z-axis in imu-frame */
00170   struct FloatVect3  c2 = { RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 0,2),
00171                             RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 1,2),
00172                             RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 2,2)};
00173 
00174   struct FloatVect3 imu_accel_float;
00175   ACCELS_FLOAT_OF_BFP(imu_accel_float, imu.accel);
00176 
00177   struct FloatVect3 residual;
00178 
00179   if (ahrs_impl.correct_gravity && ahrs_impl.ltp_vel_norm_valid) {
00180     /*
00181      * centrifugal acceleration in body frame
00182      * a_c_body = omega x (omega x r)
00183      * (omega x r) = tangential velocity in body frame
00184      * a_c_body = omega x vel_tangential_body
00185      * assumption: tangential velocity only along body x-axis
00186      */
00187     const struct FloatVect3 vel_tangential_body = {ahrs_impl.ltp_vel_norm, 0.0, 0.0};
00188     struct FloatVect3 acc_c_body;
00189     VECT3_RATES_CROSS_VECT3(acc_c_body, *stateGetBodyRates_f(), vel_tangential_body);
00190 
00191     /* convert centrifugal acceleration from body to imu frame */
00192     struct FloatVect3 acc_c_imu;
00193     FLOAT_RMAT_VECT3_MUL(acc_c_imu, ahrs_impl.body_to_imu_rmat, acc_c_body);
00194 
00195     /* and subtract it from imu measurement to get a corrected measurement of the gravitiy vector */
00196     struct FloatVect3 corrected_gravity;
00197     VECT3_DIFF(corrected_gravity, imu_accel_float, acc_c_imu);
00198 
00199     /* compute the residual of gravity vector in imu frame */
00200     FLOAT_VECT3_CROSS_PRODUCT(residual, corrected_gravity, c2);
00201   } else {
00202     FLOAT_VECT3_CROSS_PRODUCT(residual, imu_accel_float, c2);
00203   }
00204 
00205 #ifdef AHRS_GRAVITY_UPDATE_NORM_HEURISTIC
00206   /* heuristic on acceleration norm */
00207   const float acc_norm = FLOAT_VECT3_NORM(imu_accel_float);
00208   const float weight = Chop(1.-6*fabs((9.81-acc_norm)/9.81), 0., 1.);
00209 #else
00210   const float weight = 1.;
00211 #endif
00212 
00213   /* compute correction */
00214   const float gravity_rate_update_gain = -5e-2; // -5e-2
00215   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.rate_correction, residual, weight*gravity_rate_update_gain);
00216 
00217   const float gravity_bias_update_gain = 1e-5; // -5e-6
00218   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.gyro_bias, residual, weight*gravity_bias_update_gain);
00219 
00220   /* FIXME: saturate bias */
00221 
00222 }

00284 void ahrs_update_mag_2d_dumb(void) {
00285 
00286   /* project mag on local tangeant plane */
00287   struct FloatEulers ltp_to_imu_euler;
00288   FLOAT_EULERS_OF_RMAT(ltp_to_imu_euler, ahrs_impl.ltp_to_imu_rmat);
00289   struct FloatVect3 magf;
00290   MAGS_FLOAT_OF_BFP(magf, imu.mag);
00291   const float cphi   = cosf(ltp_to_imu_euler.phi);
00292   const float sphi   = sinf(ltp_to_imu_euler.phi);
00293   const float ctheta = cosf(ltp_to_imu_euler.theta);
00294   const float stheta = sinf(ltp_to_imu_euler.theta);
00295   const float mn = ctheta * magf.x + sphi*stheta*magf.y + cphi*stheta*magf.z;
00296   const float me =     0. * magf.x + cphi       *magf.y - sphi       *magf.z;
00297 
00298   const float res_norm = -RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 0,0)*me + RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 1,0)*mn;
00299   const struct FloatVect3 r2 = {RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 2,0),
00300                                 RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 2,1),
00301                                 RMAT_ELMT(ahrs_impl.ltp_to_imu_rmat, 2,2)};
00302   const float mag_rate_update_gain = 2.5;
00303   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.rate_correction, r2, (mag_rate_update_gain*res_norm));
00304   const float mag_bias_update_gain = -2.5e-4;
00305   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.gyro_bias, r2, (mag_bias_update_gain*res_norm));
00306 
00307 }

00341 void ahrs_update_heading(float heading) {
00342 
00343   FLOAT_ANGLE_NORMALIZE(heading);
00344 
00345   struct FloatRMat* ltp_to_body_rmat = stateGetNedToBodyRMat_f();
00346   // row 0 of ltp_to_body_rmat = body x-axis in ltp frame
00347   // we only consider x and y
00348   struct FloatVect2 expected_ltp =
00349     { RMAT_ELMT(*ltp_to_body_rmat, 0, 0),
00350       RMAT_ELMT(*ltp_to_body_rmat, 0, 1) };
00351 
00352   // expected_heading cross measured_heading
00353   struct FloatVect3 residual_ltp =
00354     { 0,
00355       0,
00356       expected_ltp.x * sinf(heading) - expected_ltp.y * cosf(heading)};
00357 
00358   struct FloatVect3 residual_imu;
00359   FLOAT_RMAT_VECT3_MUL(residual_imu, ahrs_impl.ltp_to_imu_rmat, residual_ltp);
00360 
00361   const float heading_rate_update_gain = 2.5;
00362   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.rate_correction, residual_imu, heading_rate_update_gain);
00363 
00364   float heading_bias_update_gain;
00365   /* crude attempt to only update bias if deviation is small
00366    * e.g. needed when you only have gps providing heading
00367    * and the inital heading is totally different from
00368    * the gps course information you get once you have a gps fix.
00369    * Otherwise the bias will be falsely "corrected".
00370    */
00371   if (fabs(residual_ltp.z) < sinf(RadOfDeg(5.)))
00372     heading_bias_update_gain = -2.5e-4;
00373   else
00374     heading_bias_update_gain = 0.0;
00375   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.gyro_bias, residual_imu, heading_bias_update_gain);
00376 }

00379 void ahrs_realign_heading(float heading) {
00380   FLOAT_ANGLE_NORMALIZE(heading);
00381 
00382   /* quaternion representing only the heading rotation from ltp to body */
00383   struct FloatQuat q_h_new;
00384   q_h_new.qx = 0.0;
00385   q_h_new.qy = 0.0;
00386   q_h_new.qz = sinf(heading/2.0);
00387   q_h_new.qi = cosf(heading/2.0);
00388 
00389   struct FloatQuat* ltp_to_body_quat = stateGetNedToBodyQuat_f();
00390   /* quaternion representing current heading only */
00391   struct FloatQuat q_h;
00392   QUAT_COPY(q_h, *ltp_to_body_quat);
00393   q_h.qx = 0.;
00394   q_h.qy = 0.;
00395   FLOAT_QUAT_NORMALIZE(q_h);
00396 
00397   /* quaternion representing rotation from current to new heading */
00398   struct FloatQuat q_c;
00399   FLOAT_QUAT_INV_COMP_NORM_SHORTEST(q_c, q_h, q_h_new);
00400 
00401   /* correct current heading in body frame */
00402   struct FloatQuat q;
00403   FLOAT_QUAT_COMP_NORM_SHORTEST(q, q_c, *ltp_to_body_quat);
00404 
00405   /* compute ltp to imu rotation quaternion and matrix */
00406   FLOAT_QUAT_COMP(ahrs_impl.ltp_to_imu_quat, q, ahrs_impl.body_to_imu_quat);
00407   FLOAT_RMAT_OF_QUAT(ahrs_impl.ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);
00408 
00409   /* set state */
00410   stateSetNedToBodyQuat_f(&q);
00411 
00412   ahrs_impl.heading_aligned = TRUE;
00413 }
