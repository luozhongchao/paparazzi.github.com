00200 void h_ctl_course_loop ( void ) {
00201   static float last_err;
00202 
00203   // Ground path error
00204   float err = h_ctl_course_setpoint - (*stateGetHorizontalSpeedDir_f());
00205   NormRadAngle(err);
00206 
00207   float d_err = err - last_err;
00208   last_err = err;
00209   NormRadAngle(d_err);
00210 
00211   float speed_depend_nav = (*stateGetHorizontalSpeedNorm_f())/NOMINAL_AIRSPEED;
00212   Bound(speed_depend_nav, 0.66, 1.5);
00213 
00214   h_ctl_roll_setpoint = h_ctl_course_pre_bank_correction * h_ctl_course_pre_bank
00215     + h_ctl_course_pgain * speed_depend_nav * err
00216     + h_ctl_course_dgain * d_err;
00217 
00218   BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
00219 }

00251 inline static void h_ctl_roll_loop( void ) {
00252 
00253   static float cmd_fb = 0.;

00265   }
00266   else if (h_ctl_ref_roll_rate < - H_CTL_REF_MAX_P) {
00267     h_ctl_ref_roll_rate = - H_CTL_REF_MAX_P;
00268     if (h_ctl_ref_roll_accel < 0.) h_ctl_ref_roll_accel = 0.;
00269   }
00270 #else
00271   h_ctl_ref_roll_angle = h_ctl_roll_setpoint;
00272   h_ctl_ref_roll_rate = 0.;

00274 #endif
00275 
00276 #ifdef USE_KFF_UPDATE
00277   // update Kff gains
00278   h_ctl_roll_Kffa += KFFA_UPDATE * h_ctl_ref_roll_accel * cmd_fb / (H_CTL_REF_MAX_P_DOT*H_CTL_REF_MAX_P_DOT);
00279   h_ctl_roll_Kffd += KFFD_UPDATE * h_ctl_ref_roll_rate  * cmd_fb / (H_CTL_REF_MAX_P*H_CTL_REF_MAX_P);
00280 #ifdef SITL

00286

00288   float err = h_ctl_ref_roll_angle - stateGetNedToBodyEulers_f()->phi;
00289   struct FloatRates* body_rate = stateGetBodyRates_f();
00290 #ifdef SITL
00291   static float last_err = 0;

00293   last_err = err;
00294 #endif
00295   float d_err = h_ctl_ref_roll_rate - body_rate->p;
00296 
00297   if (pprz_mode == PPRZ_MODE_MANUAL || launch == 0) {
00298     h_ctl_roll_sum_err = 0.;
00299   }
00300   else {
00301     if (h_ctl_roll_igain > 0.) {
00302       h_ctl_roll_sum_err += err * H_CTL_REF_DT;
00303       BoundAbs(h_ctl_roll_sum_err, H_CTL_ROLL_SUM_ERR_MAX / h_ctl_roll_igain);
00304     } else {
00305       h_ctl_roll_sum_err = 0.;
00306     }
00307   }
00308 
00309   cmd_fb = h_ctl_roll_attitude_gain * err;
00310   float cmd = - h_ctl_roll_Kffa * h_ctl_ref_roll_accel
00311     - h_ctl_roll_Kffd * h_ctl_ref_roll_rate
00312     - cmd_fb
00313     - h_ctl_roll_rate_gain * d_err
00314     - h_ctl_roll_igain * h_ctl_roll_sum_err
00315     + v_ctl_throttle_setpoint * h_ctl_aileron_of_throttle;
00316 
00317   cmd /= airspeed_ratio2;
00318 
00319   // Set aileron commands
