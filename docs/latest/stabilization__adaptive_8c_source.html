200 void h_ctl_course_loop ( void ) {
201  static float last_err;
202 
203  // Ground path error
204  float err = h_ctl_course_setpoint - (*stateGetHorizontalSpeedDir_f());
205  NormRadAngle(err);
206 
207  float d_err = err - last_err;
208  last_err = err;
209  NormRadAngle(d_err);
210 
211  float speed_depend_nav = (*stateGetHorizontalSpeedNorm_f())/NOMINAL_AIRSPEED;
212  Bound(speed_depend_nav, 0.66, 1.5);
213 
214  h_ctl_roll_setpoint = h_ctl_course_pre_bank_correction * h_ctl_course_pre_bank
215  + h_ctl_course_pgain * speed_depend_nav * err
216  + h_ctl_course_dgain * d_err;
217 
218  BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
219 }

222 static inline void compute_airspeed_ratio( void ) {
223  if (use_airspeed_ratio) {
224  // low pass airspeed
225  static float airspeed = 0.;
226  airspeed = ( 15*airspeed + (*stateGetAirspeed_f()) ) / 16;
227  // compute ratio
228  float airspeed_ratio = airspeed / NOMINAL_AIRSPEED;
229  Bound(airspeed_ratio, AIRSPEED_RATIO_MIN, AIRSPEED_RATIO_MAX);
230  airspeed_ratio2 = airspeed_ratio*airspeed_ratio;
231  } else {
232  airspeed_ratio2 = 1.;
233  }
234 }

248 #define KFFA_UPDATE 0.1
249 #define KFFD_UPDATE 0.05

251 inline static void h_ctl_roll_loop( void ) {
252 
253  static float cmd_fb = 0.;
254 
255 #if USE_ANGLE_REF
256  // Update reference setpoints for roll
257  h_ctl_ref_roll_angle += h_ctl_ref_roll_rate * H_CTL_REF_DT;
258  h_ctl_ref_roll_rate += h_ctl_ref_roll_accel * H_CTL_REF_DT;
259  h_ctl_ref_roll_accel = H_CTL_REF_W*H_CTL_REF_W * (h_ctl_roll_setpoint - h_ctl_ref_roll_angle) - 2*H_CTL_REF_XI*H_CTL_REF_W * h_ctl_ref_roll_rate;
260  // Saturation on references
261  BoundAbs(h_ctl_ref_roll_accel, H_CTL_REF_MAX_P_DOT);
262  if (h_ctl_ref_roll_rate > H_CTL_REF_MAX_P) {
263  h_ctl_ref_roll_rate = H_CTL_REF_MAX_P;
264  if (h_ctl_ref_roll_accel > 0.) h_ctl_ref_roll_accel = 0.;
265  }
266  else if (h_ctl_ref_roll_rate < - H_CTL_REF_MAX_P) {
267  h_ctl_ref_roll_rate = - H_CTL_REF_MAX_P;
268  if (h_ctl_ref_roll_accel < 0.) h_ctl_ref_roll_accel = 0.;
269  }
270 #else
271  h_ctl_ref_roll_angle = h_ctl_roll_setpoint;
272  h_ctl_ref_roll_rate = 0.;
273  h_ctl_ref_roll_accel = 0.;
274 #endif
275 
276 #ifdef USE_KFF_UPDATE
277  // update Kff gains
278  h_ctl_roll_Kffa += KFFA_UPDATE * h_ctl_ref_roll_accel * cmd_fb / (H_CTL_REF_MAX_P_DOT*H_CTL_REF_MAX_P_DOT);
279  h_ctl_roll_Kffd += KFFD_UPDATE * h_ctl_ref_roll_rate * cmd_fb / (H_CTL_REF_MAX_P*H_CTL_REF_MAX_P);
280 #ifdef SITL
281  printf("%f %f %f\n", h_ctl_roll_Kffa, h_ctl_roll_Kffd, cmd_fb);
282 #endif
283  h_ctl_roll_Kffa = Max(h_ctl_roll_Kffa, 0);
284  h_ctl_roll_Kffd = Max(h_ctl_roll_Kffd, 0);
285 #endif
286 
287  // Compute errors
288  float err = h_ctl_ref_roll_angle - stateGetNedToBodyEulers_f()->phi;
289  struct FloatRates* body_rate = stateGetBodyRates_f();
290 #ifdef SITL
291  static float last_err = 0;
292  body_rate->p = (err - last_err)/(1/60.); // FIXME should be done in ahrs sim
293  last_err = err;
294 #endif
295  float d_err = h_ctl_ref_roll_rate - body_rate->p;
296 
297  if (pprz_mode == PPRZ_MODE_MANUAL || launch == 0) {
298  h_ctl_roll_sum_err = 0.;
299  }
300  else {
301  if (h_ctl_roll_igain > 0.) {
302  h_ctl_roll_sum_err += err * H_CTL_REF_DT;
303  BoundAbs(h_ctl_roll_sum_err, H_CTL_ROLL_SUM_ERR_MAX / h_ctl_roll_igain);
304  } else {
305  h_ctl_roll_sum_err = 0.;
306  }
307  }
308 
309  cmd_fb = h_ctl_roll_attitude_gain * err;
310  float cmd = - h_ctl_roll_Kffa * h_ctl_ref_roll_accel
311  - h_ctl_roll_Kffd * h_ctl_ref_roll_rate
312  - cmd_fb
313  - h_ctl_roll_rate_gain * d_err
314  - h_ctl_roll_igain * h_ctl_roll_sum_err
315  + v_ctl_throttle_setpoint * h_ctl_aileron_of_throttle;
316 
317  cmd /= airspeed_ratio2;
318 
319  // Set aileron commands
320  h_ctl_aileron_setpoint = TRIM_PPRZ(cmd);
321 }

324 #if USE_PITCH_TRIM
325 inline static void loiter(void) {
326  float pitch_trim;
327 
328 #if USE_AIRSPEED
329  if (stateGetAirspeed_f() > NOMINAL_AIRSPEED) {
330  pitch_trim = v_ctl_pitch_dash_trim * (airspeed_ratio2-1) / ((AIRSPEED_RATIO_MAX * AIRSPEED_RATIO_MAX) - 1);
331  } else {
332  pitch_trim = v_ctl_pitch_loiter_trim * (airspeed_ratio2-1) / ((AIRSPEED_RATIO_MIN * AIRSPEED_RATIO_MIN) - 1);
333  }
334 #else
335  float throttle_diff = v_ctl_throttle_setpoint / (float)MAX_PPRZ - v_ctl_auto_throttle_nominal_cruise_throttle;
336  if (throttle_diff > 0) {
337  float max_diff = Max(V_CTL_AUTO_THROTTLE_MAX_CRUISE_THROTTLE - v_ctl_auto_throttle_nominal_cruise_throttle, 0.1);
338  pitch_trim = throttle_diff / max_diff * v_ctl_pitch_dash_trim;
339  } else {
340  float max_diff = Max(v_ctl_auto_throttle_nominal_cruise_throttle - V_CTL_AUTO_THROTTLE_MIN_CRUISE_THROTTLE, 0.1);
341  pitch_trim = -throttle_diff / max_diff * v_ctl_pitch_loiter_trim;
342  }
343 #endif
344 
345  h_ctl_pitch_loop_setpoint += pitch_trim;
346 }
347 #endif

350 inline static void h_ctl_pitch_loop( void ) {
351 #if !USE_GYRO_PITCH_RATE
352  static float last_err;
353 #endif
354 
355  /* sanity check */
356  if (h_ctl_pitch_of_roll <0.)
357  h_ctl_pitch_of_roll = 0.;
358 
359  h_ctl_pitch_loop_setpoint = h_ctl_pitch_setpoint + h_ctl_pitch_of_roll * fabs(stateGetNedToBodyEulers_f()->phi);
360 #if USE_PITCH_TRIM
361  loiter();
362 #endif
363 
364 #if USE_ANGLE_REF
365  // Update reference setpoints for pitch
366  h_ctl_ref_pitch_angle += h_ctl_ref_pitch_rate * H_CTL_REF_DT;
367  h_ctl_ref_pitch_rate += h_ctl_ref_pitch_accel * H_CTL_REF_DT;
368  h_ctl_ref_pitch_accel = H_CTL_REF_W*H_CTL_REF_W * (h_ctl_pitch_loop_setpoint - h_ctl_ref_pitch_angle) - 2*H_CTL_REF_XI*H_CTL_REF_W * h_ctl_ref_pitch_rate;
369  // Saturation on references
370  BoundAbs(h_ctl_ref_pitch_accel, H_CTL_REF_MAX_Q_DOT);
371  if (h_ctl_ref_pitch_rate > H_CTL_REF_MAX_Q) {
372  h_ctl_ref_pitch_rate = H_CTL_REF_MAX_Q;
373  if (h_ctl_ref_pitch_accel > 0.) h_ctl_ref_pitch_accel = 0.;
374  }
375  else if (h_ctl_ref_pitch_rate < - H_CTL_REF_MAX_Q) {
376  h_ctl_ref_pitch_rate = - H_CTL_REF_MAX_Q;
377  if (h_ctl_ref_pitch_accel < 0.) h_ctl_ref_pitch_accel = 0.;
378  }
379 #else
380  h_ctl_ref_pitch_angle = h_ctl_pitch_loop_setpoint;
381  h_ctl_ref_pitch_rate = 0.;
382  h_ctl_ref_pitch_accel = 0.;
383 #endif
384 
385  // Compute errors
386  float err = h_ctl_ref_pitch_angle - stateGetNedToBodyEulers_f()->theta;
387 #if USE_GYRO_PITCH_RATE
388  float d_err = h_ctl_ref_pitch_rate - stateGetBodyRates_f()->q;
389 #else // soft derivation
390  float d_err = (err - last_err)/H_CTL_REF_DT - h_ctl_ref_pitch_rate;
391  last_err = err;
392 #endif
393 
394  if (pprz_mode == PPRZ_MODE_MANUAL || launch == 0) {
395  h_ctl_pitch_sum_err = 0.;
396  }
397  else {
398  if (h_ctl_pitch_igain > 0.) {
399  h_ctl_pitch_sum_err += err * H_CTL_REF_DT;
400  BoundAbs(h_ctl_pitch_sum_err, H_CTL_PITCH_SUM_ERR_MAX / h_ctl_pitch_igain);
401  } else {
402  h_ctl_pitch_sum_err = 0.;
403  }
404  }
405 
406  float cmd = - h_ctl_pitch_Kffa * h_ctl_ref_pitch_accel
407  - h_ctl_pitch_Kffd * h_ctl_ref_pitch_rate
408  + h_ctl_pitch_pgain * err
409  + h_ctl_pitch_dgain * d_err
410  + h_ctl_pitch_igain * h_ctl_pitch_sum_err;
411 
412  cmd /= airspeed_ratio2;
413 
414  h_ctl_elevator_setpoint = TRIM_PPRZ(cmd);
415 }
