116 void nav_run(void) {
117 
118  /* compute a vector to the waypoint */
119  struct Int32Vect2 path_to_waypoint;
120  VECT2_DIFF(path_to_waypoint, navigation_target, *stateGetPositionEnu_i());
121 
122  /* saturate it */
123  VECT2_STRIM(path_to_waypoint, -(1<<15), (1<<15));
124 
125 #if !GUIDANCE_H_USE_REF
126  int32_t dist_to_waypoint;
127  INT32_VECT2_NORM(dist_to_waypoint, path_to_waypoint);
128 
129  if (dist_to_waypoint < CLOSE_TO_WAYPOINT) {
130  VECT2_COPY(navigation_carrot, navigation_target);
131  }
132  else {
133  struct Int32Vect2 path_to_carrot;
134  VECT2_SMUL(path_to_carrot, path_to_waypoint, CARROT_DIST);
135  VECT2_SDIV(path_to_carrot, path_to_carrot, dist_to_waypoint);
136  VECT2_SUM(navigation_carrot, path_to_carrot, *stateGetPositionEnu_i());
137  }
138 #else
139  // if H_REF is used, CARROT_DIST is not used
140  VECT2_COPY(navigation_carrot, navigation_target);
141 #endif
142 
143  nav_set_altitude();
144 }

146 void nav_circle(uint8_t wp_center, int32_t radius) {
147  if (radius == 0) {
148  VECT2_COPY(navigation_target, waypoints[wp_center]);
149  }
150  else {
151  struct Int32Vect2 pos_diff;
152  VECT2_DIFF(pos_diff, *stateGetPositionEnu_i(), waypoints[wp_center]);
153  // go back to half metric precision or values are too large
154  //INT32_VECT2_RSHIFT(pos_diff,pos_diff,INT32_POS_FRAC/2);
155  // store last qdr
156  int32_t last_qdr = nav_circle_qdr;
157  // compute qdr
158  INT32_ATAN2(nav_circle_qdr, pos_diff.y, pos_diff.x);
159  // increment circle radians
160  if (nav_circle_radians != 0) {
161  int32_t angle_diff = nav_circle_qdr - last_qdr;
162  INT32_ANGLE_NORMALIZE(angle_diff);
163  nav_circle_radians += angle_diff;
164  }
165  else {
166  // Smallest angle to increment at next step
167  nav_circle_radians = 1;
168  }
169 
170  // direction of rotation
171  int8_t sign_radius = radius > 0 ? 1 : -1;
172  // absolute radius
173  int32_t abs_radius = abs(radius);
174  // carrot_angle
175  int32_t carrot_angle = ((CARROT_DIST<<INT32_ANGLE_FRAC) / abs_radius);
176  Bound(carrot_angle, (INT32_ANGLE_PI / 16), INT32_ANGLE_PI_4);
177  carrot_angle = nav_circle_qdr - sign_radius * carrot_angle;
178  int32_t s_carrot, c_carrot;
179  PPRZ_ITRIG_SIN(s_carrot, carrot_angle);
180  PPRZ_ITRIG_COS(c_carrot, carrot_angle);
181  // compute setpoint
182  VECT2_ASSIGN(pos_diff, abs_radius * c_carrot, abs_radius * s_carrot);
183  INT32_VECT2_RSHIFT(pos_diff, pos_diff, INT32_TRIG_FRAC);
184  VECT2_SUM(navigation_target, waypoints[wp_center], pos_diff);
185  }
186  nav_circle_centre = wp_center;
187  nav_circle_radius = radius;
188  horizontal_mode = HORIZONTAL_MODE_CIRCLE;
189 }

193 void nav_route(uint8_t wp_start, uint8_t wp_end) {
194  struct Int32Vect2 wp_diff,pos_diff;
195  VECT2_DIFF(wp_diff, waypoints[wp_end],waypoints[wp_start]);
196  VECT2_DIFF(pos_diff, *stateGetPositionEnu_i(), waypoints[wp_start]);
197  // go back to metric precision or values are too large
198  INT32_VECT2_RSHIFT(wp_diff,wp_diff,INT32_POS_FRAC);
199  INT32_VECT2_RSHIFT(pos_diff,pos_diff,INT32_POS_FRAC);
200  int32_t leg_length2 = Max((wp_diff.x * wp_diff.x + wp_diff.y * wp_diff.y),1);
201  INT32_SQRT(nav_leg_length,leg_length2);
202  nav_leg_progress = (pos_diff.x * wp_diff.x + pos_diff.y * wp_diff.y) / nav_leg_length;
203  int32_t progress = Max((CARROT_DIST >> INT32_POS_FRAC), 0);
204  nav_leg_progress += progress;
205  int32_t prog_2 = nav_leg_length;// + progress / 2;
206  Bound(nav_leg_progress, 0, prog_2);
207  struct Int32Vect2 progress_pos;
208  VECT2_SMUL(progress_pos, wp_diff, nav_leg_progress);
209  VECT2_SDIV(progress_pos, progress_pos, nav_leg_length);
210  INT32_VECT2_LSHIFT(progress_pos,progress_pos,INT32_POS_FRAC);
211  VECT2_SUM(navigation_target,waypoints[wp_start],progress_pos);
212  //printf("target %d %d | p %d %d | s %d %d | l %d %d %d\n",
213  // navigation_target.x,
214  // navigation_target.y,
215  // progress_pos.x,
216  // progress_pos.y,
217  // waypoints[wp_start].x,
218  // waypoints[wp_start].y,
219  // leg_length, leg_length2, nav_leg_progress);
220  //fflush(stdout);
221 
222  nav_segment_start = wp_start;
223  nav_segment_end = wp_end;
224  horizontal_mode = HORIZONTAL_MODE_ROUTE;
225 }

227 bool_t nav_approaching_from(uint8_t wp_idx, uint8_t from_idx) {
228  int32_t dist_to_point;
229  struct Int32Vect2 diff;
230  static uint8_t time_at_wp = 0;
231  VECT2_DIFF(diff, waypoints[wp_idx], *stateGetPositionEnu_i());
232  INT32_VECT2_RSHIFT(diff,diff,INT32_POS_FRAC);
233  INT32_VECT2_NORM(dist_to_point, diff);
234  //printf("dist %d | %d %d\n", dist_to_point,diff.x,diff.y);
235  //fflush(stdout);
236  //if (dist_to_point < (ARRIVED_AT_WAYPOINT >> INT32_POS_FRAC)) return TRUE;
237  if (dist_to_point < (ARRIVED_AT_WAYPOINT >> INT32_POS_FRAC)) time_at_wp++;
238  else time_at_wp = 0;
239  if (time_at_wp > 20) return TRUE;
240  if (from_idx > 0 && from_idx < NB_WAYPOINT) {
241  struct Int32Vect2 from_diff;
242  VECT2_DIFF(from_diff, waypoints[wp_idx],waypoints[from_idx]);
243  INT32_VECT2_RSHIFT(from_diff,from_diff,INT32_POS_FRAC);
244  return (diff.x * from_diff.x + diff.y * from_diff.y < 0);
245  }
246  else return FALSE;
247 }

308 void navigation_update_wp_from_speed(uint8_t wp, struct Int16Vect3 speed_sp, int16_t heading_rate_sp ) {
309  // MY_ASSERT(wp < nb_waypoint); FIXME
310  int32_t s_heading, c_heading;
311  PPRZ_ITRIG_SIN(s_heading, nav_heading);
312  PPRZ_ITRIG_COS(c_heading, nav_heading);
313  // FIXME : scale POS to SPEED
314  struct Int32Vect3 delta_pos;
315  VECT3_SDIV(delta_pos, speed_sp,NAV_FREQ); /* fixme :make sure the division is really a >> */
316  INT32_VECT3_RSHIFT(delta_pos, delta_pos, (INT32_SPEED_FRAC-INT32_POS_FRAC));
317  waypoints[wp].x += (s_heading * delta_pos.x + c_heading * delta_pos.y) >> INT32_TRIG_FRAC;
318  waypoints[wp].y += (c_heading * delta_pos.x - s_heading * delta_pos.y) >> INT32_TRIG_FRAC;
319  waypoints[wp].z += delta_pos.z;
320  int32_t delta_heading = heading_rate_sp / NAV_FREQ;
321  delta_heading = delta_heading >> (INT32_SPEED_FRAC-INT32_POS_FRAC);
322  nav_heading += delta_heading;
323 
324  INT32_COURSE_NORMALIZE(nav_heading);
325  RunOnceEvery(10,DOWNLINK_SEND_WP_MOVED_ENU(DefaultChannel, DefaultDevice, &wp, &(waypoints[wp].x), &(waypoints[wp].y), &(waypoints[wp].z)));
326 }
