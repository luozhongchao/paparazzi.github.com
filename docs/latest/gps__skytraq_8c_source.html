101  if (gps_skytraq.msg_id == SKYTRAQ_ID_NAVIGATION_DATA) {

107  gps.ecef_vel.z = SKYTRAQ_NAVIGATION_DATA_ECEFVZ(gps_skytraq.msg_buf);
108  gps.lla_pos.lat = RadOfDeg(SKYTRAQ_NAVIGATION_DATA_LAT(gps_skytraq.msg_buf));
109  gps.lla_pos.lon = RadOfDeg(SKYTRAQ_NAVIGATION_DATA_LON(gps_skytraq.msg_buf));
110  gps.lla_pos.alt = SKYTRAQ_NAVIGATION_DATA_AEL(gps_skytraq.msg_buf)*10;

117

179 void gps_skytraq_parse(uint8_t c) {
180  if (gps_skytraq.status < GOT_PAYLOAD) {
181  gps_skytraq.checksum ^= c;
182  }
183  switch (gps_skytraq.status) {
184  case UNINIT:
185  if (c == SKYTRAQ_SYNC1)
186  gps_skytraq.status = GOT_SYNC1;
187  break;
188  case GOT_SYNC1:
189  if (c != SKYTRAQ_SYNC2) {
190  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
191  goto error;
192  }
193  gps_skytraq.status = GOT_SYNC2;
194  break;
195  case GOT_SYNC2:
196  gps_skytraq.len = c<<8;
197  gps_skytraq.status = GOT_LEN1;
198  break;
199  case GOT_LEN1:
200  gps_skytraq.len += c;
201  gps_skytraq.status = GOT_LEN2;
202  if (gps_skytraq.len > GPS_SKYTRAQ_MAX_PAYLOAD) {
203  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_MSG_TOO_LONG;
204  goto error;
205  }
206  break;
207  case GOT_LEN2:
208  gps_skytraq.msg_id = c;
209  gps_skytraq.msg_idx = 0;
210  gps_skytraq.checksum = c;
211  gps_skytraq.status = GOT_ID;
212  break;
213  case GOT_ID:
214  gps_skytraq.msg_buf[gps_skytraq.msg_idx] = c;
215  gps_skytraq.msg_idx++;
216  if (gps_skytraq.msg_idx >= gps_skytraq.len-1) {
217  gps_skytraq.status = GOT_PAYLOAD;
218  }
219  break;
220  case GOT_PAYLOAD:
221  if (c != gps_skytraq.checksum) {
222  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_CHECKSUM;
223  goto error;
224  }
225  gps_skytraq.status = GOT_CHECKSUM;
226  break;
227  case GOT_CHECKSUM:
228  if (c != SKYTRAQ_SYNC3) {
229  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
230  goto error;
231  }
232  gps_skytraq.status = GOT_SYNC3;
233  break;
234  case GOT_SYNC3:
235  gps_skytraq.msg_available = TRUE;
236  goto restart;
237  default:
238  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_UNEXPECTED;
239  goto error;
240  }
241  return;
242  error:
243  gps_skytraq.error_cnt++;
244  restart:
245  gps_skytraq.status = UNINIT;
246  return;
247 }

249 static int distance_too_great(struct EcefCoor_i *ecef_ref, struct EcefCoor_i *ecef_pos) {
250  int32_t xdiff = abs(ecef_ref->x - ecef_pos->x);
251  if (xdiff > MAX_DISTANCE) {
252  return TRUE;
253  }
254  int32_t ydiff = abs(ecef_ref->y - ecef_pos->y);
255  if (ydiff > MAX_DISTANCE) {
256  return TRUE;
257  }
258  int32_t zdiff = abs(ecef_ref->z - ecef_pos->z);
259  if (zdiff > MAX_DISTANCE) {
260  return TRUE;
261  }
262 
263  return FALSE;
264 }
