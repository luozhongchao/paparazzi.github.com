80  gps.lla_pos.lon = RadOfDeg(SKYTRAQ_NAVIGATION_DATA_LON(gps_skytraq.msg_buf));

154 void gps_skytraq_parse(uint8_t c) {
155  if (gps_skytraq.status < GOT_PAYLOAD)
156  gps_skytraq.checksum ^= c;
157  switch (gps_skytraq.status) {
158  case UNINIT:
159  if (c == SKYTRAQ_SYNC1)
160  gps_skytraq.status = GOT_SYNC1;
161  break;
162  case GOT_SYNC1:
163  if (c != SKYTRAQ_SYNC2) {
164  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
165  goto error;
166  }
167  gps_skytraq.status = GOT_SYNC2;
168  break;
169  case GOT_SYNC2:
170  gps_skytraq.len = c<<8;
171  gps_skytraq.status = GOT_LEN1;
172  break;
173  case GOT_LEN1:
174  gps_skytraq.len += c;
175  gps_skytraq.status = GOT_LEN2;
176  if (gps_skytraq.len > GPS_SKYTRAQ_MAX_PAYLOAD) {
177  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_MSG_TOO_LONG;
178  goto error;
179  }
180  break;
181  case GOT_LEN2:
182  gps_skytraq.msg_id = c;
183  gps_skytraq.msg_idx = 0;
184  gps_skytraq.checksum = c;
185  gps_skytraq.status = GOT_ID;
186  break;
187  case GOT_ID:
188  gps_skytraq.msg_buf[gps_skytraq.msg_idx] = c;
189  gps_skytraq.msg_idx++;
190  if (gps_skytraq.msg_idx >= gps_skytraq.len-1) {
191  gps_skytraq.status = GOT_PAYLOAD;
192  }
193  break;
194  case GOT_PAYLOAD:
195  if (c != gps_skytraq.checksum) {
196  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_CHECKSUM;
197  goto error;
198  }
199  gps_skytraq.status = GOT_CHECKSUM;
200  break;
201  case GOT_CHECKSUM:
202  if (c != SKYTRAQ_SYNC3) {
203  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
204  goto error;
205  }
206  gps_skytraq.status = GOT_SYNC3;
207  break;
208  case GOT_SYNC3:
209  gps_skytraq.msg_available = TRUE;
210  goto restart;
211  default:
212  gps_skytraq.error_last = GPS_SKYTRAQ_ERR_UNEXPECTED;
213  goto error;
214  }
215  return;
216  error:
217  gps_skytraq.error_cnt++;
218  restart:
219  gps_skytraq.status = UNINIT;
220  return;
221 }

223 static int distance_too_great( struct EcefCoor_i *ecef_ref, struct EcefCoor_i *ecef_pos ) {
224  int32_t xdiff = abs(ecef_ref->x - ecef_pos->x);
225  if ( xdiff > MAX_DISTANCE ) return TRUE;
226  int32_t ydiff = abs(ecef_ref->y - ecef_pos->y);
227  if ( ydiff > MAX_DISTANCE ) return TRUE;
228  int32_t zdiff = abs(ecef_ref->z - ecef_pos->z);
229  if ( zdiff > MAX_DISTANCE ) return TRUE;
230 
231  return FALSE;
232 }
