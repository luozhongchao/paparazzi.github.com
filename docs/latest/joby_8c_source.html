00031 static void handle_channel(void (* callback)(void))
00032 {
00033  if (parser.parser_normal_buf == RC_JOBY_MAGIC_START) {
00034     // got start channel, look for channel 0 next
00035     parser.current_channel = 0;
00036   } else if (parser.current_channel == -1) {
00037     // looking for start channel byte but didn't get it, reset
00038     parser.current_byte = READING_HIGH_BYTE;
00039     parser.current_inverted = READING_NORMAL;
00040   } else {
00041     // valid channel, store and look for next
00042     radio_control.values[parser.current_channel] = rc_joby_signs[parser.current_channel] * parser.parser_normal_buf;
00043     parser.current_channel++;
00044     if (parser.current_channel == RADIO_CONTROL_NB_CHANNEL) {
00045       // all channels read, reset parser and handle message
00046       parser.current_channel = -1;
00047       radio_control.frame_cpt++;
00048       radio_control.status = RC_OK;
00049       radio_control.time_since_last_frame = 0;
00050       if (callback != NULL)
00051     callback();
00052     }
00053   }
00054 }

00056 static void handle_tuple(void (* callback)(void))
00057 {
00058   if (parser.current_inverted == READING_NORMAL) {
00059     parser.parser_normal_buf = ((parser.high_byte_buf << 8) | parser.low_byte_buf);
00060     parser.current_inverted = READING_INVERTED;
00061   } else if (parser.current_inverted == READING_INVERTED) {
00062     parser.parser_inverted_buf = ((parser.high_byte_buf << 8) | parser.low_byte_buf);
00063     parser.current_inverted = READING_NORMAL;
00064     if (parser.parser_normal_buf == ~parser.parser_inverted_buf) {
00065       handle_channel(callback);
00066     } else {
00067       // normal didn't match inverted, error, reset
00068       parser.current_inverted = READING_NORMAL;
00069       parser.current_byte = READING_HIGH_BYTE;
00070       parser.current_channel = -1;
00071       parser.error_counter++;
00072     }
00073   }
00074 }

00076 void rc_joby_parse(int8_t c, void (* callback)(void))
00077 {
00078   if (parser.current_byte == READING_HIGH_BYTE) {
00079     parser.high_byte_buf = c;
00080     if (parser.current_channel >= 0 || parser.high_byte_buf == (RC_JOBY_MAGIC_START >> 8) || parser.current_inverted == READING_INVERTED) {
00081        // only advance parser state to low byte if we're not looking for a sync byte which we didn't find
00082       parser.current_byte = READING_LOW_BYTE;
00083     }
00084   } else { // READING_LOW_BYTE
00085     parser.low_byte_buf = c;
00086     parser.current_byte = READING_HIGH_BYTE;
00087     handle_tuple(callback);
00088   }
00089 }
