00029 static void handle_channel(void (* callback)(void))
00030 {
00031  if (parser.parser_normal_buf == RC_JOBY_MAGIC_START) {
00032     // got start channel, look for channel 0 next
00033     parser.current_channel = 0;
00034   } else if (parser.current_channel == -1) {
00035     // looking for start channel byte but didn't get it, reset
00036     parser.current_byte = READING_HIGH_BYTE;
00037     parser.current_inverted = READING_NORMAL;
00038   } else {
00039     // valid channel, store and look for next
00040     radio_control.values[parser.current_channel] = rc_joby_signs[parser.current_channel] * parser.parser_normal_buf;
00041     parser.current_channel++;
00042     if (parser.current_channel == RADIO_CONTROL_NB_CHANNEL) {
00043       // all channels read, reset parser and handle message
00044       parser.current_channel = -1;
00045       radio_control.frame_cpt++;
00046       radio_control.status = RC_OK;
00047       radio_control.time_since_last_frame = 0;
00048       if (callback != NULL)
00049     callback();
00050     }
00051   }
00052 }

00054 static void handle_tuple(void (* callback)(void))
00055 {
00056   if (parser.current_inverted == READING_NORMAL) {
00057     parser.parser_normal_buf = ((parser.high_byte_buf << 8) | parser.low_byte_buf);
00058     parser.current_inverted = READING_INVERTED;
00059   } else if (parser.current_inverted == READING_INVERTED) {
00060     parser.parser_inverted_buf = ((parser.high_byte_buf << 8) | parser.low_byte_buf);
00061     parser.current_inverted = READING_NORMAL;
00062     if (parser.parser_normal_buf == ~parser.parser_inverted_buf) {
00063       handle_channel(callback);
00064     } else {
00065       // normal didn't match inverted, error, reset
00066       parser.current_inverted = READING_NORMAL;
00067       parser.current_byte = READING_HIGH_BYTE;
00068       parser.current_channel = -1;
00069       parser.error_counter++;
00070     }
00071   }
00072 }

00074 void rc_joby_parse(int8_t c, void (* callback)(void))
00075 {
00076   if (parser.current_byte == READING_HIGH_BYTE) {
00077     parser.high_byte_buf = c;
00078     if (parser.current_channel >= 0 || parser.high_byte_buf == (RC_JOBY_MAGIC_START >> 8) || parser.current_inverted == READING_INVERTED) {
00079        // only advance parser state to low byte if we're not looking for a sync byte which we didn't find
00080       parser.current_byte = READING_LOW_BYTE;
00081     }
00082   } else { // READING_LOW_BYTE
00083     parser.low_byte_buf = c;
00084     parser.current_byte = READING_HIGH_BYTE;
00085     handle_tuple(callback);
00086   }
00087 }
