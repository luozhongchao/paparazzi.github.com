439 static inline void stateSetLocalOrigin_i(struct LtpDef_i* ltp_def) {
440  LTP_DEF_COPY(state.ned_origin_i, *ltp_def);
441  state.ned_initialized_i = TRUE;
442  ECEF_FLOAT_OF_BFP(state.ned_origin_f.ecef, state.ned_origin_i.ecef);
443  LLA_FLOAT_OF_BFP(state.ned_origin_f.lla, state.ned_origin_i.lla);
444  RMAT_FLOAT_OF_BFP(state.ned_origin_f.ltp_of_ecef, state.ned_origin_i.ltp_of_ecef);
445  state.ned_origin_f.hmsl = M_OF_MM(state.ned_origin_f.hmsl);
446  state.ned_initialized_f = TRUE;
447 
448  /* clear bits for all local frame representations */
449  ClearBit(state.pos_status, POS_NED_I);
450  ClearBit(state.pos_status, POS_ENU_I);
451  ClearBit(state.pos_status, POS_NED_F);
452  ClearBit(state.pos_status, POS_ENU_F);
453  ClearBit(state.speed_status, SPEED_NED_I);
454  ClearBit(state.speed_status, SPEED_ENU_I);
455  ClearBit(state.speed_status, SPEED_NED_F);
456  ClearBit(state.speed_status, SPEED_ENU_F);
457  ClearBit(state.accel_status, ACCEL_NED_I);
458  ClearBit(state.accel_status, ACCEL_NED_F);
459 }
