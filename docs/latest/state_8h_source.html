00441   state.ned_initialized_i = TRUE;
00442   ECEF_FLOAT_OF_BFP(state.ned_origin_f.ecef, state.ned_origin_i.ecef);
00443   LLA_FLOAT_OF_BFP(state.ned_origin_f.lla, state.ned_origin_i.lla);
00444   RMAT_FLOAT_OF_BFP(state.ned_origin_f.ltp_of_ecef, state.ned_origin_i.ltp_of_ecef);
00445   state.ned_origin_f.hmsl = M_OF_MM(state.ned_origin_f.hmsl);
00446   state.ned_initialized_f = TRUE;
00447

00450   ClearBit(state.pos_status, POS_ENU_I);
00451   ClearBit(state.pos_status, POS_NED_F);
00452   ClearBit(state.pos_status, POS_ENU_F);

00454   ClearBit(state.speed_status, SPEED_ENU_I);

00456   ClearBit(state.speed_status, SPEED_ENU_F);

00458   ClearBit(state.accel_status, ACCEL_NED_F);
