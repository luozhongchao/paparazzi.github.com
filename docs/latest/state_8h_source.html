440 static inline void stateSetLocalOrigin_i(struct LtpDef_i* ltp_def) {
441  memcpy(&state.ned_origin_i, ltp_def, sizeof(struct LtpDef_i));
442  /* convert to float */
443  ECEF_FLOAT_OF_BFP(state.ned_origin_f.ecef, state.ned_origin_i.ecef);
444  LLA_FLOAT_OF_BFP(state.ned_origin_f.lla, state.ned_origin_i.lla);
445  RMAT_FLOAT_OF_BFP(state.ned_origin_f.ltp_of_ecef, state.ned_origin_i.ltp_of_ecef);
446  state.ned_origin_f.hmsl = M_OF_MM(state.ned_origin_i.hmsl);
447 
448  /* clear bits for all local frame representations */
449  state.pos_status &= ~(POS_LOCAL_COORD);
450  state.speed_status &= ~(SPEED_LOCAL_COORD);
451  ClearBit(state.accel_status, ACCEL_NED_I);
452  ClearBit(state.accel_status, ACCEL_NED_F);
453 
454  state.ned_initialized_i = TRUE;
455  state.ned_initialized_f = TRUE;
456 }
