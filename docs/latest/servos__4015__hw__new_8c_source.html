00058   SERV1_CLOCK_PINSEL |= SERV1_CLOCK_PINSEL_VAL << SERV1_CLOCK_PINSEL_BIT;

00062   PWMMR_SERV1 = 0XFFF;
00063   PWMMR_SERV0 = 0X0;

00095 void PWM_ISR ( void ) {
00096   ISR_ENTRY();
00097   if (servos_idx == 0) {
00098     /* lower serv0 reset */
00099     IO1CLR = _BV(SERV0_RESET_PIN);
00100     /* raise serv0 data  */
00101     IO1SET = _BV(SERV0_DATA_PIN);
00102     /* start pulsing serv0 */
00103     PWMMR_SERV0 = 0XFFF;
00104     /* stop pulsing serv1 */
00105     PWMMR_SERV1 = 0X0;
00106     PWMMR0 = servos_values[servos_idx];
00107     servos_delay = SERVO_REFRESH_TICS - servos_values[servos_idx];
00108     PWMLER = PWMLER_LATCH0 | PWMLER_LATCH_SERV0 | PWMLER_LATCH_SERV1;
00109     /* enable serv0 match interrupt, disable serv1 match interrupt */
00110     PWMMCR = PWMMCR_MR0R | PWMMCR_MRI_SERV0;
00111     servos_idx++;
00112   }
00113   else if (servos_idx < 4) {
00114     /* lower serv0 data */
00115     IO1CLR = _BV(SERV0_DATA_PIN);
00116     PWMMR0 = servos_values[servos_idx];
00117     servos_delay -= servos_values[servos_idx];
00118     PWMLER = PWMLER_LATCH0;
00119     servos_idx++;
00120   }
00121   else if (servos_idx == 4) {
00122     /* raise serv0 reset */
00123     IO1SET = _BV(SERV1_RESET_PIN);
00124     /* lower serv1 reset */
00125     IO1CLR = _BV(SERV1_RESET_PIN);
00126     /* raise serv1 data */
00127     IO1SET = _BV(SERV1_DATA_PIN);
00128     /* stop pulsing serv0 */
00129     PWMMR_SERV0 = 0;
00130     /* start pulsing serv1 */
00131     PWMMR_SERV1 = 0XFFF;
00132     /* disable serv0 interrupt, enable serv1 match interrupt */
00133     PWMMCR = PWMMCR_MR0R | PWMMCR_MRI_SERV1;
00134     /* fill next servo value */
00135     PWMMR0 = servos_values[servos_idx];
00136     servos_delay -= servos_values[servos_idx];
00137     /* latch PWM values */
00138     PWMLER = PWMLER_LATCH0 | PWMLER_LATCH_SERV0 | PWMLER_LATCH_SERV1;
00139     servos_idx++;
00140   }
00141   else if (servos_idx < _4015_NB_CHANNELS) {
00142     /* clear serv1 data */
00143     IO1CLR = _BV(SERV1_DATA_PIN);
00144     PWMMR0 = servos_values[servos_idx];
00145     servos_delay -= servos_values[servos_idx];
00146     PWMLER = PWMLER_LATCH0;
00147     servos_idx++;
00148   }
00149   else {
00150     /* raise serv1 reset */
00151     IO1SET = _BV(SERV1_RESET_PIN);
00152     /* command the delay */
00153     PWMMR0 = servos_delay;
00154     PWMLER = PWMLER_LATCH0;
00155     servos_idx = 0;
00156   }
00157   /* clear the interrupt */
00158   PWMIR = PWMIR_MRI_SERV1;
00159   PWMIR = PWMIR_MRI_SERV0;
00160   VICVectAddr = 0x00000000;
00161   ISR_EXIT();
00162 }
