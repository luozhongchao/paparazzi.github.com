52 void dl_parse_msg(void) {
53 
54  datalink_time = 0;
55 
56  uint8_t msg_id = IdOfMsg(dl_buffer);
57  switch (msg_id) {
58 
59  case DL_PING:
60  {
61  DOWNLINK_SEND_PONG(DefaultChannel, DefaultDevice);
62  }
63  break;
64 
65  case DL_SETTING :
66  {
67  if (DL_SETTING_ac_id(dl_buffer) != AC_ID) break;
68  uint8_t i = DL_SETTING_index(dl_buffer);
69  float var = DL_SETTING_value(dl_buffer);
70  DlSetting(i, var);
71  DOWNLINK_SEND_DL_VALUE(DefaultChannel, DefaultDevice, &i, &var);
72  }
73  break;
74 
75  case DL_GET_SETTING :
76  {
77  if (DL_GET_SETTING_ac_id(dl_buffer) != AC_ID) break;
78  uint8_t i = DL_GET_SETTING_index(dl_buffer);
79  float val = settings_get_value(i);
80  DOWNLINK_SEND_DL_VALUE(DefaultChannel, DefaultDevice, &i, &val);
81  }
82  break;
83 
84 #if defined USE_NAVIGATION
85  case DL_BLOCK :
86  {
87  if (DL_BLOCK_ac_id(dl_buffer) != AC_ID) break;
88  nav_goto_block(DL_BLOCK_block_id(dl_buffer));
89  }
90  break;
91 
92  case DL_MOVE_WP :
93  {
94  uint8_t ac_id = DL_MOVE_WP_ac_id(dl_buffer);
95  if (ac_id != AC_ID) break;
96  uint8_t wp_id = DL_MOVE_WP_wp_id(dl_buffer);
97  struct LlaCoor_i lla;
98  struct EnuCoor_i enu;
99  lla.lat = INT32_RAD_OF_DEG(DL_MOVE_WP_lat(dl_buffer));
100  lla.lon = INT32_RAD_OF_DEG(DL_MOVE_WP_lon(dl_buffer));
101  /* WP_alt is in cm, lla.alt in mm */
102  lla.alt = DL_MOVE_WP_alt(dl_buffer)*10 - ins_ltp_def.hmsl + ins_ltp_def.lla.alt;
103  enu_of_lla_point_i(&enu,&ins_ltp_def,&lla);
104  enu.x = POS_BFP_OF_REAL(enu.x)/100;
105  enu.y = POS_BFP_OF_REAL(enu.y)/100;
106  enu.z = POS_BFP_OF_REAL(enu.z)/100;
107  VECT3_ASSIGN(waypoints[wp_id], enu.x, enu.y, enu.z);
108  DOWNLINK_SEND_WP_MOVED_ENU(DefaultChannel, DefaultDevice, &wp_id, &enu.x, &enu.y, &enu.z);
109  }
110  break;
111 #endif /* USE_NAVIGATION */
112  default:
113  break;
114  }
115  /* Parse modules datalink */
116  modules_parse_datalink(msg_id);
117 }
