148 static inline void DetectGroundEvent(void) {
149  if (autopilot_mode == AP_MODE_FAILSAFE || autopilot_detect_ground_once) {
150  struct NedCoor_f* accel = stateGetAccelNed_f();
151  if (accel->z < -THRESHOLD_GROUND_DETECT ||
152  accel->z > THRESHOLD_GROUND_DETECT) {
153  autopilot_detect_ground = TRUE;
