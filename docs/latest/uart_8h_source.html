38 /**
39  * UART peripheral
40  */
41 struct uart_periph {
42  /* Receive buffer */
43  uint8_t rx_buf[UART_RX_BUFFER_SIZE];
44  uint16_t rx_insert_idx;
45  uint16_t rx_extract_idx;
46  /* Transmit buffer */
47  uint8_t tx_buf[UART_TX_BUFFER_SIZE];
48  uint16_t tx_insert_idx;
49  uint16_t tx_extract_idx;
50  uint8_t tx_running;
51  /* UART Register */
52  void* reg_addr;
53  /* UART Dev (linux) */
54  char dev[UART_DEV_NAME_SIZE];
55 };

63 #define UartChAvailable(_p) (_p.rx_insert_idx != _p.rx_extract_idx)
64 
65 #define UartGetch(_p) ({ \
66  uint8_t ret = _p.rx_buf[_p.rx_extract_idx]; \
67  _p.rx_extract_idx = (_p.rx_extract_idx + 1)%UART_RX_BUFFER_SIZE; \
68  ret; \
69 })
