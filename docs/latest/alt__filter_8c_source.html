00130 void kalmanEstimation(TypeKalman *k, float accVert){

00132 
00133   int i,j;
00134   float I[3][3]; // matrices temporaires
00135   float J[3][2];
00136 
00137   // Calcul de X
00138   // X(k+1) = Ad*X(k) + Bd*U(k)
00139   k->X[0] = k->Ad[0][0]*k->X[0] + k->Ad[0][1]*k->X[1] + k->Ad[0][2]*k->X[2] + k->Bd[0]*accVert;
00140   k->X[1] = k->Ad[1][0]*k->X[0] + k->Ad[1][1]*k->X[1] + k->Ad[1][2]*k->X[2] + k->Bd[1]*accVert;
00141   k->X[2] = k->Ad[2][0]*k->X[0] + k->Ad[2][1]*k->X[1] + k->Ad[2][2]*k->X[2] + k->Bd[2]*accVert;
00142 
00143   // Calcul de P
00144   // P(k+1) = Ad*P(k)*Ad' + Md*W*Md'
00145   for(i=0;i<3;i++){
00146     for (j=0;j<3;j++){
00147       I[i][j] = k->Ad[i][0]*k->P[0][j] + k->Ad[i][1]*k->P[1][j] + k->Ad[i][2]*k->P[2][j];
00148     }
00149   }
00150 
00151   for(i=0;i<3;i++){
00152     for (j=0;j<3;j++){
00153       k->P[i][j] = I[i][0]*k->Ad[j][0] + I[i][1]*k->Ad[j][1] + I[i][2]*k->Ad[j][2];
00154     }

00177 void kalmanCorrectionGPS(TypeKalman *k, float altitude_gps){ // altitude_gps est l'altitude telle qu'elle est mesurée par le GPS
00178 
00179   int i, j, div;
00180   float I[3][3]; // matrice temporaire
00181 
00182   float Kf[3] = { 0., 0., 0. };
00183 
00184   // calcul de Kf
00185   // C = [1 0 0]
00186   // div = C*P*C' + R
00187   div = k->P[0][0] + SigAltiGPS*SigAltiGPS;
00188 
00189   if (fabs(div) > 1e-5) {
00190     // Kf = P*C'*inv(div)
00191     Kf[0] = k->P[0][0] / div;
00192     Kf[1] = k->P[1][0] / div;
00193     Kf[2] = k->P[2][0] / div;
00194 
00195     // calcul de X
00196     // X = X + Kf*(meas - C*X)
00197     float constante = k->X[0];
00198     for(i=0;i<3;i++){
00199       k->X[i] = k->X[i]+Kf[i]*(altitude_gps - constante);
00200     }
00201 
00202     // calcul de P
00203     // P = P - Kf*C*P
00204     I[0][0] = Kf[0]; I[0][1] = 0; I[0][2] = 0;
00205     I[1][0] = Kf[1]; I[1][1] = 0; I[1][2] = 0;
00206     I[2][0] = Kf[2]; I[2][1] = 0; I[2][2] = 0;
00207 
00208     for(i=0;i<3;i++){
00209       for (j=0;j<3;j++){
00210         k->P[i][j] = k->P[i][j] - I[i][0]*k->P[0][j] - I[i][1]*k->P[1][j] - I[i][2]*k->P[2][j];
