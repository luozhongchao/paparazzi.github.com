00134 // Generic Event Macro
00135 #define _Ads1114Event(_p) {\
00136   if (!_p.config_done) { \
00137     if (_p.trans.status == I2CTransSuccess) { _p.config_done = TRUE; _p.trans.status = I2CTransDone; } \
00138     if (_p.trans.status == I2CTransFailed) { _p.trans.status = I2CTransDone; } \
00139   } else { \
00140     if (_p.trans.status == I2CTransSuccess) { _p.data_available = TRUE; _p.trans.status = I2CTransDone; } \
00141     if (_p.trans.status == I2CTransFailed) { _p.trans.status = I2CTransDone; } \
00142   } \
00143 }
