00055 int8_t humid_sht_crc(volatile uint8_t* data) {
00056   uint8_t i, bit, crc = 0;
00057 
00058   for (i = 0; i < 2; i++) {
00059     crc ^= (data[i]);
00060     for (bit = 8; bit > 0; bit--) {
00061       if (crc & 0x80)
00062         crc = (crc << 1) ^ 0x131;
00063       else
00064         crc = (crc << 1);
00065     }
00066   }
00067   if (crc != data[2])
00068     return -1;
00069   else
00070     return 0;
00071 }
