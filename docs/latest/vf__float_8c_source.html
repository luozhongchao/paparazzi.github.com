00076 /*
00077 
00078  F = [ 1 dt -dt^2/2
00079        0  1 -dt
00080        0  0   1     ];
00081 
00082  B = [ dt^2/2 dt 0]';

00085        0     0.01  0

00089 
00090  Pk1 = F * Pk0 * F' + Q;
00091 
00092 */
00093 void vff_propagate(float accel) {
00094   /* update state (Xk1) */
00095   vff_zdotdot = accel + 9.81 - vff_bias;
00096   vff_z = vff_z + DT_VFILTER * vff_zdot;

00098   /* update covariance (Pk1) */
00099   const float FPF00 = vff_P[0][0] + DT_VFILTER * ( vff_P[1][0] + vff_P[0][1] + DT_VFILTER * vff_P[1][1] );
00100   const float FPF01 = vff_P[0][1] + DT_VFILTER * ( vff_P[1][1] - vff_P[0][2] - DT_VFILTER * vff_P[1][2] );
00101   const float FPF02 = vff_P[0][2] + DT_VFILTER * ( vff_P[1][2] );
00102   const float FPF10 = vff_P[1][0] + DT_VFILTER * (-vff_P[2][0] + vff_P[1][1] - DT_VFILTER * vff_P[2][1] );
00103   const float FPF11 = vff_P[1][1] + DT_VFILTER * (-vff_P[2][1] - vff_P[1][2] + DT_VFILTER * vff_P[2][2] );
00104   const float FPF12 = vff_P[1][2] + DT_VFILTER * (-vff_P[2][2] );
00105   const float FPF20 = vff_P[2][0] + DT_VFILTER * ( vff_P[2][1] );
00106   const float FPF21 = vff_P[2][1] + DT_VFILTER * (-vff_P[2][2] );
00107   const float FPF22 = vff_P[2][2];
00108 
00109   vff_P[0][0] = FPF00 + Qzz;
