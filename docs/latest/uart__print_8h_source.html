36 #define _PrintHex(out_fun, c) { \
37  const uint8_t hex[16] = { '0', '1', '2', '3', '4', '5', '6', '7', \
38  '8', '9', 'A', 'B', 'C', 'D', 'E', 'F' }; \
39  uint8_t high = (c & 0xF0)>>4; \
40  uint8_t low = c & 0x0F; \
41  out_fun(hex[high]); \
42  out_fun(hex[low]); \
43 } \
