57 // Configuration function called once before normal use
58 static void adxl345_send_config(void)
59 {
60  switch (adxl345_init_status) {
61  case ADXL_CONF_RATE:
62  adxl345_i2c_trans.buf[0] = ADXL345_REG_BW_RATE;
63  adxl345_i2c_trans.buf[1] = ADXL345_BW_RATE;
64  I2CTransmit(ADXL345_I2C_DEVICE, adxl345_i2c_trans, ADXL345_I2C_ADDR, 2);
65  adxl345_init_status++;
66  break;
67  case ADXL_CONF_POWER:
68  adxl345_i2c_trans.buf[0] = ADXL345_REG_POWER_CTL;
69  adxl345_i2c_trans.buf[1] = ADXL345_POWER_CTL;
70  I2CTransmit(ADXL345_I2C_DEVICE, adxl345_i2c_trans, ADXL345_I2C_ADDR, 2);
71  adxl345_init_status++;
72  break;
73  case ADXL_CONF_INT:
74  adxl345_i2c_trans.buf[0] = ADXL345_REG_INT_ENABLE;
75  adxl345_i2c_trans.buf[1] = ADXL345_INT_ENABLE;
76  I2CTransmit(ADXL345_I2C_DEVICE, adxl345_i2c_trans, ADXL345_I2C_ADDR, 2);
77  adxl345_init_status++;
78  break;
79  case ADXL_CONF_FORMAT:
80  adxl345_i2c_trans.buf[0] = ADXL345_REG_DATA_FORMAT;
81  adxl345_i2c_trans.buf[1] = ADXL345_DATA_FORMAT;
82  I2CTransmit(ADXL345_I2C_DEVICE, adxl345_i2c_trans, ADXL345_I2C_ADDR, 2);
83  adxl345_init_status++;
84  break;
85  case ADXL_CONF_DONE:
86  adxl345_initialized = TRUE;
87  adxl345_i2c_trans.status = I2CTransDone;
88  break;
89  default:
90  break;
91  }
92 }

94 // Configure
95 void adxl345_configure(void)
96 {
97  if (adxl345_init_status == ADXL_CONF_UNINIT) {
98  adxl345_init_status++;
99  if (adxl345_i2c_trans.status == I2CTransSuccess || adxl345_i2c_trans.status == I2CTransDone) {
100  adxl345_send_config();
101  }
102  }

105 // Normal reading
106 void adxl345_read(void)
107 {
108  if (adxl345_initialized && adxl345_i2c_trans.status == I2CTransDone) {
109  adxl345_i2c_trans.buf[0] = ADXL345_REG_DATA_X0;

114 #define Int16FromBuf(_buf,_idx) ((int16_t)((_buf[_idx+1]<<8) | _buf[_idx]))

116 void adxl345_event(void)
117 {
118  if (adxl345_initialized) {
119  if (adxl345_i2c_trans.status == I2CTransFailed) {
120  adxl345_i2c_trans.status = I2CTransDone;
121  }
122  else if (adxl345_i2c_trans.status == I2CTransSuccess) {
123  // Successfull reading
124  adxl345_data.x = Int16FromBuf(adxl345_i2c_trans.buf,0);
125  adxl345_data.y = Int16FromBuf(adxl345_i2c_trans.buf,2);
126  adxl345_data.z = Int16FromBuf(adxl345_i2c_trans.buf,4);
127  adxl345_data_available = TRUE;
128  adxl345_i2c_trans.status = I2CTransDone;
129  }
130  }
131  else if (!adxl345_initialized && adxl345_init_status != ADXL_CONF_UNINIT) { // Configuring
132  if (adxl345_i2c_trans.status == I2CTransSuccess || adxl345_i2c_trans.status == I2CTransDone) {
133  adxl345_i2c_trans.status = I2CTransDone;
134  adxl345_send_config();
135  }
136  if (adxl345_i2c_trans.status == I2CTransFailed) {
137  adxl345_init_status--;
138  adxl345_i2c_trans.status = I2CTransDone;
139  adxl345_send_config(); // Retry config (TODO max retry)
140  }
141  }
142 }
