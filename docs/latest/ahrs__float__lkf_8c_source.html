00161 #define FLOAT_MAT33_INVERT(_mo, _mi) {                      \
00162   float _det = 0.;                              \
00163   _det =   _mi[0][0] * (_mi[2][2] * _mi[1][1] - _mi[2][1] * _mi[1][2])    \
00164          - _mi[1][0] * (_mi[2][2] * _mi[0][1] - _mi[2][1] * _mi[0][2])    \
00165          + _mi[2][0] * (_mi[1][2] * _mi[0][1] - _mi[1][1] * _mi[0][2]);   \
00166   if (_det != 0.) { /* ? test for zero? */              \
00167     _mo[0][0] = (_mi[1][1] * _mi[2][2] - _mi[1][2] * _mi[2][1]) / _det;   \
00168     _mo[0][1] = (_mi[0][2] * _mi[2][1] - _mi[0][1] * _mi[2][2]) / _det;   \
00169     _mo[0][2] = (_mi[0][1] * _mi[1][2] - _mi[0][2] * _mi[1][1]) / _det;   \

00268 /*
00269  * Propagate our dynamic system in time
00270  *
00271  * Run the strapdown computation and the predict step of the filter
00272  *
00273  *
00274  *      quat_dot = Wxq(pqr) * quat
00275  *      bias_dot = 0
00276  *
00277  * Wxq is the quaternion omega matrix:
00278  *
00279  *              [ 0, -p, -q, -r ]
00280  *      1/2 *   [ p,  0,  r, -q ]
00281  *              [ q, -r,  0,  p ]
00282  *              [ r,  q, -p,  0 ]
00283  *
00284  */
00285 void ahrs_propagate(void) {
00286   int i, j, k;
00287 
00288     ahrs_lowpass_accel();
00289 
00290   /* compute unbiased rates */
00291   RATES_FLOAT_OF_BFP(bafl_rates, imu.gyro);
00292   RATES_SUB(bafl_rates, bafl_bias);
00293 
00294 
00295   /* run strapdown computation
00296    *
00297    */
00298 
00299   /* multiplicative quaternion update */
00300   /* compute correction quaternion */
00301   QUAT_ASSIGN(bafl_qr, 1., bafl_rates.p * BAFL_DT / 2, bafl_rates.q * BAFL_DT / 2, bafl_rates.r * BAFL_DT / 2);
00302   /* normalize it. maybe not necessary?? */
00303   float q_sq;
00304   q_sq = (bafl_qr.qx * bafl_qr.qx +bafl_qr.qy * bafl_qr.qy + bafl_qr.qz * bafl_qr.qz) / 4;
00305   if (q_sq > 1) { /* this should actually never happen */
00306     FLOAT_QUAT_SMUL(bafl_qr, bafl_qr, 1 / sqrtf(1 + q_sq));
00307   } else {
00308     bafl_qr.qi = sqrtf(1 - q_sq);
00309   }
00310   /* propagate correction quaternion */
00311   FLOAT_QUAT_COMP(bafl_qtemp, bafl_quat, bafl_qr);
00312   FLOAT_QUAT_COPY(bafl_quat, bafl_qtemp);
00313 
00314   bafl_qnorm = FLOAT_QUAT_NORM(bafl_quat);
00315   //TODO check if broot force normalization is good, use lagrange normalization?
00316   FLOAT_QUAT_NORMALIZE(bafl_quat);
00317 
00318 
00319   /*
00320    *  compute all representations
00321    */
00322   /* maintain rotation matrix representation */
00323   FLOAT_RMAT_OF_QUAT(bafl_dcm, bafl_quat);
00324   /* maintain euler representation */
00325   FLOAT_EULERS_OF_RMAT(bafl_eulers, bafl_dcm);
00326   AHRS_TO_BFP();
00327   AHRS_LTP_TO_BODY();
00328 
00329 
00330   /* error KF-Filter
00331    * propagate only the error covariance matrix since error is corrected after each measurement
00332    *
00333    * F = [ 0  0  0          ]
00334    *     [ 0  0  0   -Cbn   ]
00335    *     [ 0  0  0          ]
00336    *     [ 0  0  0  0  0  0 ]
00337    *     [ 0  0  0  0  0  0 ]
00338    *     [ 0  0  0  0  0  0 ]
00339    *
00340    * T = e^(dt * F)
00341    *
00342    * T = [ 1   0   0             ]
00343    *     [ 0   1   0   -Cbn*dt   ]
00344    *     [ 0   0   1             ]
00345    *     [ 0   0   0   1   0   0 ]
00346    *     [ 0   0   0   0   1   0 ]
00347    *     [ 0   0   0   0   0   1 ]
00348    *
00349    * P_prio = T * P * T_T + Q
00350    *
00351    */
00352 
00353   /*
00354    *  compute state transition matrix T
00355    *
00356    *  diagonal elements of T are always one
00357    */
00358   for (i = 0; i < 3; i++) {
00359     for (j = 0; j < 3; j++) {
00360       bafl_T[i][j + 3] = -RMAT_ELMT(bafl_dcm, j, i); /* inverted bafl_dcm */
00361     }
00362   }
00363 
00364 
00365   /*
00366    * estimate the a priori error covariance matrix P_k = T * P_k-1 * T_T + Q
00367    */
00368   /* Temporary multiplication temp(6x6) = T(6x6) * P(6x6)      */
00369   for (i = 0; i < BAFL_SSIZE; i++) {
00370     for (j = 0; j < BAFL_SSIZE; j++) {
00371       bafl_tempP[i][j] = 0.;
00372       for (k = 0; k < BAFL_SSIZE; k++) {
00373         bafl_tempP[i][j] += bafl_T[i][k] * bafl_P[k][j];
00374       }
00375     }
00376   }
00377 
00378   /* P_k(6x6) = temp(6x6) * T_T(6x6) + Q  */
00379   for (i = 0; i < BAFL_SSIZE; i++) {
00380     for (j = 0; j < BAFL_SSIZE; j++) {
00381       bafl_P[i][j] = 0.;
00382       for (k = 0; k < BAFL_SSIZE; k++) {
00383         bafl_P[i][j] += bafl_tempP[i][k] * bafl_T[j][k]; //T[j][k] = T_T[k][j]
00384       }
00385     }
00386     if (i<3) {
00387       bafl_P[i][i] += bafl_Q_att;
00388     } else {
00389       bafl_P[i][i] += bafl_Q_gyro;
00390     }
00391   }
00392 
00393 #ifdef LKF_PRINT_P
00394   printf("Pp =");
00395   for (i = 0; i < 6; i++) {
00396     printf("[");
00397     for (j = 0; j < 6; j++) {
00398       printf("%f\t", bafl_P[i][j]);
00399     }
00400     printf("]\n    ");
00401   }
00402   printf("\n");
00403 #endif
00404 }

00410 static void ahrs_do_update_accel(void) {
00411   int i, j, k;
00412 
00413 #ifdef BAFL_DEBUG2
00414   printf("Accel update.\n");
00415 #endif
00416 
00417   /* P_prio = P */
00418   for (i = 0; i < BAFL_SSIZE; i++) {
00419     for (j = 0; j < BAFL_SSIZE; j++) {
00420       bafl_Pprio[i][j] = bafl_P[i][j];
00421     }
00422   }
00423 
00424   /*
00425    * set up measurement matrix
00426    *
00427    * g = 9.81
00428    * gx = [ 0 -g  0
00429    *        g  0  0
00430    *        0  0  0 ]
00431    * H = [          0 0 0 ]
00432    *     [ -Cnb*gx  0 0 0 ]
00433    *     [          0 0 0 ]
00434    *
00435    * */
00436   bafl_H[0][0] = -RMAT_ELMT(bafl_dcm, 0, 1) * BAFL_g;
00437   bafl_H[0][1] =  RMAT_ELMT(bafl_dcm, 0, 0) * BAFL_g;
00438   bafl_H[1][0] = -RMAT_ELMT(bafl_dcm, 1, 1) * BAFL_g;
00439   bafl_H[1][1] =  RMAT_ELMT(bafl_dcm, 1, 0) * BAFL_g;
00440   bafl_H[2][0] = -RMAT_ELMT(bafl_dcm, 2, 1) * BAFL_g;
00441   bafl_H[2][1] =  RMAT_ELMT(bafl_dcm, 2, 0) * BAFL_g;
00442   /* rest is zero
00443   bafl_H[0][2] = 0.;
00444   bafl_H[1][2] = 0.;
00445   bafl_H[2][2] = 0.;*/
00446 
00447   /**********************************************
00448    * compute Kalman gain K
00449    *
00450    * K = P_prio * H_T * inv(S)
00451    * S = H * P_prio * HT + R
00452    *
00453    * K = P_prio * HT * inv(H * P_prio * HT + R)
00454    *
00455    **********************************************/
00456 
00457   /* covariance residual S = H * P_prio * HT + R    */
00458 
00459   /* temp_S(3x6) = H(3x6) * P_prio(6x6)
00460    * last 4 columns of H are zero
00461    */
00462   for (i = 0; i < 3; i++) {
00463     for (j = 0; j < BAFL_SSIZE; j++) {
00464       bafl_tempS[i][j]  = bafl_H[i][0] * bafl_Pprio[0][j];
00465       bafl_tempS[i][j] += bafl_H[i][1] * bafl_Pprio[1][j];
00466     }
00467   }
00468 
00469   /* S(3x3) = temp_S(3x6) * HT(6x3) + R(3x3)
00470    *
00471    * bottom 4 rows of HT are zero
00472    */
00473   for (i = 0; i < 3; i++) {
00474     for (j = 0; j < 3; j++) {
00475       bafl_S[i][j]  = bafl_tempS[i][0] * bafl_H[j][0]; /* H[j][0] = HT[0][j] */
00476       bafl_S[i][j] += bafl_tempS[i][1] * bafl_H[j][1]; /* H[j][0] = HT[0][j] */
00477     }
00478     bafl_S[i][i] += bafl_R_accel;
00479   }
00480 
00481   /* invert S
00482    */
00483   FLOAT_MAT33_INVERT(bafl_invS, bafl_S);
00484 
00485 
00486   /* temp_K(6x3) = P_prio(6x6) * HT(6x3)
00487    *
00488    * bottom 4 rows of HT are zero
00489    */
00490   for (i = 0; i < BAFL_SSIZE; i++) {
00491     for (j = 0; j < 3; j++) {
00492       /* not computing zero elements */
00493       bafl_tempK[i][j]  = bafl_Pprio[i][0] * bafl_H[j][0]; /* H[j][0] = HT[0][j] */
00494       bafl_tempK[i][j] += bafl_Pprio[i][1] * bafl_H[j][1]; /* H[j][1] = HT[1][j] */
00495     }
00496   }
00497 
00498   /* K(6x3) = temp_K(6x3) * invS(3x3)
00499    */
00500   for (i = 0; i < BAFL_SSIZE; i++) {
00501     for (j = 0; j < 3; j++) {
00502       bafl_K[i][j]  = bafl_tempK[i][0] * bafl_invS[0][j];
00503       bafl_K[i][j] += bafl_tempK[i][1] * bafl_invS[1][j];
00504       bafl_K[i][j] += bafl_tempK[i][2] * bafl_invS[2][j];
00505     }
00506   }
00507 
00508   /**********************************************
00509    * Update filter state.
00510    *
00511    *  The a priori filter state is zero since the errors are corrected after each update.
00512    *
00513    *  X = X_prio + K (y - H * X_prio)
00514    *  X = K * y
00515    **********************************************/
00516
