00161 static inline void parse_xbee( struct xbee_transport * t, uint8_t c ) {
00162   switch (t->status) {
00163   case XBEE_UNINIT:
00164     if (c == XBEE_START)
00165       t->status++;
00166     break;
00167   case XBEE_GOT_START:
00168     if (t->trans.msg_received) {
00169       t->trans.ovrn++;
00170       goto error;
00171     }
00172     t->trans.payload_len = c<<8;
00173     t->status++;
00174     break;
00175   case XBEE_GOT_LENGTH_MSB:
00176     t->trans.payload_len |= c;
00177     t->status++;
00178     t->payload_idx = 0;
00179     t->cs=0;
00180     break;
00181   case XBEE_GOT_LENGTH_LSB:
00182     t->trans.payload[t->payload_idx] = c;
00183     t->cs += c;
00184     t->payload_idx++;
00185     if (t->payload_idx == t->trans.payload_len)
00186       t->status++;
00187     break;
00188   case XBEE_GOT_PAYLOAD:
00189     if (c + t->cs != 0xff)
00190       goto error;
00191     t->trans.msg_received = TRUE;
00192     goto restart;
00193     break;
00194   default:
00195     goto error;
00196   }
00197   return;
00198  error:
00199   t->trans.error++;
00200  restart:
00201   t->status = XBEE_UNINIT;
00202   return;
00203 }
