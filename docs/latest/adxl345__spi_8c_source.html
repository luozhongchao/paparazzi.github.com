76 // Configuration function called once before normal use
77 static void adxl345_spi_send_config(struct Adxl345_Spi *adxl)
78 {
79  switch (adxl->init_status) {
80  case ADXL_CONF_RATE:
81  adxl345_spi_write_to_reg(adxl, ADXL345_REG_BW_RATE, adxl->config.rate);
82  adxl->init_status++;
83  break;
84  case ADXL_CONF_POWER:
85  /* enable measurement, is in standby after power up */
86  adxl345_spi_write_to_reg(adxl, ADXL345_REG_POWER_CTL, (0x1<<3));
87  adxl->init_status++;
88  break;
89  case ADXL_CONF_INT:
90  adxl345_spi_write_to_reg(adxl, ADXL345_REG_INT_ENABLE, (adxl->config.drdy_int_enable << 7));
91  adxl->init_status++;
92  break;
93  case ADXL_CONF_FORMAT:
94  adxl345_spi_write_to_reg(adxl, ADXL345_REG_DATA_FORMAT, adxl345_data_format(&adxl->config));
95  adxl->init_status++;
96  break;
97  case ADXL_CONF_DONE:
98  adxl->initialized = TRUE;
99  adxl->spi_trans.status = SPITransDone;
100  break;
101  default:
102  break;
103  }
104 }

129 void adxl345_spi_event(struct Adxl345_Spi *adxl)
130 {
131  if (adxl->initialized) {
132  if (adxl->spi_trans.status == SPITransFailed) {
133  adxl->spi_trans.status = SPITransDone;
134  }
135  else if (adxl->spi_trans.status == SPITransSuccess) {
136  // Successfull reading
137  adxl->data.vect.x = Int16FromBuf(adxl->rx_buf,1);
138  adxl->data.vect.y = Int16FromBuf(adxl->rx_buf,3);
139  adxl->data.vect.z = Int16FromBuf(adxl->rx_buf,5);
140  adxl->data_available = TRUE;
141  adxl->spi_trans.status = SPITransDone;
142  }
143  }
144  else if (adxl->init_status != ADXL_CONF_UNINIT) { // Configuring but not yet initialized
145  switch (adxl->spi_trans.status) {
146  case SPITransFailed:
147  adxl->init_status--; // Retry config (TODO max retry)
148  case SPITransSuccess:
149  case SPITransDone:
150  adxl->spi_trans.status = SPITransDone;
151  adxl345_spi_send_config(adxl);
152  break;
153  default:
154  break;
155  }
156  }
