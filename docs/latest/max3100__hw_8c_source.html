00074 void max3100_init( void ) {
00075   max3100_status = MAX3100_STATUS_IDLE;
00076   max3100_data_available = false;
00077   max3100_transmit_buffer_empty = true;
00078   max3100_tx_insert_idx = 0;
00079   max3100_tx_extract_idx = 0;
00080   max3100_rx_insert_idx = 0;
00081   max3100_rx_extract_idx = 0;
00082 
00083   /* setup pins for SSP (SCK, MISO, MOSI) */
00084   PINSEL1 |= PINSEL1_SCK | PINSEL1_MISO | PINSEL1_MOSI;
00085 
00086   /* setup SSP */
00087   SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
00088   SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
00089   SSPCPSR = SSPCPSR_VAL; /* Prescaler */
00090 
00091 
00092   /* From arm7/max1167_hw.c */
00093 
00094   /* SS pin is output */
00095   SetBit(MAX3100_SS_IODIR, MAX3100_SS_PIN);
00096   /* unselected max3100 */
00097   Max3100Unselect();
00098 
00099   /* connect extint (IRQ) */
00100   MAX3100_IRQ_PINSEL |= MAX3100_IRQ_PINSEL_VAL << MAX3100_IRQ_PINSEL_BIT;
00101   /* extint is edge trigered */
00102   SetBit(EXTMODE, MAX3100_IRQ_EINT);
00103   /* extint is trigered on falling edge */
00104   ClearBit(EXTPOLAR, MAX3100_IRQ_EINT);
00105   /* clear pending extint0 before enabling interrupts */
00106   SetBit(EXTINT, MAX3100_IRQ_EINT);
00107 
00108   /* Configure interrupt vector for external pin interrupt */
00109   VICIntSelect &= ~VIC_BIT( MAX3100_VIC_EINT );    // EXTINT selected as IRQ
00110   VICIntEnable = VIC_BIT( MAX3100_VIC_EINT );             // EXTINT interrupt enabled
00111   VICVectCntl8 = VIC_ENABLE | MAX3100_VIC_EINT;
00112   VICVectAddr8 = (uint32_t)EXTINT_ISR;   // address of the ISR
00113 
00114   /* Configure interrupt vector for SPI */
00115   VICIntSelect &= ~VIC_BIT(VIC_SPI1);   /* SPI1 selected as IRQ */
00116   VICIntEnable = VIC_BIT(VIC_SPI1);     /* SPI1 interrupt enabled */
00117   VICVectCntl7 = VIC_ENABLE | VIC_SPI1;
00118   VICVectAddr7 = (uint32_t)SPI1_ISR;    /* address of the ISR */
00119 
00120   /* Write configuration */
00121   //Max3100TransmitConf(MAX3100_BAUD_RATE | MAX3100_BIT_NOT_TM);
00122   Max3100TransmitConf(MAX3100_BAUD_RATE | MAX3100_BIT_NOT_RM | MAX3100_BIT_NOT_TM);
00123 }

00138 void SPI1_ISR(void) {
00139   ISR_ENTRY();
00140 
00141   while (bit_is_set(SSPSR, RNE)) {
00142     uint16_t data = SSPDR;
00143 
00144     if (bit_is_set(data, MAX3100_R_BIT)) { /* Data available */
00145       max3100_rx_buf[max3100_rx_insert_idx] = data & 0xff;
00146       max3100_rx_insert_idx++;  // automatic overflow because len=256
00147       read_bytes = true;
00148     }
00149     if (bit_is_set(data, MAX3100_T_BIT) && (max3100_status == MAX3100_STATUS_READING)) { /* transmit buffer empty */
00150       max3100_transmit_buffer_empty = true;
00151     }
00152   }
00153   SpiClearRti();                  /* clear interrupt */
00154   SpiDisableRti();
00155   SpiDisable ();
00156   Max3100Unselect();
00157   max3100_status = MAX3100_STATUS_IDLE;
00158 
00159   VICVectAddr = 0x00000000; /* clear this interrupt from the VIC */
00160   ISR_EXIT();
00161 }
