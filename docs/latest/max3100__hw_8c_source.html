72 void max3100_init( void ) {
73  max3100_status = MAX3100_STATUS_IDLE;
74  max3100_data_available = false;
75  max3100_transmit_buffer_empty = true;
76  max3100_tx_insert_idx = 0;
77  max3100_tx_extract_idx = 0;
78  max3100_rx_insert_idx = 0;
79  max3100_rx_extract_idx = 0;
80 
81  /* setup pins for SSP (SCK, MISO, MOSI) */
82  PINSEL1 |= PINSEL1_SCK | PINSEL1_MISO | PINSEL1_MOSI;
83 
84  /* setup SSP */
85  SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
86  SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
87  SSPCPSR = SSPCPSR_VAL; /* Prescaler */
88 
89 
90  /* From arm7/max1167_hw.c */
91 
92  /* SS pin is output */
93  SetBit(MAX3100_SS_IODIR, MAX3100_SS_PIN);
94  /* unselected max3100 */
95  Max3100Unselect();
96 
97  /* connect extint (IRQ) */
98  MAX3100_IRQ_PINSEL |= MAX3100_IRQ_PINSEL_VAL << MAX3100_IRQ_PINSEL_BIT;
99  /* extint is edge trigered */
100  SetBit(EXTMODE, MAX3100_IRQ_EINT);
101  /* extint is trigered on falling edge */
102  ClearBit(EXTPOLAR, MAX3100_IRQ_EINT);
103  /* clear pending extint0 before enabling interrupts */
104  SetBit(EXTINT, MAX3100_IRQ_EINT);
105 
106  /* Configure interrupt vector for external pin interrupt */
107  VICIntSelect &= ~VIC_BIT( MAX3100_VIC_EINT ); // EXTINT selected as IRQ
108  VICIntEnable = VIC_BIT( MAX3100_VIC_EINT ); // EXTINT interrupt enabled
109  VICVectCntl8 = VIC_ENABLE | MAX3100_VIC_EINT;
110  VICVectAddr8 = (uint32_t)EXTINT_ISR; // address of the ISR
111 
112  /* Configure interrupt vector for SPI */
113  VICIntSelect &= ~VIC_BIT(VIC_SPI1); /* SPI1 selected as IRQ */
114  VICIntEnable = VIC_BIT(VIC_SPI1); /* SPI1 interrupt enabled */
115  VICVectCntl7 = VIC_ENABLE | VIC_SPI1;
116  VICVectAddr7 = (uint32_t)SPI1_ISR; /* address of the ISR */
117 
118  /* Write configuration */
119  //Max3100TransmitConf(MAX3100_BAUD_RATE | MAX3100_BIT_NOT_TM);
120  Max3100TransmitConf(MAX3100_BAUD_RATE | MAX3100_BIT_NOT_RM | MAX3100_BIT_NOT_TM);
121 }

136 void SPI1_ISR(void) {
137  ISR_ENTRY();
138 
139  while (bit_is_set(SSPSR, RNE)) {
140  uint16_t data = SSPDR;
141 
142  if (bit_is_set(data, MAX3100_R_BIT)) { /* Data available */
143  max3100_rx_buf[max3100_rx_insert_idx] = data & 0xff;
144  max3100_rx_insert_idx++; // automatic overflow because len=256
145  read_bytes = true;
146  }
147  if (bit_is_set(data, MAX3100_T_BIT) && (max3100_status == MAX3100_STATUS_READING)) { /* transmit buffer empty */
148  max3100_transmit_buffer_empty = true;
149  }
150  }
151  SpiClearRti(); /* clear interrupt */
152  SpiDisableRti();
153  SpiDisable ();
154  Max3100Unselect();
155  max3100_status = MAX3100_STATUS_IDLE;
156 
157  VICVectAddr = 0x00000000; /* clear this interrupt from the VIC */
158  ISR_EXIT();
159 }
