00239   static uint8_t uart2_sent = 0;
00240   static uint8_t uart3_sent = 0;
00241   static uint8_t uart1_has_data = 0;
00242   static uint8_t uart2_has_data = 0;
00243   static uint8_t uart3_has_data = 0;
00244   static uint8_t trigger_send = 0;

00246   static uint8_t uart1_enabled = 1;
00247   static uint8_t uart2_enabled = 1;
00248   static uint8_t uart3_enabled = 1;
00249 
00250   // We cache data availability, so it doesn't change between checks:
00251   uart1_has_data = Uart1ChAvailable();
00252   uart2_has_data = Uart2ChAvailable();
00253   uart3_has_data = Uart3ChAvailable();
00254 
00255   // Fill stage: Read data from UARTs into buffers, or increment
00256   // their timeouts if no data is available:
00257   if(!uart1_sent && uart1_has_data) {
00258     spistream_uart1_msg.uart_data[uart1_num_rx_bytes] = Uart1Getch();
00259     timeout_uart1 = 0;
00260     if(uart1_num_rx_bytes < SPISTREAM_MAX_MESSAGE_LENGTH)
00261       { uart1_num_rx_bytes++; }
00262   } else { if(timeout_uart1 < timeout_trig) { timeout_uart1++; } }
00263 
00264   if(!uart2_sent && uart2_has_data) {
00265     spistream_uart2_msg.uart_data[uart2_num_rx_bytes] = Uart2Getch();
00266     timeout_uart2 = 0;
00267     if(uart2_num_rx_bytes < SPISTREAM_MAX_MESSAGE_LENGTH)
00268       { uart2_num_rx_bytes++; }
00269   } else { if(timeout_uart2 < timeout_trig) { timeout_uart2++; } }
00270 
00271   if(!uart3_sent && uart3_has_data) {
00272     spistream_uart3_msg.uart_data[uart3_num_rx_bytes] = Uart3Getch();
00273     timeout_uart3 = 0;
00274     if(uart3_num_rx_bytes < SPISTREAM_MAX_MESSAGE_LENGTH)
00275       { uart3_num_rx_bytes++; }
00276   } else { if(timeout_uart3 < timeout_trig) { timeout_uart3++; } }
00277 
00278   trigger_send = ((!uart1_enabled ||
00279                   (timeout_uart1 >= timeout_trig)) &&
00280                  (!uart2_enabled ||
00281                   (timeout_uart2 >= timeout_trig)) &&
00282                  (!uart3_enabled ||
00283                   (timeout_uart3 >= timeout_trig)));
00284 
00285   // Send stage: If all UART timeouts reach the timeout
00286   // trigger value and have accumulated data to send
00287   if(trigger_send) {
00288 
00289     // If there was no new data on any UART for some time
00290     // and there is data in every rx buffer:
00291 
00292     if(spistream_wait_for_num_transfers > 0)
00293     {
00294       // Warning LED: Could not finish all transactions
00295       // from last call.
00296       LED_ON(6);
00297     }
00298     else
00299     {
00300       LED_OFF(6);
00301 
00302       uart1_sent = !uart1_enabled; // If we set uartX_sent to 1 here, it
00303       uart2_sent = !uart2_enabled; // is just ignored for every read poll
00304       uart3_sent = !uart3_enabled; // as it seems to have been read already.
00305 
00306       if(!uart1_sent && uart1_num_rx_bytes > 0) {
00307         if(spistream_send_msg((uint8_t *)&spistream_uart1_msg,
00308                               uart1_num_rx_bytes+1, // +1 for UART id
00309                               SPISTREAM_WAIT_FOR_READ)) {
00310            uart1_sent = 1;
00311            spistream_wait_for_num_transfers++;
00312          }
00313       }
00314 
00315       if(!uart2_sent && uart1_num_rx_bytes > 0) {
00316         if(spistream_send_msg((uint8_t *)&spistream_uart2_msg,
00317                               uart1_num_rx_bytes+1, // +1 for UART id
00318                               SPISTREAM_WAIT_FOR_READ)) {
00319            uart2_sent = 1;
00320            spistream_wait_for_num_transfers++;
00321         }
00322       }
00323 
00324       if(!uart3_sent && uart3_num_rx_bytes > 0) {
00325         if(spistream_send_msg((uint8_t *)&spistream_uart3_msg,
00326                               uart3_num_rx_bytes+1, // +1 for UART id
00327                               SPISTREAM_WAIT_FOR_READ)) {
00328           uart3_sent = 1;
00329           spistream_wait_for_num_transfers++;
00330         }
00331       }
00332 
00333       // Transaction completed, reset state.
00334       // Note: Only reset when all uart buffers have been transmitted,
00335       // otherwise the timeout would start from the beginning and the
00336       // loop phase shifts (aka "you're in the deep").
00337       uart1_num_rx_bytes = 0;
00338       uart2_num_rx_bytes = 0;
00339       uart3_num_rx_bytes = 0;
00340       timeout_uart1 = 0;
00341       timeout_uart2 = 0;
00342       timeout_uart3 = 0;
00343       uart1_sent = 0;
00344       uart2_sent = 0;
00345       uart3_sent = 0;
