169 static void ac_char_update(float throttle, float pitch, float climb, float accelerate)
170 {
171  if ((accelerate > -0.02) && (accelerate < 0.02))
172  {
173  if (throttle >= 1.0f)
174  {
175  ac_char_climb_count++;
176  ac_char_average(&ac_char_climb_pitch, pitch * 57.6f, ac_char_climb_count );
177  ac_char_average(&ac_char_climb_max , stateGetSpeedEnu_f()->z, ac_char_climb_count );
178  }
179  else if (throttle <= 0.0f)
180  {
181  ac_char_descend_count++;
182  ac_char_average(&ac_char_descend_pitch, pitch * 57.6f , ac_char_descend_count );
183  ac_char_average(&ac_char_descend_max , stateGetSpeedEnu_f()->z , ac_char_descend_count );
184  }
185  else if ((climb > -0.125) && (climb < 0.125))
186  {
187  ac_char_cruise_count++;
188  ac_char_average(&ac_char_cruise_throttle , throttle , ac_char_cruise_count );
189  ac_char_average(&ac_char_cruise_pitch , pitch * 57.6f , ac_char_cruise_count );
190  }
191  }

265 void v_ctl_altitude_loop( void )
266 {
267  // Imput Checks
268  if (v_ctl_auto_airspeed_setpoint <= STALL_AIRSPEED * 1.23) v_ctl_auto_airspeed_setpoint = STALL_AIRSPEED * 1.23;
269 
270  // Altitude Controller
271  v_ctl_altitude_error = v_ctl_altitude_setpoint - stateGetPositionUtm_f()->alt;
272  float sp = v_ctl_altitude_pgain * v_ctl_altitude_error + v_ctl_altitude_pre_climb ;
273  BoundAbs(sp, v_ctl_max_climb);
274 
275  float incr = sp - v_ctl_climb_setpoint;
276  BoundAbs(incr, 2.0 / 4.0);
277  v_ctl_climb_setpoint += incr;
278 }

290 static float low_pass_vdot(float v);
291 static float low_pass_vdot(float v)
292 {
293  lp_vdot[4] += (v - lp_vdot[4]) / 3;
294  lp_vdot[3] += (lp_vdot[4] - lp_vdot[3]) / 3;
295  lp_vdot[2] += (lp_vdot[3] - lp_vdot[2]) / 3;
296  lp_vdot[1] += (lp_vdot[2] - lp_vdot[1]) / 3;
297  lp_vdot[0] += (lp_vdot[1] - lp_vdot[0]) / 3;
298 
299  return lp_vdot[0];
300 }

302 void v_ctl_climb_loop( void )
303 {
304 #ifdef AIRSPEED_SETPOINT_SLEW
305  // airspeed_setpoint ratelimiter:
306  float airspeed_incr = v_ctl_auto_airspeed_setpoint - v_ctl_auto_airspeed_setpoint_slew; // FIXME
307  BoundAbs(airspeed_incr, AIRSPEED_SETPOINT_SLEW * NOMINAL_AIRSPEED);
308  v_ctl_auto_airspeed_setpoint_slew += airspeed_incr;
309 #endif
310 
311 #ifdef V_CTL_AUTO_GROUNDSPEED_SETPOINT
312  // Ground speed control loop (input: groundspeed error, output: airspeed controlled)
313  float err_groundspeed = (v_ctl_auto_groundspeed_setpoint - (*stateGetHorizontalSpeedNorm_f()));
314  v_ctl_auto_groundspeed_sum_err += err_groundspeed;
315  BoundAbs(v_ctl_auto_groundspeed_sum_err, V_CTL_AUTO_GROUNDSPEED_MAX_SUM_ERR);
316  v_ctl_auto_airspeed_controlled = (err_groundspeed + v_ctl_auto_groundspeed_sum_err * v_ctl_auto_groundspeed_igain) * v_ctl_auto_groundspeed_pgain;
317 
318  // Do not allow controlled airspeed below the setpoint
319  if (v_ctl_auto_airspeed_controlled < v_ctl_auto_airspeed_setpoint) {
320  v_ctl_auto_airspeed_controlled = v_ctl_auto_airspeed_setpoint;
321  // reset integrator of ground speed loop
322  v_ctl_auto_groundspeed_sum_err = v_ctl_auto_airspeed_controlled/(v_ctl_auto_groundspeed_pgain*v_ctl_auto_groundspeed_igain);
323  }
324 #else
325  v_ctl_auto_airspeed_controlled = v_ctl_auto_airspeed_setpoint_slew;
326 #endif
327 
328  // Airspeed outerloop: positive means we need to accelerate
329  float speed_error = v_ctl_auto_airspeed_controlled - (*stateGetAirspeed_f());
330 
331  // Speed Controller to PseudoControl: gain 1 -> 5m/s error = 0.5g acceleration
332  v_ctl_desired_acceleration = speed_error * v_ctl_airspeed_pgain / 9.81f;
333  BoundAbs(v_ctl_desired_acceleration, v_ctl_max_acceleration);
334 
335  // Actual Acceleration from IMU: attempt to reconstruct the actual kinematic acceleration
336 #ifndef SITL
337  struct Int32Vect3 accel_meas_body;
338  INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
339  float vdot = ACCEL_FLOAT_OF_BFP(accel_meas_body.x) / 9.81f - sinf(stateGetNedToBodyEulers_f()->theta);
340 #else
341  float vdot = 0;
342 #endif
343 
344  // Acceleration Error: positive means UAV needs to accelerate: needs extra energy
345  float vdot_err = low_pass_vdot( v_ctl_desired_acceleration - vdot );
346 
347  // Flight Path Outerloop: positive means needs to climb more: needs extra energy
348  float gamma_err = (v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z) / v_ctl_auto_airspeed_setpoint;
349 
350  // Total Energy Error: positive means energy should be added
351  float en_tot_err = gamma_err + vdot_err;
352 
353  // Energy Distribution Error: positive means energy should go from overspeed to altitude = pitch up
354  float en_dis_err = gamma_err - vdot_err;
355 
356  // Auto Cruise Throttle
357  if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
358  {
359  v_ctl_auto_throttle_nominal_cruise_throttle +=
360  v_ctl_auto_throttle_of_airspeed_igain * speed_error * dt
361  + en_tot_err * v_ctl_energy_total_igain * dt;
362  Bound(v_ctl_auto_throttle_nominal_cruise_throttle,0.0f,1.0f);
363  }
364 
365  // Total Controller
366  float controlled_throttle = v_ctl_auto_throttle_nominal_cruise_throttle
367  + v_ctl_auto_throttle_climb_throttle_increment * v_ctl_climb_setpoint
368  + v_ctl_auto_throttle_of_airspeed_pgain * speed_error
369  + v_ctl_energy_total_pgain * en_tot_err;
370 
371 
372  if ((controlled_throttle >= 1.0f) || (controlled_throttle <= 0.0f))
373  {
374  // If your energy supply is not sufficient, then neglect the climb requirement
375  en_dis_err = -vdot_err;
376  }
377 
378  /* pitch pre-command */
379  if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
380  {
381  v_ctl_auto_throttle_nominal_cruise_pitch += v_ctl_auto_pitch_of_airspeed_igain * (-speed_error) * dt
382  + v_ctl_energy_diff_igain * en_dis_err * dt;
383  Bound(v_ctl_auto_throttle_nominal_cruise_pitch,H_CTL_PITCH_MIN_SETPOINT, H_CTL_PITCH_MAX_SETPOINT);
384  }
385  float v_ctl_pitch_of_vz =
386  + (v_ctl_climb_setpoint /*+ d_err * v_ctl_auto_throttle_pitch_of_vz_dgain*/) * v_ctl_auto_throttle_pitch_of_vz_pgain
387  - v_ctl_auto_pitch_of_airspeed_pgain * speed_error
388  + v_ctl_auto_pitch_of_airspeed_dgain * vdot
389  + v_ctl_energy_diff_pgain * en_dis_err
390  + v_ctl_auto_throttle_nominal_cruise_pitch;
391 
392  nav_pitch = v_ctl_pitch_of_vz;
393  Bound(nav_pitch,H_CTL_PITCH_MIN_SETPOINT,H_CTL_PITCH_MAX_SETPOINT)
394 
395  ac_char_update(controlled_throttle, v_ctl_pitch_of_vz, v_ctl_climb_setpoint, v_ctl_desired_acceleration);
396 
397  v_ctl_throttle_setpoint = TRIM_UPPRZ(controlled_throttle * MAX_PPRZ);
398 }
