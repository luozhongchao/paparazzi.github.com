00265 void v_ctl_altitude_loop( void )
00266 {
00267   // Imput Checks
00268   if (v_ctl_auto_airspeed_setpoint <= STALL_AIRSPEED * 1.23) v_ctl_auto_airspeed_setpoint = STALL_AIRSPEED * 1.23;
00269 
00270   // Altitude Controller
00271   v_ctl_altitude_error = v_ctl_altitude_setpoint - stateGetPositionUtm_f()->alt;
00272   float sp = v_ctl_altitude_pgain * v_ctl_altitude_error + v_ctl_altitude_pre_climb ;
00273   BoundAbs(sp, v_ctl_max_climb);
00274 
00275   float incr = sp - v_ctl_climb_setpoint;
00276   BoundAbs(incr, 2.0 / 4.0);
00277   v_ctl_climb_setpoint += incr;

00302 void v_ctl_climb_loop( void )
00303 {
00304 #ifdef AIRSPEED_SETPOINT_SLEW
00305   // airspeed_setpoint ratelimiter:
00306   float airspeed_incr = v_ctl_auto_airspeed_setpoint - v_ctl_auto_airspeed_setpoint_slew; // FIXME
00307   BoundAbs(airspeed_incr, AIRSPEED_SETPOINT_SLEW * NOMINAL_AIRSPEED);
00308   v_ctl_auto_airspeed_setpoint_slew += airspeed_incr;
00309 #endif
00310 
00311 #ifdef V_CTL_AUTO_GROUNDSPEED_SETPOINT
00312  // Ground speed control loop (input: groundspeed error, output: airspeed controlled)
00313   float err_groundspeed = (v_ctl_auto_groundspeed_setpoint - (*stateGetHorizontalSpeedNorm_f()));
00314   v_ctl_auto_groundspeed_sum_err += err_groundspeed;
00315   BoundAbs(v_ctl_auto_groundspeed_sum_err, V_CTL_AUTO_GROUNDSPEED_MAX_SUM_ERR);
00316   v_ctl_auto_airspeed_controlled = (err_groundspeed + v_ctl_auto_groundspeed_sum_err * v_ctl_auto_groundspeed_igain) * v_ctl_auto_groundspeed_pgain;
00317 
00318   // Do not allow controlled airspeed below the setpoint
00319   if (v_ctl_auto_airspeed_controlled < v_ctl_auto_airspeed_setpoint) {
00320     v_ctl_auto_airspeed_controlled = v_ctl_auto_airspeed_setpoint;
00321     // reset integrator of ground speed loop
00322     v_ctl_auto_groundspeed_sum_err = v_ctl_auto_airspeed_controlled/(v_ctl_auto_groundspeed_pgain*v_ctl_auto_groundspeed_igain);
00323   }
00324 #else
00325     v_ctl_auto_airspeed_controlled = v_ctl_auto_airspeed_setpoint_slew;
00326 #endif
00327 
00328   // Airspeed outerloop: positive means we need to accelerate
00329   float speed_error = v_ctl_auto_airspeed_controlled - (*stateGetAirspeed_f());
00330 
00331   // Speed Controller to PseudoControl: gain 1 -> 5m/s error = 0.5g acceleration
00332   v_ctl_desired_acceleration = speed_error * v_ctl_airspeed_pgain / 9.81f;
00333   BoundAbs(v_ctl_desired_acceleration, v_ctl_max_acceleration);
00334 
00335   // Actual Acceleration from IMU: attempt to reconstruct the actual kinematic acceleration
00336 #ifndef SITL
00337   struct Int32Vect3 accel_meas_body;
00338   INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
00339   float vdot = ACCEL_FLOAT_OF_BFP(accel_meas_body.x) / 9.81f - sinf(stateGetNedToBodyEulers_f()->theta);
00340 #else
00341   float vdot = 0;
00342 #endif
00343 
00344   // Acceleration Error: positive means UAV needs to accelerate: needs extra energy
00345   float vdot_err = low_pass_vdot( v_ctl_desired_acceleration - vdot );
00346 
00347   // Flight Path Outerloop: positive means needs to climb more: needs extra energy
00348   float gamma_err  = (v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z) / v_ctl_auto_airspeed_setpoint;
00349 
00350   // Total Energy Error: positive means energy should be added
00351   float en_tot_err = gamma_err + vdot_err;
00352 
00353   // Energy Distribution Error: positive means energy should go from overspeed to altitude = pitch up
00354   float en_dis_err = gamma_err - vdot_err;
00355 
00356   // Auto Cruise Throttle
00357   if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
00358   {
00359     v_ctl_auto_throttle_nominal_cruise_throttle +=
00360         v_ctl_auto_throttle_of_airspeed_igain * speed_error * dt
00361       + en_tot_err * v_ctl_energy_total_igain * dt;
00362     Bound(v_ctl_auto_throttle_nominal_cruise_throttle,0.0f,1.0f);
00363   }
00364 
00365   // Total Controller
00366   float controlled_throttle = v_ctl_auto_throttle_nominal_cruise_throttle
00367     + v_ctl_auto_throttle_climb_throttle_increment * v_ctl_climb_setpoint
00368     + v_ctl_auto_throttle_of_airspeed_pgain * speed_error
00369     + v_ctl_energy_total_pgain * en_tot_err;
00370 
00371 
00372   if ((controlled_throttle >= 1.0f) || (controlled_throttle <= 0.0f))
00373   {
00374     // If your energy supply is not sufficient, then neglect the climb requirement
00375     en_dis_err = -vdot_err;
00376   }
00377 
00378   /* pitch pre-command */
00379   if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
00380   {
00381     v_ctl_auto_throttle_nominal_cruise_pitch +=  v_ctl_auto_pitch_of_airspeed_igain * (-speed_error) * dt
00382                           + v_ctl_energy_diff_igain * en_dis_err * dt;
00383     Bound(v_ctl_auto_throttle_nominal_cruise_pitch,H_CTL_PITCH_MIN_SETPOINT, H_CTL_PITCH_MAX_SETPOINT);
00384   }
00385   float v_ctl_pitch_of_vz =
00386         + (v_ctl_climb_setpoint /*+ d_err * v_ctl_auto_throttle_pitch_of_vz_dgain*/) * v_ctl_auto_throttle_pitch_of_vz_pgain
00387         - v_ctl_auto_pitch_of_airspeed_pgain * speed_error
00388                 + v_ctl_auto_pitch_of_airspeed_dgain * vdot
00389                 + v_ctl_energy_diff_pgain * en_dis_err
00390                 + v_ctl_auto_throttle_nominal_cruise_pitch;
00391 
00392   nav_pitch = v_ctl_pitch_of_vz;
00393   Bound(nav_pitch,H_CTL_PITCH_MIN_SETPOINT,H_CTL_PITCH_MAX_SETPOINT)
00394 
00395   ac_char_update(controlled_throttle, v_ctl_pitch_of_vz, v_ctl_climb_setpoint, v_ctl_desired_acceleration);
00396 
00397   v_ctl_throttle_setpoint = TRIM_UPPRZ(controlled_throttle * MAX_PPRZ);
00398 }
