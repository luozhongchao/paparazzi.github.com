100 void imu_umarim_event( void )
101 {
102  // If the itg3200 I2C transaction has succeeded: convert the data
103  itg3200_event(&imu_umarim.itg);
104  if (imu_umarim.itg.data_available) {
105  RATES_COPY(imu.gyro_unscaled, imu_umarim.itg.data.rates);
106  imu_umarim.itg.data_available = FALSE;
107  imu_umarim.gyr_valid = TRUE;
108  }
109 
110  // If the adxl345 I2C transaction has succeeded: convert the data
111  adxl345_i2c_event(&imu_umarim.adxl);
112  if (imu_umarim.adxl.data_available) {
113  VECT3_ASSIGN(imu.accel_unscaled, imu_umarim.adxl.data.vect.y, -imu_umarim.adxl.data.vect.x, imu_umarim.adxl.data.vect.z);
114  imu_umarim.adxl.data_available = FALSE;
115  imu_umarim.acc_valid = TRUE;
116  }
