00145 void motor_mixing_run(bool_t motors_on, bool_t override_on, pprz_t in_cmd[] ) {
00146   uint8_t i;
00147   if (motors_on) {
00148     int32_t min_cmd = INT32_MAX;
00149     int32_t max_cmd = INT32_MIN;
00150     /* do the mixing in float to avoid overflows, implicitly casted back to int32_t */
00151     for (i=0; i<MOTOR_MIXING_NB_MOTOR; i++) {
00152       motor_mixing.commands[i] = MOTOR_MIXING_MIN_MOTOR +
00153         (thrust_coef[i] * in_cmd[COMMAND_THRUST] +
00154          roll_coef[i]   * in_cmd[COMMAND_ROLL]   +
00155          pitch_coef[i]  * in_cmd[COMMAND_PITCH]  +
00156          yaw_coef[i]    * in_cmd[COMMAND_YAW]    +
00157          motor_mixing.trim[i]) / MOTOR_MIXING_SCALE *
00158         (MOTOR_MIXING_MAX_MOTOR - MOTOR_MIXING_MIN_MOTOR) / MAX_PPRZ;
00159       if (motor_mixing.commands[i] < min_cmd)
00160         min_cmd = motor_mixing.commands[i];
00161       if (motor_mixing.commands[i] > max_cmd)
00162         max_cmd = motor_mixing.commands[i];
00163     }
00164     if (min_cmd < MOTOR_MIXING_MIN_MOTOR && max_cmd > MOTOR_MIXING_MAX_MOTOR)
00165       motor_mixing.nb_failure++;
00166     if (min_cmd < MOTOR_MIXING_MIN_MOTOR)
00167       offset_commands(-(min_cmd - MOTOR_MIXING_MIN_MOTOR));
00168     if (max_cmd > MOTOR_MIXING_MAX_MOTOR)
00169       offset_commands(-(max_cmd - MOTOR_MIXING_MAX_MOTOR));
00170 
00171     /* For testing motor failure */
00172     if (motors_on && override_on) {
00173       for (i = 0; i < MOTOR_MIXING_NB_MOTOR; i++) {
00174         if (motor_mixing.override_enabled[i])
00175           motor_mixing.commands[i] = motor_mixing.override_value[i];
00176       }
00177     }
00178     bound_commands();
00179     bound_commands_step();
00180   }
00181   else {
00182     for (i=0; i<MOTOR_MIXING_NB_MOTOR; i++) {
00183       motor_mixing.commands[i] = MOTOR_MIXING_STOP_MOTOR;
