74 void motor_mixing_init(void) {
75  uint8_t i;
76  for (i=0; i<MOTOR_MIXING_NB_MOTOR; i++) {
77  motor_mixing.commands[i] = 0;
78  motor_mixing.trim[i] =
79  roll_coef[i] * MOTOR_MIXING_TRIM_ROLL +
80  pitch_coef[i] * MOTOR_MIXING_TRIM_PITCH +
81  yaw_coef[i] * MOTOR_MIXING_TRIM_YAW;
82  motor_mixing.override_enabled[i] = FALSE;
83  motor_mixing.override_value[i] = MOTOR_MIXING_STOP_MOTOR;
84  }
85  motor_mixing.nb_failure = 0;
86 }

126 void motor_mixing_run_spinup(uint32_t counter, uint32_t max_counter)
127 {
128  int i;
129  for (i = 0; i < MOTOR_MIXING_NB_MOTOR; i++) {
130 #ifdef MOTOR_MIXING_STARTUP_DELAY
131  if (counter > i * max_counter / (MOTOR_MIXING_NB_MOTOR + MOTOR_MIXING_STARTUP_DELAY)) {
132  if (counter > MOTOR_MIXING_NB_MOTOR * max_counter / (MOTOR_MIXING_NB_MOTOR + MOTOR_MIXING_STARTUP_DELAY)) {
133  motor_mixing.commands[i] = MOTOR_MIXING_MIN_MOTOR_STARTUP + (MOTOR_MIXING_MIN_MOTOR - MOTOR_MIXING_MIN_MOTOR_STARTUP) * counter / max_counter;
134  } else {
135  motor_mixing.commands[i] = MOTOR_MIXING_MIN_MOTOR_STARTUP;
136  }
137  } else {
138  motor_mixing.commands[i] = 0;
139  }
140 #else
141  if (counter < i * max_counter / MOTOR_MIXING_NB_MOTOR) {
142  motor_mixing.commands[i] = MOTOR_MIXING_MIN_MOTOR_STARTUP;
143  }
144 #endif
145  }

148 void motor_mixing_run(bool_t motors_on, bool_t override_on, pprz_t in_cmd[] ) {
149  uint8_t i;
150  if (motors_on) {
151  int32_t min_cmd = INT32_MAX;
152  int32_t max_cmd = INT32_MIN;
153  /* do the mixing in float to avoid overflows, implicitly casted back to int32_t */
154  for (i=0; i<MOTOR_MIXING_NB_MOTOR; i++) {
155  motor_mixing.commands[i] = MOTOR_MIXING_MIN_MOTOR +
156  (thrust_coef[i] * in_cmd[COMMAND_THRUST] +
157  roll_coef[i] * in_cmd[COMMAND_ROLL] +
158  pitch_coef[i] * in_cmd[COMMAND_PITCH] +
159  yaw_coef[i] * in_cmd[COMMAND_YAW] +
160  motor_mixing.trim[i]) / MOTOR_MIXING_SCALE *
161  (MOTOR_MIXING_MAX_MOTOR - MOTOR_MIXING_MIN_MOTOR) / MAX_PPRZ;
162  if (motor_mixing.commands[i] < min_cmd)
163  min_cmd = motor_mixing.commands[i];
164  if (motor_mixing.commands[i] > max_cmd)
165  max_cmd = motor_mixing.commands[i];
166  }
167  if (min_cmd < MOTOR_MIXING_MIN_MOTOR && max_cmd > MOTOR_MIXING_MAX_MOTOR)
168  motor_mixing.nb_failure++;
169  if (min_cmd < MOTOR_MIXING_MIN_MOTOR)
170  offset_commands(-(min_cmd - MOTOR_MIXING_MIN_MOTOR));
171  if (max_cmd > MOTOR_MIXING_MAX_MOTOR)
172  offset_commands(-(max_cmd - MOTOR_MIXING_MAX_MOTOR));
173 
174  /* For testing motor failure */
175  if (motors_on && override_on) {
176  for (i = 0; i < MOTOR_MIXING_NB_MOTOR; i++) {
177  if (motor_mixing.override_enabled[i])
178  motor_mixing.commands[i] = motor_mixing.override_value[i];
179  }
180  }
181  bound_commands();
182  bound_commands_step();
183  }
184  else {
185  for (i=0; i<MOTOR_MIXING_NB_MOTOR; i++) {
186  motor_mixing.commands[i] = MOTOR_MIXING_STOP_MOTOR;
187  }
188  }
