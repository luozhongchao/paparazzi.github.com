00089 static inline enum tcas_resolve tcas_test_direction(uint8_t id) {
00090   struct ac_info_ * ac = get_ac_info(id);
00091   float dz = ac->alt - stateGetPositionEnu_f()->z;
00092   if (dz > tcas_alim/2) return RA_DESCEND;
00093   else if (dz < -tcas_alim/2) return RA_CLIMB;
00094   else // AC with the smallest ID descend
00095   {
00096     if (AC_ID < id) return RA_DESCEND;
00097     else return RA_CLIMB;
00098   }

00103 void tcas_periodic_task_1Hz( void ) {
00104   // no TCAS under security_height
00105   if (stateGetPositionEnu_f()->z < GROUND_ALT + SECURITY_HEIGHT) {
00106     uint8_t i;
00107     for (i = 0; i < NB_ACS; i++) tcas_acs_status[i].status = TCAS_NO_ALARM;
00108     return;
00109   }
00110   // test possible conflicts
00111   float tau_min = tcas_tau_ta;
00112   uint8_t ac_id_close = AC_ID;
00113   uint8_t i;
00114   float vx = (*stateGetHorizontalSpeedNorm_f()) * sinf((*stateGetHorizontalSpeedDir_f()));
00115   float vy = (*stateGetHorizontalSpeedNorm_f()) * cosf((*stateGetHorizontalSpeedDir_f()));
00116   for (i = 2; i < NB_ACS; i++) {
00117     if (the_acs[i].ac_id == 0) continue; // no AC data
00118     uint32_t dt = gps.tow - the_acs[i].itow;
00119     if (dt > 3*TCAS_DT_MAX) {
00120       tcas_acs_status[i].status = TCAS_NO_ALARM; // timeout, reset status
00121       continue;
00122     }
00123     if (dt > TCAS_DT_MAX) continue; // lost com but keep current status
00124     float dx = the_acs[i].east - stateGetPositionEnu_f()->x;
00125     float dy = the_acs[i].north - stateGetPositionEnu_f()->y;
00126     float dz = the_acs[i].alt - stateGetPositionEnu_f()->z;
00127     float dvx = vx - the_acs[i].gspeed * sinf(the_acs[i].course);
00128     float dvy = vy - the_acs[i].gspeed * cosf(the_acs[i].course);
00129     float dvz = stateGetSpeedEnu_f()->z - the_acs[i].climb;
00130     float scal = dvx*dx + dvy*dy + dvz*dz;
00131     float ddh = dx*dx + dy*dy;
00132     float ddv = dz*dz;
00133     float tau = TCAS_HUGE_TAU;
00134     if (scal > 0.) tau = (ddh + ddv) / scal;
00135     // monitor conflicts
00136     uint8_t inside = TCAS_IsInside();
00137     //enum tcas_resolve test_dir = RA_NONE;
00138     switch (tcas_acs_status[i].status) {
00139       case TCAS_RA:
00140         if (tau >= TCAS_HUGE_TAU && !inside) {
00141           tcas_acs_status[i].status = TCAS_NO_ALARM; // conflict is now resolved
00142           tcas_acs_status[i].resolve = RA_NONE;
00143           DOWNLINK_SEND_TCAS_RESOLVED(DefaultChannel, DefaultDevice,&(the_acs[i].ac_id));
00144         }
00145         break;
00146       case TCAS_TA:
00147         if (tau < tcas_tau_ra || inside) {
00148           tcas_acs_status[i].status = TCAS_RA; // TA -> RA
00149           // Downlink alert
00150           //test_dir = tcas_test_direction(the_acs[i].ac_id);
00151           //DOWNLINK_SEND_TCAS_RA(DefaultChannel, DefaultDevice,&(the_acs[i].ac_id),&test_dir);// FIXME only one closest AC ???
00152           break;
00153         }
00154         if (tau > tcas_tau_ta && !inside)
00155           tcas_acs_status[i].status = TCAS_NO_ALARM; // conflict is now resolved
00156           tcas_acs_status[i].resolve = RA_NONE;
00157           DOWNLINK_SEND_TCAS_RESOLVED(DefaultChannel, DefaultDevice,&(the_acs[i].ac_id));
00158         break;
00159       case TCAS_NO_ALARM:
00160         if (tau < tcas_tau_ta || inside) {
00161           tcas_acs_status[i].status = TCAS_TA; // NO_ALARM -> TA
00162           // Downlink warning
00163           DOWNLINK_SEND_TCAS_TA(DefaultChannel, DefaultDevice,&(the_acs[i].ac_id));
00164         }
00165         if (tau < tcas_tau_ra || inside) {
00166           tcas_acs_status[i].status = TCAS_RA; // NO_ALARM -> RA = big problem ?
00167           // Downlink alert
00168           //test_dir = tcas_test_direction(the_acs[i].ac_id);
00169           //DOWNLINK_SEND_TCAS_RA(DefaultChannel, DefaultDevice,&(the_acs[i].ac_id),&test_dir);
00170         }
00171         break;
00172     }
00173     // store closest AC
00174     if (tau < tau_min) {
00175       tau_min = tau;
00176       ac_id_close = the_acs[i].ac_id;
00177 
00178     }
00179   }
00180   // set current conflict mode
00181   if (tcas_status == TCAS_RA && tcas_ac_RA != AC_ID && tcas_acs_status[the_acs_id[tcas_ac_RA]].status == TCAS_RA) {
00182     ac_id_close = tcas_ac_RA; // keep RA until resolved
00183   }
00184   tcas_status = tcas_acs_status[the_acs_id[ac_id_close]].status;
00185   // at least one in conflict, deal with closest one
00186   if (tcas_status == TCAS_RA) {
00187     tcas_ac_RA = ac_id_close;
00188     tcas_resolve = tcas_test_direction(tcas_ac_RA);
00189     uint8_t ac_resolve = tcas_acs_status[the_acs_id[tcas_ac_RA]].resolve;
00190     if (ac_resolve != RA_NONE) { // first resolution, no message received
00191       if (ac_resolve == tcas_resolve) { // same direction, lowest id go down
00192         if (AC_ID < tcas_ac_RA) tcas_resolve = RA_DESCEND;
00193         else tcas_resolve = RA_CLIMB;
00194       }
00195       tcas_acs_status[the_acs_id[tcas_ac_RA]].resolve = RA_LEVEL; // assuming level flight for now
00196     }
00197     else { // second resolution or message received
00198       if (ac_resolve != RA_LEVEL) { // message received
00199         if (ac_resolve == tcas_resolve) { // same direction, lowest id go down
00200           if (AC_ID < tcas_ac_RA) tcas_resolve = RA_DESCEND;
00201           else tcas_resolve = RA_CLIMB;
00202         }
00203       }
00204       else { // no message
00205         if (tcas_resolve == RA_CLIMB && the_acs[the_acs_id[tcas_ac_RA]].climb > 1.0) tcas_resolve = RA_DESCEND; // revert resolve
00206         else if (tcas_resolve == RA_DESCEND && the_acs[the_acs_id[tcas_ac_RA]].climb < -1.0) tcas_resolve = RA_CLIMB; // revert resolve
00207       }
00208     }
00209     // Downlink alert
00210     DOWNLINK_SEND_TCAS_RA(DefaultChannel, DefaultDevice,&tcas_ac_RA,&tcas_resolve);
00211   }
00212   else tcas_ac_RA = AC_ID; // no conflict
00213 #ifdef TCAS_DEBUG
00214   if (tcas_status == TCAS_RA) DOWNLINK_SEND_TCAS_DEBUG(DefaultChannel, DefaultDevice,&ac_id_close,&tau_min);
00215 #endif
00216 }

00220 void tcas_periodic_task_4Hz( void ) {
00221   // set alt setpoint
00222   if (stateGetPositionEnu_f()->z > GROUND_ALT + SECURITY_HEIGHT && tcas_status == TCAS_RA) {
00223     struct ac_info_ * ac = get_ac_info(tcas_ac_RA);
00224     switch (tcas_resolve) {
00225       case RA_CLIMB :
00226         tcas_alt_setpoint = Max(nav_altitude, ac->alt + tcas_alim);
00227         break;
00228       case RA_DESCEND :
00229         tcas_alt_setpoint = Min(nav_altitude, ac->alt - tcas_alim);
00230         break;
00231       case RA_LEVEL :
00232       case RA_NONE :
00233         tcas_alt_setpoint = nav_altitude;
00234         break;
00235     }
00236     // Bound alt
00237     tcas_alt_setpoint = Max(GROUND_ALT + SECURITY_HEIGHT, tcas_alt_setpoint);
00238   }
00239   else {
00240     tcas_alt_setpoint = nav_altitude;
00241     tcas_resolve = RA_NONE;
