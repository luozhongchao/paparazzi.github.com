00041 inline void init_median_filter(struct MedianFilterInt * filter) {
00042   int i;
00043   for (i = 0; i < MEDIAN_DATASIZE; i++) {
00044     filter->data[i] = 0;
00045     filter->sortData[i] = 0;
00046   }
00047   filter->dataIndex = 0;

00050 inline int32_t update_median_filter(struct MedianFilterInt * filter, int32_t new_data) {
00051   int temp, i, j; // used to sort array
00052 
00053   // Insert new data into raw data array round robin style
00054   filter->data[filter->dataIndex] = new_data;
00055   if (filter->dataIndex < (MEDIAN_DATASIZE-1)) {
00056     filter->dataIndex++;
00057   }
00058   else {
00059     filter->dataIndex = 0;
00060   }
00061 
00062   // Copy raw data to sort data array
00063   memcpy(filter->sortData, filter->data, sizeof(filter->data));
00064 
00065   // Insertion Sort
00066   for(i = 1; i <= (MEDIAN_DATASIZE-1); i++) {
00067     temp = filter->sortData[i];
00068     j = i-1;
00069     while(temp < filter->sortData[j] && j>=0) {
00070       filter->sortData[j+1] = filter->sortData[j];
00071       j = j-1;
00072     }
00073     filter->sortData[j+1] = temp;
00074   }
00075   return filter->sortData[(MEDIAN_DATASIZE)>>1]; // return data value in middle of sorted array
00076 }
00077 
00078 inline int32_t get_median_filter(struct MedianFilterInt * filter) {
00079   return filter->sortData[(MEDIAN_DATASIZE)>>1];
00080 }
