00167   static const float a = 6378137.0;           /* earth semimajor axis in meters */
00168   static const float f = 1./298.257223563;    /* reciprocal flattening          */
00169   const float b = a*(1.-f);                   /* semi-minor axis                */
00170   const float b2 = b*b;
00171 
00172   const float e2 = 2.*f-(f*f);                /* first eccentricity squared     */
00173   const float ep2 = f*(2.-f)/((1.-f)*(1.-f)); /* second eccentricity squared    */
00174   const float E2 = a*a - b2;
00175 
00176 
00177   const float z2 = in->z*in->z;
00178   const float r2 = in->x*in->x+in->y*in->y;
00179   const float r = sqrtf(r2);
00180   const float F = 54.*b2*z2;
00181   const float G = r2 + (1-e2)*z2 - e2*E2;
00182   const float c = (e2*e2*F*r2)/(G*G*G);
00183   const float s = powf( (1 + c + sqrtf(c*c + 2*c)), 1./3.);
00184   const float s1 = 1+s+1/s;
00185   const float P = F/(3*s1*s1*G*G);
00186   const float Q = sqrtf(1+2*e2*e2*P);
00187   const float ro = -(e2*P*r)/(1+Q) + sqrtf((a*a/2)*(1+1/Q) - ((1-e2)*P*z2)/(Q*(1+Q)) - P*r2/2);
00188   const float tmp = (r - e2*ro)*(r - e2*ro);
00189   const float U = sqrtf( tmp + z2 );
00190   const float V = sqrtf( tmp + (1-e2)*z2 );
00191   const float zo = (b2*in->z)/(a*V);
00192 
00193   out->alt = U*(1 - b2/(a*V));
00194   out->lat = atanf((in->z + ep2*zo)/r);
00195   out->lon = atan2f(in->y,in->x);
