00052 void pwm_input_init ( void )
00053 {
00054   // initialize the arrays to 0 to avoid junk
00055   for (int i=0; i < PWM_INPUT_NB; i++)
00056   {
00057     pwm_input_duty_tics[i] = 0;
00058     pwm_input_duty_valid[i] = 0;
00059     pwm_input_period_tics[i] = 0;
00060     pwm_input_period_valid[i] = 0;
00061   }
00062   /* select pin for capture */
00063 #ifdef USE_PWM_INPUT1
00064   PWM_INPUT1_PINSEL = (PWM_INPUT1_PINSEL & ~PWM_INPUT1_PINSEL_MASK) | PWM_INPUT1_PINSEL_VAL;
00065   //enable capture 0.3 on rising edge + trigger interrupt
00066   T0CCR |= TCCR_CR3_R | TCCR_CR3_I;
00067 #endif
00068 #ifdef USE_PWM_INPUT2
00069   PWM_INPUT2_PINSEL = (PWM_INPUT2_PINSEL & ~PWM_INPUT2_PINSEL_MASK) | PWM_INPUT2_PINSEL_VAL;
00070   //enable capture 0.0 on rising edge + trigger interrupt
00071   T0CCR |= TCCR_CR0_R | TCCR_CR0_I;
00072 #endif
00073 }
