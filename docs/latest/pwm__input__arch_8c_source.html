00050 void pwm_input_init ( void )
00051 {
00052   // initialize the arrays to 0 to avoid junk
00053   for (int i=0; i < PWM_INPUT_NB; i++)
00054   {
00055     pwm_input_duty_tics[i] = 0;
00056     pwm_input_duty_valid[i] = 0;
00057     pwm_input_period_tics[i] = 0;
00058     pwm_input_period_valid[i] = 0;
00059   }
00060   /* select pin for capture */
00061 #ifdef USE_PWM_INPUT1
00062   PWM_INPUT1_PINSEL = (PWM_INPUT1_PINSEL & ~PWM_INPUT1_PINSEL_MASK) | PWM_INPUT1_PINSEL_VAL;
00063   //enable capture 0.3 on rising edge + trigger interrupt
00064   T0CCR |= TCCR_CR3_R | TCCR_CR3_I;
00065 #endif
00066 #ifdef USE_PWM_INPUT2
00067   PWM_INPUT2_PINSEL = (PWM_INPUT2_PINSEL & ~PWM_INPUT2_PINSEL_MASK) | PWM_INPUT2_PINSEL_VAL;
00068   //enable capture 0.0 on rising edge + trigger interrupt
00069   T0CCR |= TCCR_CR0_R | TCCR_CR0_I;
00070 #endif
00071 }
