63 void actuators_pwm_arch_init(void) {

65  /*-----------------------------------
66  * Configure timer peripheral clocks
67  *-----------------------------------*/
68  /* TIM3, TIM4 and TIM5 clock enable */
69  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_TIM3EN);
70 #if REMAP_SERVOS_5AND6
71  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_TIM5EN);
72 #else
73  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_TIM4EN);
74 #endif
75 #if USE_SERVOS_7AND8
76  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_TIM4EN);
77 #endif
78 
79  /*----------------
80  * Configure GPIO
81  *----------------*/
82  /* GPIO A,B and C clock enable */
83  rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_IOPAEN |
84  RCC_APB2ENR_IOPBEN |
85  RCC_APB2ENR_IOPCEN |
86  RCC_APB2ENR_AFIOEN);
87 
88  /* TIM3 GPIO for PWM1..4 */
89  AFIO_MAPR |= AFIO_MAPR_TIM3_REMAP_FULL_REMAP;
90  gpio_set_mode(GPIO_BANK_TIM3_FR,
91  GPIO_MODE_OUTPUT_50_MHZ,
92  GPIO_CNF_OUTPUT_ALTFN_PUSHPULL,
93  GPIO_TIM3_FR_CH1 |
94  GPIO_TIM3_FR_CH2 |
95  GPIO_TIM3_FR_CH3 |
96  GPIO_TIM3_FR_CH4);
97 
98  /* TIM4 GPIO for PWM7..8 */
99 #if USE_SERVOS_7AND8
100  gpio_set_mode(GPIO_BANK_TIM4,
101  GPIO_MODE_OUTPUT_50_MHZ,
102  GPIO_CNF_OUTPUT_ALTFN_PUSHPULL,
103  GPIO_TIM4_CH1 |
104  GPIO_TIM4_CH2);
105 #endif
106 
107  /* TIM4/5 GPIO for PWM6..7 */
108 #if REMAP_SERVOS_5AND6
109  gpio_set_mode(GPIO_BANK_TIM5,
110  GPIO_MODE_OUTPUT_50_MHZ,
111  GPIO_CNF_OUTPUT_ALTFN_PUSHPULL,
112  GPIO_TIM5_CH1 |
113  GPIO_TIM5_CH2);
114 #else
115  gpio_set_mode(GPIO_BANK_TIM4,
116  GPIO_MODE_OUTPUT_50_MHZ,

118  GPIO_TIM4_CH3 |
119  GPIO_TIM4_CH4);
120 #endif

122  /*---------------
123  * Timer 3 setup
124  *---------------*/
125  timer_reset(TIM3);
126 
127  /* Timer global mode:
128  * - No divider.
129  * - Alignement edge.
130  * - Direction up.
131  */
132  timer_set_mode(TIM3, TIM_CR1_CKD_CK_INT,
133  TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
134 
135  timer_set_prescaler(TIM3, (PCLK / ONE_MHZ_CLK) - 1); // 1uS
136 
137  timer_disable_preload(TIM3);
138 
139  timer_continuous_mode(TIM3);
140 
141  timer_set_period(TIM3, (ONE_MHZ_CLK / SERVO_HZ) - 1);
142 
143  /* Disable outputs. */
144  timer_disable_oc_output(TIM3, TIM_OC1);
145  timer_disable_oc_output(TIM3, TIM_OC2);
146  timer_disable_oc_output(TIM3, TIM_OC3);
147  timer_disable_oc_output(TIM3, TIM_OC4);
148 
149  /* -- Channel configuration -- */

151  actuators_pwm_arch_channel_init(TIM3, TIM_OC2);
152  actuators_pwm_arch_channel_init(TIM3, TIM_OC3);
153  actuators_pwm_arch_channel_init(TIM3, TIM_OC4);

316 /* set pulse widths from actuator values, assumed to be in us */
317 void actuators_pwm_commit(void) {
318  timer_set_oc_value(TIM3, TIM_OC1, actuators_pwm_values[0]);
319  timer_set_oc_value(TIM3, TIM_OC2, actuators_pwm_values[1]);
320  timer_set_oc_value(TIM3, TIM_OC3, actuators_pwm_values[2]);
321  timer_set_oc_value(TIM3, TIM_OC4, actuators_pwm_values[3]);
322 
323 #if USE_SERVOS_7AND8
324  timer_set_oc_value(TIM4, TIM_OC1, actuators_pwm_values[6]);
325  timer_set_oc_value(TIM4, TIM_OC2, actuators_pwm_values[7]);
326 #endif
327 #if REMAP_SERVOS_5AND6
328  timer_set_oc_value(TIM5, TIM_OC1, actuators_pwm_values[4]);
329  timer_set_oc_value(TIM5, TIM_OC2, actuators_pwm_values[5]);
330 #else
331  timer_set_oc_value(TIM4, TIM_OC3, actuators_pwm_values[4]);
