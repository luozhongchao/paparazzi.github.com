00048 void compute_dist2_to_home(void) {
00049   struct EnuCoor_f* pos = stateGetPositionEnu_f();
00050   float ph_x = waypoints[WP_HOME].x - pos->x;
00051   float ph_y = waypoints[WP_HOME].y - pos->y;
00052   dist2_to_home = ph_x*ph_x + ph_y *ph_y;
00053   too_far_from_home = dist2_to_home > (MAX_DIST_FROM_HOME*MAX_DIST_FROM_HOME);
00054 #if defined InAirspace
00055   too_far_from_home = too_far_from_home || !(InAirspace(pos_x, pos_y));
00056 #endif
00057 }

00105 void nav_move_waypoint(uint8_t wp_id, float ux, float uy, float alt) {
00106   if (wp_id < nb_waypoint) {
00107     float dx, dy;
00108     dx = ux - nav_utm_east0 - waypoints[WP_HOME].x;
00109     dy = uy - nav_utm_north0 - waypoints[WP_HOME].y;
00110     BoundAbs(dx, max_dist_from_home);
00111     BoundAbs(dy, max_dist_from_home);
00112     waypoints[wp_id].x = waypoints[WP_HOME].x + dx;
00113     waypoints[wp_id].y = waypoints[WP_HOME].y + dy;
