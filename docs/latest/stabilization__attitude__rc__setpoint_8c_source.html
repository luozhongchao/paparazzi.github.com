57 /* This is a different way to obtain yaw. It will not switch when going beyond 90 degrees pitch.
58  However, when rolling more then 90 degrees in combination with pitch it switches. For a
59  transition vehicle this is better as 90 degrees pitch will occur, but more than 90 degrees roll probably not. */
60 int32_t stabilization_attitude_get_heading_i(void) {
61  struct Int32Eulers* att = stateGetNedToBodyEulers_i();
62 
63  int32_t heading;
64 
65  if(abs(att->phi) < INT32_ANGLE_PI_2) {
66  int32_t sin_theta;
67  PPRZ_ITRIG_SIN(sin_theta, att->theta);
68  heading = att->psi - INT_MULT_RSHIFT(sin_theta, att->phi, INT32_TRIG_FRAC);
69  }
70  else if(ANGLE_FLOAT_OF_BFP(att->theta) > 0)
71  heading = att->psi - att->phi;
72  else
73  heading = att->psi + att->phi;
74 
75  return heading;
76 }

98 void stabilization_attitude_read_rc_setpoint_eulers(struct Int32Eulers *sp, bool_t in_flight) {
99 
100  sp->phi = ((int32_t) radio_control.values[RADIO_ROLL] * SP_MAX_PHI / MAX_PPRZ);
101  sp->theta = ((int32_t) radio_control.values[RADIO_PITCH] * SP_MAX_THETA / MAX_PPRZ);
102 
103  if (in_flight) {
104  if (YAW_DEADBAND_EXCEEDED()) {
105  sp->psi += ((int32_t) radio_control.values[RADIO_YAW] * SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ);
106  INT32_ANGLE_NORMALIZE(sp->psi);
107  }
108  if (autopilot_mode == AP_MODE_FORWARD) {
109  //Coordinated turn
110  //feedforward estimate angular rotation omega = g*tan(phi)/v
111  //Take v = 9.81/1.3 m/s
112  int32_t omega;
113  const int32_t max_phi = ANGLE_BFP_OF_REAL(RadOfDeg(85.0));
114  if(abs(sp->phi) < max_phi)
115  omega = ANGLE_BFP_OF_REAL(1.3*tanf(ANGLE_FLOAT_OF_BFP(sp->phi)));
116  else //max 60 degrees roll, then take constant omega
117  omega = ANGLE_BFP_OF_REAL(1.3*1.72305* ((sp->phi > 0) - (sp->phi < 0)));
118 
119  sp->psi += omega/RC_UPDATE_FREQ;
120  }
121 #ifdef STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT
122  // Make sure the yaw setpoint does not differ too much from the real yaw
123  // to prevent a sudden switch at 180 deg
124  int32_t heading = stabilization_attitude_get_heading_i();
125 
126  int32_t delta_psi = sp->psi - heading;
127  int32_t delta_limit = ANGLE_BFP_OF_REAL(STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT);
128  INT32_ANGLE_NORMALIZE(delta_psi);
129  if (delta_psi > delta_limit){
130  sp->psi = heading + delta_limit;
131  }
132  else if (delta_psi < -delta_limit){
133  sp->psi = heading - delta_limit;
134  }
135  INT32_ANGLE_NORMALIZE(sp->psi);
136 #endif
137  //Care Free mode
138  if (guidance_h_mode == GUIDANCE_H_MODE_CARE_FREE) {
139  //care_free_heading has been set to current psi when entering care free mode.
140  int32_t cos_psi;
141  int32_t sin_psi;
142  int32_t temp_theta;
143  int32_t care_free_delta_psi_i;
144 
145  care_free_delta_psi_i = sp->psi - ANGLE_BFP_OF_REAL(care_free_heading);
146 
147  INT32_ANGLE_NORMALIZE(care_free_delta_psi_i);
148 
149  PPRZ_ITRIG_SIN(sin_psi, care_free_delta_psi_i);
150  PPRZ_ITRIG_COS(cos_psi, care_free_delta_psi_i);
151 
152  temp_theta = INT_MULT_RSHIFT(cos_psi, sp->theta, INT32_ANGLE_FRAC) - INT_MULT_RSHIFT(sin_psi, sp->phi, INT32_ANGLE_FRAC);
153  sp->phi = INT_MULT_RSHIFT(cos_psi, sp->phi, INT32_ANGLE_FRAC) - INT_MULT_RSHIFT(sin_psi, sp->theta, INT32_ANGLE_FRAC);
154 
155  sp->theta = temp_theta;
156  }
157  }
158  else { /* if not flying, use current yaw as setpoint */
159  sp->psi = stateGetNedToBodyEulers_i()->psi;
160  }
161 }

164 void stabilization_attitude_read_rc_setpoint_eulers_f(struct FloatEulers *sp, bool_t in_flight) {
165  sp->phi = (radio_control.values[RADIO_ROLL] * SP_MAX_PHI / MAX_PPRZ);
166  sp->theta = (radio_control.values[RADIO_PITCH] * SP_MAX_THETA / MAX_PPRZ);
167 
168  if (in_flight) {
169  if (YAW_DEADBAND_EXCEEDED()) {
170  sp->psi += (radio_control.values[RADIO_YAW] * SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ);
171  FLOAT_ANGLE_NORMALIZE(sp->psi);
172  }
173  if (autopilot_mode == AP_MODE_FORWARD) {
174  //Coordinated turn
175  //feedforward estimate angular rotation omega = g*tan(phi)/v
176  //Take v = 9.81/1.3 m/s
177  float omega;
178  const float max_phi = RadOfDeg(85.0);
179  if(abs(sp->phi) < max_phi)
180  omega = 1.3*tanf(sp->phi);
181  else //max 60 degrees roll, then take constant omega
182  omega = 1.3*1.72305* ((sp->phi > 0) - (sp->phi < 0));
183 
184  sp->psi += omega/RC_UPDATE_FREQ;
185  }
186 #ifdef STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT
187  // Make sure the yaw setpoint does not differ too much from the real yaw
188  // to prevent a sudden switch at 180 deg
189  float heading = stabilization_attitude_get_heading_f();
190 
191  float delta_psi = sp->psi - heading;
192  FLOAT_ANGLE_NORMALIZE(delta_psi);
193  if (delta_psi > STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT){
194  sp->psi = heading + STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT;
195  }
196  else if (delta_psi < -STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT){
197  sp->psi = heading - STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT;
198  }
199  FLOAT_ANGLE_NORMALIZE(sp->psi);
200 #endif
201  //Care Free mode
202  if (guidance_h_mode == GUIDANCE_H_MODE_CARE_FREE) {
203  //care_free_heading has been set to current psi when entering care free mode.
204  float cos_psi;
205  float sin_psi;
206  float temp_theta;
207 
208  float care_free_delta_psi_f = sp->psi - care_free_heading;
209 
210  FLOAT_ANGLE_NORMALIZE(care_free_delta_psi_f);
211 
212  sin_psi = sinf(care_free_delta_psi_f);
213  cos_psi = cosf(care_free_delta_psi_f);
214 
215  temp_theta = cos_psi*sp->theta - sin_psi*sp->phi;
216  sp->phi = cos_psi*sp->phi - sin_psi*sp->theta;
217 
218  sp->theta = temp_theta;
219  }
220  }
221  else { /* if not flying, use current yaw as setpoint */
222  sp->psi = stateGetNedToBodyEulers_f()->psi;
223  }
224 }

231 void stabilization_attitude_read_rc_roll_pitch_quat_f(struct FloatQuat* q) {
232  q->qx = radio_control.values[RADIO_ROLL] * STABILIZATION_ATTITUDE_SP_MAX_PHI / MAX_PPRZ / 2;
233  q->qy = radio_control.values[RADIO_PITCH] * STABILIZATION_ATTITUDE_SP_MAX_THETA / MAX_PPRZ / 2;
234  q->qz = 0.0;
235 
236  /* normalize */
237  float norm = sqrtf(1.0 + SQUARE(q->qx)+ SQUARE(q->qy));
238  q->qi = 1.0 / norm;
239  q->qx /= norm;
240  q->qy /= norm;
241 }

247 void stabilization_attitude_read_rc_roll_pitch_earth_quat_f(struct FloatQuat* q) {
248  /* only non-zero entries for roll quaternion */
249  float roll2 = radio_control.values[RADIO_ROLL] * STABILIZATION_ATTITUDE_SP_MAX_PHI / MAX_PPRZ / 2;
250  float qx_roll = sinf(roll2);
251  float qi_roll = cosf(roll2);
252 
253  //An offset is added if in forward mode
254  /* only non-zero entries for pitch quaternion */
255  float pitch2 = (ANGLE_FLOAT_OF_BFP(transition_theta_offset) + radio_control.values[RADIO_PITCH] * STABILIZATION_ATTITUDE_SP_MAX_THETA / MAX_PPRZ) / 2;
256  float qy_pitch = sinf(pitch2);
257  float qi_pitch = cosf(pitch2);
258 
259  /* only multiply non-zero entries of FLOAT_QUAT_COMP(*q, q_roll, q_pitch) */
260  q->qi = qi_roll * qi_pitch;
261  q->qx = qx_roll * qi_pitch;
262  q->qy = qi_roll * qy_pitch;
263  q->qz = qx_roll * qy_pitch;
264 }

267 
268  // FIXME: remove me, do in quaternion directly
269  // is currently still needed, since the yaw setpoint integration is done in eulers
270 #if defined STABILIZATION_ATTITUDE_TYPE_INT
271  stabilization_attitude_read_rc_setpoint_eulers(&stab_att_sp_euler, in_flight);
272 #else

276  struct FloatQuat q_rp_cmd;
277  stabilization_attitude_read_rc_roll_pitch_quat_f(&q_rp_cmd);
278 
279  /* get current heading */
280  const struct FloatVect3 zaxis = {0., 0., 1.};
281  struct FloatQuat q_yaw;
282 
283  //Care Free mode
284  if (guidance_h_mode == GUIDANCE_H_MODE_CARE_FREE) {
285  //care_free_heading has been set to current psi when entering care free mode.
286  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw, zaxis, care_free_heading);
287  }
288  else {
289  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw, zaxis, stateGetNedToBodyEulers_f()->psi);
290  }
291 
292  /* roll/pitch commands applied to to current heading */
293  struct FloatQuat q_rp_sp;
294  FLOAT_QUAT_COMP(q_rp_sp, q_yaw, q_rp_cmd);
295  FLOAT_QUAT_NORMALIZE(q_rp_sp);
296 
297  if (in_flight)
298  {
299  /* get current heading setpoint */
300  struct FloatQuat q_yaw_sp;
301 #if defined STABILIZATION_ATTITUDE_TYPE_INT
302  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw_sp, zaxis, ANGLE_FLOAT_OF_BFP(stab_att_sp_euler.psi));
303 #else
304  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw_sp, zaxis, stab_att_sp_euler.psi);
305 #endif
306 
307  /* rotation between current yaw and yaw setpoint */
308  struct FloatQuat q_yaw_diff;
309  FLOAT_QUAT_COMP_INV(q_yaw_diff, q_yaw_sp, q_yaw);
310 
311  /* compute final setpoint with yaw */
312  FLOAT_QUAT_COMP_NORM_SHORTEST(*q_sp, q_rp_sp, q_yaw_diff);
313  } else {
314  QUAT_COPY(*q_sp, q_rp_sp);
315  }
316 }
