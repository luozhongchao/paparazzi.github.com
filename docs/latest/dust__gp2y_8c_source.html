58 void dust_gp2y_periodic( void ) {
59  if (dust_gp2y_status == DUST_GP2Y_IDLE) {
60  I2CReceive(GP2Y_I2C_DEV, gp2y_trans, GP2Y_SLAVE_ADDR, 2);
61  }
62  else if (dust_gp2y_status == DUST_GP2Y_UNINIT && cpu_time_sec > 1) {
63  dust_gp2y_status = DUST_GP2Y_IDLE;
64  }
65 }

67 void dust_gp2y_event( void ) {
68  if (gp2y_trans.status == I2CTransSuccess) {
69  /* read two byte particle density */
70  dust_gp2y_density = gp2y_trans.buf[0] << 8;
71  dust_gp2y_density |= gp2y_trans.buf[1];
72 
73  /* "just for reference and not for guarantee" */
74  dust_gp2y_density_f = ((dust_gp2y_density / 1024.) * 3.3 * (51. / 33.) - 0.6) * (0.5 / 3.);
75  if (dust_gp2y_density_f < 0)
76  dust_gp2y_density_f = 0;
77 
78  DOWNLINK_SEND_GP2Y_STATUS(DefaultChannel, DefaultDevice, &dust_gp2y_density, &dust_gp2y_density_f);
79 
80  gp2y_trans.status = I2CTransDone;
81  }
82 }
