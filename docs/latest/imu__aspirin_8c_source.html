76 static void send_i2c_msg_with_retry(struct i2c_transaction* t) {
77  uint8_t max_retry = 8;
78  uint8_t nb_retry = 0;
79  do {
80  i2c_submit(&i2c2, t);
81  while((I2C2_SR2 & I2C_SR2_BUSY));
82  //while(I2C_GetFlagStatus(I2C2, I2C_FLAG_BUSY));
83  while (t->status == I2CTransPending || t->status == I2CTransRunning);
84  if (t->status == I2CTransFailed)
85  nb_retry++;
86  }
87  while (t->status != I2CTransSuccess && nb_retry < max_retry);
88 }

158 /* sends a series of I2C commands to configure the ITG3200 gyro */
159 static void configure_gyro(void) {
160  /* set gyro range to 2000deg/s and low pass at 256Hz */
161  imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_DLPF_FS;
162  imu_aspirin.i2c_trans_gyro.buf[1] = (ITG3200_FS_SEL<<3)|(ITG3200_DLPF_CFG);
163  //I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
164  I2C2TransmitWithRetry(imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
165 
166  /* set sample rate to 533Hz */
167  imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_SMPLRT_DIV;
168  imu_aspirin.i2c_trans_gyro.buf[1] = ITG3200_SMPLRT_DIV;
169  //I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
170  I2C2TransmitWithRetry(imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
171 
172  /* switch to gyroX clock */
173  imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_PWR_MGM;
174  imu_aspirin.i2c_trans_gyro.buf[1] = ITG3200_CLK_SEL;
175  //I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
176  I2C2TransmitWithRetry(imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
177 
178  /* enable interrupt on data ready, idle high, latch until read any register */
179  imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_INT_CFG;
180  imu_aspirin.i2c_trans_gyro.buf[1] = (0x01 | (0x1<<4) | (0x1<<5) | 0x01<<7);
181  //I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
182  I2C2TransmitWithRetry(imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
183 }

194 static void configure_accel(void) {
195  /* set data rate to 800Hz and bandwidth to 400Hz (unless set otherwise) */
196  adxl345_write_to_reg(ADXL345_REG_BW_RATE, ASPIRIN_ACCEL_RATE);
197  /* switch to measurememnt mode */
198  adxl345_write_to_reg(ADXL345_REG_POWER_CTL, 1<<3);
199  /* enable data ready interrupt */
200  adxl345_write_to_reg(ADXL345_REG_INT_ENABLE, 1<<7);
201  /* Enable full res with +-16g range and interrupt active low */
202  adxl345_write_to_reg(ADXL345_REG_DATA_FORMAT, ADXL345_FULL_RES|ADXL345_RANGE_16G|ADXL345_INT_INVERT);
203 }
