00148 /* sends a serie of I2C commands to configure the ITG3200 gyro */
00149 static void configure_gyro(void) {
00150   /* set gyro range to 2000deg/s and low pass at 256Hz */
00151   imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_DLPF_FS;
00152   imu_aspirin.i2c_trans_gyro.buf[1] = (ITG3200_FS_SEL<<3)|(ITG3200_DLPF_CFG);
00153   I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
00154 
00155   /* set sample rate to 533Hz */
00156   imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_SMPLRT_DIV;
00157   imu_aspirin.i2c_trans_gyro.buf[1] = ITG3200_SMPLRT_DIV;
00158   I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
00159 
00160   /* switch to gyroX clock */
00161   imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_PWR_MGM;
00162   imu_aspirin.i2c_trans_gyro.buf[1] = ITG3200_CLK_SEL;
00163   I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
00164 
00165   /* enable interrupt on data ready, idle high, latch until read any register */
00166   imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_INT_CFG;
00167   imu_aspirin.i2c_trans_gyro.buf[1] = (0x01 | (0x1<<4) | (0x1<<5) | 0x01<<7);
00168   I2CTransmit(ITG3200_I2C_DEV, imu_aspirin.i2c_trans_gyro, ITG3200_I2C_ADDR, 2);
00169 }

00180 static void configure_accel(void) {
00181   /* set data rate to 800Hz and bandwidth to 400Hz (unless set otherwise) */
00182   adxl345_write_to_reg(ADXL345_REG_BW_RATE, ASPIRIN_ACCEL_RATE);
00183   /* switch to measurememnt mode */
00184   adxl345_write_to_reg(ADXL345_REG_POWER_CTL, 1<<3);
00185   /* enable data ready interrupt */
00186   adxl345_write_to_reg(ADXL345_REG_INT_ENABLE, 1<<7);
