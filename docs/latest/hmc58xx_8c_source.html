00070       hmc58xx_init_status++;

00073       hmc58xx_i2c_trans.buf[0] = HMC58XX_REG_MODE;

00109 void hmc58xx_event(void)

00122   }
00123   else if (!hmc58xx_initialized && hmc58xx_init_status != HMC_CONF_UNINIT) { // Configuring
00124     if (hmc58xx_i2c_trans.status == I2CTransSuccess || hmc58xx_i2c_trans.status == I2CTransDone) {
00125       hmc58xx_i2c_trans.status = I2CTransDone;
00126       hmc58xx_send_config();

00128     if (hmc58xx_i2c_trans.status == I2CTransFailed) {
00129       hmc58xx_init_status--;
00130       hmc58xx_i2c_trans.status = I2CTransDone;
