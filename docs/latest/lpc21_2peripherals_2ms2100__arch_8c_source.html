00035 void ms2100_arch_init( void ) {
00036 
00037   /* configure RESET pin */
00038   Ms2100Reset();                                /* pin idles low  */
00039   SetBit(MS2100_RESET_IODIR, MS2100_RESET_PIN); /* pin is output  */
00040 
00041   /* configure DRDY pin */
00042   /* connected pin to EXINT */
00043   MS2100_DRDY_PINSEL |= MS2100_DRDY_PINSEL_VAL << MS2100_DRDY_PINSEL_BIT;
00044   SetBit(EXTMODE, MS2100_DRDY_EINT);  /* EINT is edge trigered */
00045   SetBit(EXTPOLAR, MS2100_DRDY_EINT); /* EINT is trigered on rising edge */
00046   SetBit(EXTINT, MS2100_DRDY_EINT);   /* clear pending EINT */
00047 
00048   /* initialize interrupt vector */
00049   VICIntSelect &= ~VIC_BIT( MS2100_DRDY_VIC_IT ); /* select EINT as IRQ source */
00050   VICIntEnable = VIC_BIT( MS2100_DRDY_VIC_IT );   /* enable it                 */
00051   _VIC_CNTL(MS2100_DRDY_VIC_SLOT) = VIC_ENABLE | MS2100_DRDY_VIC_IT;
00052   _VIC_ADDR(MS2100_DRDY_VIC_SLOT) = (uint32_t)EXTINT_ISR;         // address of the ISR
00053 
00054 }

00056 void EXTINT_ISR(void) {
00057   ISR_ENTRY();
00058   /* no, we won't do anything asynchronously, so just notify */
00059   ms2100_status = MS2100_GOT_EOC;
00060   /* clear EINT */
00061   EXTINT = (1<<MS2100_DRDY_EINT);
00062   VICVectAddr = 0x00000000;    /* clear this interrupt from the VIC */
00063   ISR_EXIT();
00064 }

00066 void ms2100_reset_cb( struct spi_transaction * t __attribute__ ((unused)) ) {
00067   // set RESET pin high for at least 100 nsec
00068   // busy wait should not harm
00069   // storing start and dt is probably long enough...
00070   Ms2100Set();
00071   uint32_t start = T0TC;
00072   uint32_t dt = CPU_TICKS_OF_NSEC(110);
00073   while ((uint32_t)(T0TC - start) < dt);
00074   Ms2100Reset();
00075 }
