00118 void gps_ubx_ucenter_event(void)
00119 {
00120   // Save processing time inflight
00121   if (gps_ubx_ucenter.status == GPS_UBX_UCENTER_STATUS_STOPPED)

00123 
00124   // Read Configuration Reply's
00125   if (gps_ubx.msg_class == UBX_ACK_ID) {

00129     else
00130     {
00131       gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NACK;
00132     }
00133   }
00134   // Version info
00135   else if (gps_ubx.msg_class == UBX_MON_ID) {
00136     if (gps_ubx.msg_id == UBX_MON_VER_ID ) {
00137       gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_VERSION;
00138       gps_ubx_ucenter.sw_ver_h = UBX_MON_VER_c(gps_ubx.msg_buf,0) - '0';
00139       gps_ubx_ucenter.sw_ver_l = 10*(UBX_MON_VER_c(gps_ubx.msg_buf,2) - '0');
00140       gps_ubx_ucenter.sw_ver_l += UBX_MON_VER_c(gps_ubx.msg_buf,3) - '0';
00141       gps_ubx_ucenter.hw_ver_h = UBX_MON_VER_c(gps_ubx.msg_buf,33) - '0';
00142       gps_ubx_ucenter.hw_ver_h += 10*(UBX_MON_VER_c(gps_ubx.msg_buf,32) - '0');
00143       gps_ubx_ucenter.hw_ver_l = UBX_MON_VER_c(gps_ubx.msg_buf,37) - '0';
00144       gps_ubx_ucenter.hw_ver_l += 10*(UBX_MON_VER_c(gps_ubx.msg_buf,36) - '0');

00155 static bool_t gps_ubx_ucenter_autobaud(uint8_t nr)
00156 {
00157   switch (nr)
00158   {
00159   case 0:
00160   case 1:
00161     // Very important for some modules:
00162     // Give the GPS some time to boot (up to 0.75 second)
00163     break;
00164   case 2:
00165     gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
00166     GpsUartSetBaudrate(B38400); // Try the most common first?
00167     UbxSend_CFG_MSG(UBX_NAV_ID, UBX_NAV_VELNED_ID, 0, 1, 0, 0);
00168     break;
00169   case 3:
00170     if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK)
00171     {
00172       gps_ubx_ucenter.baud_init = 38400;
00173       return FALSE;
00174     }
00175     gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
00176     GpsUartSetBaudrate(B9600); // Maybe the factory default?
00177     UbxSend_CFG_MSG(UBX_NAV_ID, UBX_NAV_VELNED_ID, 0, 1, 0, 0);
00178     break;
00179   case 4:
00180     if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK)
00181     {
00182       gps_ubx_ucenter.baud_init = 9600;
00183       return FALSE;
00184     }
00185     gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
00186     GpsUartSetBaudrate(B57600); // The high-rate default?
00187     UbxSend_CFG_MSG(UBX_NAV_ID, UBX_NAV_VELNED_ID, 0, 1, 0, 0);
00188     break;
00189   case 5:
00190     if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK)
00191     {
00192       gps_ubx_ucenter.baud_init = 57600;
00193       return FALSE;
00194      }
00195     gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
00196     GpsUartSetBaudrate(B4800); // Default NMEA baudrate finally?
00197     UbxSend_CFG_MSG(UBX_NAV_ID, UBX_NAV_VELNED_ID, 0, 1, 0, 0);
00198     break;
00199   case 6:
00200     if (gps_ubx_ucenter.reply == GPS_UBX_UCENTER_REPLY_ACK)
00201     {
00202       gps_ubx_ucenter.baud_init = 4800;
00203       return FALSE;
00204      }
00205 
00206     // Autoconfig Failed... let's setup the failsafe baudrate
00207     // Should we try even a different baudrate?
00208     gps_ubx_ucenter.baud_init = 0;
00209     GpsUartSetBaudrate(B9600);
00210     return FALSE;
00211   default:
00212     break;
00213   }
00214   return TRUE;
00215 }

00328 static bool_t gps_ubx_ucenter_configure(uint8_t nr)
00329 {
00330   // Store the reply of the last configuration step and reset
00331   if (nr < GPS_UBX_UCENTER_CONFIG_STEPS)
00332     gps_ubx_ucenter.replies[nr] = gps_ubx_ucenter.reply;
00333 
00334   gps_ubx_ucenter.reply = GPS_UBX_UCENTER_REPLY_NONE;
00335 
00336   switch (nr) {
00337   case 0:
00338     UbxSend_MON_GET_VER();
00339     break;
00340   case 1:
00341   case 2:
00342   case 3:
00343   case 4:
00344     // UBX_G5010 takes 0.7 seconds to answer a firmware request
00345     // Version info is important for porper configuration as different firmwares have different settings
00346     break;
00347   case 5:
00348     // Send some debugging info: detected baudrate, software version etc...
00349     gps_ubx_ucenter.replies[0] = (gps_ubx_ucenter.baud_init/1000);
00350     gps_ubx_ucenter.replies[1] = (gps_ubx_ucenter.baud_init - 1000 * gps_ubx_ucenter.replies[0]) / 100;
00351     gps_ubx_ucenter.replies[2] = gps_ubx_ucenter.sw_ver_h;
00352     gps_ubx_ucenter.replies[3] = gps_ubx_ucenter.sw_ver_l;
00353     gps_ubx_ucenter.replies[4] = gps_ubx_ucenter.hw_ver_h;
00354     gps_ubx_ucenter.replies[5] = gps_ubx_ucenter.hw_ver_l;
00355     DOWNLINK_SEND_DEBUG(DefaultChannel, DefaultDevice,6,gps_ubx_ucenter.replies);
00356 
00358     // Actual configuration start
00359 
00360     // Use old baudrate to issue a baudrate change command
00361     gps_ubx_ucenter_config_port();
00362     break;
00363   case 6:
00364     // Now the GPS baudrate should have changed
00365     GpsUartSetBaudrate(B38400);
00366     gps_ubx_ucenter.baud_run = 38400;
00367     gps_ubx_ucenter_config_nav();
00368     break;
00369   case 7:
00370     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_POSLLH_ID,1);
00371     break;
00372   case 8:
00373     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_VELNED_ID, 1);
00374     break;
00375   case 9:
00376     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_STATUS_ID, 1);
00377     break;
00378   case 10:
00379     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_SVINFO_ID, 4);
00380     break;
00381   case 11:
00382 #if defined FIRMWARE && FIRMWARE == ROTORCRAFT
00383     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_SOL_ID, 1);
00384 #else
00385     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_SOL_ID, 8);
00386 #endif
00387     break;
00388   case 12:
00389     // Disable UTM on old Lea4P
00390     gps_ubx_ucenter_enable_msg(UBX_NAV_ID, UBX_NAV_POSUTM_ID, 0);
00391     break;
00392   case 13:
00393     gps_ubx_ucenter_config_sbas();
00394     break;
00395   case 14:
00396     UbxSend_CFG_RATE(0x00FA, 0x0001, 0x0000);
00397     break;
00398   case 15:
00399     // Try to save on non-ROM devices...
00400     UbxSend_CFG_CFG(0x00000000,0xffffffff,0x00000000);
00401     break;
00402   case 16:
00403     // Debug Downlink the result of all configuration steps: see messages
00404     DOWNLINK_SEND_DEBUG(DefaultChannel, DefaultDevice,GPS_UBX_UCENTER_CONFIG_STEPS,gps_ubx_ucenter.replies);
00405     return FALSE;
00406   default:
00407     break;
00408   }
00409   return TRUE; // Continue, except for the last case
00410 }
