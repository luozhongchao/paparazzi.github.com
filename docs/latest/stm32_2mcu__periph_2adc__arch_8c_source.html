214 /* Configure and enable RCC for peripherals (ADC1, ADC2, Timer) */
215 static inline void adc_init_rcc( void )
216 {
217 #if defined (USE_AD1) || defined (USE_AD2)
218  uint32_t timer;
219  volatile uint32_t *rcc_apbenr;
220  uint32_t rcc_apb;
221 #if defined(USE_AD_TIM4)
222  timer = TIM4;
223  rcc_apbenr = &RCC_APB1ENR;
224  rcc_apb = RCC_APB1ENR_TIM4EN;
225 #elif defined(USE_AD_TIM1)
226  timer = TIM1;
227  rcc_apbenr = &RCC_APB2ENR;
228  rcc_apb = RCC_APB2ENR_TIM1EN;
229 #else
230  timer = TIM2;
231  rcc_apbenr = &RCC_APB1ENR;
232  rcc_apb = RCC_APB1ENR_TIM2EN;
233 #endif
234 
235  /*
236  * Historic Note:
237  * Previously in libstm32 we were setting the ADC clock here.
238  * It was being set to PCLK2 DIV2 resulting in 36MHz clock on the ADC. I am
239  * pretty sure that this is wrong as based on the datasheet the ADC clock
240  * must not exceed 14MHz! Now the clock is being set by the clock init
241  * routine in libopencm3 so we don't have to set up this clock ourselves any
242  * more. This comment is here just as a reminder and may be removed in the
243  * future when we know that everything is working properly.
244  * (by Esden the historian :D)
245  */
246 
247  /* Timer peripheral clock enable. */
248  rcc_peripheral_enable_clock(rcc_apbenr, rcc_apb);
249  /* GPIO peripheral clock enable. */
250  rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_IOPBEN |
251  RCC_APB2ENR_IOPCEN);
252 
253  /* Enable ADC peripheral clocks. */
254 #ifdef USE_AD1
255  rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC1EN);
256 #endif
257 #ifdef USE_AD2
258  rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC2EN);
259 #endif
260 
261  /* Time Base configuration */
262  timer_reset(timer);
263  timer_set_mode(timer, TIM_CR1_CKD_CK_INT,
264  TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
265  timer_set_period(timer, 0xFF);
266  timer_set_prescaler(timer, 0x8);
267  timer_set_clock_division(timer, 0x0);
268  /* Generate TRGO on every update. */
269  timer_set_master_mode(timer, TIM_CR2_MMS_UPDATE);
270  timer_enable_counter(timer);
271 
272 #endif // defined (USE_AD1) || defined (USE_AD2)
273 }

291 static inline void adc_init_single(uint32_t adc,
292  uint8_t chan1, uint8_t chan2,
293  uint8_t chan3, uint8_t chan4)
294 {
295  uint8_t num_channels, rank;
296  uint8_t channels[4];
297 
298  // Paranoia, must be down for 2+ ADC clock cycles before calibration
299  adc_off(adc);
300 
301  /* enable adc clock */
302  if (adc == ADC1) {
303 #ifdef USE_AD1
304  num_channels = NB_ADC1_CHANNELS;
305  ADC1_GPIO_INIT();
306 #endif
307  }
308  else if (adc == ADC2) {
309 #ifdef USE_AD2
310  num_channels = NB_ADC2_CHANNELS;
311  ADC2_GPIO_INIT();
312 #endif
313  }
314 
315  /* Configure ADC */
316 
317  /* Explicitly setting most registers, reset/default values are correct for most */
318 
319  /* Set CR1 register. */
320 
321  /* Clear AWDEN */
322  adc_disable_analog_watchdog_regular(adc);
323  /* Clear JAWDEN */
324  adc_disable_analog_watchdog_injected(adc);
325  /* Clear DISCEN */
326  adc_disable_discontinuous_mode_regular(adc);
327  /* Clear JDISCEN */
328  adc_disable_discontinuous_mode_injected(adc);
329  /* Clear JAUTO */
330  adc_disable_automatic_injected_group_conversion(adc);
331  /* Set SCAN */
332  adc_enable_scan_mode(adc);
333  /* Enable ADC<X> JEOC interrupt (Set JEOCIE) */
334  adc_enable_eoc_interrupt_injected(adc);
335  /* Clear AWDIE */
336  adc_disable_awd_interrupt(adc);
337  /* Clear EOCIE */
338  adc_disable_eoc_interrupt(adc);
339 
340  /* Set CR2 register. */
341 
342  /* Clear TSVREFE */
343  adc_disable_temperature_sensor(adc);
344  /* Clear EXTTRIG */
345  adc_disable_external_trigger_regular(adc);
346  /* Clear ALIGN */
347  adc_set_right_aligned(adc);
348  /* Clear DMA */
349  adc_disable_dma(adc);
350  /* Clear CONT */
351  adc_set_single_conversion_mode(adc);
352 
353  rank = 0;
354  if (chan1) {
355  adc_set_sample_time(adc, adc_channel_map[0], ADC_SMPR1_SMP_41DOT5CYC);
356  channels[rank] = adc_channel_map[0];
357  rank++;
358  }
359  if (chan2) {
360  adc_set_sample_time(adc, adc_channel_map[1], ADC_SMPR1_SMP_41DOT5CYC);
361  channels[rank] = adc_channel_map[1];
362  rank++;
363  }
364  if (chan3) {
365  adc_set_sample_time(adc, adc_channel_map[2], ADC_SMPR1_SMP_41DOT5CYC);
366  channels[rank] = adc_channel_map[2];
367  rank++;
368  }
369  if (chan4) {
370  adc_set_sample_time(adc, adc_channel_map[3], ADC_SMPR1_SMP_41DOT5CYC);
371  channels[rank] = adc_channel_map[3];
372  }
373 
374  adc_set_injected_sequence(adc, num_channels, channels);
375 
376 #if USE_AD_TIM4
377 PRINT_CONFIG_MSG("Info: Using TIM4 for ADC")
378  adc_enable_external_trigger_injected(adc, ADC_CR2_JEXTSEL_TIM4_TRGO);
379 #elif USE_AD_TIM1
380 PRINT_CONFIG_MSG("Info: Using TIM1 for ADC")
381  adc_enable_external_trigger_injected(adc, ADC_CR2_JEXTSEL_TIM1_TRGO);
382 #else
383 PRINT_CONFIG_MSG("Info: Using default TIM2 for ADC")
384  adc_enable_external_trigger_injected(adc, ADC_CR2_JEXTSEL_TIM2_TRGO);
385 #endif
386 
387  /* Enable ADC<X> */
388  adc_power_on(adc);
389 
390  /* Enable ADC<X> reset calibaration register */
391  adc_reset_calibration(adc);
392  /* Check the end of ADC<X> reset calibration */
393  while ((ADC_CR2(adc) & ADC_CR2_RSTCAL) != 0);
394  /* Start ADC<X> calibaration */
395  adc_calibration(adc);
396  /* Check the end of ADC<X> calibration */
397  while ((ADC_CR2(adc) & ADC_CR2_CAL) != 0);
398 
399 } // adc_init_single

402 void adc_init( void ) {
403 
404  /* initialize buffer pointers with 0 (not set).
405  buffer null pointers will be ignored in interrupt
406  handler, which is important as there are no
407  buffers registered at the time the ADC trigger
408  interrupt is enabled.
409  */
410  uint8_t channel;
411 #ifdef USE_AD1
412  for(channel = 0; channel < NB_ADC1_CHANNELS; channel++)
413  adc1_buffers[channel] = NULL;
414  adc_injected_channels[0] = &ADC_JDR1(ADC1);
415  adc_injected_channels[1] = &ADC_JDR2(ADC1);
416  adc_injected_channels[2] = &ADC_JDR3(ADC1);
417  adc_injected_channels[3] = &ADC_JDR4(ADC1);
418 #endif
419 #ifdef USE_AD2
420  for(channel = 0; channel < NB_ADC2_CHANNELS; channel++)
421  adc2_buffers[channel] = NULL;
422  adc_injected_channels[0] = &ADC_JDR1(ADC2);
423  adc_injected_channels[1] = &ADC_JDR2(ADC2);
424  adc_injected_channels[2] = &ADC_JDR3(ADC2);
425  adc_injected_channels[3] = &ADC_JDR4(ADC2);
426 #endif
427 
428  adc_new_data_trigger = FALSE;
429  adc_channel_map[0] = BOARD_ADC_CHANNEL_1;
430  adc_channel_map[1] = BOARD_ADC_CHANNEL_2;
431  adc_channel_map[2] = BOARD_ADC_CHANNEL_3;
432  adc_channel_map[3] = BOARD_ADC_CHANNEL_4;
433 
434  adc_init_rcc();
435  adc_init_irq();

490 static inline void adc_push_sample(struct adc_buf * buf, uint16_t value) {
491  uint8_t new_head = buf->head + 1;
492 
493  if (new_head >= buf->av_nb_sample) {
494  new_head = 0;
495  }
496  buf->sum -= buf->values[new_head];
497  buf->values[new_head] = value;
498  buf->sum += value;
499  buf->head = new_head;
500 }

505 void adc1_2_isr(void)
506 {
507  uint8_t channel = 0;
508  uint16_t value = 0;
509  struct adc_buf * buf;
510 
511 #ifdef USE_AD1
512  // Clear Injected End Of Conversion
513  ADC_SR(ADC1) &= ~ADC_SR_JEOC;
514  for(channel = 0; channel < NB_ADC1_CHANNELS; channel++) {
515  buf = adc1_buffers[channel];
516  if(buf) {
517  value = *adc_injected_channels[channel];
518  adc_push_sample(buf, value);
519  }
520  }
521  adc_new_data_trigger = 1;
522 #endif
523 #ifdef USE_AD2
524  // Clear Injected End Of Conversion
525  ADC_SR(ADC2) &= ~ADC_SR_JEOC;
526  for(channel = 0; channel < NB_ADC2_CHANNELS; channel++) {
527  buf = adc2_buffers[channel];
528  if(buf) {
529  value = *adc_injected_channels[channel];
530  adc_push_sample(buf, value);
531  }
532  }
533  adc_new_data_trigger = 1;
534 #endif
