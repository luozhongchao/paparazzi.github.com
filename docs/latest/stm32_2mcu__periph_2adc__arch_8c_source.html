00291                    uint8_t chan1, uint8_t chan2,
00292                    uint8_t chan3, uint8_t chan4)
00293 {
00294     uint8_t num_channels, rank;
00295     uint8_t channels[4];
00296 
00297     // Paranoia, must be down for 2+ ADC clock cycles before calibration
00298     adc_off(adc);
00299 
00300     /* enable adc clock */
00301     if (adc == ADC1) {
00302 #ifdef USE_AD1
00303         num_channels = NB_ADC1_CHANNELS;
00304         ADC1_GPIO_INIT();
00305 #endif
00306     }
00307     else if (adc == ADC2) {
00308 #ifdef USE_AD2
00309         num_channels = NB_ADC2_CHANNELS;
00310         ADC2_GPIO_INIT();
00311 #endif
00312     }
00313 
00314     /* Configure ADC */
00315     //adc_reset(adc);
00316 
00317     /* XXX: This register fiddeling code should be moved to libopencm3! */
00318     /* Set CR1 register. */
00319     {
00320       uint32_t tmpreg = ADC_CR1(adc);
00321 
00322       /* Clear DUALMOD and SCAN. */
00323       tmpreg &= ~(ADC_CR1_DUALMOD_MASK | ADC_CR1_SCAN);
00324 
00325       /* Set correct DUALMOD and SCAN. */
00326       tmpreg |= ADC_CR1_DUALMOD_IND | ADC_CR1_SCAN;
00327 
00328       ADC_CR1(adc) = tmpreg;
00329     }
00330 
00331     /* Set CR2 register. */
00332     {
00333       uint32_t tmpreg = ADC_CR2(adc);
00334 
00335       /* Clear CONT, ALIGN and EXTSEL. */
00336       tmpreg &= ~(ADC_CR2_CONT | ADC_CR2_ALIGN | ADC_CR2_EXTSEL_MASK);
00337 
00338       /* Set correct CONT, ALIGN and EXTSEL. */
00339       tmpreg |= ADC_CR2_EXTSEL_SWSTART;
00340 
00341       ADC_CR2(adc) = tmpreg;
00342     }
00343 
00344     /* Set SQR1 register. */
00345     {
00346       uint32_t tmpreg = ADC_SQR1(adc);
00347 
00348       /* Clear regular channel sequence length. */
00349       tmpreg &= ~(0xF); //~(ADC_SQR1_L_MSK);
00350 
00351       /* Set regular channel sequence length. */
00352       tmpreg |= 0;
00353 
00354       ADC_SQR1(adc) = tmpreg;
00355     }
00356 
00357     /* Set JSQR1 injected sequence length. */
00358     {
00359       uint32_t tmpreg = ADC_JSQR(adc);
00360 
00361       /* Clear injected channel sequence length. */
00362       tmpreg &= ~(0x2 << 20); //~(ADC_JSQR_JL_MSK);
00363 
00364       /* Set injected channel sequence length. */
00365       tmpreg |= (num_channels << ADC_JSQR_JL_LSB);
00366 
00367       ADC_JSQR(adc) = tmpreg;
00368     }
00369 
00370     rank = 0;
00371     if (chan1) {
00372   adc_set_conversion_time(adc, adc_channel_map[0], ADC_SMPR1_SMP_41DOT5CYC);
00373   channels[rank] = adc_channel_map[0];
00374         rank++;
00375     }
00376     if (chan2) {
00377   adc_set_conversion_time(adc, adc_channel_map[1], ADC_SMPR1_SMP_41DOT5CYC);
00378   channels[rank] = adc_channel_map[1];
00379         rank++;
00380     }
00381     if (chan3) {
00382   adc_set_conversion_time(adc, adc_channel_map[2], ADC_SMPR1_SMP_41DOT5CYC);
00383   channels[rank] = adc_channel_map[2];
00384         rank++;
00385     }
00386     if (chan4) {
00387   adc_set_conversion_time(adc, adc_channel_map[3], ADC_SMPR1_SMP_41DOT5CYC);
00388   channels[rank] = adc_channel_map[3];
00389     }
00390 
00391     adc_set_injected_sequence(adc, num_channels, channels);
00392 
00393 #if defined(USE_AD_TIM4)
00394     adc_enable_external_trigger_injected(adc, ADC_CR2_JEXTSEL_TIM4_TRGO);
00395 #elif defined(USE_AD_TIM1)
00396     adc_enable_external_trigger_injected(adc, ADC_CR2_JEXTSEL_TIM1_TRGO);
00397 #else
00398     adc_enable_external_trigger_injected(adc, ADC_CR2_JEXTSEL_TIM2_TRGO);
00399 #endif
00400 
00401     /* Enable ADC<X> JEOC interrupt */
00402     adc_enable_jeoc_interrupt(adc);
00403 
00404     /* Enable ADC<X> */
00405     adc_on(adc);
00406 
00407     /* Enable ADC<X> reset calibaration register */
00408     adc_reset_calibration(adc);
00409 
00410     /* Check the end of ADC<X> reset calibration */
00411     while ((ADC_CR2(adc) & ADC_CR2_RSTCAL) != 0);
00412     /* Start ADC<X> calibaration */
00413     adc_calibration(adc);
00414     /* Check the end of ADC<X> calibration */
00415     while ((ADC_CR2(adc) & ADC_CR2_CAL) != 0);

00511 static inline void adc_push_sample(struct adc_buf * buf, uint16_t value) {
00512     uint8_t new_head = buf->head + 1;
00513 
00514     if (new_head >= buf->av_nb_sample) { new_head = 0; }
00515     buf->sum -= buf->values[new_head];
00516     buf->values[new_head] = value;
00517     buf->sum += value;
00518     buf->head = new_head;
00519 }
