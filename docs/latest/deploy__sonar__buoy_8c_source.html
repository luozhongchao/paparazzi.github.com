00038 void deploy_sonar_buoy_init(void) {
00039   /* initialise peripheral clock for port C */
00040   RCC_APB2PeriphClockCmd( RCC_APB2Periph_GPIOC, ENABLE);
00041   GPIO_InitTypeDef GPIO_InitStructure;
00042   GPIO_InitStructure.GPIO_Pin = GPIO_Pin_5;
00043   GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
00044   GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
