82 void imu_impl_init(void)
83 {
84  imu_aspirin.accel_valid = FALSE;
85  imu_aspirin.gyro_valid = FALSE;
86  imu_aspirin.mag_valid = FALSE;
87 
88  /* Set accel configuration */
89  adxl345_i2c_init(&imu_aspirin.acc_adxl, &(ASPIRIN_I2C_DEV), ADXL345_ADDR);
90  // set the data rate
91  imu_aspirin.acc_adxl.config.rate = ASPIRIN_ACCEL_RATE;
93  //imu_aspirin.acc_adxl.config.drdy_int_enable = TRUE;
94 
95  // With CS tied high to VDD I/O, the ADXL345 is in I2C mode
96  GPIO_ARCH_SET_SPI_CS_HIGH();
97 
98  /* Gyro configuration and initalization */
99  itg3200_init(&imu_aspirin.gyro_itg, &(ASPIRIN_I2C_DEV), ITG3200_ADDR);
100  /* change the default config */
101  // Aspirin sample rate divider
102  imu_aspirin.gyro_itg.config.smplrt_div = ASPIRIN_GYRO_SMPLRT_DIV;
103  // digital low pass filter
104  imu_aspirin.gyro_itg.config.dlpf_cfg = ASPIRIN_GYRO_LOWPASS;
105 
107  /* interrupt on data ready, idle high, latch until read any register */
108  //itg_conf.int_cfg = (0x01 | (0x1<<4) | (0x1<<5) | 0x01<<7);
109 
110  /* initialize mag and set default options */
111  hmc58xx_init(&imu_aspirin.mag_hmc, &(ASPIRIN_I2C_DEV), HMC58XX_ADDR);
112 #ifdef IMU_ASPIRIN_VERSION_1_0
113  imu_aspirin.mag_hmc.type = HMC_TYPE_5843;
114 #endif
115 }

118 void imu_periodic(void)
119 {
120  adxl345_i2c_periodic(&imu_aspirin.acc_adxl);
121 
122  // Start reading the latest gyroscope data
123  itg3200_periodic(&imu_aspirin.gyro_itg);
124 
125  // Read HMC58XX at 50Hz (main loop for rotorcraft: 512Hz)
126  RunOnceEvery(10, hmc58xx_periodic(&imu_aspirin.mag_hmc));
127 }

129 void imu_aspirin_i2c_event(void)
130 {
131  adxl345_i2c_event(&imu_aspirin.acc_adxl);
132  if (imu_aspirin.acc_adxl.data_available) {
133  VECT3_COPY(imu.accel_unscaled, imu_aspirin.acc_adxl.data.vect);
134  imu_aspirin.acc_adxl.data_available = FALSE;
135  imu_aspirin.accel_valid = TRUE;
136  }
137 
138  /* If the itg3200 I2C transaction has succeeded: convert the data */
139  itg3200_event(&imu_aspirin.gyro_itg);
140  if (imu_aspirin.gyro_itg.data_available) {
141  RATES_COPY(imu.gyro_unscaled, imu_aspirin.gyro_itg.data.rates);
142  imu_aspirin.gyro_itg.data_available = FALSE;
143  imu_aspirin.gyro_valid = TRUE;
144  }
145 
146  /* HMC58XX event task */
147  hmc58xx_event(&imu_aspirin.mag_hmc);
148  if (imu_aspirin.mag_hmc.data_available) {
149 #ifdef IMU_ASPIRIN_VERSION_1_0
150  VECT3_COPY(imu.mag_unscaled, imu_aspirin.mag_hmc.data.vect);
151 #else // aspirin 1.5 with hmc5883
152  imu.mag_unscaled.x = imu_aspirin.mag_hmc.data.vect.y;
153  imu.mag_unscaled.y = -imu_aspirin.mag_hmc.data.vect.x;
154  imu.mag_unscaled.z = imu_aspirin.mag_hmc.data.vect.z;
155 #endif
156  imu_aspirin.mag_hmc.data_available = FALSE;
157  imu_aspirin.mag_valid = TRUE;
158  }
