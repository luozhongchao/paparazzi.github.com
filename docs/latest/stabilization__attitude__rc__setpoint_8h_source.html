00072 static inline void stabilization_attitude_read_rc_setpoint_eulers(struct Int32Eulers *sp, bool_t in_flight) {
00073 
00074   sp->phi = ((int32_t) radio_control.values[RADIO_ROLL]  * SP_MAX_PHI / MAX_PPRZ);
00075   sp->theta = ((int32_t) radio_control.values[RADIO_PITCH] * SP_MAX_THETA / MAX_PPRZ);
00076 
00077   if (in_flight) {
00078     if (YAW_DEADBAND_EXCEEDED()) {
00079       sp->psi += ((int32_t) radio_control.values[RADIO_YAW] * SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ);
00080       INT32_ANGLE_NORMALIZE(sp->psi);
00081     }
00082 #ifdef STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT
00083     // Make sure the yaw setpoint does not differ too much from the real yaw to prevent a sudden switch at 180 deg
00084     int32_t delta_psi = sp->psi - stateGetNedToBodyEulers_i()->psi;
00085     int32_t delta_limit = ANGLE_BFP_OF_REAL(STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT);
00086     INT32_ANGLE_NORMALIZE(delta_psi);
00087     if (delta_psi > delta_limit){
00088       sp->psi = stateGetNedToBodyEulers_i()->psi + delta_limit;
00089     }
00090     else if (delta_psi < -delta_limit){
00091       sp->psi = stateGetNedToBodyEulers_i()->psi - delta_limit;
00092     }
00093     INT32_ANGLE_NORMALIZE(sp->psi);
00094 #endif
00095   }
00096   else { /* if not flying, use current yaw as setpoint */
00097     sp->psi = stateGetNedToBodyEulers_i()->psi;
00098   }
00099 }

00134 static inline void stabilization_attitude_read_rc_roll_pitch_quat_f(struct FloatQuat* q) {
00135   q->qx = radio_control.values[RADIO_ROLL] * STABILIZATION_ATTITUDE_SP_MAX_PHI / MAX_PPRZ / 2;
00136   q->qy = radio_control.values[RADIO_PITCH] * STABILIZATION_ATTITUDE_SP_MAX_THETA / MAX_PPRZ / 2;
00137   q->qz = 0.0;
00138 
00139   /* normalize */
00140   float norm = sqrtf(1.0 + SQUARE(q->qx)+ SQUARE(q->qy));
00141   q->qi = 1.0 / norm;
00142   q->qx /= norm;
00143   q->qy /= norm;
00144 }

00150 static inline void stabilization_attitude_read_rc_roll_pitch_earth_quat_f(struct FloatQuat* q) {
00151   /* only non-zero entries for roll quaternion */
00152   float roll2 = radio_control.values[RADIO_ROLL] * STABILIZATION_ATTITUDE_SP_MAX_PHI / MAX_PPRZ / 2;
00153   float qx_roll = sinf(roll2);
00154   float qi_roll = cosf(roll2);
00155 
00156   /* only non-zero entries for pitch quaternion */
00157   float pitch2 = radio_control.values[RADIO_PITCH] * STABILIZATION_ATTITUDE_SP_MAX_THETA / MAX_PPRZ / 2;
00158   float qy_pitch = sinf(pitch2);
00159   float qi_pitch = cosf(pitch2);
00160 
00161   /* only multiply non-zero entries of FLOAT_QUAT_COMP(*q, q_roll, q_pitch) */
00162   q->qi = qi_roll * qi_pitch;
00163   q->qx = qx_roll * qi_pitch;
00164   q->qy = qi_roll * qy_pitch;
00165   q->qz = qx_roll * qy_pitch;
00166 }
