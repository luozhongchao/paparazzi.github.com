72 static inline void stabilization_attitude_read_rc_setpoint_eulers(struct Int32Eulers *sp, bool_t in_flight) {
73 
74  sp->phi = ((int32_t) radio_control.values[RADIO_ROLL] * SP_MAX_PHI / MAX_PPRZ);
75  sp->theta = ((int32_t) radio_control.values[RADIO_PITCH] * SP_MAX_THETA / MAX_PPRZ);
76 
77  if (in_flight) {
78  if (YAW_DEADBAND_EXCEEDED()) {
79  sp->psi += ((int32_t) radio_control.values[RADIO_YAW] * SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ);
80  INT32_ANGLE_NORMALIZE(sp->psi);
81  }
82 #ifdef STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT
83  // Make sure the yaw setpoint does not differ too much from the real yaw to prevent a sudden switch at 180 deg
84  int32_t delta_psi = sp->psi - stateGetNedToBodyEulers_i()->psi;
85  int32_t delta_limit = ANGLE_BFP_OF_REAL(STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT);
86  INT32_ANGLE_NORMALIZE(delta_psi);
87  if (delta_psi > delta_limit){
88  sp->psi = stateGetNedToBodyEulers_i()->psi + delta_limit;
89  }
90  else if (delta_psi < -delta_limit){
91  sp->psi = stateGetNedToBodyEulers_i()->psi - delta_limit;
92  }
93  INT32_ANGLE_NORMALIZE(sp->psi);
94 #endif
95  }
96  else { /* if not flying, use current yaw as setpoint */
97  sp->psi = stateGetNedToBodyEulers_i()->psi;
98  }
99 }

103  sp->phi = (radio_control.values[RADIO_ROLL] * SP_MAX_PHI / MAX_PPRZ);
104  sp->theta = (radio_control.values[RADIO_PITCH] * SP_MAX_THETA / MAX_PPRZ);
105 
106  if (in_flight) {
107  if (YAW_DEADBAND_EXCEEDED()) {
108  sp->psi += (radio_control.values[RADIO_YAW] * SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ);
109  FLOAT_ANGLE_NORMALIZE(sp->psi);
110  }
111 #ifdef STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT
112  // Make sure the yaw setpoint does not differ too much from the real yaw to prevent a sudden switch at 180 deg
113  float delta_psi = sp->psi - stateGetNedToBodyEulers_f()->psi;
114  FLOAT_ANGLE_NORMALIZE(delta_psi);
115  if (delta_psi > STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT){
116  sp->psi = stateGetNedToBodyEulers_f()->psi + STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT;
117  }
118  else if (delta_psi < -STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT){
119  sp->psi = stateGetNedToBodyEulers_f()->psi - STABILIZATION_ATTITUDE_SP_PSI_DELTA_LIMIT;
120  }
121  FLOAT_ANGLE_NORMALIZE(sp->psi);
122 #endif
123  }
124  else { /* if not flying, use current yaw as setpoint */
125  sp->psi = stateGetNedToBodyEulers_f()->psi;
126  }
127 }

130 /** Read roll/pitch command from RC as quaternion.
131  * Interprets the stick positions as axes.
132  * @param[out] q quaternion representing the RC roll/pitch input
133  */
134 static inline void stabilization_attitude_read_rc_roll_pitch_quat_f(struct FloatQuat* q) {
135  q->qx = radio_control.values[RADIO_ROLL] * STABILIZATION_ATTITUDE_SP_MAX_PHI / MAX_PPRZ / 2;
136  q->qy = radio_control.values[RADIO_PITCH] * STABILIZATION_ATTITUDE_SP_MAX_THETA / MAX_PPRZ / 2;
137  q->qz = 0.0;
138 
139  /* normalize */
140  float norm = sqrtf(1.0 + SQUARE(q->qx)+ SQUARE(q->qy));
141  q->qi = 1.0 / norm;
142  q->qx /= norm;
143  q->qy /= norm;
144 }

146 /** Read roll/pitch command from RC as quaternion.
147  * Both angles are are interpreted relative to to the horizontal plane (earth bound).
148  * @param[out] q quaternion representing the RC roll/pitch input
149  */
150 static inline void stabilization_attitude_read_rc_roll_pitch_earth_quat_f(struct FloatQuat* q) {
151  /* only non-zero entries for roll quaternion */
152  float roll2 = radio_control.values[RADIO_ROLL] * STABILIZATION_ATTITUDE_SP_MAX_PHI / MAX_PPRZ / 2;
153  float qx_roll = sinf(roll2);
154  float qi_roll = cosf(roll2);
155 
156  /* only non-zero entries for pitch quaternion */
157  float pitch2 = radio_control.values[RADIO_PITCH] * STABILIZATION_ATTITUDE_SP_MAX_THETA / MAX_PPRZ / 2;
158  float qy_pitch = sinf(pitch2);
159  float qi_pitch = cosf(pitch2);
160 
161  /* only multiply non-zero entries of FLOAT_QUAT_COMP(*q, q_roll, q_pitch) */
162  q->qi = qi_roll * qi_pitch;
163  q->qx = qx_roll * qi_pitch;
164  q->qy = qi_roll * qy_pitch;
165  q->qz = qx_roll * qy_pitch;
166 }

168 static inline void stabilization_attitude_read_rc_setpoint_quat_f(struct FloatQuat* q_sp, bool_t in_flight) {
169 
170  // FIXME: remove me, do in quaternion directly
171  // is currently still needed, since the yaw setpoint integration is done in eulers
172 #if defined STABILIZATION_ATTITUDE_TYPE_INT
173  stabilization_attitude_read_rc_setpoint_eulers(&stab_att_sp_euler, in_flight);
174 #else
175  stabilization_attitude_read_rc_setpoint_eulers_f(&stab_att_sp_euler, in_flight);
176 #endif
177 
178  struct FloatQuat q_rp_cmd;
179 #if USE_EARTH_BOUND_RC_SETPOINT
180  stabilization_attitude_read_rc_roll_pitch_earth_quat_f(&q_rp_cmd);
181 #else
182  stabilization_attitude_read_rc_roll_pitch_quat_f(&q_rp_cmd);
183 #endif
184 
185  /* get current heading */
186  const struct FloatVect3 zaxis = {0., 0., 1.};
187  struct FloatQuat q_yaw;
188  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw, zaxis, stateGetNedToBodyEulers_f()->psi);
189 
190  /* roll/pitch commands applied to to current heading */
191  struct FloatQuat q_rp_sp;
192  FLOAT_QUAT_COMP(q_rp_sp, q_yaw, q_rp_cmd);
193  FLOAT_QUAT_NORMALIZE(q_rp_sp);
194 
195  if (in_flight)
196  {
197  /* get current heading setpoint */
198  struct FloatQuat q_yaw_sp;
199 #if defined STABILIZATION_ATTITUDE_TYPE_INT
200  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw_sp, zaxis, ANGLE_FLOAT_OF_BFP(stab_att_sp_euler.psi));
201 #else
202  FLOAT_QUAT_OF_AXIS_ANGLE(q_yaw_sp, zaxis, stab_att_sp_euler.psi);
203 #endif
204 
205  /* rotation between current yaw and yaw setpoint */
206  struct FloatQuat q_yaw_diff;
207  FLOAT_QUAT_COMP_INV(q_yaw_diff, q_yaw_sp, q_yaw);
208 
209  /* compute final setpoint with yaw */
210  FLOAT_QUAT_COMP_NORM_SHORTEST(*q_sp, q_rp_sp, q_yaw_diff);
211  } else {
212  QUAT_COPY(*q_sp, q_rp_sp);
213  }
214 }
