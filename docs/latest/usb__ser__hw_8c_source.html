00029 /*
00030   Minimal implementation of a USB serial port, using the CDC class.
00031   This example application simply echoes everything it receives right back
00032   to the host.
00033 
00034   Windows:
00035   Extract the usbser.sys file from .cab file in C:\WINDOWS\Driver Cache\i386
00036   and store it somewhere (C:\temp is a good place) along with the usbser.inf
00037   file. Then plug in the LPC214x and direct windows to the usbser driver.
00038   Windows then creates an extra COMx port that you can open in a terminal
00039   program, like hyperterminal.
00040 
00041   Linux:
00042   The device should be recognised automatically by the cdc_acm driver,
00043   which creates a /dev/ttyACMx device file that acts just like a regular
00044   serial port.
00045 
00046 */

00434 {
00435   int i, iLen;
00436 
00437   if (fifo_free(&rxfifo) < MAX_PACKET_SIZE) {
00438     // may not fit into fifo

00462 static void BulkIn(U8 bEP, U8 bEPStatus)
00463 {
00464   int i, iLen;
00465 
00466   if (fifo_avail(&txfifo) == 0) {
00467     // no more data, disable further NAK interrupts until next USB frame
00468     USBHwNakIntEnable(0);
00469     return;
00470   }
00471 
00472   // get bytes from transmit FIFO into intermediate buffer
00473   for (i = 0; i < MAX_PACKET_SIZE; i++) {
00474     if (!fifo_get(&txfifo, &abBulkBuf[i])) {
00475       break;
00476     }
00477   }
00478   iLen = i;
00479 
00480   // send over USB
00481   if (iLen > 0) {
00482     USBHwEPWrite(bEP, abBulkBuf, iLen);
00483   }
