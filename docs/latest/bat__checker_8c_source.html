00056 void bat_checker_periodic(void) {
00057   static uint8_t bat_low_counter = 0;
00058   if(electrical.vsupply < WARN_BAT_LEVEL1) {
00059     if(bat_low_counter)
00060       bat_low_counter--;
00061   } else {
00062     bat_low_counter = BAT_CHECKER_DELAY * BAT_CHECKER_PERIODIC_FREQ;
00063   }
00064 
00065   if(!bat_low_counter) {
00066     if(electrical.vsupply < WARN_BAT_LEVEL2) {
00067       LED_ON(BAT_CHECKER_LED);
00068     } else if(electrical.vsupply < WARN_BAT_LEVEL1) {
00069       LED_TOGGLE(BAT_CHECKER_LED);
00070     }
00071   } else {
00072     LED_OFF(BAT_CHECKER_LED);
00073   }
