00111 static inline int imu_from_buff(volatile uint8_t *buf)
00112 {
00113   int32_t x, y, z, p, q, r, Mx, My, Mz;
00114 
00115 #define MPU_OFFSET_STATUS 1
00116   if (!(buf[MPU_OFFSET_STATUS] & 0x01)) {
00117     return 0;
00118   }
00119 
00120 #define MPU_OFFSET_GYRO 10
00121   p = (int16_t) ((buf[0+MPU_OFFSET_GYRO] << 8) | buf[1+MPU_OFFSET_GYRO]);
00122   q = (int16_t) ((buf[2+MPU_OFFSET_GYRO] << 8) | buf[3+MPU_OFFSET_GYRO]);
00123   r = (int16_t) ((buf[4+MPU_OFFSET_GYRO] << 8) | buf[5+MPU_OFFSET_GYRO]);
00124 
00125 #define MPU_OFFSET_ACC 2
00126   x = (int16_t) ((buf[0+MPU_OFFSET_ACC] << 8) | buf[1+MPU_OFFSET_ACC]);
00127   y = (int16_t) ((buf[2+MPU_OFFSET_ACC] << 8) | buf[3+MPU_OFFSET_ACC]);
00128   z = (int16_t) ((buf[4+MPU_OFFSET_ACC] << 8) | buf[5+MPU_OFFSET_ACC]);
00129 
00130 #define MPU_OFFSET_MAG 16
00131   Mx = (int16_t) ((buf[0+MPU_OFFSET_MAG] << 8) | buf[1+MPU_OFFSET_MAG]);
00132   My = (int16_t) ((buf[2+MPU_OFFSET_MAG] << 8) | buf[3+MPU_OFFSET_MAG]);
00133   Mz = (int16_t) ((buf[4+MPU_OFFSET_MAG] << 8) | buf[5+MPU_OFFSET_MAG]);
00134 
00135 #ifdef LISA_M_LONGITUDINAL_X
00136   RATES_ASSIGN(imu.gyro_unscaled, q, -p, r);
00137   VECT3_ASSIGN(imu.accel_unscaled, y, -x, z);
00138   VECT3_ASSIGN(imu.mag_unscaled, -Mx, -Mz, My);
00139 #else
00140   RATES_ASSIGN(imu.gyro_unscaled, p, q, r);
00141   VECT3_ASSIGN(imu.accel_unscaled, x, y, z);
00142   VECT3_ASSIGN(imu.mag_unscaled, Mz, -Mx, My);
00143 #endif
00144 
00145   return 1;
00146 }

00150 {
00151   if (imu_aspirin2.status == Aspirin2StatusUninit) return;
