111 static inline int imu_from_buff(volatile uint8_t *buf)
112 {
113  int32_t x, y, z, p, q, r, Mx, My, Mz;
114 
115 #define MPU_OFFSET_STATUS 1
116  if (!(buf[MPU_OFFSET_STATUS] & 0x01)) {
117  return 0;
118  }
119 
120 #define MPU_OFFSET_GYRO 10
121  p = (int16_t) ((buf[0+MPU_OFFSET_GYRO] << 8) | buf[1+MPU_OFFSET_GYRO]);
122  q = (int16_t) ((buf[2+MPU_OFFSET_GYRO] << 8) | buf[3+MPU_OFFSET_GYRO]);
123  r = (int16_t) ((buf[4+MPU_OFFSET_GYRO] << 8) | buf[5+MPU_OFFSET_GYRO]);
124 
125 #define MPU_OFFSET_ACC 2
126  x = (int16_t) ((buf[0+MPU_OFFSET_ACC] << 8) | buf[1+MPU_OFFSET_ACC]);
127  y = (int16_t) ((buf[2+MPU_OFFSET_ACC] << 8) | buf[3+MPU_OFFSET_ACC]);
128  z = (int16_t) ((buf[4+MPU_OFFSET_ACC] << 8) | buf[5+MPU_OFFSET_ACC]);
129 
130 #define MPU_OFFSET_MAG 16
131  Mx = (int16_t) ((buf[0+MPU_OFFSET_MAG] << 8) | buf[1+MPU_OFFSET_MAG]);
132  My = (int16_t) ((buf[2+MPU_OFFSET_MAG] << 8) | buf[3+MPU_OFFSET_MAG]);
133  Mz = (int16_t) ((buf[4+MPU_OFFSET_MAG] << 8) | buf[5+MPU_OFFSET_MAG]);
134 
135 #ifdef LISA_M_LONGITUDINAL_X
136  RATES_ASSIGN(imu.gyro_unscaled, q, -p, r);
137  VECT3_ASSIGN(imu.accel_unscaled, y, -x, z);
138  VECT3_ASSIGN(imu.mag_unscaled, -Mx, -Mz, My);
139 #else
140  RATES_ASSIGN(imu.gyro_unscaled, p, q, r);
141  VECT3_ASSIGN(imu.accel_unscaled, x, y, z);
142  VECT3_ASSIGN(imu.mag_unscaled, Mz, -Mx, My);
143 #endif
144 
145  return 1;
146 }

149 static inline void imu_aspirin2_event(void (* _gyro_handler)(void), void (* _accel_handler)(void), void (* _mag_handler)(void))
150 {
151  if (imu_aspirin2.status == Aspirin2StatusUninit) return;
152 
153  if (imu_aspirin2.imu_available) {
154  imu_aspirin2.imu_available = FALSE;
155  if (imu_from_buff(imu_aspirin2.input_buf_p)) {
156  _gyro_handler();
157  _accel_handler();
158  _mag_handler();
159  }
160  }
161 }

163 #define ImuEvent(_gyro_handler, _accel_handler, _mag_handler) { \

165 }
