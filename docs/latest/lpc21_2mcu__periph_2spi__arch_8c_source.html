00180 __attribute__ ((always_inline)) static inline void SpiTransmit(struct spi_periph* p, struct spi_transaction* t) {
00181   // when all byte are sent, continue until tx_idx reach input_length
00182   // needed when input_length is bigger than output_length
00183   uint8_t max_idx = Max(t->output_length, t->input_length);
00184   while (p->tx_idx_buf < max_idx && bit_is_set(((sspRegs_t *)(p->reg_addr))->sr, TNF)) {
00185     if (p->tx_idx_buf < t->output_length) {
00186       if (t->dss == SPIDss8bit) {
00187         SpiSend(p, t->output_buf[p->tx_idx_buf]);
00188       }
00189       else if (t->dss == SPIDss16bit) {
00190         uint16_t tmp1 = t->output_buf[2*p->tx_idx_buf]; // LSB
00191         uint16_t tmp2 = t->output_buf[2*p->tx_idx_buf+1]<<8; // MSB
00192         SpiSend(p, tmp1 | tmp2);
00193       }
00194     }
00195     else {
00196       SpiSend(p, 0);
00197     }
00198     p->tx_idx_buf++;
00199   }
00200   if (p->tx_idx_buf == max_idx) {
00201     SpiDisableTxi(p);
00202   }
00203 }

00205 __attribute__ ((always_inline)) static inline void SpiReceive(struct spi_periph* p, struct spi_transaction* t) {
00206   while (bit_is_set(((sspRegs_t *)(p->reg_addr))->sr, RNE)) {
00207     if (p->rx_idx_buf < t->input_length) {
00208       uint16_t r;
00209       SpiRead(p, &r);
00210       if (t->dss == SPIDss8bit) {
00211         t->input_buf[p->rx_idx_buf] = (uint8_t)r;
00212       }
00213       else if (t->dss == SPIDss16bit) {
00214         t->input_buf[2*p->rx_idx_buf] = (uint8_t)r;
00215         t->input_buf[2*p->rx_idx_buf+1] = (uint8_t)(r>>8);
00216       }
00217       p->rx_idx_buf++;
00218     }
00219     else {
00220       uint16_t foo;
00221       SpiRead(p, &foo);
00222     }
00223   }

00261 __attribute__ ((always_inline)) static inline void SpiEndOfTransaction(struct spi_periph* p, struct spi_transaction* t) {
00262   // callback function after transaction
00263   if (t->after_cb != 0) t->after_cb(t);
00264 
00265   // handle slave unselect
00266   if (t->select == SPISelectUnselect || t->select == SPIUnselect) {
00267     SpiSlaveUnselect(t->slave_idx);

00269

00271   // end transaction with success

00280     p->status = SPIIdle;
00281   }
00282   else {

00288 __attribute__ ((always_inline)) static inline void SpiAutomaton(struct spi_periph* p) {
00289   struct spi_transaction* trans = p->trans[p->trans_extract_idx];
00290 
00291   /* Tx fifo is half empty */
00292   if (bit_is_set(((sspRegs_t *)(p->reg_addr))->mis, TXMIS)) {
00293     SpiTransmit(p, trans);
00294     SpiReceive(p, trans);
00295     // tx_idx can be greater than output_length (when input_length > output_length)
00296     if (p->tx_idx_buf >= trans->output_length && p->rx_idx_buf == trans->input_length) {
00297       if (bit_is_set(((sspRegs_t *)(p->reg_addr))->sr, BSY)) {
00298         SpiEnableTxi(p); // FIXME in case Rti is not called
00299       }
00300       else {
00301         SpiDisableRti(p);
00302         SpiClearRti(p);                /* clear interrupt */
00303         SpiEndOfTransaction(p, trans);
00304       }
00305     }
00306   }
00307 
00308   /* Rx fifo is not empty and no receive took place in the last 32 bits period */
00309   if (bit_is_set(((sspRegs_t *)(p->reg_addr))->mis, RTMIS)) {
00310     SpiReceive(p, trans);
00311     SpiDisableRti(p);
00312     SpiClearRti(p);                /* clear interrupt */
00313     SpiEndOfTransaction(p, trans);
00314   }

00469 bool_t spi_submit(struct spi_periph* p, struct spi_transaction* t) {
00470   unsigned cpsr;
00471 
00472   uint8_t idx;
00473   idx = p->trans_insert_idx + 1;
00474   if (idx >= SPI_TRANSACTION_QUEUE_LEN) idx = 0;
00475   if (idx == p->trans_extract_idx) {
00476     t->status = SPITransFailed;
00477     return FALSE; /* queue full */
00478   }
00479   t->status = SPITransPending;
00480 
00481   // Disable interrupts
00482   uint8_t* vic = (uint8_t*)(p->init_struct);
00483   cpsr = disableIRQ();                                // disable global interrupts
00484   VICIntEnClear = VIC_BIT(*vic);
00485   restoreIRQ(cpsr);                                   // restore global interrupts
00486 
00487   p->trans[p->trans_insert_idx] = t;
00488   p->trans_insert_idx = idx;
00489   /* if peripheral is idle and not locked, start the transaction */
00490   if (p->status == SPIIdle && !p->suspend) {
00491     SpiStart(p,p->trans[p->trans_extract_idx]);
00492   }
00493 
00494   cpsr = disableIRQ();                                // disable global interrupts
00495   VICIntEnable = VIC_BIT(*vic);
00496   restoreIRQ(cpsr);                                   // restore global interrupts
00497   return TRUE;
00498 }
