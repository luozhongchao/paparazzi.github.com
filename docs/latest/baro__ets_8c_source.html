114  i2c_receive(&BARO_ETS_I2C_DEV, &baro_ets_i2c_trans, BARO_ETS_ADDR, 2);
115 #else // SITL
116  /* fake an offset so sim works for under hmsl as well */

130 void baro_ets_read_event( void ) {
131  // Get raw altimeter from buffer
132  baro_ets_adc = ((uint16_t)(baro_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(baro_ets_i2c_trans.buf[0]);
133  // Check if this is valid altimeter
134  if (baro_ets_adc == 0)
135  baro_ets_valid = FALSE;
136  else
137  baro_ets_valid = TRUE;
138 
139  // Continue only if a new altimeter value was received
140  if (baro_ets_valid) {
141  // Calculate offset average if not done already
142  if (!baro_ets_offset_init) {
143  --baro_ets_cnt;
144  // Check if averaging completed
145  if (baro_ets_cnt == 0) {
146  // Calculate average
147  baro_ets_offset = (uint16_t)(baro_ets_offset_tmp / BARO_ETS_OFFSET_NBSAMPLES_AVRG);
148  // Limit offset
149  if (baro_ets_offset < BARO_ETS_OFFSET_MIN)
150  baro_ets_offset = BARO_ETS_OFFSET_MIN;
151  if (baro_ets_offset > BARO_ETS_OFFSET_MAX)
152  baro_ets_offset = BARO_ETS_OFFSET_MAX;
153  baro_ets_offset_init = TRUE;
154  }
155  // Check if averaging needs to continue
156  else if (baro_ets_cnt <= BARO_ETS_OFFSET_NBSAMPLES_AVRG)
157  baro_ets_offset_tmp += baro_ets_adc;
158  }
159  // Convert raw to m/s
160  if (baro_ets_offset_init) {
161  baro_ets_altitude = ground_alt + BARO_ETS_SCALE * (float)(baro_ets_offset-baro_ets_adc);
162  // New value available
163 #ifdef BARO_ETS_TELEMETRY
164  DOWNLINK_SEND_BARO_ETS(DefaultChannel, DefaultDevice, &baro_ets_adc, &baro_ets_offset, &baro_ets_altitude);
165 #endif
166  } else {
167  baro_ets_altitude = 0.0;
168  }
169  } else {
170  baro_ets_altitude = 0.0;
171  }
172 
173  // Transaction has been read
174  baro_ets_i2c_trans.status = I2CTransDone;
175 }
