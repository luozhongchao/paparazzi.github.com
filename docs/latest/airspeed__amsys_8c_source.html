00113 void airspeed_amsys_read_event( void ) {
00114 
00115   // Get raw airspeed from buffer
00116   airspeed_amsys_raw = 0;
00117   airspeed_amsys_raw = (airspeed_amsys_i2c_trans.buf[0]<<8) | airspeed_amsys_i2c_trans.buf[1];
00118 #ifdef MEASURE_AMSYS_TEMPERATURE
00119   tempAS_amsys_raw = (airspeed_amsys_i2c_trans.buf[2]<<8) | airspeed_amsys_i2c_trans.buf[3];
00120   airspeed_temperature = (float)((float)(tempAS_amsys_raw-TEMPERATURE_AMSYS_OFFSET_MIN)/((float)(TEMPERATURE_AMSYS_OFFSET_MAX-TEMPERATURE_AMSYS_OFFSET_MIN)/TEMPERATURE_AMSYS_MAX)+TEMPERATURE_AMSYS_MIN);// Tmin=-25, Tmax=85
00121 #endif
00122 
00123   // Check if this is valid airspeed
00124   if (airspeed_amsys_raw == 0)
00125     airspeed_amsys_valid = FALSE;
00126   else
00127     airspeed_amsys_valid = TRUE;
00128 
00129   // Continue only if a new airspeed value was received
00130   if (airspeed_amsys_valid) {
00131 
00132     // raw not under offest min
00133     if (airspeed_amsys_raw<AIRSPEED_AMSYS_OFFSET_MIN)
00134       airspeed_amsys_raw = AIRSPEED_AMSYS_OFFSET_MIN;
00135     // raw not over offest max
00136     if (airspeed_amsys_raw>AIRSPEED_AMSYS_OFFSET_MAX)
00137       airspeed_amsys_raw = AIRSPEED_AMSYS_OFFSET_MAX;
00138 
00139     // calculate raw to pressure
00140     pressure_amsys = (float)(airspeed_amsys_raw-AIRSPEED_AMSYS_OFFSET_MIN)*AIRSPEED_AMSYS_MAXPRESURE/(float)(AIRSPEED_AMSYS_OFFSET_MAX-AIRSPEED_AMSYS_OFFSET_MIN);
00141 
00142     airspeed_tmp = sqrtf(2*(pressure_amsys)*airspeed_scale/1.2041); //without offset
00143 
00144   //  Lowpass filter
00145     airspeed_amsys = airspeed_filter * airspeed_old + (1 - airspeed_filter) * airspeed_tmp;
00146     airspeed_old = airspeed_amsys;
00147 
00148 #if USE_AIRSPEED
00149     stateSetAirspeed_f(&airspeed_amsys);
00150 #endif
00151 #ifdef SENSOR_SYNC_SEND
00152     DOWNLINK_SEND_AMSYS_AIRSPEED(DefaultChannel, DefaultDevice, &airspeed_amsys_raw, &pressure_amsys, &airspeed_tmp, &airspeed_amsys, &airspeed_temperature);
00153 #else
00154     RunOnceEvery(10, DOWNLINK_SEND_AMSYS_AIRSPEED(DefaultChannel, DefaultDevice, &airspeed_amsys_raw, &pressure_amsys, &airspeed_tmp, &airspeed_amsys, &airspeed_temperature));
00155 #endif
00156   }
00157 
00158   // Transaction has been read
00159   airspeed_amsys_i2c_trans.status = I2CTransDone;
00160 }
