113 void airspeed_amsys_read_event( void ) {
114 
115  // Get raw airspeed from buffer
116  airspeed_amsys_raw = 0;
117  airspeed_amsys_raw = (airspeed_amsys_i2c_trans.buf[0]<<8) | airspeed_amsys_i2c_trans.buf[1];
118 #ifdef MEASURE_AMSYS_TEMPERATURE
119  tempAS_amsys_raw = (airspeed_amsys_i2c_trans.buf[2]<<8) | airspeed_amsys_i2c_trans.buf[3];
120  airspeed_temperature = (float)((float)(tempAS_amsys_raw-TEMPERATURE_AMSYS_OFFSET_MIN)/((float)(TEMPERATURE_AMSYS_OFFSET_MAX-TEMPERATURE_AMSYS_OFFSET_MIN)/TEMPERATURE_AMSYS_MAX)+TEMPERATURE_AMSYS_MIN);// Tmin=-25, Tmax=85
121 #endif
122 
123  // Check if this is valid airspeed
124  if (airspeed_amsys_raw == 0)
125  airspeed_amsys_valid = FALSE;
126  else
127  airspeed_amsys_valid = TRUE;
128 
129  // Continue only if a new airspeed value was received
130  if (airspeed_amsys_valid) {
131 
132  // raw not under offest min
133  if (airspeed_amsys_raw<AIRSPEED_AMSYS_OFFSET_MIN)
134  airspeed_amsys_raw = AIRSPEED_AMSYS_OFFSET_MIN;
135  // raw not over offest max
136  if (airspeed_amsys_raw>AIRSPEED_AMSYS_OFFSET_MAX)
137  airspeed_amsys_raw = AIRSPEED_AMSYS_OFFSET_MAX;
138 
139  // calculate raw to pressure
140  pressure_amsys = (float)(airspeed_amsys_raw-AIRSPEED_AMSYS_OFFSET_MIN)*AIRSPEED_AMSYS_MAXPRESURE/(float)(AIRSPEED_AMSYS_OFFSET_MAX-AIRSPEED_AMSYS_OFFSET_MIN);
141 
142  airspeed_tmp = sqrtf(2*(pressure_amsys)*airspeed_scale/1.2041); //without offset
143 
144  // Lowpass filter
145  airspeed_amsys = airspeed_filter * airspeed_old + (1 - airspeed_filter) * airspeed_tmp;
146  airspeed_old = airspeed_amsys;
147 
148 #if USE_AIRSPEED
149  stateSetAirspeed_f(&airspeed_amsys);
150 #endif
151 #ifdef SENSOR_SYNC_SEND
152  DOWNLINK_SEND_AMSYS_AIRSPEED(DefaultChannel, DefaultDevice, &airspeed_amsys_raw, &pressure_amsys, &airspeed_tmp, &airspeed_amsys, &airspeed_temperature);
153 #else
154  RunOnceEvery(10, DOWNLINK_SEND_AMSYS_AIRSPEED(DefaultChannel, DefaultDevice, &airspeed_amsys_raw, &pressure_amsys, &airspeed_tmp, &airspeed_amsys, &airspeed_temperature));
155 #endif
156  }
157 
158  // Transaction has been read
159  airspeed_amsys_i2c_trans.status = I2CTransDone;
160 }
