00197 void parse_cam_buffer( uint8_t c ) {
00198   char bla[1];
00199   bla[1] = c;
00200   //DOWNLINK_SEND_DEBUG(DefaultChannel, DefaultDevice,1,bla);
00201   switch (cam_status) {
00202   case UNINIT:
00203     if (c != CAM_START_1)
00204       goto error;
00205     cam_status++;
00206     break;
00207   case GOT_START_1:
00208     if (c != CAM_START_2)
00209       goto error;
00210     cam_status++;
00211     break;
00212   case GOT_START_2:
00213     cam_data_len = c;
00214     if (cam_data_len > CAM_MAX_PAYLOAD)
00215       goto error;
00216     cam_data_idx = 0;
00217     cam_status++;
00218     break;
00219   case GOT_LEN:
00220     cam_data_buf[cam_data_idx] = c;
00221     cam_data_idx++;
00222     if (cam_data_idx >= cam_data_len)
00223       cam_status++;
00224     break;
00225   case GOT_DATA:
00226     if (c != CAM_END)
00227       goto error;
00228     cam_msg_received = TRUE;
00229     goto restart;
00230     break;
00231   }
00232   return;
00233  error:
00234  restart:
00235   cam_status = UNINIT;
00236   return;
