41 static uint8_t xbee_text_reply_is_ok(void)
42 {
43  char c[2];
44  int count = 0;
45 
46  while (TransportLink(XBEE_UART,ChAvailable()))
47  {
48  char cc = TransportLink(XBEE_UART,Getch());
49  if (count < 2)
50  c[count] = cc;
51  count++;
52  }
53 
54  if ((count > 2) && (c[0] == 'O') && (c[1] == 'K'))
55  return TRUE;
56 
57  return FALSE;
58 }
