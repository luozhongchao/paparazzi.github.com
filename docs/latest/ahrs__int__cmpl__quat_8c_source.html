00131 void ahrs_propagate(void) {
00132 
00133   /* unbias gyro             */
00134   struct Int32Rates omega;
00135   RATES_DIFF(omega, imu.gyro_prev, ahrs_impl.gyro_bias);
00136 
00137   /* low pass rate */
00138 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
00139   RATES_SMUL(ahrs_impl.imu_rate, ahrs_impl.imu_rate,2);
00140   RATES_ADD(ahrs_impl.imu_rate, omega);
00141   RATES_SDIV(ahrs_impl.imu_rate, ahrs_impl.imu_rate, 3);
00142 #else
00143   RATES_COPY(ahrs_impl.imu_rate, omega);
00144 #endif
00145 
00146   /* add correction     */
00147   RATES_ADD(omega, ahrs_impl.rate_correction);
00148   /* and zeros it */
00149   INT_RATES_ZERO(ahrs_impl.rate_correction);
00150 
00151   /* integrate quaternion */
00152   INT32_QUAT_INTEGRATE_FI(ahrs_impl.ltp_to_imu_quat, ahrs_impl.high_rez_quat, omega, AHRS_PROPAGATE_FREQUENCY);
00153   INT32_QUAT_NORMALIZE(ahrs_impl.ltp_to_imu_quat);
00154 
00155   set_body_state_from_quat();

00162 void ahrs_update_accel(void) {
00163 
00164   // c2 = ltp z-axis in imu-frame
00165   struct Int32RMat ltp_to_imu_rmat;
00166   INT32_RMAT_OF_QUAT(ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);
00167   struct Int32Vect3 c2 = { RMAT_ELMT(ltp_to_imu_rmat, 0,2),
00168                            RMAT_ELMT(ltp_to_imu_rmat, 1,2),
00169                            RMAT_ELMT(ltp_to_imu_rmat, 2,2)};
00170   struct Int32Vect3 residual;
00171 
00172   struct Int32Vect3 pseudo_gravity_measurement;
00173 
00174   if (ahrs_impl.correct_gravity && ahrs_impl.ltp_vel_norm_valid) {
00175     /*
00176      * centrifugal acceleration in body frame
00177      * a_c_body = omega x (omega x r)
00178      * (omega x r) = tangential velocity in body frame
00179      * a_c_body = omega x vel_tangential_body
00180      * assumption: tangential velocity only along body x-axis
00181      */
00182 
00183     // FIXME: check overflows !
00184 #define COMPUTATION_FRAC 16
00185 
00186     const struct Int32Vect3 vel_tangential_body = {ahrs_impl.ltp_vel_norm >> COMPUTATION_FRAC, 0, 0};
00187     struct Int32Vect3 acc_c_body;
00188     VECT3_RATES_CROSS_VECT3(acc_c_body, (*stateGetBodyRates_i()), vel_tangential_body);
00189     INT32_VECT3_RSHIFT(acc_c_body, acc_c_body, INT32_SPEED_FRAC+INT32_RATE_FRAC-INT32_ACCEL_FRAC-COMPUTATION_FRAC);
00190 
00191     /* convert centrifucal acceleration from body to imu frame */
00192     struct Int32Vect3 acc_c_imu;
00193     INT32_RMAT_VMULT(acc_c_imu, imu.body_to_imu_rmat, acc_c_body);
00194 
00195     /* and subtract it from imu measurement to get a corrected measurement of the gravitiy vector */
00196     INT32_VECT3_DIFF(pseudo_gravity_measurement, imu.accel, acc_c_imu);
00197   } else {
00198     VECT3_COPY(pseudo_gravity_measurement, imu.accel);
00199   }
00200 
00201   /* compute the residual of the pseudo gravity vector in imu frame */
00202   INT32_VECT3_CROSS_PRODUCT(residual, pseudo_gravity_measurement, c2);
00203 
00204 
00205   int32_t inv_weight;
00206   if (ahrs_impl.use_gravity_heuristic) {
00207     /* heuristic on acceleration norm */
00208 
00209     /* FIR filtered pseudo_gravity_measurement */
00210     static struct Int32Vect3 filtered_gravity_measurement = {0, 0, 0};
00211     VECT3_SMUL(filtered_gravity_measurement, filtered_gravity_measurement, 7);
00212     VECT3_ADD(filtered_gravity_measurement, pseudo_gravity_measurement);
00213     VECT3_SDIV(filtered_gravity_measurement, filtered_gravity_measurement, 8);
00214 
00215     int32_t acc_norm;
00216     INT32_VECT3_NORM(acc_norm, filtered_gravity_measurement);
00217     const int32_t acc_norm_d = ABS(ACCEL_BFP_OF_REAL(9.81)-acc_norm);
00218     inv_weight = Chop(50*acc_norm_d/ACCEL_BFP_OF_REAL(9.81), 1, 50);
00219   }
00220   else {
00221     inv_weight = 1;
00222   }
00223 
00224   // residual FRAC : ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
00225   // rate_correction FRAC = RATE_FRAC = 12
00226   // 2^12 / 2^24 * 5e-2 = 1/81920
00227   ahrs_impl.rate_correction.p += -residual.x/82000/inv_weight;
00228   ahrs_impl.rate_correction.q += -residual.y/82000/inv_weight;
00229   ahrs_impl.rate_correction.r += -residual.z/82000/inv_weight;
00230 
00231   // residual FRAC = ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
00232   // high_rez_bias = RATE_FRAC+28 = 40
00233   // 2^40 / 2^24 * 5e-6 = 1/3.05
00234 
00235   //  ahrs_impl.high_rez_bias.p += residual.x*3;
00236   //  ahrs_impl.high_rez_bias.q += residual.y*3;
00237   //  ahrs_impl.high_rez_bias.r += residual.z*3;
00238 
00239   ahrs_impl.high_rez_bias.p += residual.x/(2*inv_weight);
00240   ahrs_impl.high_rez_bias.q += residual.y/(2*inv_weight);
00241   ahrs_impl.high_rez_bias.r += residual.z/(2*inv_weight);
00242 
00243 
00244   /*                        */
00245   INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
00246 
00247 }

00363 void ahrs_update_heading(int32_t heading) {
00364 
00365   INT32_ANGLE_NORMALIZE(heading);
00366 
00367   // row 0 of ltp_to_body_rmat = body x-axis in ltp frame
00368   // we only consider x and y
00369   struct Int32RMat* ltp_to_body_rmat = stateGetNedToBodyRMat_i();
00370   struct Int32Vect2 expected_ltp =
00371     { RMAT_ELMT((*ltp_to_body_rmat), 0, 0),
00372       RMAT_ELMT((*ltp_to_body_rmat), 0, 1) };
00373 
00374   int32_t heading_x, heading_y;
00375   PPRZ_ITRIG_COS(heading_x, heading); // measured course in x-direction
00376   PPRZ_ITRIG_SIN(heading_y, heading); // measured course in y-direction
00377 
00378   // expected_heading cross measured_heading ??
00379   struct Int32Vect3 residual_ltp =
00380     { 0,
00381       0,
00382       (expected_ltp.x * heading_y - expected_ltp.y * heading_x)/(1<<INT32_ANGLE_FRAC)};
00383 
00384   struct Int32Vect3 residual_imu;
00385   struct Int32RMat ltp_to_imu_rmat;
00386   INT32_RMAT_OF_QUAT(ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);
00387   INT32_RMAT_VMULT(residual_imu, ltp_to_imu_rmat, residual_ltp);
00388 
00389   // residual FRAC = TRIG_FRAC + TRIG_FRAC = 14 + 14 = 28
00390   // rate_correction FRAC = RATE_FRAC = 12
00391   // 2^12 / 2^28 * 4.0 = 1/2^14
00392   // (1<<INT32_ANGLE_FRAC)/2^14 = 1/4
00393   ahrs_impl.rate_correction.p += residual_imu.x/4;
00394   ahrs_impl.rate_correction.q += residual_imu.y/4;
00395   ahrs_impl.rate_correction.r += residual_imu.z/4;
00396 
00397 
00398   /* crude attempt to only update bias if deviation is small
00399    * e.g. needed when you only have gps providing heading
00400    * and the inital heading is totally different from
00401    * the gps course information you get once you have a gps fix.
00402    * Otherwise the bias will be falsely "corrected".
00403    */
00404   int32_t sin_max_angle_deviation;
00405   PPRZ_ITRIG_SIN(sin_max_angle_deviation, TRIG_BFP_OF_REAL(RadOfDeg(5.)));
00406   if (ABS(residual_ltp.z) < sin_max_angle_deviation)
00407   {
00408     // residual_ltp FRAC = 2 * TRIG_FRAC = 28
00409     // high_rez_bias = RATE_FRAC+28 = 40
00410     // 2^40 / 2^28 * 2.5e-4 = 1
00411     ahrs_impl.high_rez_bias.p -= residual_imu.x*(1<<INT32_ANGLE_FRAC);
00412     ahrs_impl.high_rez_bias.q -= residual_imu.y*(1<<INT32_ANGLE_FRAC);
00413     ahrs_impl.high_rez_bias.r -= residual_imu.z*(1<<INT32_ANGLE_FRAC);
00414 
00415     INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
00416   }
00417 }

00420 
00421   struct Int32Quat ltp_to_body_quat = *stateGetNedToBodyQuat_i();
00422 
00423   /* quaternion representing only the heading rotation from ltp to body */
00424   struct Int32Quat q_h_new;
00425   q_h_new.qx = 0;
00426   q_h_new.qy = 0;
00427   PPRZ_ITRIG_SIN(q_h_new.qz, heading/2);
00428   PPRZ_ITRIG_COS(q_h_new.qi, heading/2);
00429 
00430   /* quaternion representing current heading only */
00431   struct Int32Quat q_h;
00432   QUAT_COPY(q_h, ltp_to_body_quat);
00433   q_h.qx = 0;
00434   q_h.qy = 0;
00435   INT32_QUAT_NORMALIZE(q_h);
00436 
00437   /* quaternion representing rotation from current to new heading */
00438   struct Int32Quat q_c;
00439   INT32_QUAT_INV_COMP_NORM_SHORTEST(q_c, q_h, q_h_new);
00440 
00441   /* correct current heading in body frame */
00442   struct Int32Quat q;
00443   INT32_QUAT_COMP_NORM_SHORTEST(q, q_c, ltp_to_body_quat);
00444   QUAT_COPY(ltp_to_body_quat, q);
00445 
00446   /* compute ltp to imu rotations */
00447   INT32_QUAT_COMP(ahrs_impl.ltp_to_imu_quat, ltp_to_body_quat, imu.body_to_imu_quat);

00450   stateSetNedToBodyQuat_i(&ltp_to_body_quat);
