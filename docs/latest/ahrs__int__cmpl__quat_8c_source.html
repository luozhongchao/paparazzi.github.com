131 void ahrs_propagate(void) {
132 
133  /* unbias gyro */
134  struct Int32Rates omega;
135  RATES_DIFF(omega, imu.gyro_prev, ahrs_impl.gyro_bias);
136 
137  /* low pass rate */
138 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
139  RATES_SMUL(ahrs_impl.imu_rate, ahrs_impl.imu_rate,2);
140  RATES_ADD(ahrs_impl.imu_rate, omega);
141  RATES_SDIV(ahrs_impl.imu_rate, ahrs_impl.imu_rate, 3);
142 #else
143  RATES_COPY(ahrs_impl.imu_rate, omega);
144 #endif
145 
146  /* add correction */
147  RATES_ADD(omega, ahrs_impl.rate_correction);
148  /* and zeros it */
149  INT_RATES_ZERO(ahrs_impl.rate_correction);
150 
151  /* integrate quaternion */
152  INT32_QUAT_INTEGRATE_FI(ahrs_impl.ltp_to_imu_quat, ahrs_impl.high_rez_quat, omega, AHRS_PROPAGATE_FREQUENCY);
153  INT32_QUAT_NORMALIZE(ahrs_impl.ltp_to_imu_quat);
154 
155  set_body_state_from_quat();
156 
157 }

162 void ahrs_update_accel(void) {
163 
164  // c2 = ltp z-axis in imu-frame
165  struct Int32RMat ltp_to_imu_rmat;
166  INT32_RMAT_OF_QUAT(ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);
167  struct Int32Vect3 c2 = { RMAT_ELMT(ltp_to_imu_rmat, 0,2),
168  RMAT_ELMT(ltp_to_imu_rmat, 1,2),
169  RMAT_ELMT(ltp_to_imu_rmat, 2,2)};
170  struct Int32Vect3 residual;
171 
172  struct Int32Vect3 pseudo_gravity_measurement;
173 
174  if (ahrs_impl.correct_gravity && ahrs_impl.ltp_vel_norm_valid) {
175  /*
176  * centrifugal acceleration in body frame
177  * a_c_body = omega x (omega x r)
178  * (omega x r) = tangential velocity in body frame
179  * a_c_body = omega x vel_tangential_body
180  * assumption: tangential velocity only along body x-axis
181  */
182 
183  // FIXME: check overflows !
184 #define COMPUTATION_FRAC 16
185 
186  const struct Int32Vect3 vel_tangential_body = {ahrs_impl.ltp_vel_norm >> COMPUTATION_FRAC, 0, 0};
187  struct Int32Vect3 acc_c_body;
188  VECT3_RATES_CROSS_VECT3(acc_c_body, (*stateGetBodyRates_i()), vel_tangential_body);
189  INT32_VECT3_RSHIFT(acc_c_body, acc_c_body, INT32_SPEED_FRAC+INT32_RATE_FRAC-INT32_ACCEL_FRAC-COMPUTATION_FRAC);
190 
191  /* convert centrifucal acceleration from body to imu frame */
192  struct Int32Vect3 acc_c_imu;
193  INT32_RMAT_VMULT(acc_c_imu, imu.body_to_imu_rmat, acc_c_body);
194 
195  /* and subtract it from imu measurement to get a corrected measurement of the gravitiy vector */
196  INT32_VECT3_DIFF(pseudo_gravity_measurement, imu.accel, acc_c_imu);
197  } else {
198  VECT3_COPY(pseudo_gravity_measurement, imu.accel);
199  }
200 
201  /* compute the residual of the pseudo gravity vector in imu frame */
202  INT32_VECT3_CROSS_PRODUCT(residual, pseudo_gravity_measurement, c2);
203 
204 
205  int32_t inv_weight;
206  if (ahrs_impl.use_gravity_heuristic) {
207  /* heuristic on acceleration norm */
208 
209  /* FIR filtered pseudo_gravity_measurement */
210  static struct Int32Vect3 filtered_gravity_measurement = {0, 0, 0};
211  VECT3_SMUL(filtered_gravity_measurement, filtered_gravity_measurement, 7);
212  VECT3_ADD(filtered_gravity_measurement, pseudo_gravity_measurement);
213  VECT3_SDIV(filtered_gravity_measurement, filtered_gravity_measurement, 8);
214 
215  int32_t acc_norm;
216  INT32_VECT3_NORM(acc_norm, filtered_gravity_measurement);
217  const int32_t acc_norm_d = ABS(ACCEL_BFP_OF_REAL(9.81)-acc_norm);
218  inv_weight = Chop(50*acc_norm_d/ACCEL_BFP_OF_REAL(9.81), 1, 50);
219  }
220  else {
221  inv_weight = 1;
222  }
223 
224  // residual FRAC : ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
225  // rate_correction FRAC = RATE_FRAC = 12
226  // 2^12 / 2^24 * 5e-2 = 1/81920
227  ahrs_impl.rate_correction.p += -residual.x/82000/inv_weight;
228  ahrs_impl.rate_correction.q += -residual.y/82000/inv_weight;
229  ahrs_impl.rate_correction.r += -residual.z/82000/inv_weight;
230 
231  // residual FRAC = ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
232  // high_rez_bias = RATE_FRAC+28 = 40
233  // 2^40 / 2^24 * 5e-6 = 1/3.05
234 
235  // ahrs_impl.high_rez_bias.p += residual.x*3;
236  // ahrs_impl.high_rez_bias.q += residual.y*3;
237  // ahrs_impl.high_rez_bias.r += residual.z*3;
238 
239  ahrs_impl.high_rez_bias.p += residual.x/(2*inv_weight);
240  ahrs_impl.high_rez_bias.q += residual.y/(2*inv_weight);
241  ahrs_impl.high_rez_bias.r += residual.z/(2*inv_weight);
242 
243 
244  /* */
245  INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
246 
247 }

287  INT32_RMAT_OF_QUAT(ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);

293  { 0,

299 
300  // residual_ltp FRAC = 2 * MAG_FRAC = 22
301  // rate_correction FRAC = RATE_FRAC = 12
302  // 2^12 / 2^22 * 2.5 = 1/410
303 
304  // ahrs_impl.rate_correction.p += residual_imu.x*(1<<5)/410;
305  // ahrs_impl.rate_correction.q += residual_imu.y*(1<<5)/410;
306  // ahrs_impl.rate_correction.r += residual_imu.z*(1<<5)/410;
307 
308  ahrs_impl.rate_correction.p += residual_imu.x/16;
309  ahrs_impl.rate_correction.q += residual_imu.y/16;
310  ahrs_impl.rate_correction.r += residual_imu.z/16;
311 
312 
313  // residual_ltp FRAC = 2 * MAG_FRAC = 22
314  // high_rez_bias = RATE_FRAC+28 = 40

318  // ahrs_impl.high_rez_bias.q -= residual_imu.y*(1<<5)*655;
319  // ahrs_impl.high_rez_bias.r -= residual_imu.z*(1<<5)*655;
320 
321  ahrs_impl.high_rez_bias.p -= residual_imu.x*1024;
322  ahrs_impl.high_rez_bias.q -= residual_imu.y*1024;
323  ahrs_impl.high_rez_bias.r -= residual_imu.z*1024;
324 
325

330 void ahrs_update_gps(void) {
331 #if AHRS_GRAVITY_UPDATE_COORDINATED_TURN && USE_GPS
332  if (gps.fix == GPS_FIX_3D) {
333  ahrs_impl.ltp_vel_norm = SPEED_BFP_OF_REAL(gps.speed_3d / 100.);
334  ahrs_impl.ltp_vel_norm_valid = TRUE;
335  } else {
336  ahrs_impl.ltp_vel_norm_valid = FALSE;
337  }
338 #endif
339 
340 #if AHRS_USE_GPS_HEADING && USE_GPS
341  //got a 3d fix,ground speed > 0.5 m/s and course accuracy is better than 10deg
342  if(gps.fix == GPS_FIX_3D &&
343  gps.gspeed >= 500 &&
344  gps.cacc <= RadOfDeg(10*1e7)) {
345 
346  // gps.course is in rad * 1e7, we need it in rad * 2^INT32_ANGLE_FRAC
347  int32_t course = gps.course * ((1<<INT32_ANGLE_FRAC) / 1e7);
348 
349  /* the assumption here is that there is no side-slip, so heading=course */
350 
351  if (ahrs_impl.heading_aligned) {
352  ahrs_update_heading(course);
353  }
354  else {
355  /* hard reset the heading if this is the first measurement */
356  ahrs_realign_heading(course);
357  }
358  }
359 #endif
360 }

363 void ahrs_update_heading(int32_t heading) {
364 
365  INT32_ANGLE_NORMALIZE(heading);
366 
367  // row 0 of ltp_to_body_rmat = body x-axis in ltp frame
368  // we only consider x and y
369  struct Int32RMat* ltp_to_body_rmat = stateGetNedToBodyRMat_i();
370  struct Int32Vect2 expected_ltp =
371  { RMAT_ELMT((*ltp_to_body_rmat), 0, 0),
372  RMAT_ELMT((*ltp_to_body_rmat), 0, 1) };
373 
374  int32_t heading_x, heading_y;
375  PPRZ_ITRIG_COS(heading_x, heading); // measured course in x-direction
376  PPRZ_ITRIG_SIN(heading_y, heading); // measured course in y-direction
377 
378  // expected_heading cross measured_heading ??
379  struct Int32Vect3 residual_ltp =
380  { 0,
381  0,
382  (expected_ltp.x * heading_y - expected_ltp.y * heading_x)/(1<<INT32_ANGLE_FRAC)};
383 
384  struct Int32Vect3 residual_imu;
385  struct Int32RMat ltp_to_imu_rmat;
386  INT32_RMAT_OF_QUAT(ltp_to_imu_rmat, ahrs_impl.ltp_to_imu_quat);
387  INT32_RMAT_VMULT(residual_imu, ltp_to_imu_rmat, residual_ltp);
388 
389  // residual FRAC = TRIG_FRAC + TRIG_FRAC = 14 + 14 = 28
390  // rate_correction FRAC = RATE_FRAC = 12
391  // 2^12 / 2^28 * 4.0 = 1/2^14
392  // (1<<INT32_ANGLE_FRAC)/2^14 = 1/4
393  ahrs_impl.rate_correction.p += residual_imu.x/4;
394  ahrs_impl.rate_correction.q += residual_imu.y/4;
395  ahrs_impl.rate_correction.r += residual_imu.z/4;
396 
397 
398  /* crude attempt to only update bias if deviation is small
399  * e.g. needed when you only have gps providing heading
400  * and the inital heading is totally different from
401  * the gps course information you get once you have a gps fix.
402  * Otherwise the bias will be falsely "corrected".
403  */
404  int32_t sin_max_angle_deviation;
405  PPRZ_ITRIG_SIN(sin_max_angle_deviation, TRIG_BFP_OF_REAL(RadOfDeg(5.)));
406  if (ABS(residual_ltp.z) < sin_max_angle_deviation)
407  {
408  // residual_ltp FRAC = 2 * TRIG_FRAC = 28
409  // high_rez_bias = RATE_FRAC+28 = 40
410  // 2^40 / 2^28 * 2.5e-4 = 1
411  ahrs_impl.high_rez_bias.p -= residual_imu.x*(1<<INT32_ANGLE_FRAC);
412  ahrs_impl.high_rez_bias.q -= residual_imu.y*(1<<INT32_ANGLE_FRAC);
413  ahrs_impl.high_rez_bias.r -= residual_imu.z*(1<<INT32_ANGLE_FRAC);
414 
415  INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
416  }

419 void ahrs_realign_heading(int32_t heading) {
420 
421  struct Int32Quat ltp_to_body_quat = *stateGetNedToBodyQuat_i();
422 
423  /* quaternion representing only the heading rotation from ltp to body */
424  struct Int32Quat q_h_new;
425  q_h_new.qx = 0;
426  q_h_new.qy = 0;
427  PPRZ_ITRIG_SIN(q_h_new.qz, heading/2);
428  PPRZ_ITRIG_COS(q_h_new.qi, heading/2);
429 
430  /* quaternion representing current heading only */
431  struct Int32Quat q_h;
432  QUAT_COPY(q_h, ltp_to_body_quat);
433  q_h.qx = 0;
434  q_h.qy = 0;
435  INT32_QUAT_NORMALIZE(q_h);
436 
437  /* quaternion representing rotation from current to new heading */
438  struct Int32Quat q_c;
439  INT32_QUAT_INV_COMP_NORM_SHORTEST(q_c, q_h, q_h_new);
440 
441  /* correct current heading in body frame */
442  struct Int32Quat q;
443  INT32_QUAT_COMP_NORM_SHORTEST(q, q_c, ltp_to_body_quat);
444  QUAT_COPY(ltp_to_body_quat, q);
445 
446  /* compute ltp to imu rotations */
447  INT32_QUAT_COMP(ahrs_impl.ltp_to_imu_quat, ltp_to_body_quat, imu.body_to_imu_quat);
448 
449  /* Set state */
450  stateSetNedToBodyQuat_i(&ltp_to_body_quat);
451 
452  ahrs_impl.heading_aligned = TRUE;
453 }

456 /* Rotate angles and rates from imu to body frame and set state */
457 __attribute__ ((always_inline)) static inline void set_body_state_from_quat(void) {
458  /* Compute LTP to BODY quaternion */
459  struct Int32Quat ltp_to_body_quat;
460  INT32_QUAT_COMP_INV(ltp_to_body_quat, ahrs_impl.ltp_to_imu_quat, imu.body_to_imu_quat);
461  /* Set state */
462 #ifdef AHRS_UPDATE_FW_ESTIMATOR
463  struct Int32Eulers neutrals_to_body_eulers = {
464  ANGLE_BFP_OF_REAL(ins_roll_neutral),
465  ANGLE_BFP_OF_REAL(ins_pitch_neutral),
466  0 };
467  struct Int32Quat neutrals_to_body_quat, ltp_to_neutrals_quat;
468  INT32_QUAT_OF_EULERS(neutrals_to_body_quat, neutrals_to_body_eulers);
469  INT32_QUAT_NORMALIZE(neutrals_to_body_quat);
470  INT32_QUAT_COMP_INV(ltp_to_neutrals_quat, ltp_to_body_quat, neutrals_to_body_quat);
471  stateSetNedToBodyQuat_i(&ltp_to_neutrals_quat);
472 #else
473  stateSetNedToBodyQuat_i(&ltp_to_body_quat);
474 #endif
475 
476  /* compute body rates */
477  struct Int32Rates body_rate;
478  INT32_RMAT_TRANSP_RATEMULT(body_rate, imu.body_to_imu_rmat, ahrs_impl.imu_rate);
479  /* Set state */
480  stateSetBodyRates_i(&body_rate);
481 }
