32 static struct Int32Rates gyro_sum;

38 void ahrs_aligner_init(void) {
39 
40  ahrs_aligner.status = AHRS_ALIGNER_RUNNING;
41  INT_RATES_ZERO(gyro_sum);
42  INT_VECT3_ZERO(accel_sum);
43  INT_VECT3_ZERO(mag_sum);
44  samples_idx = 0;
45  ahrs_aligner.noise = 0;
46  ahrs_aligner.low_noise_cnt = 0;
47 }

50 #define LOW_NOISE_THRESHOLD 90000
51 #endif
52 #ifndef LOW_NOISE_TIME
53 #define LOW_NOISE_TIME 5
54 #endif

56 void ahrs_aligner_run(void) {
57 
58  RATES_ADD(gyro_sum, imu.gyro);
59  VECT3_ADD(accel_sum, imu.accel);
60  VECT3_ADD(mag_sum, imu.mag);
61 
62  ref_sensor_samples[samples_idx] = imu.accel.z;
63  samples_idx++;
64 
65 #ifdef AHRS_ALIGNER_LED
66  RunOnceEvery(50, {LED_TOGGLE(AHRS_ALIGNER_LED);});
67 #endif
68 
69  if (samples_idx >= SAMPLES_NB) {
70  int32_t avg_ref_sensor = accel_sum.z;
71  if ( avg_ref_sensor >= 0)
72  avg_ref_sensor += SAMPLES_NB / 2;
73  else
74  avg_ref_sensor -= SAMPLES_NB / 2;
75  avg_ref_sensor /= SAMPLES_NB;
76 
77  ahrs_aligner.noise = 0;
78  int i;
79  for (i=0; i<SAMPLES_NB; i++) {
80  int32_t diff = ref_sensor_samples[i] - avg_ref_sensor;
81  ahrs_aligner.noise += abs(diff);
82  }
83 
84  RATES_SDIV(ahrs_aligner.lp_gyro, gyro_sum, SAMPLES_NB);
85  VECT3_SDIV(ahrs_aligner.lp_accel, accel_sum, SAMPLES_NB);
86  VECT3_SDIV(ahrs_aligner.lp_mag, mag_sum, SAMPLES_NB);
87 
88  INT_RATES_ZERO(gyro_sum);
89  INT_VECT3_ZERO(accel_sum);
90  INT_VECT3_ZERO(mag_sum);
91  samples_idx = 0;
92 
93  if (ahrs_aligner.noise < LOW_NOISE_THRESHOLD)
94  ahrs_aligner.low_noise_cnt++;
95  else
96  if ( ahrs_aligner.low_noise_cnt > 0)
97  ahrs_aligner.low_noise_cnt--;
98 
99  if (ahrs_aligner.low_noise_cnt > LOW_NOISE_TIME) {
100  ahrs_aligner.status = AHRS_ALIGNER_LOCKED;
101 #ifdef AHRS_ALIGNER_LED
102  LED_ON(AHRS_ALIGNER_LED);
103 #endif
104  }
105  }
