00098 void stabilization_attitude_run(bool_t  in_flight) {
00099 
00100   stabilization_attitude_ref_update();
00101 
00102   /* Compute feedforward */
00103   stabilization_att_ff_cmd[COMMAND_ROLL] =
00104     stabilization_gains.dd.x * stab_att_ref_accel.p / 32.;
00105   stabilization_att_ff_cmd[COMMAND_PITCH] =
00106     stabilization_gains.dd.y * stab_att_ref_accel.q / 32.;
00107   stabilization_att_ff_cmd[COMMAND_YAW] =
00108     stabilization_gains.dd.z * stab_att_ref_accel.r / 32.;
00109 
00110   /* Compute feedback                  */
00111   /* attitude error            */
00112   struct FloatEulers att_float* = stateGetNedToBodyEulers_f();
00113   struct FloatEulers att_err;
00114   EULERS_DIFF(att_err, stab_att_ref_euler, *att_float);
00115   FLOAT_ANGLE_NORMALIZE(att_err.psi);
00116 
00117   if (in_flight) {
00118     /* update integrator */
00119     EULERS_ADD(stabilization_att_sum_err, att_err);
00120     EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
00121   }
00122   else {
00123     FLOAT_EULERS_ZERO(stabilization_att_sum_err);
00124   }
00125 
00126   /*  rate error                */
00127   struct FloatRates* rate_float = stateGetBodyRates_f();
00128   struct FloatRates rate_err;
00129   RATES_DIFF(rate_err, stab_att_ref_rate, *rate_float);
00130 
00131   /*  PID                  */
00132 
00133   stabilization_att_fb_cmd[COMMAND_ROLL] =
00134     stabilization_gains.p.x  * att_err.phi +
00135     stabilization_gains.d.x  * rate_err.p +
00136     stabilization_gains.i.x  * stabilization_att_sum_err.phi / 1024.;
00137 
00138   stabilization_att_fb_cmd[COMMAND_PITCH] =
00139     stabilization_gains.p.y  * att_err.theta +
00140     stabilization_gains.d.y  * rate_err.q +
00141     stabilization_gains.i.y  * stabilization_att_sum_err.theta / 1024.;
00142 
00143   stabilization_att_fb_cmd[COMMAND_YAW] =
00144     stabilization_gains.p.z  * att_err.psi +
00145     stabilization_gains.d.z  * rate_err.r +
00146     stabilization_gains.i.z  * stabilization_att_sum_err.psi / 1024.;
00147 
00148 
00149   stabilization_cmd[COMMAND_ROLL] =
00150     (stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL])/16.;
00151   stabilization_cmd[COMMAND_PITCH] =
00152     (stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH])/16.;
00153   stabilization_cmd[COMMAND_YAW] =
00154     (stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW])/16.;
00155 
00156 }
