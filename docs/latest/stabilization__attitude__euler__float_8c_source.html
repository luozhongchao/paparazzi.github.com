84 void stabilization_attitude_run(bool_t in_flight) {
85 
86  stabilization_attitude_ref_update();
87 
88  /* Compute feedforward */
89  stabilization_att_ff_cmd[COMMAND_ROLL] =
90  stabilization_gains.dd.x * stab_att_ref_accel.p / 32.;
91  stabilization_att_ff_cmd[COMMAND_PITCH] =
92  stabilization_gains.dd.y * stab_att_ref_accel.q / 32.;
93  stabilization_att_ff_cmd[COMMAND_YAW] =
94  stabilization_gains.dd.z * stab_att_ref_accel.r / 32.;
95 
96  /* Compute feedback */
97  /* attitude error */
98  struct FloatEulers att_float* = stateGetNedToBodyEulers_f();
99  struct FloatEulers att_err;
100  EULERS_DIFF(att_err, stab_att_ref_euler, *att_float);
101  FLOAT_ANGLE_NORMALIZE(att_err.psi);
102 
103  if (in_flight) {
104  /* update integrator */
105  EULERS_ADD(stabilization_att_sum_err, att_err);
106  EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
107  }
108  else {
109  FLOAT_EULERS_ZERO(stabilization_att_sum_err);
110  }
111 
112  /* rate error */
113  struct FloatRates* rate_float = stateGetBodyRates_f();
114  struct FloatRates rate_err;
115  RATES_DIFF(rate_err, stab_att_ref_rate, *rate_float);
116 
117  /* PID */
118 
119  stabilization_att_fb_cmd[COMMAND_ROLL] =
120  stabilization_gains.p.x * att_err.phi +
121  stabilization_gains.d.x * rate_err.p +
122  stabilization_gains.i.x * stabilization_att_sum_err.phi / 1024.;
123 
124  stabilization_att_fb_cmd[COMMAND_PITCH] =
125  stabilization_gains.p.y * att_err.theta +
126  stabilization_gains.d.y * rate_err.q +
127  stabilization_gains.i.y * stabilization_att_sum_err.theta / 1024.;
128 
129  stabilization_att_fb_cmd[COMMAND_YAW] =
130  stabilization_gains.p.z * att_err.psi +
131  stabilization_gains.d.z * rate_err.r +
132  stabilization_gains.i.z * stabilization_att_sum_err.psi / 1024.;
133 
134 
135  stabilization_cmd[COMMAND_ROLL] =
136  (stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL])/16.;
137  stabilization_cmd[COMMAND_PITCH] =
138  (stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH])/16.;
139  stabilization_cmd[COMMAND_YAW] =
140  (stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW])/16.;
141 
142 }
