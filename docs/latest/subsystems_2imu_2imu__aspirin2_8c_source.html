66 void imu_periodic(void)
67 {
68 
69  if (imu_aspirin2.status == Aspirin2StatusUninit) {
70  mpu_configure();
71  imu_aspirin2.status = Aspirin2StatusIdle;
72 
73  aspirin2_mpu60x0.output_length = 22;
74  aspirin2_mpu60x0.input_length = 22;
75  aspirin2_mpu60x0.output_buf[0] = MPU60X0_REG_INT_STATUS + MPU60X0_SPI_READ;
76  for (int i=1; i<aspirin2_mpu60x0.output_length; i++) {
77  aspirin2_mpu60x0.output_buf[i] = 0;
78  }
79  }
80  else {
81  spi_submit(&(MPU6000_SPI_DEV), &aspirin2_mpu60x0);
82  }
83 }

101 static inline void mpu_wait_slave4_ready(void)
102 {
103  uint8_t ret = 0x80;
104  while (ret & 0x80)
105  {
106  aspirin2_mpu60x0.output_buf[0] = MPU60X0_REG_I2C_SLV4_CTRL | MPU60X0_SPI_READ ;
107  aspirin2_mpu60x0.output_buf[1] = 0;
108  spi_submit(&(MPU6000_SPI_DEV), &aspirin2_mpu60x0);
109 
110  // FIXME: no busy waiting! if really needed add a timeout!!!!
111  while(aspirin2_mpu60x0.status != SPITransSuccess);
112 
113  ret = aspirin2_mpu60x0.input_buf[1];
114  }
115 }

117 static void mpu_configure(void)
118 {
119  aspirin2_mpu60x0.output_length = 2;
120  aspirin2_mpu60x0.input_length = 2;
121 
122  ///////////////////
123  // Reset the MPU
124  mpu_set( MPU60X0_REG_PWR_MGMT_1,
125  0x01 << 7); // -device reset
126  mpu_set( MPU60X0_REG_USER_CTRL,
127  (1 << 2) | // Trigger a FIFO_RESET
128  (1 << 1) | // Trigger a I2C_MST_RESET
129  (1 << 0) ); // Trigger a SIG_COND_RESET
130 
131  ///////////////////
132  // Configure power:
133 
134  // MPU60X0_REG_PWR_MGMT_1
135  mpu_set( MPU60X0_REG_PWR_MGMT_1,
136  0x01); // -switch to gyroX clock
137 
138  // Wait for the new clock to stabilize.
139  // FIXME: This must not be a delay!
140  // It should be done using the MPU-6000 interrupt!
141  {for (int i = 0; i < 1000000; i++) { asm("nop"); }}
142 
143  // MPU60X0_REG_PWR_MGMT_2: Nothing should be in standby: default OK
144  // -No standby and no wake timer
145 
146  /////////////////////////
147  // Measurement Settings
148 
149 #if PERIODIC_FREQUENCY == 60
150 // Accelerometer: Bandwidth 44Hz, Delay 4.9ms
151 // Gyroscope: Bandwidth 42Hz, Delay 4.8ms sampling 1Khz
152 # define MPU_DIG_FILTER 3
153 // -100Hz output = 1kHz / (9 + 1)
154 # define MPU_SMPLRT_DIV 9
155 #else
156 # if PERIODIC_FREQUENCY == 120
157 // Accelerometer: Bandwidth 44Hz, Delay 4.9ms
158 // Gyroscope: Bandwidth 42Hz, Delay 4.8ms sampling 1Khz
159 # define MPU_DIG_FILTER 3
160 // -100Hz output = 1kHz / (9 + 1)
161 # define MPU_SMPLRT_DIV 9
162 # else
163 # if PERIODIC_FREQUENCY == 512
164 // Accelerometer: Bandwidth 260Hz, Delay 0ms
165 // Gyroscope: Bandwidth 256Hz, Delay 0.89ms sampling 8Khz
166 # define MPU_DIG_FILTER 0
167 // -500Hz output = 1kHz / (1 + 1)
168 # define MPU_SMPLRT_DIV 1
169 # else
170 # error PERIODIC_FREQUENCY should be either 60Hz, 120Hz or 512Hz. Otherwise manually fix the sensor rates
171 # endif
172 # endif
173 #endif
174  aspirin2_mpu60x0.output_buf[1] = (2 << 3) | (MPU_DIG_FILTER << 0);
175  spi_submit(&(MPU6000_SPI_DEV), &aspirin2_mpu60x0);
176  mpu_set( MPU60X0_REG_CONFIG,
177  (2 << 3) | // Fsync / ext sync on gyro X (bit 3->6)
178  (MPU_DIG_FILTER << 0) ); // Low-Pass Filter
179 
180  // MPU60X0_REG_SMPLRT_DIV
181  mpu_set( MPU60X0_REG_SMPLRT_DIV, MPU_SMPLRT_DIV);
182 
183  // MPU60X0_REG_GYRO_CONFIG
184  mpu_set( MPU60X0_REG_GYRO_CONFIG,
185  (3 << 3) ); // -2000deg/sec
186 
187  // MPU60X0_REG_ACCEL_CONFIG
188  mpu_set( MPU60X0_REG_ACCEL_CONFIG,
189  (0 << 0) | // No HPFL
190  (3 << 3) ); // Full Scale = 16g
191 
192 #ifndef MPU6000_NO_SLAVES
193 
194  /////////////////////////////////////
195  // SPI Slave Configuration Section
196 
197  // Power the Aux I2C Circuit:
198  // MPU60X0_REG_AUX_VDDIO = 0 (good on startup): (0 << 7); // MPU6000: 0=Vdd. MPU6050 : 0=VLogic 1=Vdd
199 
200  // MPU60X0_REG_USER_CTRL:
201  mpu_set( MPU60X0_REG_USER_CTRL,
202  (1 << 5) | // I2C_MST_EN: Enable Aux I2C Master Mode
203  (1 << 4) | // I2C_IF_DIS: Disable I2C on primary interface
204  (0 << 1) ); // Trigger a I2C_MST_RESET
205 
206  // Enable the aux i2c
207  mpu_set( MPU60X0_REG_I2C_MST_CTRL,
208  (0 << 7) | // no multimaster
209  (0 << 6) | // do not delay IRQ waiting for all external slaves
210  (0 << 5) | // no slave 3 FIFO
211  (0 << 4) | // restart or stop/start from one slave to another: read -> write is always stop/start
212  (8 << 0) ); // 0=348kHz 8=256kHz, 9=500kHz
213 
214  mpu_set( MPU60X0_REG_I2C_MST_DELAY,
215  (0 << 2) | // No Delay Slave 2
216  (1 << 3) ); // Delay Slave 3
