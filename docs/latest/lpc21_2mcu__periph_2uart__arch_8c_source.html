00067 void uart_transmit(struct uart_periph* p, uint8_t data ) {
00068   uint16_t temp;
00069   unsigned cpsr;
00070 
00071   temp = (p->tx_insert_idx + 1) % UART_TX_BUFFER_SIZE;
00072 
00073   if (temp == p->tx_extract_idx) {
00074     return;                          // no room
00075   }
00076 
00077   cpsr = disableIRQ();                                // disable global interrupts
00078   ((uartRegs_t *)(p->reg_addr))->ier &= ~UIER_ETBEI;  // disable TX interrupts
00079   restoreIRQ(cpsr);                                   // restore global interrupts
00080 
00081   // check if in process of sending data
00082   if (p->tx_running) {
00083     // add to queue
00084     p->tx_buf[p->tx_insert_idx] = data;
00085     p->tx_insert_idx = temp;
00086   } else {
00087     // set running flag and write to output register
00088     p->tx_running = 1;
00089     ((uartRegs_t *)(p->reg_addr))->thr = data;
00090   }
00091 
00092   cpsr = disableIRQ();                              // disable global interrupts
00093   ((uartRegs_t *)(p->reg_addr))->ier |= UIER_ETBEI; // enable TX interrupts
00094   restoreIRQ(cpsr);                                 // restore global interrupts
00095 }

00097 static inline void uart_ISR(struct uart_periph* p)
00098 {
00099   uint8_t iid;
00100   // loop until not more interrupt sources
00101   while (((iid = ((uartRegs_t *)(p->reg_addr))->iir) & UIIR_NO_INT) == 0)
00102   {
00103     // identify & process the highest priority interrupt
00104     switch (iid & UIIR_ID_MASK)
00105     {
00106       case UIIR_RLS_INT:                // Receive Line Status
00107         ((uartRegs_t *)(p->reg_addr))->lsr; // read LSR to clear
00108         break;
00109 
00110       case UIIR_CTI_INT:                // Character Timeout Indicator
00111       case UIIR_RDA_INT:                // Receive Data Available
00112         do
00113         {
00114           uint16_t temp;
00115 
00116           // calc next insert index & store character
00117           temp = (p->rx_insert_idx + 1) % UART_RX_BUFFER_SIZE;
00118           p->rx_buf[p->rx_insert_idx] = ((uartRegs_t *)(p->reg_addr))->rbr;
00119 
00120           // check for more room in queue
00121           if (temp != p->rx_extract_idx)
00122             p->rx_insert_idx = temp; // update insert index
00123         }
00124         while (((uartRegs_t *)(p->reg_addr))->lsr & ULSR_RDR);
00125 
00126         break;
00127 
00128       case UIIR_THRE_INT:               // Transmit Holding Register Empty
00129         while (((uartRegs_t *)(p->reg_addr))->lsr & ULSR_THRE)
00130         {
00131           // check if more data to send
00132           if (p->tx_insert_idx != p->tx_extract_idx)
00133           {
00134             ((uartRegs_t *)(p->reg_addr))->thr = p->tx_buf[p->tx_extract_idx];
00135             p->tx_extract_idx++;
00136             p->tx_extract_idx %= UART_TX_BUFFER_SIZE;
00137           }
00138           else
00139           {
00140             // no
00141             p->tx_running = 0;       // clear running flag
00142             break;
00143           }
00144         }
00145 
00146         break;
00147 
00148       default:                          // Unknown
00149         ((uartRegs_t *)(p->reg_addr))->lsr;
00150         ((uartRegs_t *)(p->reg_addr))->rbr;
00151         break;
00152     }
00153   }
00154 }
