00032 static inline void uart_disable_interrupts(struct uart_periph* p) {
00033   // disable interrups
00034   ((uartRegs_t *)(p->reg_addr))->ier = 0x00;  // disable all interrupts
00035   ((uartRegs_t *)(p->reg_addr))->iir;         // clear interrupt ID
00036   ((uartRegs_t *)(p->reg_addr))->rbr;         // clear receive register
00037   ((uartRegs_t *)(p->reg_addr))->lsr;         // clear line status register
00038 }

00045 static inline void uart_set_baudrate(struct uart_periph* p, uint32_t baud) {
00046   // set the baudrate
00047   ((uartRegs_t *)(p->reg_addr))->lcr = ULCR_DLAB_ENABLE;     // select divisor latches
00048   ((uartRegs_t *)(p->reg_addr))->dll = (uint8_t)baud;        // set for baud low byte
00049   ((uartRegs_t *)(p->reg_addr))->dlm = (uint8_t)(baud >> 8); // set for baud high byte
00050 
00051   // set the number of characters and other
00052   // user specified operating parameters
00053   // For now: hard wired configuration 8 bits 1 stop no parity
00054   //          fifo triger -> 8 bytes
00055   ((uartRegs_t *)(p->reg_addr))->lcr = (UART_8N1 & ~ULCR_DLAB_ENABLE);
00056   ((uartRegs_t *)(p->reg_addr))->fcr = UART_FIFO_8;

00065 void uart_transmit(struct uart_periph* p, uint8_t data ) {
00066   uint16_t temp;
00067   unsigned cpsr;
00068 
00069   temp = (p->tx_insert_idx + 1) % UART_TX_BUFFER_SIZE;
00070 
00071   if (temp == p->tx_extract_idx)
00072     return;                          // no room
00073 
00074   cpsr = disableIRQ();                                // disable global interrupts
00075   ((uartRegs_t *)(p->reg_addr))->ier &= ~UIER_ETBEI;  // disable TX interrupts
00076   restoreIRQ(cpsr);                                   // restore global interrupts
00077 
00078   // check if in process of sending data
00079   if (p->tx_running) {
00080     // add to queue
00081     p->tx_buf[p->tx_insert_idx] = data;
00082     p->tx_insert_idx = temp;
00083   } else {
00084     // set running flag and write to output register
00085     p->tx_running = 1;
00086     ((uartRegs_t *)(p->reg_addr))->thr = data;
00087   }
00088 
00089   cpsr = disableIRQ();                              // disable global interrupts
00090   ((uartRegs_t *)(p->reg_addr))->ier |= UIER_ETBEI; // enable TX interrupts
00091   restoreIRQ(cpsr);                                 // restore global interrupts
00092 }

00094 static inline void uart_ISR(struct uart_periph* p)
00095 {
00096   uint8_t iid;
00097 
00098   // loop until not more interrupt sources
00099   while (((iid = ((uartRegs_t *)(p->reg_addr))->iir) & UIIR_NO_INT) == 0)
00100   {
00101     // identify & process the highest priority interrupt
00102     switch (iid & UIIR_ID_MASK)
00103     {
00104       case UIIR_RLS_INT:                // Receive Line Status
00105         ((uartRegs_t *)(p->reg_addr))->lsr; // read LSR to clear
00106         break;
00107 
00108       case UIIR_CTI_INT:                // Character Timeout Indicator
00109       case UIIR_RDA_INT:                // Receive Data Available
00110         do
00111         {
00112           uint16_t temp;
00113 
00114           // calc next insert index & store character
00115           temp = (p->rx_insert_idx + 1) % UART_RX_BUFFER_SIZE;
00116           p->rx_buf[p->rx_insert_idx] = ((uartRegs_t *)(p->reg_addr))->rbr;
00117 
00118           // check for more room in queue
00119           if (temp != p->rx_extract_idx)
00120             p->rx_insert_idx = temp; // update insert index
00121         }
00122         while (((uartRegs_t *)(p->reg_addr))->lsr & ULSR_RDR);
00123 
00124         break;
00125 
00126       case UIIR_THRE_INT:               // Transmit Holding Register Empty
00127         while (((uartRegs_t *)(p->reg_addr))->lsr & ULSR_THRE)
00128         {
00129           // check if more data to send
00130           if (p->tx_insert_idx != p->tx_extract_idx)
00131           {
00132             ((uartRegs_t *)(p->reg_addr))->thr = p->tx_buf[p->tx_extract_idx];
00133             p->tx_extract_idx++;
00134             p->tx_extract_idx %= UART_TX_BUFFER_SIZE;
00135           }
00136           else
00137           {
00138             // no
00139             p->tx_running = 0;       // clear running flag
00140             break;
00141           }
00142         }
00143 
00144         break;
00145 
00146       default:                          // Unknown
00147         ((uartRegs_t *)(p->reg_addr))->lsr;
00148         ((uartRegs_t *)(p->reg_addr))->rbr;
00149         break;
00150     }
00151   }
00152 }
