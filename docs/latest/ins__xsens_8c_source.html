00263 void xsens_periodic(void) {
00264   if (xsens_configured > 0)
00265     {
00266       switch (xsens_configured)
00267         {
00268         case 20:
00269           /* send mode and settings to MT */
00270           XSENS_GoToConfig();
00271           XSENS_SetOutputMode(xsens_output_mode);
00272           XSENS_SetOutputSettings(xsens_output_settings);
00273           break;
00274         case 18:
00275           // Give pulses on SyncOut
00276           XSENS_SetSyncOutSettings(0,0x0002);
00277           break;
00278         case 17:
00279           // 1 pulse every 100 samples
00280           XSENS_SetSyncOutSettings(1,100);
00281           break;
00282         case 2:
00283           XSENS_ReqLeverArmGps();
00284           break;
00285 
00286         case 6:
00287           XSENS_ReqMagneticDeclination();
00288           break;
00289 
00290         case 13:
00291 #ifdef AHRS_H_X
00292 #pragma message "Sending XSens Magnetic Declination."
00293           xsens_declination = atan2(AHRS_H_Y, AHRS_H_X);
00294           XSENS_SetMagneticDeclination(xsens_declination);
00295 #endif
00296           break;
00297         case 12:
00298 #ifdef GPS_IMU_LEVER_ARM_X
00299 #pragma message "Sending XSens GPS Arm."
00300           XSENS_SetLeverArmGps(GPS_IMU_LEVER_ARM_X,GPS_IMU_LEVER_ARM_Y,GPS_IMU_LEVER_ARM_Z);
00301 #endif
00302           break;
00303         case 10:
00304           {
00305             uint8_t baud = 1;
00306             XSENS_SetBaudrate(baud);
00307           }
00308           break;
00309 
00310         case 1:
00311           XSENS_GoToMeasurment();
00312           break;
00313 
00314         default:
00315           break;
00316         }
00317       xsens_configured--;
00318       return;
00319     }
00320 
00321   RunOnceEvery(100,XSENS_ReqGPSStatus());
00322 }
