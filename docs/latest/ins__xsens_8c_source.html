116 /* output mode : calibrated, orientation, position, velocity, status
117  * -----------
118  *
119  * bit 0 temp
120  * bit 1 calibrated
121  * bit 2 orentation
122  * bit 3 aux
123  *
124  * bit 4 position
125  * bit 5 velocity
126  * bit 6-7 Reserved
127  *
128  * bit 8-10 Reserved
129  * bit 11 status
130  *
131  * bit 12 GPS PVT+baro
132  * bit 13 Reserved
133  * bit 14 Raw
134  * bit 15 Reserved
135  */

137 #ifndef XSENS_OUTPUT_MODE
138 #define XSENS_OUTPUT_MODE 0x1836
139 #endif
140 /* output settings : timestamp, euler, acc, rate, mag, float, no aux, lla, m/s, NED
141  * -----------
142  *
143  * bit 01 0=none, 1=sample counter, 2=utc, 3=sample+utc
144  * bit 23 0=quaternion, 1=euler, 2=DCM
145  *
146  * bit 4 1=disable acc output
147  * bit 5 1=disable gyro output
148  * bit 6 1=disable magneto output
149  * bit 7 Reserved
150  *
151  * bit 89 0=float, 1=fixedpoint12.20, 2=fixedpoint16.32
152  * bit 10 1=disable aux analog 1
153  * bit 11 1=disable aux analog 2
154  *
155  * bit 12-15 0-only: 14-16 WGS84
156  *
157  * bit 16-19 0-only: 16-18 m/s XYZ
158  *
159  * bit 20-23 Reserved
160  *
161  * bit 24-27 Reseverd
162  *
163  * bit 28-30 Reseverd
164  * bit 31 0=X-North-Z-Up, 1=North-East-Down
165  */

269 void xsens_periodic(void) {
270  if (xsens_configured > 0)
271  {
272  switch (xsens_configured)
273  {
274  case 20:
275  /* send mode and settings to MT */
276  XSENS_GoToConfig();
277  XSENS_SetOutputMode(xsens_output_mode);
278  XSENS_SetOutputSettings(xsens_output_settings);
279  break;
280  case 18:
281  // Give pulses on SyncOut
282  XSENS_SetSyncOutSettings(0,0x0002);
283  break;
284  case 17:
285  // 1 pulse every 100 samples
286  XSENS_SetSyncOutSettings(1,100);
287  break;
288  case 2:
289  XSENS_ReqLeverArmGps();
290  break;
291 
292  case 6:
293  XSENS_ReqMagneticDeclination();
294  break;
295 
296  case 13:
297 #ifdef AHRS_H_X
298 #pragma message "Sending XSens Magnetic Declination."
299  xsens_declination = atan2(AHRS_H_Y, AHRS_H_X);
300  XSENS_SetMagneticDeclination(xsens_declination);
301 #endif
302  break;
303  case 12:
304 #ifdef GPS_IMU_LEVER_ARM_X
305 #pragma message "Sending XSens GPS Arm."
306  XSENS_SetLeverArmGps(GPS_IMU_LEVER_ARM_X,GPS_IMU_LEVER_ARM_Y,GPS_IMU_LEVER_ARM_Z);
307 #endif
308  break;
309  case 10:
310  {
311  uint8_t baud = 1;
312  XSENS_SetBaudrate(baud);
313  }
314  break;
315 
316  case 1:
317  XSENS_GoToMeasurment();
318  break;
319 
320  default:
321  break;
322  }
323  xsens_configured--;
324  return;
325  }
326 
327  RunOnceEvery(100,XSENS_ReqGPSStatus());
328 }

363 void handle_ins_msg(void) {
364 
365 #if USE_INS_MODULE
366  update_fw_estimator();
367 #endif
368 
369 #if USE_IMU
370 #ifdef XSENS_BACKWARDS
371  if (imu_xsens.gyro_available) {
372  RATES_ASSIGN(imu.gyro_unscaled, -RATE_BFP_OF_REAL(ins_p), -RATE_BFP_OF_REAL(ins_q), RATE_BFP_OF_REAL(ins_r));
373  }
374  if (imu_xsens.accel_available) {
375  VECT3_ASSIGN(imu.accel_unscaled, -ACCEL_BFP_OF_REAL(ins_ax), -ACCEL_BFP_OF_REAL(ins_ay), ACCEL_BFP_OF_REAL(ins_az));
376  }
377  if (imu_xsens.mag_available) {
378  VECT3_ASSIGN(imu.mag_unscaled, -MAG_BFP_OF_REAL(ins_mx), -MAG_BFP_OF_REAL(ins_my), MAG_BFP_OF_REAL(ins_mz));
379  }
380 #else
381  if (imu_xsens.gyro_available) {
382  RATES_ASSIGN(imu.gyro_unscaled, RATE_BFP_OF_REAL(ins_p), RATE_BFP_OF_REAL(ins_q), RATE_BFP_OF_REAL(ins_r));
383  }
384  if (imu_xsens.accel_available) {
385  VECT3_ASSIGN(imu.accel_unscaled, ACCEL_BFP_OF_REAL(ins_ax), ACCEL_BFP_OF_REAL(ins_ay), ACCEL_BFP_OF_REAL(ins_az));
386  }
387  if (imu_xsens.mag_available) {
388  VECT3_ASSIGN(imu.mag_unscaled, MAG_BFP_OF_REAL(ins_mx), MAG_BFP_OF_REAL(ins_my), MAG_BFP_OF_REAL(ins_mz));
389  }
390 #endif /* XSENS_BACKWARDS */
391 #endif /* USE_IMU */
392 
393 #if USE_GPS_XSENS
394 
395  // Horizontal speed
396  float fspeed = sqrt(ins_vx*ins_vx + ins_vy*ins_vy);
397  if (gps.fix != GPS_FIX_3D) {
398  fspeed = 0;
399  }
400  gps.gspeed = fspeed * 100.;
401  gps.speed_3d = (uint16_t)(sqrt(ins_vx*ins_vx + ins_vy*ins_vy + ins_vz*ins_vz) * 100);
402 
403  float fcourse = atan2f((float)ins_vy, (float)ins_vx);
404  gps.course = fcourse * 1e7;
405 #endif // USE_GPS_XSENS
406 }

408 void parse_ins_msg(void) {
409  uint8_t offset = 0;
410  if (xsens_id == XSENS_ReqOutputModeAck_ID) {
411  xsens_output_mode = XSENS_ReqOutputModeAck_mode(xsens_msg_buf);
412  }
413  else if (xsens_id == XSENS_ReqOutputSettings_ID) {
414  xsens_output_settings = XSENS_ReqOutputSettingsAck_settings(xsens_msg_buf);
415  }
416  else if (xsens_id == XSENS_ReqMagneticDeclinationAck_ID) {
417  xsens_declination = DegOfRad (XSENS_ReqMagneticDeclinationAck_declination(xsens_msg_buf) );
418 
419  DOWNLINK_SEND_IMU_MAG_SETTINGS(DefaultChannel, DefaultDevice,&xsens_declination,&xsens_declination,&xsens_gps_arm_x,&xsens_gps_arm_y,&xsens_gps_arm_z);
420  }
421  else if (xsens_id == XSENS_ReqLeverArmGpsAck_ID) {
422  xsens_gps_arm_x = XSENS_ReqLeverArmGpsAck_x(xsens_msg_buf);
423  xsens_gps_arm_y = XSENS_ReqLeverArmGpsAck_y(xsens_msg_buf);
424  xsens_gps_arm_z = XSENS_ReqLeverArmGpsAck_z(xsens_msg_buf);
425 
426  DOWNLINK_SEND_IMU_MAG_SETTINGS(DefaultChannel, DefaultDevice,&xsens_declination,&xsens_declination,&xsens_gps_arm_x,&xsens_gps_arm_y,&xsens_gps_arm_z);
427  }
428  else if (xsens_id == XSENS_Error_ID) {
429  xsens_errorcode = XSENS_Error_errorcode(xsens_msg_buf);
430  }
431 #if USE_GPS_XSENS
432  else if (xsens_id == XSENS_GPSStatus_ID) {
433  gps.nb_channels = XSENS_GPSStatus_nch(xsens_msg_buf);
434  gps.num_sv = 0;
435 
436  uint8_t i;
437  // Do not write outside buffer
438  for(i = 0; i < Min(gps.nb_channels, GPS_NB_CHANNELS); i++) {
439  uint8_t ch = XSENS_GPSStatus_chn(xsens_msg_buf,i);
440  if (ch > gps.nb_channels) continue;
441  gps.svinfos[ch].svid = XSENS_GPSStatus_svid(xsens_msg_buf, i);
442  gps.svinfos[ch].flags = XSENS_GPSStatus_bitmask(xsens_msg_buf, i);
443  gps.svinfos[ch].qi = XSENS_GPSStatus_qi(xsens_msg_buf, i);
444  gps.svinfos[ch].cno = XSENS_GPSStatus_cnr(xsens_msg_buf, i);
445  if (gps.svinfos[ch].flags > 0)
446  {
447  gps.num_sv++;
448  }
449  }
450  }
451 #endif
452  else if (xsens_id == XSENS_MTData_ID) {
453  /* test RAW modes else calibrated modes */
454  //if ((XSENS_MASK_RAWInertial(xsens_output_mode)) || (XSENS_MASK_RAWGPS(xsens_output_mode))){
455  if (XSENS_MASK_RAWInertial(xsens_output_mode)) {
456  ins_p = XSENS_DATA_RAWInertial_gyrX(xsens_msg_buf,offset);
457  ins_q = XSENS_DATA_RAWInertial_gyrY(xsens_msg_buf,offset);
458  ins_r = XSENS_DATA_RAWInertial_gyrZ(xsens_msg_buf,offset);
459 #if USE_IMU
460  imu_xsens.gyro_available = TRUE;
461 #endif
462  offset += XSENS_DATA_RAWInertial_LENGTH;
463  }
464  if (XSENS_MASK_RAWGPS(xsens_output_mode)) {
465 #if USE_GPS_XSENS_RAW_DATA && USE_GPS_XSENS
466 
467  gps.week = 0; // FIXME
468  gps.tow = XSENS_DATA_RAWGPS_itow(xsens_msg_buf,offset) * 10;
469  gps.lla_pos.lat = RadOfDeg(XSENS_DATA_RAWGPS_lat(xsens_msg_buf,offset));
470  gps.lla_pos.lon = RadOfDeg(XSENS_DATA_RAWGPS_lon(xsens_msg_buf,offset));
471  gps.lla_pos.alt = XSENS_DATA_RAWGPS_alt(xsens_msg_buf,offset);
472 
473  /* Set the real UTM zone */
474  gps.utm_pos.zone = (DegOfRad(gps.lla_pos.lon/1e7)+180) / 6 + 1;
475 
476  lla_f.lat = ((float) gps.lla_pos.lat) / 1e7;
477  lla_f.lon = ((float) gps.lla_pos.lon) / 1e7;
478  utm_f.zone = nav_utm_zone0;
479  /* convert to utm */
480  utm_of_lla_f(&utm_f, &lla_f);
481  /* copy results of utm conversion */
482  gps.utm_pos.east = utm_f.east*100;
483  gps.utm_pos.north = utm_f.north*100;
484  gps.utm_pos.alt = gps.lla_pos.alt;
485 
486  ins_x = utm_f.east;
487  ins_y = utm_f.north;
488  // Altitude: Xsens LLH gives ellipsoid height
489  ins_z = -(INS_FORMAT)XSENS_DATA_RAWGPS_alt(xsens_msg_buf,offset) / 1000.;
490 
491  // Compute geoid (MSL) height
492  float hmsl;
493  WGS84_ELLIPSOID_TO_GEOID(lla_f.lat,lla_f.lon,hmsl);
494  gps.hmsl = XSENS_DATA_RAWGPS_alt(xsens_msg_buf,offset) - (hmsl * 1000.0f);
495 
496  ins_vx = ((INS_FORMAT)XSENS_DATA_RAWGPS_vel_n(xsens_msg_buf,offset)) / 100.;
497  ins_vy = ((INS_FORMAT)XSENS_DATA_RAWGPS_vel_e(xsens_msg_buf,offset)) / 100.;
498  ins_vz = ((INS_FORMAT)XSENS_DATA_RAWGPS_vel_d(xsens_msg_buf,offset)) / 100.;
499  gps.ned_vel.x = XSENS_DATA_RAWGPS_vel_n(xsens_msg_buf,offset);
500  gps.ned_vel.y = XSENS_DATA_RAWGPS_vel_e(xsens_msg_buf,offset);
501  gps.ned_vel.z = XSENS_DATA_RAWGPS_vel_d(xsens_msg_buf,offset);
502  gps.pacc = XSENS_DATA_RAWGPS_hacc(xsens_msg_buf,offset) / 100;
503  gps.sacc = XSENS_DATA_RAWGPS_sacc(xsens_msg_buf,offset) / 100;
504  gps.pdop = 5; // FIXME Not output by XSens
505 
506  gps_xsens_msg_available = TRUE;
507 #endif
508  offset += XSENS_DATA_RAWGPS_LENGTH;
509  }
510  //} else {
511  if (XSENS_MASK_Temp(xsens_output_mode)) {
512  offset += XSENS_DATA_Temp_LENGTH;
513  }
514  if (XSENS_MASK_Calibrated(xsens_output_mode)) {
515  uint8_t l = 0;
516  if (!XSENS_MASK_AccOut(xsens_output_settings)) {
517  ins_ax = XSENS_DATA_Calibrated_accX(xsens_msg_buf,offset);
518  ins_ay = XSENS_DATA_Calibrated_accY(xsens_msg_buf,offset);
519  ins_az = XSENS_DATA_Calibrated_accZ(xsens_msg_buf,offset);
520 #if USE_IMU
521  imu_xsens.accel_available = TRUE;
522 #endif
523  l++;
524  }
525  if (!XSENS_MASK_GyrOut(xsens_output_settings)) {
526  ins_p = XSENS_DATA_Calibrated_gyrX(xsens_msg_buf,offset);
527  ins_q = XSENS_DATA_Calibrated_gyrY(xsens_msg_buf,offset);
528  ins_r = XSENS_DATA_Calibrated_gyrZ(xsens_msg_buf,offset);
529 #if USE_IMU
530  imu_xsens.gyro_available = TRUE;
531 #endif
532  l++;
533  }
534  if (!XSENS_MASK_MagOut(xsens_output_settings)) {
535  ins_mx = XSENS_DATA_Calibrated_magX(xsens_msg_buf,offset);
536  ins_my = XSENS_DATA_Calibrated_magY(xsens_msg_buf,offset);
537  ins_mz = XSENS_DATA_Calibrated_magZ(xsens_msg_buf,offset);
538 #if USE_IMU
539  imu_xsens.mag_available = TRUE;
540 #endif
541  l++;
542  }
543  offset += l * XSENS_DATA_Calibrated_LENGTH / 3;
544  }
545  if (XSENS_MASK_Orientation(xsens_output_mode)) {
546  if (XSENS_MASK_OrientationMode(xsens_output_settings) == 0x00) {
547  float q0 = XSENS_DATA_Quaternion_q0(xsens_msg_buf,offset);
548  float q1 = XSENS_DATA_Quaternion_q1(xsens_msg_buf,offset);
549  float q2 = XSENS_DATA_Quaternion_q2(xsens_msg_buf,offset);
550  float q3 = XSENS_DATA_Quaternion_q3(xsens_msg_buf,offset);
551  float dcm00 = 1.0 - 2 * (q2*q2 + q3*q3);
552  float dcm01 = 2 * (q1*q2 + q0*q3);
553  float dcm02 = 2 * (q1*q3 - q0*q2);
554  float dcm12 = 2 * (q2*q3 + q0*q1);
555  float dcm22 = 1.0 - 2 * (q1*q1 + q2*q2);
556  ins_phi = atan2(dcm12, dcm22);
557  ins_theta = -asin(dcm02);
558  ins_psi = atan2(dcm01, dcm00);
559  offset += XSENS_DATA_Quaternion_LENGTH;
560  }
561  if (XSENS_MASK_OrientationMode(xsens_output_settings) == 0x01) {
562  ins_phi = XSENS_DATA_Euler_roll(xsens_msg_buf,offset) * M_PI / 180;
563  ins_theta = XSENS_DATA_Euler_pitch(xsens_msg_buf,offset) * M_PI / 180;
564  ins_psi = XSENS_DATA_Euler_yaw(xsens_msg_buf,offset) * M_PI / 180;
565  offset += XSENS_DATA_Euler_LENGTH;
566  }
567  if (XSENS_MASK_OrientationMode(xsens_output_settings) == 0x10) {
568  offset += XSENS_DATA_Matrix_LENGTH;
569  }

653 void parse_ins_buffer(uint8_t c) {
654  ck += c;
655  switch (xsens_status) {
656  case UNINIT:
657  if (c != XSENS_START)
658  goto error;
659  xsens_status++;
660  ck = 0;
661  break;
662  case GOT_START:
663  if (c != XSENS_BID)
664  goto error;
665  xsens_status++;
666  break;
667  case GOT_BID:
668  xsens_id = c;
669  xsens_status++;
670  break;
671  case GOT_MID:
672  xsens_len = c;
673  if (xsens_len > XSENS_MAX_PAYLOAD)
674  goto error;
675  xsens_msg_idx = 0;
676  xsens_status++;
677  break;
678  case GOT_LEN:
679  xsens_msg_buf[xsens_msg_idx] = c;
680  xsens_msg_idx++;
681  if (xsens_msg_idx >= xsens_len)
682  xsens_status++;
683  break;
684  case GOT_DATA:
685  if (ck != 0)
686  goto error;
687  ins_msg_received = TRUE;
688  goto restart;
689  break;
690  default:
691  break;
692  }
693  return;
694  error:
695  restart:
696  xsens_status = UNINIT;
697  return;
698 }
