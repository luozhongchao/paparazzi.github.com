00104 static inline void inter_mcu_fill_fbw_state (void) {
00105   uint8_t status = 0;
00106 
00107 #ifdef RADIO_CONTROL
00108   uint8_t i;
00109   for(i = 0; i < RADIO_CONTROL_NB_CHANNEL; i++)
00110     fbw_state->channels[i] = radio_control.values[i];
00111 
00112   fbw_state->ppm_cpt = radio_control.frame_rate;
00113 
00114   status = (radio_control.status == RC_OK ? _BV(STATUS_RADIO_OK) : 0);
00115   status |= (radio_control.status == RC_REALLY_LOST ? _BV(STATUS_RADIO_REALLY_LOST) : 0);
00116   status |= (radio_control.status == RC_OK ? _BV(AVERAGED_CHANNELS_SENT) : 0); // Any valid frame contains averaged channels
00117 #endif // RADIO_CONTROL
00118 
00119   status |= (fbw_mode == FBW_MODE_AUTO ? _BV(STATUS_MODE_AUTO) : 0);
00120   status |= (fbw_mode == FBW_MODE_FAILSAFE ? _BV(STATUS_MODE_FAILSAFE) : 0);
00121   fbw_state->status  = status;
00122 
00123   fbw_state->vsupply = electrical.vsupply;
00124   fbw_state->current = electrical.current;
00125 #if defined SINGLE_MCU
00126 
00127   inter_mcu_received_fbw = TRUE;
