59 void humid_htm_start( void ) {
60  if (sys_time.nb_sec > 1) {
61  /* measurement request: wake up sensor, sample temperature/humidity */
62  i2c_transmit(&HTM_I2C_DEV, &htm_trans, HTM_SLAVE_ADDR, 0);

68 void humid_htm_read( void ) {
69  if (htm_status == HTM_MR_OK) {
70  /* read humid and temp*/
71  htm_status = HTM_READ_DATA;
72  i2c_receive(&HTM_I2C_DEV, &htm_trans, HTM_SLAVE_ADDR, 4);
73  }
74 }

76 void humid_htm_event( void ) {
77  if (htm_trans.status == I2CTransSuccess) {
78  switch (htm_status) {
79 
80  case HTM_MR:
81  htm_status = HTM_MR_OK;
82  htm_trans.status = I2CTransDone;
83  break;
84 
85  case HTM_READ_DATA:
86  /* check stale status */
87  if (((htm_trans.buf[0] >> 6) & 0x3) == 0) {
88  /* humidity */
89  humidhtm = ((htm_trans.buf[0] & 0x3F) << 8) | htm_trans.buf[1];
90  fhumidhtm = humidhtm / 163.83;
91  /* temperature */
92  temphtm = (htm_trans.buf[2] << 6) | (htm_trans.buf[3] >> 2);
93  ftemphtm = -40.00 + 0.01 * temphtm;
94  DOWNLINK_SEND_HTM_STATUS(DefaultChannel, DefaultDevice, &humidhtm, &temphtm, &fhumidhtm, &ftemphtm);
95  }
96  htm_trans.status = I2CTransDone;
97  break;
98 
99  default:
100  htm_trans.status = I2CTransDone;
101  break;
102  }
103  }
