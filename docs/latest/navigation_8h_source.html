131 #define NavGlide(_last_wp, _wp) { \
132  int32_t start_alt = waypoints[_last_wp].z; \
133  int32_t diff_alt = waypoints[_wp].z - start_alt; \
134  int32_t alt = start_alt + ((diff_alt * nav_leg_progress) / nav_leg_length); \

136 }
