66 void tmp102_init(void) {

70  tmp_trans.buf[1] = TMP102_CONF1;

81 void tmp102_event( void ) {

83  if ((tmp_trans.status == I2CTransSuccess) && (tmp_meas_started == TRUE)) {
84 
85  uint16_t tmp_temperature;

87  /* read two byte temperature */
