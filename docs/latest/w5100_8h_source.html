00188 static inline void parse_w5100(struct w5100_transport * t, uint8_t c ) {
00189   switch (t->status) {
00190   case UNINIT:
00191     if (c == STX)
00192       t->status++;
00193     break;
00194   case GOT_STX:
00195     if (t->trans.msg_received) {
00196       t->trans.ovrn++;
00197       goto error;
00198     }
00199     t->trans.payload_len = c-4; /* Counting STX, LENGTH and CRC1 and CRC2 */
00200     t->ck_a = t->ck_b = c;
00201     t->status++;
00202     t->payload_idx = 0;
00203     break;
00204   case GOT_LENGTH:
00205     t->trans.payload[t->payload_idx] = c;
00206     t->ck_a += c; t->ck_b += t->ck_a;
00207     t->payload_idx++;
00208     if (t->payload_idx == t->trans.payload_len)
00209       t->status++;
00210     break;
00211   case GOT_PAYLOAD:
00212     if (c != t->ck_a)
00213       goto error;
00214     t->status++;
00215     break;
00216   case GOT_CRC1:
00217     if (c != t->ck_b)
00218       goto error;
00219     t->trans.msg_received = TRUE;
00220     goto restart;
00221   default:
00222     goto error;
00223   }
00224   return;
00225  error:
00226   t->trans.error++;
00227  restart:
00228   t->status = UNINIT;
00229   return;
00230 }
