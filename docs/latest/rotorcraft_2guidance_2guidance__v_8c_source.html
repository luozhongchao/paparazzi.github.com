128

131

149 
150  }

163 void guidance_v_run(bool_t in_flight) {
164 
165  // FIXME... SATURATIONS NOT TAKEN INTO ACCOUNT
166  // AKA SUPERVISION and co
167  if (in_flight) {
168  gv_adapt_run(stateGetAccelNed_i()->z, stabilization_cmd[COMMAND_THRUST], guidance_v_zd_ref);
169  }
170 
171  switch (guidance_v_mode) {
172 
173  case GUIDANCE_V_MODE_RC_DIRECT:
174  guidance_v_z_sp = stateGetPositionNed_i()->z; // for display only
175  stabilization_cmd[COMMAND_THRUST] = guidance_v_rc_delta_t;
176  break;
177 
178  case GUIDANCE_V_MODE_RC_CLIMB:
179  guidance_v_zd_sp = guidance_v_rc_zd_sp;
180  gv_update_ref_from_zd_sp(guidance_v_zd_sp);
181  run_hover_loop(in_flight);
182  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
183  break;
184 
185  case GUIDANCE_V_MODE_CLIMB:
186 #if USE_FMS
187  if (fms.enabled && fms.input.v_mode == GUIDANCE_V_MODE_CLIMB) {
188  guidance_v_zd_sp = fms.input.v_sp.climb;
189  }
190 #endif
191  gv_update_ref_from_zd_sp(guidance_v_zd_sp);
192  run_hover_loop(in_flight);
193 #if NO_RC_THRUST_LIMIT
194  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
195 #else
196  // saturate max authority with RC stick
197  stabilization_cmd[COMMAND_THRUST] = Min(guidance_v_rc_delta_t, guidance_v_delta_t);
198 #endif
199  break;
200 
201  case GUIDANCE_V_MODE_HOVER:
202 #if USE_FMS
203  if (fms.enabled && fms.input.v_mode == GUIDANCE_V_MODE_HOVER)
204  guidance_v_z_sp = fms.input.v_sp.height;
205 #endif
206  gv_update_ref_from_z_sp(guidance_v_z_sp);
207  run_hover_loop(in_flight);
208 #if NO_RC_THRUST_LIMIT
209  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
210 #else
211  // saturate max authority with RC stick
212  stabilization_cmd[COMMAND_THRUST] = Min(guidance_v_rc_delta_t, guidance_v_delta_t);
213 #endif
214  break;
215 
216  case GUIDANCE_V_MODE_NAV:
217  {
218  if (vertical_mode == VERTICAL_MODE_ALT) {
219  guidance_v_z_sp = -nav_flight_altitude;
220  gv_update_ref_from_z_sp(guidance_v_z_sp);
221  run_hover_loop(in_flight);
222  }
223  else if (vertical_mode == VERTICAL_MODE_CLIMB) {
224  guidance_v_zd_sp = -nav_climb;
225  gv_update_ref_from_zd_sp(guidance_v_zd_sp);
226  nav_flight_altitude = -guidance_v_z_sp;
227  run_hover_loop(in_flight);
228  }
229  else if (vertical_mode == VERTICAL_MODE_MANUAL) {
230  guidance_v_z_sp = -nav_flight_altitude; // For display only
231  guidance_v_delta_t = nav_throttle;
232  }
233 #if NO_RC_THRUST_LIMIT
234  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
235 #else
236  /* use rc limitation if available */
237  if (radio_control.status == RC_OK)
238  stabilization_cmd[COMMAND_THRUST] = Min(guidance_v_rc_delta_t, guidance_v_delta_t);
239  else
240  stabilization_cmd[COMMAND_THRUST] = guidance_v_delta_t;
241 #endif
242  break;
243  }
244  default:
245  break;
246  }

254 __attribute__ ((always_inline)) static inline void run_hover_loop(bool_t in_flight) {
255 
256  /* convert our reference to generic representation */
257  int64_t tmp = gv_z_ref>>(GV_Z_REF_FRAC - INT32_POS_FRAC);
258  guidance_v_z_ref = (int32_t)tmp;
259  guidance_v_zd_ref = gv_zd_ref<<(INT32_SPEED_FRAC - GV_ZD_REF_FRAC);
260  guidance_v_zdd_ref = gv_zdd_ref<<(INT32_ACCEL_FRAC - GV_ZDD_REF_FRAC);
261  /* compute the error to our reference */
262  int32_t err_z = guidance_v_z_ref - stateGetPositionNed_i()->z;
263  Bound(err_z, GUIDANCE_V_MIN_ERR_Z, GUIDANCE_V_MAX_ERR_Z);
264  int32_t err_zd = guidance_v_zd_ref - stateGetSpeedNed_i()->z;
265  Bound(err_zd, GUIDANCE_V_MIN_ERR_ZD, GUIDANCE_V_MAX_ERR_ZD);
266 
267  if (in_flight) {
268  guidance_v_z_sum_err += err_z;
269  Bound(guidance_v_z_sum_err, -GUIDANCE_V_MAX_SUM_ERR, GUIDANCE_V_MAX_SUM_ERR);
270  }
271  else
272  guidance_v_z_sum_err = 0;
273 
274  /* our nominal command : (g + zdd)*m */
275 #ifdef GUIDANCE_V_NOMINAL_HOVER_THROTTLE
276  const int32_t inv_m = BFP_OF_REAL(9.81/(guidance_v_nominal_throttle*MAX_PPRZ), FF_CMD_FRAC);
277 #else
278  const int32_t inv_m = gv_adapt_X>>(GV_ADAPT_X_FRAC - FF_CMD_FRAC);
279 #endif
280  const int32_t g_m_zdd = (int32_t)BFP_OF_REAL(9.81, FF_CMD_FRAC) -
281  (guidance_v_zdd_ref<<(FF_CMD_FRAC - INT32_ACCEL_FRAC));
282 
283  guidance_v_ff_cmd = g_m_zdd / inv_m;
284  int32_t cphi,ctheta,cphitheta;
285  struct Int32Eulers* att_euler = stateGetNedToBodyEulers_i();
286  PPRZ_ITRIG_COS(cphi, att_euler->phi);
287  PPRZ_ITRIG_COS(ctheta, att_euler->theta);
288  cphitheta = (cphi * ctheta) >> INT32_TRIG_FRAC;
289  if (cphitheta < MAX_BANK_COEF) cphitheta = MAX_BANK_COEF;
290  /* feed forward command */
291  guidance_v_ff_cmd = (guidance_v_ff_cmd << INT32_TRIG_FRAC) / cphitheta;
292 
293  /* bound the nominal command to 0.9*MAX_PPRZ */
294  Bound(guidance_v_ff_cmd, 0, 8640);
295 
296 
297  /* our error feed back command */
298  /* z-axis pointing down -> positive error means we need less thrust */
299  guidance_v_fb_cmd = ((-guidance_v_kp * err_z) >> 7) +
300  ((-guidance_v_kd * err_zd) >> 16) +
301  ((-guidance_v_ki * guidance_v_z_sum_err) >> 16);
302 
303  guidance_v_delta_t = guidance_v_ff_cmd + guidance_v_fb_cmd;
304 
305  /* bound the result */
306  Bound(guidance_v_delta_t, 0, MAX_PPRZ);
307 
308 }
