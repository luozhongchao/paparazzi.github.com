00112     bx = (start_bx + i*cube_grid_x) - WaypointX(center);

00115     ey = start_ey - WaypointY(center);

00117     waypoints[tb+j].x = bx * cos_alpha - by * sin_alpha + WaypointX(center);
00118     waypoints[tb+j].y = bx * sin_alpha + by * cos_alpha + WaypointY(center);
00119     waypoints[tb+j].a = start_bz;
00120     waypoints[te+j].x = ex * cos_alpha - ey * sin_alpha + WaypointX(center);
00121     waypoints[te+j].y = ex * sin_alpha + ey * cos_alpha + WaypointY(center);
00122     waypoints[te+j].a = start_ez;
00123   }
00124 
00125   /* bug in <for from="" to=""> ? */
00126   cube_nline_x--;
00127   cube_nline_z--;
00128 
00129   return FALSE;
