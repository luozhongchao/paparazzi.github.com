00044

00061   else cube_nline_x_t = cube_size_x / cube_grid_x + 1;

00067   cube_line_x_start =  (cube_pos_x    * cube_nline_x_t)/cube_nsect_x;
00068   cube_line_x_end   = ((cube_pos_x+1) * cube_nline_x_t)/cube_nsect_x;
00069   if (cube_line_x_end > cube_nline_x_t) cube_line_x_end = cube_nline_x_t;
00070   cube_nline_x = cube_line_x_end - cube_line_x_start;
00071 
00072   /* do not do more than pre-set number of lines */
00073   if (cube_nline_x >= MAX_LINES_X) cube_nline_x = MAX_LINES_X-1;
00074 
00075   /* position and number of layers in this sector */
00076   cube_pos_z = (cube_sect-1) / cube_nsect_x;
00077   cube_line_z_start =  (cube_pos_z    * cube_nline_z_t)/cube_nsect_z;
00078   cube_line_z_end   = ((cube_pos_z+1) * cube_nline_z_t)/cube_nsect_z;
00079   cube_nline_z = cube_line_z_end - cube_line_z_start;
00080 
00081   /* do the costly stuff only once */
00082   alpha = ((360. - cube_alpha) / 360.) * 2 * M_PI;
00083   cos_alpha = cos(alpha);
00084   sin_alpha = sin(alpha);
00085 
00086   /* calculate lower left start begin/end x coord */
00087   start_bx = WaypointX(center) - (((cube_nline_x_t-1) * cube_grid_x)/2)

00102   /* reset all waypoints to the standby position */

00104     waypoints[tb+j].x = WaypointX(center) + STBY_OFFSET;
00105     waypoints[tb+j].y = WaypointY(center);
00106     waypoints[te+j].x = WaypointX(center) + STBY_OFFSET;
00107     waypoints[te+j].y = WaypointY(center);
00108   }
00109 
00110   /* set used waypoints */
00111   for (j=0; j < cube_nline_x; j++) {
00112     int i = cube_line_x_start+j;
00113     /* set waypoints and vectorize in regard to center */
00114     bx = (start_bx + i*cube_grid_x) - WaypointX(center);
00115     by = start_by - WaypointY(center);
00116     ex = (start_ex + i*cube_grid_x) - WaypointX(center);
00117     ey = start_ey - WaypointY(center);
00118     /* rotate clockwise with alpha and un-vectorize*/
00119     waypoints[tb+j].x = bx * cos_alpha - by * sin_alpha + WaypointX(center);
00120     waypoints[tb+j].y = bx * sin_alpha + by * cos_alpha + WaypointY(center);
00121     waypoints[tb+j].a = start_bz;
00122     waypoints[te+j].x = ex * cos_alpha - ey * sin_alpha + WaypointX(center);
00123     waypoints[te+j].y = ex * sin_alpha + ey * cos_alpha + WaypointY(center);
00124     waypoints[te+j].a = start_ez;
00125   }
00126 
00127   /* bug in <for from="" to=""> ? */
00128   cube_nline_x--;
00129   cube_nline_z--;
00130 
00131   return FALSE;
00132 }
