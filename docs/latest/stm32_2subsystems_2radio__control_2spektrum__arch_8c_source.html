00141 /*****************************************************************************
00142  * The bind function means that the satellite receivers believe they are
00143  * connected to a 9 channel JR-R921 24 receiver thus during the bind process
00144  * they try to get the transmitter to transmit at the highest resolution that
00145  * it can manage. The data is contained in 16 byte packets transmitted at
00146  * 115200 baud. Depending on the transmitter either 1 or 2 frames are required
00147  * to contain the data for all channels. These frames are either 11ms or 22ms
00148  * apart.
00149  *
00150  * The format of each frame for the main receiver is as follows
00151  *
00152  *  byte1:  frame loss data
00153  *  byte2:  transmitter information
00154  *  byte3:  and byte4:  channel data
00155  *  byte5:  and byte6:  channel data
00156  *  byte7:  and byte8:  channel data
00157  *  byte9:  and byte10: channel data
00158  *  byte11: and byte12: channel data
00159  *  byte13: and byte14: channel data
00160  *  byte15: and byte16: channel data
00161  *
00162  *
00163  * The format of each frame for the secondary receiver is as follows
00164  *
00165  *  byte1:  frame loss data
00166  *  byte2:  frame loss data
00167  *  byte3:  and byte4:  channel data
00168  *  byte5:  and byte6:  channel data
00169  *  byte7:  and byte8:  channel data
00170  *  byte9:  and byte10: channel data
00171  *  byte11: and byte12: channel data
00172  *  byte13: and byte14: channel data
00173  *  byte15: and byte16: channel data
00174  *
00175  * The frame loss data bytes starts out containing 0 as long as the
00176  * transmitter is switched on before the receivers. It then increments
00177  * whenever frames are dropped.
00178  *
00179  * Three values for the transmitter information byte have been seen thus far
00180  *
00181  * 0x01 From a Spektrum DX7eu which transmits a single frame containing all
00182  * channel data every 22ms with 10bit resolution.
00183  *
00184  * 0x02 From a Spektrum DM9 module which transmits two frames to carry the
00185  * data for all channels 11ms apart with 10bit resolution.
00186  *
00187  * 0x12 From a Spektrum DX7se which transmits two frames to carry the
00188  * data for all channels 11ms apart with 11bit resolution.
00189  *
00190  * 0x12 From a JR X9503 which transmits two frames to carry the
00191  * data for all channels 11ms apart with 11bit resolution.
00192  *
00193  * 0x01 From a Spektrum DX7 which transmits a single frame containing all
00194  * channel data every 22ms with 10bit resolution.
00195  *
00196  * 0x12 From a JR DSX12 which transmits two frames to carry the
00197  * data for all channels 11ms apart with 11bit resolution.
00198  *
00199  * 0x1 From a Spektru DX5e which transmits a single frame containing all
00200  * channel data every 22ms with 10bit resolution.
00201  *
00202  * 0x01 From a Spektrum DX6i which transmits a single frame containing all
00203  * channel data every 22ms with 10bit resolution.
00204  *
00205  * Currently the assumption is that the data has the form :
00206  *
00207  * [0 0 0 R 0 0 N1 N0]
00208  *
00209  * where :
00210  *
00211  * 0 means a '0' bit
00212  * R: 0 for 10 bit resolution 1 for 11 bit resolution channel data
00213  * N1 to N0 is the number of frames required to receive all channel
00214  * data.
00215  *
00216  * Channels can have either 10bit or 11bit resolution. Data from a tranmitter
00217  * with 10 bit resolution has the form:
00218  *
00219  * [F 0 C3 C2 C1 C0 D9 D8 D7 D6 D5 D4 D3 D2 D1 D0]
00220  *
00221  * Data from a tranmitter with 11 bit resolution has the form
00222  *
00223  * [F C3 C2 C1 C0 D10 D9 D8 D7 D6 D5 D4 D3 D2 D1 D0]
00224  *
00225  * where :
00226  *
00227  * 0 means a '0' bit
00228  * F: Normally 0 but set to 1 for the first channel of the 2nd frame if a
00229  * second frame is transmitted.
00230  *
00231  * C3 to C0 is the channel number, 4 bit, matching the numbers allocated in
00232  * the transmitter.
00233  *
00234  * D9 to D0 is the channel data (10 bit) 0xaa..0x200..0x356 for
00235  * 100% transmitter-travel
00236  *
00237  *
00238  * D10 to D0 is the channel data (11 bit) 0x154..0x400..0x6AC for
00239  * 100% transmitter-travel
00240  *****************************************************************************/

00248 #define SpektrumParser(_c, _SpektrumState, _receiver)  {                      \
00249                                                                               \
00250   uint16_t ChannelData;                                                       \
00251   uint8_t TimedOut;                                                           \
00252   static uint8_t TmpEncType = 0;        /* 0 = 10bit, 1 = 11 bit        */    \
00253   static uint8_t TmpExpFrames = 0;      /* # of frames for channel data */    \
00254                                                                               \
00255    TimedOut = (!_SpektrumState.SpektrumTimer) ? 1 : 0;                        \
00256                                                                               \
00257   /* If we have just started the resync process or */                         \
00258   /* if we have recieved a character before our    */                         \
00259   /* 7ms wait has finished                         */                         \
00260   if ((_SpektrumState.ReSync == 1) ||                                         \
00261       ((_SpektrumState.Sync == 0) && (!TimedOut))) {                          \
00262                                                                               \
00263     _SpektrumState.ReSync = 0;                                                \
00264     _SpektrumState.SpektrumTimer = MIN_FRAME_SPACE;                           \
00265     _SpektrumState.Sync = 0;                                                  \
00266     _SpektrumState.ChannelCnt = 0;                                            \
00267     _SpektrumState.FrameCnt = 0;                                              \
00268     _SpektrumState.SecondFrame = 0;                                           \
00269     return;                                                                   \
00270   }                                                                           \
00271                                                                               \
00272   /* the first byte of a new frame. It was received */                        \
00273   /* more than 7ms after the last received byte.    */                        \
00274   /* It represents the number of lost frames so far.*/                        \
00275   if (_SpektrumState.Sync == 0) {                                             \
00276       _SpektrumState.LostFrameCnt = _c;                                       \
00277       if(_receiver) /* secondary receiver */                                  \
00278         _SpektrumState.LostFrameCnt = _SpektrumState.LostFrameCnt << 8;       \
00279       _SpektrumState.Sync = 1;                                                \
00280       _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                          \
00281       return;                                                                 \
00282   }                                                                           \
00283                                                                               \
00284   /* all other bytes should be recieved within     */                         \
00285   /* MAX_BYTE_SPACE time of the last byte received */                         \
00286   /* otherwise something went wrong resynchronise  */                         \
00287   if(TimedOut) {                                                              \
00288     _SpektrumState.ReSync = 1;                                                \
00289     /* next frame not expected sooner than 7ms     */                         \
00290     _SpektrumState.SpektrumTimer = MIN_FRAME_SPACE;                           \
00291     return;                                                                   \
00292   }                                                                           \
00293                                                                               \
00294   /* second character determines resolution and frame rate for main */        \
00295   /* receiver or low byte of LostFrameCount for secondary receiver  */        \
00296   if(_SpektrumState.Sync == 1) {                                              \
00297     if(_receiver) {                                                           \
00298       _SpektrumState.LostFrameCnt +=_c;                                       \
00299       TmpExpFrames = ExpectedFrames;                                          \
00300     } else {                                                                  \
00301  \
00302       /* protocol that is still 10 bit but without using the full range.    */\
00303       TmpEncType =(_c & 0x10)>>4;      /* 0 = 10bit, 1 = 11 bit             */\
00304       TmpExpFrames = _c & 0x03;        /* 1 = 1 frame contains all channels */\
00305                                        /* 2 = 2 channel data in 2 frames    */\
00306     }                                                                         \
00307     _SpektrumState.Sync = 2;                                                  \
00308     _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                            \
00309     return;                                                                   \
00310   }                                                                           \
00311                                                                               \
00312   /* high byte of channel data if this is the first byte */                   \
00313   /* of channel data and the most significant bit is set */                   \
00314   /* then this is the second frame of channel data.      */                   \
00315   if(_SpektrumState.Sync == 2) {                                              \
00316     _SpektrumState.HighByte = _c;                                             \
00317     if (_SpektrumState.ChannelCnt == 0) {                                     \
00318       _SpektrumState.SecondFrame = (_SpektrumState.HighByte & 0x80) ? 1 : 0;  \
00319     }                                                                         \
00320     _SpektrumState.Sync = 3;                                                  \
00321     _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                            \
00322     return;                                                                   \
00323   }                                                                           \
00324                                                                               \
00325   /* low byte of channel data */                                              \
00326   if(_SpektrumState.Sync == 3) {                                              \
00327     _SpektrumState.Sync = 2;                                                  \
00328     _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                            \
00329     /* we overwrite the buffer now so rc data is not available now */         \
00330     _SpektrumState.RcAvailable = 0;                                           \
00331     ChannelData = ((uint16_t)_SpektrumState.HighByte << 8) | _c;              \
00332     _SpektrumState.values[_SpektrumState.ChannelCnt                           \
00333                           + (_SpektrumState.SecondFrame * 7)] = ChannelData;  \
00334     _SpektrumState.ChannelCnt ++;                                             \
00335   }                                                                           \
00336                                                                               \
00337   /* If we have a whole frame */                                              \
00338   if(_SpektrumState.ChannelCnt >= SPEKTRUM_CHANNELS_PER_FRAME) {              \
00339     /* how many frames did we expect ? */                                     \
00340     ++_SpektrumState.FrameCnt;                                                \
00341     if (_SpektrumState.FrameCnt == TmpExpFrames)                              \
00342     {                                                                         \
00343       /* set the rc_available_flag */                                         \
00344       _SpektrumState.RcAvailable = 1;                                         \
00345       _SpektrumState.FrameCnt = 0;                                            \
00346     }                                                                         \
00347     if(!_receiver) { /* main receiver */                                      \
00348       EncodingType = TmpEncType;         /* only update on a good */          \
00349       ExpectedFrames = TmpExpFrames;     /* main receiver frame   */          \
00350     }                                                                         \
00351     _SpektrumState.Sync = 0;                                                  \
00352     _SpektrumState.ChannelCnt = 0;                                            \
00353     _SpektrumState.SecondFrame = 0;                                           \
00354     _SpektrumState.SpektrumTimer = MIN_FRAME_SPACE;                           \
00355   }                                                                           \
00356 }                                                                             \

00361  * and calls callback funtion

00365 void RadioControlEventImp(void (*frame_handler)(void)) {
00366   uint8_t ChannelCnt;
00367   uint8_t ChannelNum;
00368   uint16_t ChannelData;
00369   uint8_t MaxChannelNum = 0;
00370 
00371 #ifdef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00372   /* If we have two receivers and at least one of them has new data */
00373   uint8_t BestReceiver;
00374   if ((PrimarySpektrumState.RcAvailable) ||
00375       (SecondarySpektrumState.RcAvailable)) {
00376     /* if both receivers have new data select the one  */
00377     /* that has had the least number of frames lost    */
00378     if ((PrimarySpektrumState.RcAvailable) &&
00379         (SecondarySpektrumState.RcAvailable)) {
00380       BestReceiver  = (PrimarySpektrumState.LostFrameCnt
00381                        <= SecondarySpektrumState.LostFrameCnt) ? 0 : 1;
00382     } else {
00383       /* if only one of the receivers have new data use it */
00384       BestReceiver  = (PrimarySpektrumState.RcAvailable) ? 0 : 1;
00385     }
00386     /* clear the data ready flags */
00387     PrimarySpektrumState.RcAvailable = 0;
00388     SecondarySpektrumState.RcAvailable = 0;
00389 
00390 #else
00391   /* if we have one receiver and it has new data */
00392   if(PrimarySpektrumState.RcAvailable) {
00393     PrimarySpektrumState.RcAvailable = 0;
00394 #endif
00395     ChannelCnt = 0;
00396     /* for every piece of channel data we have received */
00397     for(int i = 0; (i < SPEKTRUM_CHANNELS_PER_FRAME * ExpectedFrames); i++) {
00398 #ifndef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00399       ChannelData = PrimarySpektrumState.values[i];
00400 #else
00401       ChannelData = (!BestReceiver) ? PrimarySpektrumState.values[i] :
00402                                  SecondarySpektrumState.values[i];
00403 #endif
00404       /* find out the channel number and its value by  */
00405       /* using the EncodingType which is only received */
00406       /* from the main receiver                        */
00407       switch(EncodingType) {
00408         case(0) : /* 10 bit */
00409           ChannelNum = (ChannelData >> 10) & 0x0f;
00410           /* don't bother decoding unused channels */
00411           if (ChannelNum < RADIO_CONTROL_NB_CHANNEL) {
00412            SpektrumBuf[ChannelNum] = ChannelData & 0x3ff;
00413            SpektrumBuf[ChannelNum] -= 0x200;
00414            SpektrumBuf[ChannelNum] *= MAX_PPRZ/0x156;
00415            ChannelCnt++;
00416           }
00417           break;
00418 
00419         case(1) : /* 11 bit */
00420           ChannelNum = (ChannelData >> 11) & 0x0f;
00421           /* don't bother decoding unused channels */
00422           if (ChannelNum < RADIO_CONTROL_NB_CHANNEL) {
00423             SpektrumBuf[ChannelNum] = ChannelData & 0x7ff;
00424             SpektrumBuf[ChannelNum] -= 0x400;
00425             SpektrumBuf[ChannelNum] *= MAX_PPRZ/0x2AC;
00426             ChannelCnt++;
00427           }
00428           break;
00429 
00430         default : ChannelNum = 0x0F; break;  /* never going to get here */
00431       }
00432       /* store the value of the highest valid channel */
00433       if ((ChannelNum != 0x0F) && (ChannelNum > MaxChannelNum))
00434         MaxChannelNum = ChannelNum;
00435 
00436     }
00437 
00438     /* if we have a valid frame the pass it to the frame handler */
00439     if (ChannelCnt >= (MaxChannelNum + 1)) {
00440       radio_control.frame_cpt++;
00441       radio_control.time_since_last_frame = 0;
00442       radio_control.status = RC_OK;
00443       for (int i = 0; i < (MaxChannelNum + 1); i++) {
00444         radio_control.values[i] = SpektrumBuf[i];
00445         if (i == RADIO_THROTTLE ) {
00446           radio_control.values[i] += MAX_PPRZ;
00447           radio_control.values[i] /= 2;
00448         }
00449         radio_control.values[i] *= SpektrumSigns[i];
00450       }
00451       (*frame_handler)();
00452     }
00453   }
00454 }

00628 /*****************************************************************************
00629  *
00630  * The following functions provide functionality to allow binding of
00631  * spektrum satellite receivers. The pulse train sent to them means

00634  *****************************************************************************/
00635 /*****************************************************************************
00636  *
00637  * radio_control_spektrum_try_bind(void) must called on powerup as spektrum
00638  * satellites can only bind immediately after power up also it must be called
00639  * before the call to SpektrumUartInit as we leave them with their Rx pins set
00640  * as outputs.
00641  *
00642  *****************************************************************************/
00643 void radio_control_spektrum_try_bind(void) {
00644 
00645   /* init RCC */
00646   rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_IOPCEN);
00647 
00648   /* Init GPIO for the bind pin */
00649   gpio_set(GPIOC, GPIO3);
00650   gpio_set_mode(GPIOC, GPIO_MODE_INPUT,
00651           GPIO_CNF_INPUT_PULL_UPDOWN, GPIO3);
00652   /* exit if the BIND_PIN is high, it needs to
00653      be pulled low at startup to initiate bind */
00654   if (gpio_get(GPIOC, GPIO3) != 0)
00655     return;
00656 
00657   /* bind initiated, initialise the delay timer */
00658   SpektrumDelayInit();

00660   /* initialise the uarts rx pins as  GPIOS */
00661   rcc_peripheral_enable_clock(&RCC_APB2ENR, PrimaryUart(_RCC_GPIO));
00662 
00663   /* Master receiver Rx push-pull */
00664   gpio_set_mode(PrimaryUart(_BANK), GPIO_MODE_OUTPUT_50_MHZ,
00665     GPIO_CNF_OUTPUT_PUSHPULL, PrimaryUart(_PIN));
00666 
00667   /* Master receiver RX line, drive high */
00668   gpio_set(PrimaryUart(_BANK), PrimaryUart(_PIN));
00669 
00670 #ifdef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00671 
00672   rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_IOPDEN);
00673 
00674   /* Slave receiver Rx push-pull */
00675   gpio_set_mode(GPIO_BANK_UART5_RX, GPIO_MODE_OUTPUT_50_MHZ,
00676                 GPIO_CNF_OUTPUT_PUSHPULL, GPIO_UART5_RX);
00677 
00678   /* Slave receiver RX line, drive high */
00679   gpio_set(GPIO_BANK_UART5_RX, GPIO_UART5_RX);
00680 #endif
00681 
00682   /* We have no idea how long the window for allowing binding after
00683      power up is. This works for the moment but will need revisiting */
00684   DelayMs(61);
00685 
00686   for (int i = 0; i < MASTER_RECEIVER_PULSES ; i++)
00687   {
00688     gpio_clear(PrimaryUart(_BANK), PrimaryUart(_PIN));
00689     DelayUs(118);
00690     gpio_set(PrimaryUart(_BANK), PrimaryUart(_PIN));

00711 /* set TIM6 to run at DELAY_TIM_FREQUENCY */
00712 static void SpektrumDelayInit( void ) {
00713 
00714   /* Enable timer clock */
00715   rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_TIM6EN);
00716 
00717   /* Make sure the timer is reset to default values. */
00718   timer_reset(TIM6);
00719 
00720   /* Time base configuration */
00721   /* Mode does not need to be set as the default reset values are ok. */
00722   timer_set_period(TIM6, UINT16_MAX);
00723   timer_set_prescaler(TIM6, (AHB_CLK / DELAY_TIM_FREQUENCY) - 1);
00724 
00725   /*
00726    * Let's start the timer late in the cycle to force an update event before
00727    * we start using this timer for generating delays. Otherwise the prescaler
00728    * value does not seem to be taken over by the timer, resulting in way too
00729    * high counting frequency. There does not seem to be a force update bit on
00730    * TIM6 is there?
00731    */
00732   TIM6_CNT = 65534;
00733 
00734   /* Enable counter */
00735   timer_enable_counter(TIM6);
00736 }
00737 
00738 /* wait busy loop, microseconds */
00739 static void DelayUs( uint16_t uSecs ) {
00740   uint16_t start = TIM6_CNT;
00741 
00742   /* use 16 bit count wrap around */
00743   while((TIM6_CNT - start) <= uSecs);
00744 }
00745 
00746 /* wait busy loop, milliseconds */
00747 static void DelayMs( uint16_t mSecs ) {
00748 
00749   for(int i = 0; i < mSecs; i++) {
00750     DelayUs(DELAY_TIM_FREQUENCY / 1000);
00751   }
00752 }
