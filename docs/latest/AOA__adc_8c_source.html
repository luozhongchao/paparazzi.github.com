90 void AOA_adc_update( void ) {

92  adc_AOA_val = buf_AOA.sum / buf_AOA.av_nb_sample;
93 
94  // PT1 filter and convert to rad
95  AOA = AOA_filter * AOA_old + (1 - AOA_filter) * (adc_AOA_val*AOA_SENS - AOA_offset);
96  AOA_old = AOA;
97 #endif
