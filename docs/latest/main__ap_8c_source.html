330 static inline void telecommand_task( void ) {
331  uint8_t mode_changed = FALSE;
332  copy_from_to_fbw();
333 
334  uint8_t really_lost = bit_is_set(fbw_state->status, STATUS_RADIO_REALLY_LOST) && (pprz_mode == PPRZ_MODE_AUTO1 || pprz_mode == PPRZ_MODE_MANUAL);
335  if (pprz_mode != PPRZ_MODE_HOME && pprz_mode != PPRZ_MODE_GPS_OUT_OF_ORDER && launch) {
336  if (too_far_from_home) {
337  pprz_mode = PPRZ_MODE_HOME;
338  mode_changed = TRUE;
339  }
340  if (really_lost) {
341  pprz_mode = RC_LOST_MODE;
342  mode_changed = TRUE;
343  }
344  }
345  if (bit_is_set(fbw_state->status, AVERAGED_CHANNELS_SENT)) {
346  bool_t pprz_mode_changed = pprz_mode_update();
347  mode_changed |= pprz_mode_changed;
348 #if defined RADIO_CALIB && defined RADIO_CONTROL_SETTINGS
349  bool_t calib_mode_changed = RcSettingsModeUpdate(fbw_state->channels);
350  rc_settings(calib_mode_changed || pprz_mode_changed);
351  mode_changed |= calib_mode_changed;
352 #endif
353  }
354  mode_changed |= mcu1_status_update();
355  if ( mode_changed )
356  PERIODIC_SEND_PPRZ_MODE(DefaultChannel, DefaultDevice);
357 
358 #if defined RADIO_CONTROL || defined RADIO_CONTROL_AUTO1
359 
362  if (pprz_mode == PPRZ_MODE_AUTO1) {
364  h_ctl_roll_setpoint = FLOAT_OF_PPRZ(fbw_state->channels[RADIO_ROLL], 0., AUTO1_MAX_ROLL);

367  h_ctl_pitch_setpoint = FLOAT_OF_PPRZ(fbw_state->channels[RADIO_PITCH], 0., AUTO1_MAX_PITCH);
368  }

504  v_ctl_climb_loop();
505  }
506 
507 #if defined V_CTL_THROTTLE_IDLE

510

513  v_ctl_throttle_setpoint *= 10. * V_CTL_POWER_CTL_BAT_NOMINAL / (float)vsupply;
514  v_ctl_throttle_setpoint = TRIM_UPPRZ(v_ctl_throttle_setpoint);
515  }
516 #endif
517 
518  h_ctl_pitch_setpoint = v_ctl_pitch_setpoint; // Copy the pitch setpoint from the guidance to the stabilization control
519  Bound(h_ctl_pitch_setpoint, H_CTL_PITCH_MIN_SETPOINT, H_CTL_PITCH_MAX_SETPOINT);

521  v_ctl_throttle_setpoint = 0;

532  link_mcu_send();
