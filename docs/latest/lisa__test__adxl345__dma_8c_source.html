83 static inline void main_periodic_task( void ) {
84  RunOnceEvery(10,
85  {
86  DOWNLINK_SEND_ALIVE(DefaultChannel, DefaultDevice, 16, MD5SUM);
87  LED_PERIODIC();
88  });
89 
90  if (acc_status != CONFIGURED) {
91  /* set data rate to 800Hz */
92  write_to_reg(ADXL345_REG_BW_RATE, 0x0D);
93  /* switch to measurememnt mode */
94  write_to_reg(ADXL345_REG_POWER_CTL, 1<<3);
95  /* enable data ready interrupt */
96  write_to_reg(ADXL345_REG_INT_ENABLE, 1<<7);
97  /* Enable full res and interrupt active low */
98  write_to_reg(ADXL345_REG_DATA_FORMAT, 1<<3|1<<5);
99  /* reads data once to bring interrupt line up */
100  read_data();
101  acc_status = CONFIGURED;
102  }
103  else {
104  read_data();
105  }
106 }
