98 #define OFFSET_AND_ROUND2(_a, _b) (((_a)+(1<<((_b)-1))-((_a)<0?1:0))>>(_b))

100 #define MAX_SUM_ERR 4000000
101 
102 void stabilization_attitude_run(bool_t in_flight) {
103 
104 
105  /* update reference */
106  stabilization_attitude_ref_update();
107 
108  /* compute feedforward command */
109  stabilization_att_ff_cmd[COMMAND_ROLL] =
110  OFFSET_AND_ROUND(stabilization_gains.dd.x * stab_att_ref_accel.p, 5);
111  stabilization_att_ff_cmd[COMMAND_PITCH] =
112  OFFSET_AND_ROUND(stabilization_gains.dd.y * stab_att_ref_accel.q, 5);
113  stabilization_att_ff_cmd[COMMAND_YAW] =
114  OFFSET_AND_ROUND(stabilization_gains.dd.z * stab_att_ref_accel.r, 5);
115 
116  /* compute feedback command */
117  /* attitude error */
118  const struct Int32Eulers att_ref_scaled = {
119  OFFSET_AND_ROUND(stab_att_ref_euler.phi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
120  OFFSET_AND_ROUND(stab_att_ref_euler.theta, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
121  OFFSET_AND_ROUND(stab_att_ref_euler.psi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)) };
122  struct Int32Eulers att_err;
123  struct Int32Eulers* ltp_to_body_euler = stateGetNedToBodyEulers_i();
124  EULERS_DIFF(att_err, att_ref_scaled, (*ltp_to_body_euler));
125  INT32_ANGLE_NORMALIZE(att_err.psi);
126 
127  if (in_flight) {
128  /* update integrator */
129  EULERS_ADD(stabilization_att_sum_err, att_err);
130  EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
131  }
132  else {
133  INT_EULERS_ZERO(stabilization_att_sum_err);
134  }
135 
136  /* rate error */
137  const struct Int32Rates rate_ref_scaled = {
138  OFFSET_AND_ROUND(stab_att_ref_rate.p, (REF_RATE_FRAC - INT32_RATE_FRAC)),
139  OFFSET_AND_ROUND(stab_att_ref_rate.q, (REF_RATE_FRAC - INT32_RATE_FRAC)),
140  OFFSET_AND_ROUND(stab_att_ref_rate.r, (REF_RATE_FRAC - INT32_RATE_FRAC)) };
141  struct Int32Rates rate_err;
142  struct Int32Rates* body_rate = stateGetBodyRates_i();
143  RATES_DIFF(rate_err, rate_ref_scaled, (*body_rate));
144 
145  /* PID */
146  stabilization_att_fb_cmd[COMMAND_ROLL] =
147  stabilization_gains.p.x * att_err.phi +
148  stabilization_gains.d.x * rate_err.p +
149  OFFSET_AND_ROUND2((stabilization_gains.i.x * stabilization_att_sum_err.phi), 10);
150 
151  stabilization_att_fb_cmd[COMMAND_PITCH] =
152  stabilization_gains.p.y * att_err.theta +
153  stabilization_gains.d.y * rate_err.q +
154  OFFSET_AND_ROUND2((stabilization_gains.i.y * stabilization_att_sum_err.theta), 10);
155 
156  stabilization_att_fb_cmd[COMMAND_YAW] =
157  stabilization_gains.p.z * att_err.psi +
158  stabilization_gains.d.z * rate_err.r +
159  OFFSET_AND_ROUND2((stabilization_gains.i.z * stabilization_att_sum_err.psi), 10);
160 
161 
162  /* with P gain of 100, att_err of 180deg (3.14 rad)
163  * fb cmd: 100 * 3.14 * 2^12 / 2^CMD_SHIFT = 628
164  * max possible command is 9600
165  */
166 #define CMD_SHIFT 11
167 
168  /* sum feedforward and feedback */
169  stabilization_cmd[COMMAND_ROLL] =
170  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL]), CMD_SHIFT);
171 
172  stabilization_cmd[COMMAND_PITCH] =
173  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH]), CMD_SHIFT);
174 
175  stabilization_cmd[COMMAND_YAW] =
176  OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW]), CMD_SHIFT);
177 
178  /* bound the result */
179  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
180  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
181  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
182 
183 }
