83 void ppm_arch_init ( void ) {
84 
85  /* timer clock enable */
86  rcc_peripheral_enable_clock(PPM_RCC, PPM_PERIPHERAL);
87 
88  /* GPIOA clock enable */
89  rcc_peripheral_enable_clock(&RCC_APB2ENR, PPM_GPIO_PERIPHERAL);
90 
91  /* timer gpio configuration */
92  gpio_set_mode(PPM_GPIO_PORT, GPIO_MODE_INPUT,
93  GPIO_CNF_INPUT_FLOAT, PPM_GPIO_PIN);
94 
95  /* Time Base configuration */
96  timer_reset(PPM_TIMER);
97  timer_set_mode(PPM_TIMER, TIM_CR1_CKD_CK_INT,
98  TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
99  timer_set_period(PPM_TIMER, 0xFFFF);
100  /* run ppm timer at cpu freq / 9 = 8MHz */
101  timer_set_prescaler(PPM_TIMER, 8);
102 
103  /* TIM configuration: Input Capture mode ---------------------
104  The Rising edge is used as active edge,
105  Intput pin is either PA1 or PA10
106  ------------------------------------------------------------ */
107 #if defined PPM_PULSE_TYPE && PPM_PULSE_TYPE == PPM_PULSE_TYPE_POSITIVE
108  timer_ic_set_polarity(PPM_TIMER, PPM_CHANNEL, TIM_IC_RISING);
109 #elif defined PPM_PULSE_TYPE && PPM_PULSE_TYPE == PPM_PULSE_TYPE_NEGATIVE
110  timer_ic_set_polarity(PPM_TIMER, PPM_CHANNEL, TIM_IC_FALLING);
111 #else
112 #error "Unknown PM_PULSE_TYPE"
113 #endif
114  timer_ic_set_input(PPM_TIMER, PPM_CHANNEL, PPM_TIMER_INPUT);
115  timer_ic_set_prescaler(PPM_TIMER, PPM_CHANNEL, TIM_IC_PSC_OFF);
116  timer_ic_set_filter(PPM_TIMER, PPM_CHANNEL, TIM_IC_OFF);
117 
118  /* Enable timer Interrupt(s). */
119  nvic_set_priority(PPM_IRQ, 2);
120  nvic_enable_irq(PPM_IRQ);
121 
122 #ifdef PPM_IRQ2
123  nvic_set_priority(PPM_IRQ2, 2);
124  nvic_enable_irq(PPM_IRQ2);
125 #endif
126 
127  /* Enable the CC2 and Update interrupt requests. */
128  timer_enable_irq(PPM_TIMER, PPM_IRQ_FLAGS);
129 
130  /* Enable capture channel. */
131  timer_ic_enable(PPM_TIMER, PPM_CHANNEL);
132 
133  /* TIM enable counter */
134  timer_enable_counter(PPM_TIMER);
135 
136  ppm_last_pulse_time = 0;
137  ppm_cur_pulse = RADIO_CONTROL_NB_CHANNEL;
138  timer_rollover_cnt = 0;
139 
140 }

144 void tim2_isr(void) {
145 
146  if((TIM2_SR & TIM_SR_CC2IF) != 0) {
147  timer_clear_flag(TIM2, TIM_SR_CC2IF);
148 
149  uint32_t now = timer_get_counter(TIM2) + timer_rollover_cnt;
150  DecodePpmFrame(now);
151  }
152  else if((TIM2_SR & TIM_SR_UIF) != 0) {
153  timer_rollover_cnt+=(1<<16);

155  }
