00083 void ppm_arch_init ( void ) {
00084 
00085   /* timer clock enable */
00086   rcc_peripheral_enable_clock(PPM_RCC, PPM_PERIPHERAL);
00087 
00088   /* GPIOA clock enable */
00089   rcc_peripheral_enable_clock(&RCC_APB2ENR, PPM_GPIO_PERIPHERAL);
00090 
00091   /* timer gpio configuration */
00092   gpio_set_mode(PPM_GPIO_PORT, GPIO_MODE_INPUT,
00093     GPIO_CNF_INPUT_FLOAT, PPM_GPIO_PIN);
00094 
00095   /* Time Base configuration */
00096   timer_reset(PPM_TIMER);
00097   timer_set_mode(PPM_TIMER, TIM_CR1_CKD_CK_INT,
00098      TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
00099   timer_set_period(PPM_TIMER, 0xFFFF);
00100   timer_set_prescaler(PPM_TIMER, 0x8);
00101 
00102  /* TIM2 configuration: Input Capture mode ---------------------
00103      The external signal is connected to TIM2 CH2 pin (PA.01)
00104      The Rising edge is used as active edge,
00105   ------------------------------------------------------------ */
00106   timer_ic_set_polarity(PPM_TIMER, PPM_CHANNEL, TIM_IC_RISING);
00107   timer_ic_set_input(PPM_TIMER, PPM_CHANNEL, PPM_TIMER_INPUT);
00108   timer_ic_set_prescaler(PPM_TIMER, PPM_CHANNEL, TIM_IC_PSC_OFF);
00109   timer_ic_set_filter(PPM_TIMER, PPM_CHANNEL, TIM_IC_OFF);
00110 
00111   /* Enable timer Interrupt(s). */
00112   nvic_set_priority(PPM_IRQ, 2);
00113   nvic_enable_irq(PPM_IRQ);
00114 
00115 #ifdef PPM_IRQ2
00116   nvic_set_priority(PPM_IRQ2, 2);
00117   nvic_enable_irq(PPM_IRQ2);
00118 #endif
00119 
00120   /* Enable the CC2 and Update interrupt requests. */
00121   timer_enable_irq(PPM_TIMER, PPM_IRQ_FLAGS);
00122 
00123   /* Enable capture channel. */
00124   timer_ic_enable(PPM_TIMER, PPM_CHANNEL);
00125 
00126   /* TIM2 enable counter */
00127   timer_enable_counter(PPM_TIMER);
00128 
00129   ppm_last_pulse_time = 0;
00130   ppm_cur_pulse = RADIO_CONTROL_NB_CHANNEL;
00131   timer_rollover_cnt = 0;
00132 
00133 }

00138 
00139   if((TIM2_SR & TIM_SR_CC2IF) != 0) {
00140     timer_clear_flag(TIM2, TIM_SR_CC2IF);
00141 
00142     uint32_t now = timer_get_counter(TIM2) + timer_rollover_cnt;
00143     DecodePpmFrame(now);
