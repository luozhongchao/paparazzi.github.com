00076 void ppm_arch_init ( void ) {
00077 
00078   /* timer clock enable */
00079   rcc_peripheral_enable_clock(PPM_RCC, PPM_PERIPHERAL);
00080 
00081   /* GPIOA clock enable */
00082   rcc_peripheral_enable_clock(&RCC_APB2ENR, PPM_GPIO_PERIPHERAL);
00083 
00084   /* timer gpio configuration */
00085   gpio_set_mode(PPM_GPIO_PORT, GPIO_MODE_INPUT,
00086     GPIO_CNF_INPUT_FLOAT, PPM_GPIO_PIN);
00087 
00088   /* Time Base configuration */
00089   timer_reset(PPM_TIMER);
00090   timer_set_mode(PPM_TIMER, TIM_CR1_CKD_CK_INT,
00091      TIM_CR1_CMS_EDGE, TIM_CR1_DIR_UP);
00092   timer_set_period(PPM_TIMER, 0xFFFF);
00093   timer_set_prescaler(PPM_TIMER, 0x8);
00094 
00095  /* TIM2 configuration: Input Capture mode ---------------------
00096      The external signal is connected to TIM2 CH2 pin (PA.01)
00097      The Rising edge is used as active edge,
00098   ------------------------------------------------------------ */
00099   timer_ic_set_polarity(PPM_TIMER, PPM_CHANNEL, TIM_IC_RISING);
00100   timer_ic_set_input(PPM_TIMER, PPM_CHANNEL, PPM_TIMER_INPUT);
00101   timer_ic_set_prescaler(PPM_TIMER, PPM_CHANNEL, TIM_IC_PSC_OFF);
00102   timer_ic_set_filter(PPM_TIMER, PPM_CHANNEL, TIM_IC_OFF);
00103 
00104   /* Enable timer Interrupt(s). */
00105   nvic_set_priority(PPM_IRQ, 2);
00106   nvic_enable_irq(PPM_IRQ);
00107 
00108 #ifdef PPM_IRQ2
00109   nvic_set_priority(PPM_IRQ2, 2);
00110   nvic_enable_irq(PPM_IRQ2);
00111 #endif
00112 
00113   /* Enable the CC2 and Update interrupt requests. */
00114   timer_enable_irq(PPM_TIMER, PPM_IRQ_FLAGS);
00115 
00116   /* Enable capture channel. */
00117   timer_ic_enable(PPM_TIMER, PPM_CHANNEL);
00118 
00119   /* TIM2 enable counter */
00120   timer_enable_counter(PPM_TIMER);
00121 
00122   ppm_last_pulse_time = 0;
00123   ppm_cur_pulse = RADIO_CONTROL_NB_CHANNEL;
00124   timer_rollover_cnt = 0;
00125 
00126 }
