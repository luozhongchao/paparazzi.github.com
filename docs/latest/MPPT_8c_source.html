98 void MPPT_periodic( void ) {
99 
100  if (mppt_trans.status == I2CTransSuccess) {
101  switch (MPPT_status) {
102  case MPPT_STATUS_IDLE:
103  /* If free, change mode if needed */
104  if (MPPT_mode) {
105  mppt_trans.buf[0] = MPPT_MODE_ADDR;
106  mppt_trans.buf[1] = 0;
107  mppt_trans.buf[2] = MPPT_mode;
108  I2CTransmit(i2c0, mppt_trans, MPPT_SLAVE_ADDR, 3);
109  MPPT_mode = 0;
110  MPPT_status = MPPT_STATUS_WRITING;
111  } else {
112  MPPT_ask();
113  }
114  break;
115 
116  case MPPT_STATUS_WRITING:
117  MPPT_status = MPPT_STATUS_IDLE;
118  break;
119 
120  case MPPT_STATUS_ASKING:
121  /* The slave should send 2 bytes */
122  I2CReceive(i2c0, mppt_trans, MPPT_SLAVE_ADDR, 2);
123  MPPT_status = MPPT_STATUS_READING;
124  break;
125 
126  case MPPT_STATUS_READING:
127  /* We got 2 bytes */
128  if (data_index < NB_I2C_DATA)
129  MPPT_data[data_index] = (mppt_trans.buf[0]<<8) | mppt_trans.buf[1];
130  MPPT_status = MPPT_STATUS_IDLE;
131  break;
132  }
133  }
