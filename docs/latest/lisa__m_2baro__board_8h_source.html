00057 // Apply temp calibration and sensor calibration to raw measurement to get Pa (from BMP085 datasheet)
00058 static inline int32_t baro_apply_calibration(int32_t raw)
00059 {
00060   int32_t b6 = calibration.b5 - 4000;
00061   int x1 = (calibration.b2 * (b6 * b6 >> 12)) >> 11;
00062   int x2 = calibration.ac2 * b6 >> 11;
00063   int32_t x3 = x1 + x2;
00064   int32_t b3 = (((calibration.ac1 * 4 + x3) << BMP085_OSS) + 2)/4;
00065   x1 = calibration.ac3 * b6 >> 13;
00066   x2 = (calibration.b1 * (b6 * b6 >> 12)) >> 16;
00067   x3 = ((x1 + x2) + 2) >> 2;
00068   uint32_t b4 = (calibration.ac4 * (uint32_t) (x3 + 32768)) >> 15;
00069   uint32_t b7 = (raw - b3) * (50000 >> BMP085_OSS);
00070   int32_t p = b7 < 0x80000000 ? (b7 * 2) / b4 : (b7 / b4) * 2;
00071   x1 = (p >> 8) * (p >> 8);
00072   x1 = (x1 * 3038) >> 16;
00073   x2 = (-7357 * p) >> 16;
00074   return p + ((x1 + x2 + 3791) >> 4);
00075 }
