00187 void adc_init( void ) {
00188 
00189   /* connect pins for selected ADCs */
00190   PINSEL0 |= ADC_PINSEL0_ONES;
00191   PINSEL1 |= ADC_PINSEL1_ONES;
00192 
00193 #ifdef USE_AD0
00194   /* FIXME: this needs to be investigated, we should run just below 4.5MHz,
00195       but we are a lot slower (e.g. 58.6kHz with PCLK = 15MHz), see
00196       lpc_vor_convertions.c for right timing code */
00197   /* setup hw scan - PCLK/256 ( 58.6kHz/117.2kHz/234.4kHz ) - BURST ON */
00198   AD0CR = ADC_AD0CR_SEL_HW_SCAN | 0xFF << 8 | 1 << 16 | 0x01 << 21 ;
00199   /* AD0 selected as IRQ */
00200   VICIntSelect &= ~VIC_BIT(VIC_AD0);
00201   /* AD0 interrupt enabled */
00202   VICIntEnable = VIC_BIT(VIC_AD0);
00203   /* AD0 interrupt as VIC2 */
00204   _VIC_CNTL(AD0_VIC_SLOT) = VIC_ENABLE | VIC_AD0;
00205   _VIC_ADDR(AD0_VIC_SLOT) = (uint32_t)adcISR0;

00226 void adcISR0 ( void ) {
00227   ISR_ENTRY();
00228   uint32_t tmp = AD0GDR;
00229   uint8_t  channel = (uint8_t)(tmp >> 24) & 0x07;
00230   uint16_t value = (uint16_t)(tmp >> 6) & 0x03FF;
00231   adc0_val[channel] = value;
00232 
00233   struct adc_buf* buf = buffers[channel];
00234   if (buf) {
00235   uint8_t new_head = buf->head + 1;
00236   if (new_head >= buf->av_nb_sample) new_head = 0;
00237   buf->sum -= buf->values[new_head];
00238   buf->values[new_head] = value;
00239   buf->sum += value;
00240   buf->head = new_head;
00241   }
00242 
00243   VICVectAddr = 0x00000000;                 // clear this interrupt from the VIC
00244   ISR_EXIT();                               // recover registers and return
00245 }
