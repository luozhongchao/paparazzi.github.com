00189 void adc_init( void ) {
00190 
00191   /* connect pins for selected ADCs */
00192   PINSEL0 |= ADC_PINSEL0_ONES;
00193   PINSEL1 |= ADC_PINSEL1_ONES;
00194 
00195 #ifdef USE_AD0
00196   /* FIXME: this needs to be investigated, we should run just below 4.5MHz,
00197       but we are a lot slower (e.g. 58.6kHz with PCLK = 15MHz), see
00198       lpc_vor_convertions.c for right timing code */
00199   /* setup hw scan - PCLK/256 ( 58.6kHz/117.2kHz/234.4kHz ) - BURST ON */
00200   AD0CR = ADC_AD0CR_SEL_HW_SCAN | 0xFF << 8 | 1 << 16 | 0x01 << 21 ;
00201   /* AD0 selected as IRQ */
00202   VICIntSelect &= ~VIC_BIT(VIC_AD0);
00203   /* AD0 interrupt enabled */
00204   VICIntEnable = VIC_BIT(VIC_AD0);
00205   /* AD0 interrupt as VIC2 */
00206   _VIC_CNTL(AD0_VIC_SLOT) = VIC_ENABLE | VIC_AD0;

00228 void adcISR0 ( void ) {
00229   ISR_ENTRY();
00230   uint32_t tmp = AD0GDR;
00231   uint8_t  channel = (uint8_t)(tmp >> 24) & 0x07;
00232   uint16_t value = (uint16_t)(tmp >> 6) & 0x03FF;
00233   adc0_val[channel] = value;
00234 
00235   struct adc_buf* buf = buffers[channel];
00236   if (buf) {
00237   uint8_t new_head = buf->head + 1;
00238   if (new_head >= buf->av_nb_sample) new_head = 0;
00239   buf->sum -= buf->values[new_head];
00240   buf->values[new_head] = value;
00241   buf->sum += value;
00242   buf->head = new_head;
00243   }
00244 
00245   VICVectAddr = 0x00000000;                 // clear this interrupt from the VIC
00246   ISR_EXIT();                               // recover registers and return
00247 }
