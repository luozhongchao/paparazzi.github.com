00060 static int8_t baro_ms5611_crc(uint16_t* prom) {
00061   int32_t i, j;
00062   uint32_t res = 0;
00063   uint8_t crc = prom[7] & 0xF;
00064   prom[7] &= 0xFF00;
00065   for (i = 0; i < 16; i++) {
00066     if (i & 1) res ^= ((prom[i>>1]) & 0x00FF);
00067     else res ^= (prom[i>>1]>>8);
00068     for (j = 8; j > 0; j--) {
00069       if (res & 0x8000) res ^= 0x1800;
00070       res <<= 1;
00071     }
00072   }
00073   prom[7] |= crc;
00074   if (crc == ((res >> 12) & 0xF)) return 0;
00075   else return -1;
00076 }

00131 void baro_ms5611_event( void ) {
00132   if (ms5611_trans.status == I2CTransSuccess) {
00133     switch (ms5611_status) {
00134 
00135     case MS5611_RESET:
00136       ms5611_status = MS5611_RESET_OK;
00137       ms5611_trans.status = I2CTransDone;
00138       break;
00139 
00140     case MS5611_PROM:
00141       /* read prom data */
00142       ms5611_c[prom_cnt++] = (ms5611_trans.buf[0] << 8) | ms5611_trans.buf[1];
00143       if (prom_cnt < PROM_NB) {
00144         /* get next prom data */
00145         ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
00146         I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
00147       }
00148       else {
00149         /* done reading prom */
00150         ms5611_trans.status = I2CTransDone;
00151         /* check prom crc */
00152         if (baro_ms5611_crc(ms5611_c) == 0) {
00153           DOWNLINK_SEND_MS5611_COEFF(DefaultChannel, DefaultDevice,
00154               &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
00155               &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]);
00156           ms5611_status = MS5611_IDLE;
00157         }
00158         else {
00159           /* checksum error, try again */
00160           baro_ms5611_init();
00161         }
00162       }
00163       break;
00164 
00165     case MS5611_CONV_D1:
00166       ms5611_status = MS5611_CONV_D1_OK;
00167       ms5611_trans.status = I2CTransDone;
00168       break;
00169 
00170     case  MS5611_ADC_D1:
00171       /* read D1 (pressure) */
00172       ms5611_d1 = (ms5611_trans.buf[0] << 16) |
00173                   (ms5611_trans.buf[1] << 8) |
00174                   ms5611_trans.buf[2];
00175       /* start D2 conversion */
00176       ms5611_status = MS5611_CONV_D2;
00177       ms5611_trans.buf[0] = MS5611_START_CONV_D2;
00178       I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
00179       break;
00180 
00181     case MS5611_CONV_D2:
00182       ms5611_status = MS5611_CONV_D2_OK;
00183       ms5611_trans.status = I2CTransDone;
00184       break;
00185 
00186     case  MS5611_ADC_D2: {
00187       int64_t dt, baroms, tempms, off, sens, t2, off2, sens2;
00188       /* read D2 (temperature) */
00189       ms5611_d2 = (ms5611_trans.buf[0] << 16) |
00190                   (ms5611_trans.buf[1] << 8) |
00191                   ms5611_trans.buf[2];
00192       ms5611_status = MS5611_IDLE;
00193       ms5611_trans.status = I2CTransDone;
00194 
00195       /* difference between actual and ref temperature */
00196       dt = ms5611_d2 - (int64_t)ms5611_c[5] * (1<<8);
00197       /* actual temperature */
00198       tempms = 2000 + ((int64_t)dt * ms5611_c[6]) / (1<<23);
00199       /* offset at actual temperature */
00200       off = ((int64_t)ms5611_c[2] * (1<<16)) + ((int64_t)ms5611_c[4] * dt) / (1<<7);
00201       /* sensitivity at actual temperature */
00202       sens = ((int64_t)ms5611_c[1] * (1<<15)) + ((int64_t)ms5611_c[3] * dt) / (1<<8);
00203       /* second order temperature compensation */
00204       if (tempms < 2000) {
00205         t2 = (dt*dt) / (1<<31);
00206         off2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<1);
00207         sens2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<2);

00209           off2 = off2 + 7 * (int64_t)(tempms+1500)*(tempms+1500);
00210           sens2 = sens2 + 11 * ((int64_t)(tempms+1500)*(tempms+1500)) / (1<<1);
00211         }
00212         tempms = tempms - t2;
00213         off = off - off2;
00214         sens = sens - sens2;
