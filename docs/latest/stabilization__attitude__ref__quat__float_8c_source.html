00106 void stabilization_attitude_ref_update() {
00107 
00108   /* integrate reference attitude            */
00109   struct FloatQuat qdot;
00110   FLOAT_QUAT_DERIVATIVE(qdot, stab_att_ref_rate, stab_att_ref_quat);
00111   QUAT_SMUL(qdot, qdot, DT_UPDATE);
00112   QUAT_ADD(stab_att_ref_quat, qdot);
00113   FLOAT_QUAT_NORMALIZE(stab_att_ref_quat);
00114 
00115   /* integrate reference rotational speeds   */
00116   struct FloatRates delta_rate;
00117   RATES_SMUL(delta_rate, stab_att_ref_accel, DT_UPDATE);
00118   RATES_ADD(stab_att_ref_rate, delta_rate);
00119 
00120   /* compute reference angular accelerations */
00121   struct FloatQuat err;
00122   /* compute reference attitude error        */
00123   FLOAT_QUAT_INV_COMP(err, stab_att_sp_quat, stab_att_ref_quat);
00124   /* wrap it in the shortest direction       */
00125   FLOAT_QUAT_WRAP_SHORTEST(err);
00126   /* propagate the 2nd order linear model    */
00127   stab_att_ref_accel.p = -2.*stab_att_ref_model[ref_idx].zeta.p*stab_att_ref_model[ref_idx].omega.p*stab_att_ref_rate.p
00128     - stab_att_ref_model[ref_idx].omega.p*stab_att_ref_model[ref_idx].omega.p*err.qx;
00129   stab_att_ref_accel.q = -2.*stab_att_ref_model[ref_idx].zeta.q*stab_att_ref_model[ref_idx].omega.q*stab_att_ref_rate.q
00130     - stab_att_ref_model[ref_idx].omega.q*stab_att_ref_model[ref_idx].omega.q*err.qy;
00131   stab_att_ref_accel.r = -2.*stab_att_ref_model[ref_idx].zeta.r*stab_att_ref_model[ref_idx].omega.r*stab_att_ref_rate.r
00132     - stab_att_ref_model[ref_idx].omega.r*stab_att_ref_model[ref_idx].omega.r*err.qz;
00133 
00134   /*  saturate acceleration */
00135   const struct FloatRates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
00136   const struct FloatRates MAX_ACCEL = {  REF_ACCEL_MAX_P,  REF_ACCEL_MAX_Q,  REF_ACCEL_MAX_R };
00137   RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
00138 
00139   /* compute ref_euler */
00140   FLOAT_EULERS_OF_QUAT(stab_att_ref_euler, stab_att_ref_quat);
00141 }
