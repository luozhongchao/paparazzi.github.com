00108 void stabilization_attitude_ref_update() {
00109 
00110   /* integrate reference attitude            */
00111   struct FloatQuat qdot;
00112   FLOAT_QUAT_DERIVATIVE(qdot, stab_att_ref_rate, stab_att_ref_quat);
00113   QUAT_SMUL(qdot, qdot, DT_UPDATE);
00114   QUAT_ADD(stab_att_ref_quat, qdot);
00115   FLOAT_QUAT_NORMALIZE(stab_att_ref_quat);
00116 
00117   /* integrate reference rotational speeds   */
00118   struct FloatRates delta_rate;
00119   RATES_SMUL(delta_rate, stab_att_ref_accel, DT_UPDATE);
00120   RATES_ADD(stab_att_ref_rate, delta_rate);
00121 
00122   /* compute reference angular accelerations */
00123   struct FloatQuat err;
00124   /* compute reference attitude error        */
00125   FLOAT_QUAT_INV_COMP(err, stab_att_sp_quat, stab_att_ref_quat);
00126   /* wrap it in the shortest direction       */
00127   FLOAT_QUAT_WRAP_SHORTEST(err);
00128   /* propagate the 2nd order linear model    */
00129   stab_att_ref_accel.p = -2.*stab_att_ref_model[ref_idx].zeta.p*stab_att_ref_model[ref_idx].omega.p*stab_att_ref_rate.p
00130     - stab_att_ref_model[ref_idx].omega.p*stab_att_ref_model[ref_idx].omega.p*err.qx;
00131   stab_att_ref_accel.q = -2.*stab_att_ref_model[ref_idx].zeta.q*stab_att_ref_model[ref_idx].omega.q*stab_att_ref_rate.q
00132     - stab_att_ref_model[ref_idx].omega.q*stab_att_ref_model[ref_idx].omega.q*err.qy;
00133   stab_att_ref_accel.r = -2.*stab_att_ref_model[ref_idx].zeta.r*stab_att_ref_model[ref_idx].omega.r*stab_att_ref_rate.r
00134     - stab_att_ref_model[ref_idx].omega.r*stab_att_ref_model[ref_idx].omega.r*err.qz;
00135 
00136   /*  saturate acceleration */
00137   const struct FloatRates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
00138   const struct FloatRates MAX_ACCEL = {  REF_ACCEL_MAX_P,  REF_ACCEL_MAX_Q,  REF_ACCEL_MAX_R };
00139   RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
00140 
00141   /* compute ref_euler */
00142   FLOAT_EULERS_OF_QUAT(stab_att_ref_euler, stab_att_ref_quat);
00143 }
