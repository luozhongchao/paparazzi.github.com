109 void stabilization_attitude_ref_update() {
110 
111  /* integrate reference attitude */
112 #if STABILIZATION_ATTITUDE_FLOAT_REF_QUAT_INFINITESIMAL_STEP
113  struct FloatQuat qdot;
114  FLOAT_QUAT_DERIVATIVE(qdot, stab_att_ref_rate, stab_att_ref_quat);
115  QUAT_SMUL(qdot, qdot, DT_UPDATE);
116  QUAT_ADD(stab_att_ref_quat, qdot);
117 #else // use finite step (involves trig)
118  struct FloatQuat delta_q;
119  FLOAT_QUAT_DIFFERENTIAL(delta_q, stab_att_ref_rate, DT_UPDATE);
120  /* compose new ref_quat by quaternion multiplication of delta rotation and current ref_quat */
121  struct FloatQuat new_ref_quat;
122  FLOAT_QUAT_COMP(new_ref_quat, delta_q, stab_att_ref_quat);
123  QUAT_COPY(stab_att_ref_quat, new_ref_quat);
124 #endif
125  FLOAT_QUAT_NORMALIZE(stab_att_ref_quat);
126 
127  /* integrate reference rotational speeds */
128  struct FloatRates delta_rate;
129  RATES_SMUL(delta_rate, stab_att_ref_accel, DT_UPDATE);
130  RATES_ADD(stab_att_ref_rate, delta_rate);
131 
132  /* compute reference angular accelerations */
133  struct FloatQuat err;
134  /* compute reference attitude error */
135  FLOAT_QUAT_INV_COMP(err, stab_att_sp_quat, stab_att_ref_quat);
136  /* wrap it in the shortest direction */
137  FLOAT_QUAT_WRAP_SHORTEST(err);
138  /* propagate the 2nd order linear model */
139  stab_att_ref_accel.p = -2.*stab_att_ref_model[ref_idx].zeta.p*stab_att_ref_model[ref_idx].omega.p*stab_att_ref_rate.p
140  - stab_att_ref_model[ref_idx].omega.p*stab_att_ref_model[ref_idx].omega.p*err.qx;
141  stab_att_ref_accel.q = -2.*stab_att_ref_model[ref_idx].zeta.q*stab_att_ref_model[ref_idx].omega.q*stab_att_ref_rate.q
142  - stab_att_ref_model[ref_idx].omega.q*stab_att_ref_model[ref_idx].omega.q*err.qy;
143  stab_att_ref_accel.r = -2.*stab_att_ref_model[ref_idx].zeta.r*stab_att_ref_model[ref_idx].omega.r*stab_att_ref_rate.r
144  - stab_att_ref_model[ref_idx].omega.r*stab_att_ref_model[ref_idx].omega.r*err.qz;
145 
146  /* saturate acceleration */
147  const struct FloatRates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
148  const struct FloatRates MAX_ACCEL = { REF_ACCEL_MAX_P, REF_ACCEL_MAX_Q, REF_ACCEL_MAX_R };
149  RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
150 
151  /* compute ref_euler */
152  FLOAT_EULERS_OF_QUAT(stab_att_ref_euler, stab_att_ref_quat);
153 }
