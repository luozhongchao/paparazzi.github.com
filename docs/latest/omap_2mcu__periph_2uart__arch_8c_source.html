51 void uart_transmit(struct uart_periph* p, uint8_t data ) {
52  uint16_t temp = (p->tx_insert_idx + 1) % UART_TX_BUFFER_SIZE;
53 
54  if (temp == p->tx_extract_idx)
55  return; // no room
56 
57  // check if in process of sending data
58  if (p->tx_running) { // yes, add to queue
59  p->tx_buf[p->tx_insert_idx] = data;
60  p->tx_insert_idx = temp;
61  }
62  else { // no, set running flag and write to output register
63  p->tx_running = TRUE;
64  struct FmsSerialPort* fmssp = (struct FmsSerialPort*)(p->reg_addr);
65  write((int)(fmssp->fd),&data,1);
66  //printf("w %x\n",data);
67  }
68 }

70 static inline void uart_handler(struct uart_periph* p) {
71  unsigned char c='D';
72 
73  if (p->reg_addr == NULL) return; // device not initialized ?
74 
75  struct FmsSerialPort* fmssp = (struct FmsSerialPort*)(p->reg_addr);
76  int fd = fmssp->fd;
77 
78  // check if more data to send
79  if (p->tx_insert_idx != p->tx_extract_idx) {
80  write(fd,&(p->tx_buf[p->tx_extract_idx]),1);
81  //printf("w %x\n",p->tx_buf[p->tx_extract_idx]);
82  p->tx_extract_idx++;
83  p->tx_extract_idx %= UART_TX_BUFFER_SIZE;
84  }
85  else {
86  p->tx_running = FALSE; // clear running flag
87  }
88 
89  if(read(fd,&c,1) > 0){
90  //printf("r %x %c\n",c,c);
91  uint16_t temp = (p->rx_insert_idx + 1) % UART_RX_BUFFER_SIZE;
92  p->rx_buf[p->rx_insert_idx] = c;
93  // check for more room in queue
94  if (temp != p->rx_extract_idx)
95  p->rx_insert_idx = temp; // update insert index
96  }
97 
98 }
