00094 
00095   struct EcefCoor_i delta;
00096   VECT3_DIFF(delta, *ecef, def->ecef);
00097   const int64_t tmpx = (int64_t)def->ltp_of_ecef.m[0]*delta.x +
00098                        (int64_t)def->ltp_of_ecef.m[1]*delta.y +
00099                        0; /* this element is always zero http://en.wikipedia.org/wiki/Geodetic_system#From_ECEF_to_ENU */
00100   enu->x = (int32_t)(tmpx>>HIGH_RES_TRIG_FRAC);
00101   const int64_t tmpy = (int64_t)def->ltp_of_ecef.m[3]*delta.x +
00102                        (int64_t)def->ltp_of_ecef.m[4]*delta.y +
00103                        (int64_t)def->ltp_of_ecef.m[5]*delta.z;
00104   enu->y = (int32_t)(tmpy>>HIGH_RES_TRIG_FRAC);
