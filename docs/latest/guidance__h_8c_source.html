178 void guidance_h_run(bool_t in_flight) {
179  switch ( guidance_h_mode ) {
180 
181  case GUIDANCE_H_MODE_RC_DIRECT:
182  stabilization_none_run(in_flight);
183  break;
184 
185  case GUIDANCE_H_MODE_RATE:
186  stabilization_rate_run(in_flight);
187  break;
188 
189  case GUIDANCE_H_MODE_CARE_FREE:
190  case GUIDANCE_H_MODE_ATTITUDE:
191  stabilization_attitude_run(in_flight);
192  break;
193 
194  case GUIDANCE_H_MODE_HOVER:
195  guidance_h_update_reference(FALSE);
196 
197  /* set psi command */
198  guidance_h_command_body.psi = guidance_h_rc_sp.psi;
199  /* compute roll and pitch commands and set final attitude setpoint */
200  guidance_h_traj_run(in_flight);
201 
202  stabilization_attitude_run(in_flight);
203  break;
204 
205  case GUIDANCE_H_MODE_NAV:
206  if (!in_flight) guidance_h_nav_enter();
207 
208  if (horizontal_mode == HORIZONTAL_MODE_ATTITUDE) {
209  stab_att_sp_euler.phi = nav_roll;
210  stab_att_sp_euler.theta = nav_pitch;
211  /* FIXME: heading can't be set via attitude block yet, use current heading for now */
212  stab_att_sp_euler.psi = stateGetNedToBodyEulers_i()->psi;
213 #ifdef STABILIZATION_ATTITUDE_TYPE_QUAT
214  INT32_QUAT_OF_EULERS(stab_att_sp_quat, stab_att_sp_euler);
215  INT32_QUAT_WRAP_SHORTEST(stab_att_sp_quat);
216 #endif
217  }
218  else {
219  INT32_VECT2_NED_OF_ENU(guidance_h_pos_sp, navigation_carrot);
220 
221 #if GUIDANCE_H_USE_REF
222  guidance_h_update_reference(TRUE);
223 #else
224  guidance_h_update_reference(FALSE);
225 #endif
226  /* set psi command */
227  guidance_h_command_body.psi = nav_heading;
228  /* compute roll and pitch commands and set final attitude setpoint */
229  guidance_h_traj_run(in_flight);
230  }
231  stabilization_attitude_run(in_flight);
232  break;
233 
234  default:
235  break;
236  }

240 static inline void guidance_h_update_reference(bool_t use_ref) {
241  /* convert our reference to generic representation */
242  if (use_ref) {
243  gh_update_ref_from_pos_sp(guidance_h_pos_sp);
244  INT32_VECT2_RSHIFT(guidance_h_pos_ref, gh_pos_ref, (GH_POS_REF_FRAC - INT32_POS_FRAC));
245  INT32_VECT2_LSHIFT(guidance_h_speed_ref, gh_speed_ref, (INT32_SPEED_FRAC - GH_SPEED_REF_FRAC));
246  INT32_VECT2_LSHIFT(guidance_h_accel_ref, gh_accel_ref, (INT32_ACCEL_FRAC - GH_ACCEL_REF_FRAC));
247  } else {
248  VECT2_COPY(guidance_h_pos_ref, guidance_h_pos_sp);
249  INT_VECT2_ZERO(guidance_h_speed_ref);
250  INT_VECT2_ZERO(guidance_h_accel_ref);
251  }

255 #define MAX_POS_ERR POS_BFP_OF_REAL(16.)
256 #define MAX_SPEED_ERR SPEED_BFP_OF_REAL(16.)
257 #define MAX_POS_ERR_SUM ((int32_t)(MAX_POS_ERR)<< 12)
258 
259 /* with a pgain of 100 and a scale of 2,
260  * you get an angle of 5.6 degrees for 1m pos error */
261 #define GH_GAIN_SCALE 2
262 
264 #define TRAJ_MAX_BANK BFP_OF_REAL(GUIDANCE_H_MAX_BANK, INT32_ANGLE_FRAC)

266 static inline void guidance_h_traj_run(bool_t in_flight) {
267 
268  /* compute position error */
269  VECT2_DIFF(guidance_h_pos_err, guidance_h_pos_ref, *stateGetPositionNed_i());
270  /* saturate it */
271  VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
272 
273  /* compute speed error */
274  VECT2_DIFF(guidance_h_speed_err, guidance_h_speed_ref, *stateGetSpeedNed_i());
275  /* saturate it */
276  VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
277 
278  /* update pos error integral, zero it if not in_flight */
279  if (in_flight) {
280  VECT2_ADD(guidance_h_pos_err_sum, guidance_h_pos_err);
281  /* saturate it */
282  VECT2_STRIM(guidance_h_pos_err_sum, -MAX_POS_ERR_SUM, MAX_POS_ERR_SUM);
283  } else {
284  INT_VECT2_ZERO(guidance_h_pos_err_sum);
285  }
286 
287  /* run PID */
288  guidance_h_command_earth.x =
289  ((guidance_h_pgain * guidance_h_pos_err.x) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
290  ((guidance_h_dgain * (guidance_h_speed_err.x >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2)) +
291  ((guidance_h_igain * (guidance_h_pos_err_sum.x >> 12)) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
292  ((guidance_h_again * guidance_h_accel_ref.x) >> 8); /* feedforward gain */
293  guidance_h_command_earth.y =
294  ((guidance_h_pgain * guidance_h_pos_err.y) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
295  ((guidance_h_dgain * (guidance_h_speed_err.y >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2)) +
296  ((guidance_h_igain * (guidance_h_pos_err_sum.y >> 12)) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
297  ((guidance_h_again * guidance_h_accel_ref.y) >> 8); /* feedforward gain */
298 
299  VECT2_STRIM(guidance_h_command_earth, -TRAJ_MAX_BANK, TRAJ_MAX_BANK);
300 
301  /* Rotate to body frame */
302  int32_t s_psi, c_psi;
303  int32_t psi = stateGetNedToBodyEulers_i()->psi;
304  PPRZ_ITRIG_SIN(s_psi, psi);
305  PPRZ_ITRIG_COS(c_psi, psi);
306 
307  // Restore angle ref resolution after rotation
308  guidance_h_command_body.phi =
309  ( - s_psi * guidance_h_command_earth.x + c_psi * guidance_h_command_earth.y) >> INT32_TRIG_FRAC;
310  guidance_h_command_body.theta =
311  - ( c_psi * guidance_h_command_earth.x + s_psi * guidance_h_command_earth.y) >> INT32_TRIG_FRAC;
312 
313 
314  /* Add RC roll and pitch setpoints for emergency corrections */
315  guidance_h_command_body.phi += guidance_h_rc_sp.phi;
316  guidance_h_command_body.theta += guidance_h_rc_sp.theta;
317 
318  /* Set attitude setpoint in eulers and as quaternion */
319  EULERS_COPY(stab_att_sp_euler, guidance_h_command_body);
320 
321 #ifdef STABILIZATION_ATTITUDE_TYPE_QUAT
322  INT32_QUAT_OF_EULERS(stab_att_sp_quat, stab_att_sp_euler);
323  INT32_QUAT_WRAP_SHORTEST(stab_att_sp_quat);
324 #endif /* STABILIZATION_ATTITUDE_TYPE_QUAT */
325 
326 }
