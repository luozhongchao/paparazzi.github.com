00266 static inline void guidance_h_traj_run(bool_t in_flight) {
00267 
00268   /* compute position error    */
00269   VECT2_DIFF(guidance_h_pos_err, guidance_h_pos_ref, *stateGetPositionNed_i());
00270   /* saturate it               */
00271   VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
00272 
00273   /* compute speed error    */
00274   VECT2_DIFF(guidance_h_speed_err, guidance_h_speed_ref, *stateGetSpeedNed_i());
00275   /* saturate it               */
00276   VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
00277 
00278   /* update pos error integral, zero it if not in_flight */
00279   if (in_flight) {
00280     VECT2_ADD(guidance_h_pos_err_sum, guidance_h_pos_err);
00281     /* saturate it               */
00282     VECT2_STRIM(guidance_h_pos_err_sum, -MAX_POS_ERR_SUM, MAX_POS_ERR_SUM);
00283   } else {
00284     INT_VECT2_ZERO(guidance_h_pos_err_sum);
00285   }
00286 
00287   /* run PID */
00288   guidance_h_command_earth.x =
00289     guidance_h_pgain * (guidance_h_pos_err.x >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
00290     guidance_h_dgain * (guidance_h_speed_err.x >> (INT32_SPEED_FRAC - GH_GAIN_SCALE)) +
00291     guidance_h_igain * (guidance_h_pos_err_sum.x >> (12 + INT32_POS_FRAC - GH_GAIN_SCALE)) +
00292     guidance_h_again * (guidance_h_accel_ref.x >> 8); /* feedforward gain */
00293   guidance_h_command_earth.y =
00294     guidance_h_pgain * (guidance_h_pos_err.y >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
00295     guidance_h_dgain * (guidance_h_speed_err.y >> (INT32_SPEED_FRAC - GH_GAIN_SCALE)) +
00296     guidance_h_igain * (guidance_h_pos_err_sum.y >> (12 + INT32_POS_FRAC - GH_GAIN_SCALE)) +
00297     guidance_h_again * (guidance_h_accel_ref.y >> 8); /* feedforward gain */
00298 
00299   VECT2_STRIM(guidance_h_command_earth, -TRAJ_MAX_BANK, TRAJ_MAX_BANK);
00300 
00301   /* Rotate to body frame */
00302   int32_t s_psi, c_psi;
00303   int32_t psi = stateGetNedToBodyEulers_i()->psi;
00304   PPRZ_ITRIG_SIN(s_psi, psi);
00305   PPRZ_ITRIG_COS(c_psi, psi);
00306 
00307   // Restore angle ref resolution after rotation
00308   guidance_h_command_body.phi =
00309       ( - s_psi * guidance_h_command_earth.x + c_psi * guidance_h_command_earth.y) >> INT32_TRIG_FRAC;
00310   guidance_h_command_body.theta =
00311     - ( c_psi * guidance_h_command_earth.x + s_psi * guidance_h_command_earth.y) >> INT32_TRIG_FRAC;
00312 
00313 
00314   /* Add RC roll and pitch setpoints for emergency corrections */
00315   guidance_h_command_body.phi += guidance_h_rc_sp.phi;
00316   guidance_h_command_body.theta += guidance_h_rc_sp.theta;
00317 
00318   /* Set attitude setpoint in eulers and as quaternion */
00319   EULERS_COPY(stab_att_sp_euler, guidance_h_command_body);
00320 
00321 #ifdef STABILIZATION_ATTITUDE_TYPE_QUAT
00322   INT32_QUAT_OF_EULERS(stab_att_sp_quat, stab_att_sp_euler);
00323   INT32_QUAT_WRAP_SHORTEST(stab_att_sp_quat);
00324 #endif /* STABILIZATION_ATTITUDE_TYPE_QUAT */
00325 
00326 }
