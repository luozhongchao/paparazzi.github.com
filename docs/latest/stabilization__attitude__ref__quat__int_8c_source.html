00119 void stabilization_attitude_ref_update() {
00120 
00121   /* integrate reference attitude            */
00122   struct Int32Quat qdot;
00123   INT32_QUAT_DERIVATIVE(qdot, stab_att_ref_rate, stab_att_ref_quat);
00124   //QUAT_SMUL(qdot, qdot, DT_UPDATE);
00125   qdot.qi = qdot.qi >> F_UPDATE_RES;
00126   qdot.qx = qdot.qx >> F_UPDATE_RES;
00127   qdot.qy = qdot.qy >> F_UPDATE_RES;
00128   qdot.qz = qdot.qz >> F_UPDATE_RES;
00129   QUAT_ADD(stab_att_ref_quat, qdot);
00130   INT32_QUAT_NORMALIZE(stab_att_ref_quat);
00131 
00132   /* integrate reference rotational speeds
00133    * delta rate = ref_accel * dt
00134    * ref_rate = old_ref_rate + delta_rate
00135    */
00136   const struct Int32Rates delta_rate = {
00137          stab_att_ref_accel.p >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
00138          stab_att_ref_accel.q >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
00139          stab_att_ref_accel.r >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC)};
00140 
00141   RATES_ADD(stab_att_ref_rate, delta_rate);
00142 
00143   /* compute reference angular accelerations */
00144   struct Int32Quat err;
00145   /* compute reference attitude error        */
00146   INT32_QUAT_INV_COMP(err, stab_att_sp_quat, stab_att_ref_quat);
00147   /* wrap it in the shortest direction       */
00148   INT32_QUAT_WRAP_SHORTEST(err);
00149   /* propagate the 2nd order linear model    */
00150 
00151   const struct Int32Rates accel_rate = {
00152     ((int32_t)(-2.*ZETA_OMEGA_P) * (stab_att_ref_rate.p >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (ZETA_OMEGA_P_RES),
00153     ((int32_t)(-2.*ZETA_OMEGA_Q) * (stab_att_ref_rate.q >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (ZETA_OMEGA_Q_RES),
00154     ((int32_t)(-2.*ZETA_OMEGA_R) * (stab_att_ref_rate.r >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (ZETA_OMEGA_R_RES) };
00155 
00156   const struct Int32Rates accel_angle = {
00157     ((int32_t)(-OMEGA_2_P)* (err.qx   >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_P_RES),
00158     ((int32_t)(-OMEGA_2_Q)* (err.qy   >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_Q_RES),
00159     ((int32_t)(-OMEGA_2_R)* (err.qz   >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_R_RES) };
00160 
00161   RATES_SUM(stab_att_ref_accel, accel_rate, accel_angle);
00162
