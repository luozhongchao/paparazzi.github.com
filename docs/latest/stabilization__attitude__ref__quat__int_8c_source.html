103  INT32_QUAT_OF_EULERS(stab_att_ref_quat, ref_eul);
104  INT32_QUAT_WRAP_SHORTEST(stab_att_ref_quat);

106  /* set reference rate and acceleration to zero */

108  memset(&stab_att_ref_rate, 0, sizeof(struct Int32Rates));
109 }

114 #define DT_UPDATE (1./PERIODIC_FREQUENCY)
115 // CAUTION! Periodic frequency is assumed to be 512 Hz
116 // which is equal to >> 9
117 #define F_UPDATE_RES 9

119 void stabilization_attitude_ref_update() {
120 
121  /* integrate reference attitude */
122  struct Int32Quat qdot;
123  INT32_QUAT_DERIVATIVE(qdot, stab_att_ref_rate, stab_att_ref_quat);
124  //QUAT_SMUL(qdot, qdot, DT_UPDATE);
125  qdot.qi = qdot.qi >> F_UPDATE_RES;
126  qdot.qx = qdot.qx >> F_UPDATE_RES;
127  qdot.qy = qdot.qy >> F_UPDATE_RES;
128  qdot.qz = qdot.qz >> F_UPDATE_RES;
129  QUAT_ADD(stab_att_ref_quat, qdot);
130  INT32_QUAT_NORMALIZE(stab_att_ref_quat);
131 
132  /* integrate reference rotational speeds
133  * delta rate = ref_accel * dt
134  * ref_rate = old_ref_rate + delta_rate
135  */
136  const struct Int32Rates delta_rate = {
137  stab_att_ref_accel.p >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
138  stab_att_ref_accel.q >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
139  stab_att_ref_accel.r >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC)};
140 
141  RATES_ADD(stab_att_ref_rate, delta_rate);
142 
143  /* compute reference angular accelerations */
144  struct Int32Quat err;
145  /* compute reference attitude error */
146  INT32_QUAT_INV_COMP(err, stab_att_sp_quat, stab_att_ref_quat);
147  /* wrap it in the shortest direction */
148  INT32_QUAT_WRAP_SHORTEST(err);
149  /* propagate the 2nd order linear model */
150 
151  const struct Int32Rates accel_rate = {
152  ((int32_t)(-2.*ZETA_OMEGA_P) * (stab_att_ref_rate.p >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (ZETA_OMEGA_P_RES),
153  ((int32_t)(-2.*ZETA_OMEGA_Q) * (stab_att_ref_rate.q >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (ZETA_OMEGA_Q_RES),
154  ((int32_t)(-2.*ZETA_OMEGA_R) * (stab_att_ref_rate.r >> (REF_RATE_FRAC - REF_ACCEL_FRAC))) >> (ZETA_OMEGA_R_RES) };
155 
156  const struct Int32Rates accel_angle = {
157  ((int32_t)(-OMEGA_2_P)* (err.qx >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_P_RES),
158  ((int32_t)(-OMEGA_2_Q)* (err.qy >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_Q_RES),
159  ((int32_t)(-OMEGA_2_R)* (err.qz >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_R_RES) };
160 
161  RATES_SUM(stab_att_ref_accel, accel_rate, accel_angle);
