00079 void photogrammetry_calculator_update_camera2flightplan(void)
00080 {
00081 
00082   // Photogrammetry Goals
00083   float photogrammetry_sidelap_f = ((float) photogrammetry_sidelap) / 100.0f;
00084   float photogrammetry_overlap_f = ((float) photogrammetry_overlap) / 100.0f;
00085 
00086   // Linear Projection Camera Model
00087   float viewing_ratio_height = ((float) PHOTOGRAMMETRY_SENSOR_HEIGHT) / ((float)PHOTOGRAMMETRY_FOCAL_LENGTH);
00088   float viewing_ratio_width = ((float) PHOTOGRAMMETRY_SENSOR_WIDTH) / ((float)PHOTOGRAMMETRY_FOCAL_LENGTH);
00089   float pixel_projection_width = viewing_ratio_width / ((float)PHOTOGRAMMETRY_PIXELS_WIDTH);
00090 
00091   // Flightplan Variables
00092   photogrammetry_height = ((float) photogrammetry_resolution) / pixel_projection_width / 1000.0f;
00093 
00094   if (photogrammetry_height > photogrammetry_height_max)
00095     photogrammetry_height = photogrammetry_height_max;
00096   else if (photogrammetry_height < photogrammetry_height_min)
00097     photogrammetry_height = photogrammetry_height_min;
00098 
00099   photogrammetry_sidestep = viewing_ratio_width * photogrammetry_height * (1.0f - photogrammetry_sidelap_f);
00100   photogrammetry_triggerstep = viewing_ratio_height * photogrammetry_height * (1.0f - photogrammetry_overlap_f);
00101 }

00103 void photogrammetry_calculator_update_flightplan2camera(void)
00104 {
00105   // Linear Projection Camera Model
00106   float viewing_ratio_height = ((float) PHOTOGRAMMETRY_SENSOR_HEIGHT) / ((float)PHOTOGRAMMETRY_FOCAL_LENGTH);
00107   float viewing_ratio_width = ((float) PHOTOGRAMMETRY_SENSOR_WIDTH) / ((float)PHOTOGRAMMETRY_FOCAL_LENGTH);
00108   float pixel_projection_width = viewing_ratio_width / ((float)PHOTOGRAMMETRY_PIXELS_WIDTH);
00109 
00110   // Resolution <-> Height
00111   photogrammetry_resolution = photogrammetry_height * 1000.0f * pixel_projection_width;
00112 
00113   // Overlap <-> track width
00114   photogrammetry_sidelap = 100.0f - photogrammetry_sidestep / viewing_ratio_width / photogrammetry_height * 100.0f;
00115   photogrammetry_overlap = 100.0f - photogrammetry_triggerstep / viewing_ratio_height / photogrammetry_height * 100.0f;
