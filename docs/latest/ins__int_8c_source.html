00132 void ins_propagate() {
00133   /* untilt accels */
00134   struct Int32Vect3 accel_meas_body;
00135   INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
00136   struct Int32Vect3 accel_meas_ltp;
00137   INT32_RMAT_TRANSP_VMULT(accel_meas_ltp, (*stateGetNedToBodyRMat_i()), accel_meas_body);
00138 
00139 #if USE_VFF
00140   float z_accel_meas_float = ACCEL_FLOAT_OF_BFP(accel_meas_ltp.z);
00141   if (baro.status == BS_RUNNING && ins_baro_initialised) {
00142     vff_propagate(z_accel_meas_float);
00143     ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
00144     ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
00145     ins_ltp_pos.z   = POS_BFP_OF_REAL(vff_z);
00146   }
00147   else { // feed accel from the sensors
00148     // subtract -9.81m/s2 (acceleration measured due to gravity, but vehivle not accelerating in ltp)
00149     ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
00150   }
00151 #else
00152   ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
00153 #endif /* USE_VFF */
00154 
00155 #if USE_HFF
00156   /* propagate horizontal filter */
00157   b2_hff_propagate();
00158 #else
00159   ins_ltp_accel.x = accel_meas_ltp.x;
00160   ins_ltp_accel.y = accel_meas_ltp.y;
00161 #endif /* USE_HFF */
00162 
00163   INS_NED_TO_STATE();
00164 }

00166 void ins_update_baro() {
00167 #if USE_VFF
00168   if (baro.status == BS_RUNNING) {
00169     if (!ins_baro_initialised) {
00170       ins_qfe = baro.absolute;
00171       ins_baro_initialised = TRUE;
00172     }
00173     if (ins.vf_realign) {
00174       ins.vf_realign = FALSE;
00175       ins_qfe = baro.absolute;
00176 #if USE_SONAR
00177       ins_sonar_offset = sonar_meas;
00178 #endif
00179       vff_realign(0.);
00180       ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
00181       ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
00182       ins_ltp_pos.z   = POS_BFP_OF_REAL(vff_z);
00183     }
00184     else { /* not realigning, so normal update with baro measurement */
00185       ins_baro_alt = ((baro.absolute - ins_qfe) * INS_BARO_SENS_NUM)/INS_BARO_SENS_DEN;
00186       float alt_float = POS_FLOAT_OF_BFP(ins_baro_alt);
00187       vff_update(alt_float);
00188     }
00189   }
00190   INS_NED_TO_STATE();
00191 #endif
00192 }
