139 void ins_propagate() {
140  /* untilt accels */
141  struct Int32Vect3 accel_meas_body;
142  INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
143  struct Int32Vect3 accel_meas_ltp;
144  INT32_RMAT_TRANSP_VMULT(accel_meas_ltp, (*stateGetNedToBodyRMat_i()), accel_meas_body);
145 
146 #if USE_VFF
147  float z_accel_meas_float = ACCEL_FLOAT_OF_BFP(accel_meas_ltp.z);
148  if (baro.status == BS_RUNNING && ins_baro_initialised) {
149  vff_propagate(z_accel_meas_float);
150  ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
151  ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
152  ins_ltp_pos.z = POS_BFP_OF_REAL(vff_z);
153  }
154  else { // feed accel from the sensors
155  // subtract -9.81m/s2 (acceleration measured due to gravity, but vehivle not accelerating in ltp)
156  ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
157  }
158 #else
159  ins_ltp_accel.z = accel_meas_ltp.z + ACCEL_BFP_OF_REAL(9.81);
160 #endif /* USE_VFF */
161 
162 #if USE_HFF
163  /* propagate horizontal filter */
164  b2_hff_propagate();
165 #else
166  ins_ltp_accel.x = accel_meas_ltp.x;
167  ins_ltp_accel.y = accel_meas_ltp.y;
168 #endif /* USE_HFF */
169 
170  INS_NED_TO_STATE();
171 }

173 void ins_update_baro() {
174 #if USE_VFF
175  if (baro.status == BS_RUNNING) {
176  if (!ins_baro_initialised) {
177  ins_qfe = baro.absolute;
178  ins_baro_initialised = TRUE;
179  }
180  if (ins.vf_realign) {
181  ins.vf_realign = FALSE;
182  ins_qfe = baro.absolute;
183 #if USE_SONAR
184  ins_sonar_offset = sonar_meas;
185 #endif
186  vff_realign(0.);
187  ins_ltp_accel.z = ACCEL_BFP_OF_REAL(vff_zdotdot);
188  ins_ltp_speed.z = SPEED_BFP_OF_REAL(vff_zdot);
189  ins_ltp_pos.z = POS_BFP_OF_REAL(vff_z);
190  }
191  else { /* not realigning, so normal update with baro measurement */
192  ins_baro_alt = ((baro.absolute - ins_qfe) * INS_BARO_SENS_NUM)/INS_BARO_SENS_DEN;
193  float alt_float = POS_FLOAT_OF_BFP(ins_baro_alt);
194  vff_update(alt_float);
195  }
196  }
197  INS_NED_TO_STATE();
198 #endif
199 }

202 void ins_update_gps(void) {
203 #if USE_GPS
204  if (gps.fix == GPS_FIX_3D) {
205  if (!ins_ltp_initialised) {
206  ltp_def_from_ecef_i(&ins_ltp_def, &gps.ecef_pos);
207  ins_ltp_def.lla.alt = gps.lla_pos.alt;
208  ins_ltp_def.hmsl = gps.hmsl;
209  ins_ltp_initialised = TRUE;
210  stateSetLocalOrigin_i(&ins_ltp_def);
211  }
212  ned_of_ecef_point_i(&ins_gps_pos_cm_ned, &ins_ltp_def, &gps.ecef_pos);
213  ned_of_ecef_vect_i(&ins_gps_speed_cm_s_ned, &ins_ltp_def, &gps.ecef_vel);
214 #if USE_HFF
215  VECT2_ASSIGN(ins_gps_pos_m_ned, ins_gps_pos_cm_ned.x, ins_gps_pos_cm_ned.y);
216  VECT2_SDIV(ins_gps_pos_m_ned, ins_gps_pos_m_ned, 100.);
217  VECT2_ASSIGN(ins_gps_speed_m_s_ned, ins_gps_speed_cm_s_ned.x, ins_gps_speed_cm_s_ned.y);
218  VECT2_SDIV(ins_gps_speed_m_s_ned, ins_gps_speed_m_s_ned, 100.);
219  if (ins.hf_realign) {
220  ins.hf_realign = FALSE;
221 #ifdef SITL
222  struct FloatVect2 true_pos, true_speed;
223  VECT2_COPY(true_pos, fdm.ltpprz_pos);
224  VECT2_COPY(true_speed, fdm.ltpprz_ecef_vel);
225  b2_hff_realign(true_pos, true_speed);
226 #else
227  const struct FloatVect2 zero = {0.0, 0.0};
228  b2_hff_realign(ins_gps_pos_m_ned, zero);
229 #endif
230  }
231  b2_hff_update_gps();
232 #if !USE_VFF /* vff not used */
233  ins_ltp_pos.z = (ins_gps_pos_cm_ned.z * INT32_POS_OF_CM_NUM) / INT32_POS_OF_CM_DEN;
234  ins_ltp_speed.z = (ins_gps_speed_cm_s_ned.z * INT32_SPEED_OF_CM_S_NUM) INT32_SPEED_OF_CM_S_DEN;
235 #endif /* vff not used */
236 #endif /* hff used */
237 
238 
239 #if !USE_HFF /* hff not used */
240 #if !USE_VFF /* neither hf nor vf used */
241  INT32_VECT3_SCALE_3(ins_ltp_pos, ins_gps_pos_cm_ned, INT32_POS_OF_CM_NUM, INT32_POS_OF_CM_DEN);
242  INT32_VECT3_SCALE_3(ins_ltp_speed, ins_gps_speed_cm_s_ned, INT32_SPEED_OF_CM_S_NUM, INT32_SPEED_OF_CM_S_DEN);
243 #else /* only vff used */
244  INT32_VECT2_SCALE_2(ins_ltp_pos, ins_gps_pos_cm_ned, INT32_POS_OF_CM_NUM, INT32_POS_OF_CM_DEN);
245  INT32_VECT2_SCALE_2(ins_ltp_speed, ins_gps_speed_cm_s_ned, INT32_SPEED_OF_CM_S_NUM, INT32_SPEED_OF_CM_S_DEN);
246 #endif
247 
248 #if USE_GPS_LAG_HACK

260 void ins_update_sonar() {
261 #if USE_SONAR && USE_VFF
262  static int32_t sonar_filtered = 0;
263  sonar_filtered = (sonar_meas + 2*sonar_filtered) / 3;
264  /* update baro_qfe assuming a flat ground */
265  if (ins_update_on_agl && baro.status == BS_RUNNING) {
266  int32_t d_sonar = (((int32_t)sonar_filtered - ins_sonar_offset) * INS_SONAR_SENS_NUM) / INS_SONAR_SENS_DEN;
267  ins_qfe = baro.absolute + (d_sonar * (INS_BARO_SENS_DEN))/INS_BARO_SENS_NUM;
268  }
269 #endif
270 }
