00605 static inline void stmi2c_clear_pending_interrupts(I2C_TypeDef *regs)
00606 {

00608 
00609   regs->CR2 &= ~ I2C_CR2_BIT_ITBUFEN;     // Disable TXE, RXNE
00610 
00611

00613   if (BIT_X_IS_SET_IN_REG( I2C_SR1_BIT_SB, SR1 ) )

00616     regs->DR = 0x00;
00617   }

00626   {

00636 static inline void i2c_irq(struct i2c_periph *periph)
00637 {
00638 
00639   /*
00640   There are 7 possible reasons to get here:
00641 
00642   If IT_EV_FEN
00643   -------------------------
00644 
00645   We are always interested in all IT_EV_FEV: all are required.
00646 
00647   1) SB   // Start Condition Success in Master mode
00648   2) ADDR   // Address sent received Acknoledge
00649   [3 ADDR10]  // -- 10bit address stuff
00650   [4 STOPF] // -- only for slaves: master has no stop interrupt
00651   5) BTF    // I2C has stopped working (it is waiting for new data, all buffers are tx_empty/rx_full)
00652 
00653   // Beware: using the buffered I2C has some interesting properties:
00654     -in master receive mode: BTF only occurs after the 2nd received byte: after the first byte is received it is
00655            in RD but the I2C can still receive a second byte. Only when the 2nd byte is received while the RxNE is 1
00656      then a BTF occurs (I2C can not continue receiving bytes or they will get lost). During BTF I2C is halted (SCL held low)
00657     -in master transmitmode: when writing a byte to WD, you instantly get a new TxE interrupt while the first is not
00658      transmitted yet. The byte was pushed to the I2C shift register and the buffer is ready for more. You can already
00659      fill new data in the buffer while the first is still being transmitted for max performance transmission.
00660 
00661         // Beware: besides data buffering you can/must plan several consecutive actions. You can send 2 bytes to the buffer, ask for a stop and
00662            a new start in one go.
00663 
00664           -thanks to / because of this buffering and event sheduling there is not 1 interrupt per start / byte / stop
00665            This also means you must think more in advance and a transaction could be popped from the stack even before it is
00666            actually completely transmitted. But then you would not know the result yet so you have to keep it until the result
00667            is known.
00668 
00669   // Beware: the order in which Status is read determines how flags are cleared. You should not just read SR1 & SR2 every time
00670 
00671   If IT_EV_FEN AND IT_EV_BUF
00672   --------------------------
00673 
00674   Buffer event are not always wanted and are tipically switched on during longer data transfers. Make sure to turn off in time.
00675 
00676   6) RxNE
00677   7) TxE
00678 
00679   --------------------------------------------------------------------------------------------------
00680   // This driver uses only a subset of the pprz_i2c_states for several reasons:
00681   // -we have less interrupts than the I2CStatus states (for efficiency)
00682   // -STM32 has such a powerfull I2C engine with plenty of status register flags that
00683             only little extra status information needs to be stored.
00684 
00685        // Status is re-used (abused) to remember the last COMMAND THAT WAS SENT to the STM I2C hardware.
00686 
00687 // TODO: check which are used
00688   enum I2CStatus {
00689     I2CIdle,      // No more last command
00690 
00691     I2CStartRequested,    // Last command was start
00692     I2CRestartRequested,    // Last command was restart
00693     I2CStopRequested,   // Very important to not send double stop conditions
00694 
00695     I2CSendingByte,   // Some address/data operation
00696 
00697     // Following are not used
00698     I2CReadingByte,
00699     I2CAddrWrSent,
00700     I2CAddrRdSent,
00701     I2CSendingLastByte,
00702     I2CReadingLastByte,
00703     I2CComplete,
00704     I2CFailed
00705   };
00706 
00707   ---------
00708 
00709   The STM waits indefinately (holding SCL low) for user interaction:
00710   a) after a master-start (waiting for address)
00711   b) after an address (waiting for data)
00712      not during data sending when using buffered
00713   c) after the last byte is transmitted (waiting for either stop or restart)
00714      not during data receiving when using buffered
00715      not after the last byte is received
00716 
00717   -The STM I2C stalls indefinately when a stop condition was attempted that
00718   did not succeed. The BUSY flag remains on.
00719         -There is no STOP interrupt: use needs another way to finish.
00720 
00721    */
00722 
00723 
00725   // Reading the status:
00726   // - Caution: this clears several flags and can start transmissions etc...
00727   // - Certain flags like STOP / (N)ACK need to be guaranteed to be set before
00728   //   the transmission of the byte is finished. At higher clock rates that can be
00729   //   quite fast: so we allow no other interrupt to be triggered in between
00730   //   reading the status and setting all needed flags
00731 
00732   // Direct Access to the I2C Registers
00733   // Do not read SR2 as it might start the reading while an (n)ack bit might be needed first
00734   I2C_TypeDef *regs = (I2C_TypeDef *) periph->reg_addr;
00735 
00736 #ifdef I2C_DEBUG_LED
00737   LED1_ON();
00738   LED1_OFF();
00739 #endif
00740 
00741 
00744   //
00745   //  TRANSACTION HANDLER
00746 
00747   enum STMI2CSubTransactionStatus ret = 0;
00748 
00750   // Nothing Left To Do
00751   if (periph->trans_extract_idx == periph->trans_insert_idx)
00752   {
00753 #ifdef I2C_DEBUG_LED
00754         LED2_ON();
00755         LED1_ON();
00756   LED2_OFF();
00757   LED1_OFF();
00758 
00759     // no transaction and also an error?
00760     LED_SHOW_ACTIVE_BITS(regs);
00761 #endif
00762 
00763     // If we still get an interrupt but there are no more things to do
00764     // (which can happen if an event was sheduled just before a bus error occurs)
00765     // then its easy: just stop: clear all interrupt generating bits
00766 
00767     // Clear Running Events
00768     stmi2c_clear_pending_interrupts(regs);
00769 
00770     // Count The Errors
00771     i2c_error(periph);
00772 
00773     // Mark this as a special error
00774     periph->errors->unexpected_event_cnt++;
00775 
00776     periph->status = I2CIdle;
00777 
00778     // There are no transactions anymore:
00779     // furtheron we need a transaction pointer: so we are not allowed to continue
00780     return;
00781   }
00782 
00783   struct i2c_transaction* trans = periph->trans[periph->trans_extract_idx];
00784 
00786   // If there was an error:
00787   if (( regs->SR1 & I2C_SR1_BITS_ERR ) != 0x0000)
00788   {
00789 
00790 #ifdef I2C_DEBUG_LED
00791         LED1_ON();
00792         LED2_ON();
00793   LED1_OFF();
00794   LED2_OFF();
00795 
00796         LED_SHOW_ACTIVE_BITS(regs);
00797 #endif
00798 
00799     // Set result in transaction
00800     trans->status = I2CTransFailed;
00801 
00802     // Prepare for next
00803     ret = STMI2C_SubTra_Ready;
00804 
00805     // Make sure a TxRx does not Restart
00806     trans->type = I2CTransRx;
00807 
00808     // Clear Running Events
00809     stmi2c_clear_pending_interrupts(regs);
00810 
00811     // Count The Errors
00812     i2c_error(periph);
00813   }
00814 
00815 
00817   // Normal Event:
00818   else
00819   {
00820 
00821     if (trans->type == I2CTransRx) // TxRx are converted to Rx after the Tx Part
00822     {
00823       switch (trans->len_r)
00824       {
00825         case 1:
00826           ret = stmi2c_read1(regs,trans);
00827           break;
00828         case 2:
00829           ret = stmi2c_read2(regs,trans);
00830           break;
00831         default:
00832           ret = stmi2c_readmany(regs,periph, trans);
00833           break;
00834       }
00835     }
00836     else // TxRx or Tx
00837     {
00838       ret = stmi2c_send(regs,periph,trans);
00839     }
00840   }
00841 
00843   // Sub-transaction has finished
00844   if (ret != STMI2C_SubTra_Busy)
00845   {
00846     // If a restart is not needed
00847     if (trans->type != I2CTransTxRx)
00848     {
00849       // Ready, no stop condition set yet
00850       if (ret == STMI2C_SubTra_Ready)
00851       {
00852 
00853         // Program a stop
00854         PPRZ_I2C_SEND_STOP(regs);
00855 
00856         // Silent any BTF that would occur before STOP is executed
00857         regs->DR = 0x00;
00858       }
00859 
00860       // Jump to the next transaction
00861       periph->trans_extract_idx++;
00862       if (periph->trans_extract_idx >= I2C_TRANSACTION_QUEUE_LEN)
00863         periph->trans_extract_idx = 0;
00864 
00865       // Tell everyone we are ready
00866       periph->status = I2CIdle;
00867 
00868 
00869       // if we have no more transaction to process, stop here
00870       if (periph->trans_extract_idx == periph->trans_insert_idx)
00871       {
00872 
00873 #ifdef I2C_DEBUG_LED
00874         LED2_ON();
00875         LED1_ON();
00876   LED1_OFF();
00877         LED1_ON();
00878   LED1_OFF();
00879   LED2_OFF();
00880 #endif
00881       }
00882       // if not, start next transaction
00883       else
00884       {
00885         // Restart transaction doing the Rx part now
00886 // --- moved to idle function
00887         PPRZ_I2C_SEND_START(periph);
00888 // ------
00889      }
00890 
00891     }
00892     // RxTx -> Restart and do Rx part
00893     else
00894     {
00895       trans->type = I2CTransRx;
00896       periph->status = I2CStartRequested;
00897       regs->CR1 |= I2C_CR1_BIT_START;
00898 
00899       // Silent any BTF that would occur before SB
00900       regs->DR = 0x00;
00901     }
00902   }
00903 
00904   return;
00905 }

00907 static inline void i2c_error(struct i2c_periph *periph)
00908 {
00909   uint8_t err_nr = 0;
00910   periph->errors->er_irq_cnt;
00911   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_AF)) {       /* Acknowledge failure */
00912     periph->errors->ack_fail_cnt++;
00913     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_AF);
00914     err_nr = 1;
00915   }
00916   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_BERR)) {     /* Misplaced Start or Stop condition */
00917     periph->errors->miss_start_stop_cnt++;
00918     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_BERR);
00919     err_nr = 2;
00920   }
00921   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_ARLO)) {     /* Arbitration lost */
00922     periph->errors->arb_lost_cnt++;
00923     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_ARLO);
00924     err_nr = 3;
00925   }
00926   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_OVR)) {      /* Overrun/Underrun */
00927     periph->errors->over_under_cnt++;
00928     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_OVR);
00929     err_nr = 4;
00930   }
00931   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_PECERR)) {   /* PEC Error in reception */
00932     periph->errors->pec_recep_cnt++;
00933     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_PECERR);
00934     err_nr = 5;
00935   }
00936   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_TIMEOUT)) {  /* Timeout or Tlow error */
00937     periph->errors->timeout_tlow_cnt++;
00938     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_TIMEOUT);
00939     err_nr = 6;
00940   }
00941   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_SMBALERT)) { /* SMBus alert */
00942     periph->errors->smbus_alert_cnt++;
00943     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_SMBALERT);
00944     err_nr = 7;
00945   }
00946 
00947 #ifdef I2C_DEBUG_LED
00948   LED_ERROR(20, err_nr);
00949 #endif
00950 
00951   return;
00952

01233   uint8_t temp;
