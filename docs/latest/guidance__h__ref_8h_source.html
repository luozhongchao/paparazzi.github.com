00101 static inline void b2_gh_update_ref_from_pos_sp(struct Int32Vect2 pos_sp) {
00102 
00103   VECT2_ADD(b2_gh_pos_ref, b2_gh_speed_ref);
00104   VECT2_ADD(b2_gh_speed_ref, b2_gh_accel_ref);
00105 
00106   // compute the "speed part" of accel = -2*zeta*omega*speed -omega^2(pos - pos_sp)
00107   struct Int32Vect2 speed;
00108   INT32_VECT2_RSHIFT(speed, b2_gh_speed_ref, (B2_GH_SPEED_REF_FRAC - B2_GH_ACCEL_REF_FRAC));
00109   VECT2_SMUL(speed, speed, -2*B2_GH_ZETA_OMEGA);
00110   INT32_VECT2_RSHIFT(speed, speed, B2_GH_ZETA_OMEGA_FRAC);
00111   // compute pos error in pos_sp resolution
00112   struct Int32Vect2 pos_err;
00113   INT32_VECT2_RSHIFT(pos_err, b2_gh_pos_ref, (B2_GH_POS_REF_FRAC - INT32_POS_FRAC));
00114   VECT2_DIFF(pos_err, pos_err, pos_sp);
00115   // convert to accel resolution
00116   INT32_VECT2_RSHIFT(pos_err, pos_err, (INT32_POS_FRAC - B2_GH_ACCEL_REF_FRAC));
00117   // compute the "pos part" of accel
00118   struct Int32Vect2 pos;
00119   VECT2_SMUL(pos, pos_err, (-B2_GH_OMEGA_2));
00120   INT32_VECT2_RSHIFT(pos, pos, B2_GH_OMEGA_2_FRAC);
00121   // sum accel
00122   VECT2_SUM(b2_gh_accel_ref, speed, pos);
00123 
00124   /* Saturate accelerations */
00125   VECT2_STRIM(b2_gh_accel_ref, -B2_GH_MAX_ACCEL, B2_GH_MAX_ACCEL);
00126 
00127   /* Saturate speed and adjust acceleration accordingly */
00128   if (b2_gh_speed_ref.x <= -B2_GH_MAX_SPEED) {
00129     b2_gh_speed_ref.x = -B2_GH_MAX_SPEED;
00130     if (b2_gh_accel_ref.x < 0)
00131       b2_gh_accel_ref.x = 0;
00132   }
00133   else if (b2_gh_speed_ref.x >= B2_GH_MAX_SPEED) {
00134     b2_gh_speed_ref.x = B2_GH_MAX_SPEED;
00135     if (b2_gh_accel_ref.x > 0)
00136       b2_gh_accel_ref.x = 0;
00137   }
00138   if (b2_gh_speed_ref.y <= -B2_GH_MAX_SPEED) {
00139     b2_gh_speed_ref.y = -B2_GH_MAX_SPEED;
00140     if (b2_gh_accel_ref.y < 0)
