00209   ahrs_impl.imu_rate.r += dr;

00266     ahrs_impl.gps_speed = 0;
00267     ahrs_impl.gps_acceleration = 0;

00330 void Normalize(void)
00331 {
00332   float error=0;
00333   float temporary[3][3];
00334   float renorm=0;
00335   uint8_t problem=FALSE;
00336 
00337   // Find the non-orthogonality of X wrt Y
00338   error= -Vector_Dot_Product(&DCM_Matrix[0][0],&DCM_Matrix[1][0])*.5; //eq.19
00339 
00340   // Add half the XY error to X, and half to Y
00341   Vector_Scale(&temporary[0][0], &DCM_Matrix[1][0], error);           //eq.19
00342   Vector_Scale(&temporary[1][0], &DCM_Matrix[0][0], error);           //eq.19
00343   Vector_Add(&temporary[0][0], &temporary[0][0], &DCM_Matrix[0][0]);  //eq.19
00344   Vector_Add(&temporary[1][0], &temporary[1][0], &DCM_Matrix[1][0]);  //eq.19
00345 
00346   // The third axis is simply set perpendicular to the first 2. (there is not correction of XY based on Z)
00347   Vector_Cross_Product(&temporary[2][0],&temporary[0][0],&temporary[1][0]); // c= a x b //eq.20
00348 
00349   // Normalize lenght of X
00350   renorm= Vector_Dot_Product(&temporary[0][0],&temporary[0][0]);
00351   // a) if norm is close to 1, use the fast 1st element from the tailer expansion of SQRT
00352   // b) if the norm is further from 1, use a real sqrt
00353   // c) norm is huge: disaster! reset! mayday!
00354   if (renorm < 1.5625f && renorm > 0.64f) {
00355     renorm= .5 * (3-renorm);                                          //eq.21
00356   } else if (renorm < 100.0f && renorm > 0.01f) {
00357     renorm= 1. / sqrt(renorm);
00358 #if PERFORMANCE_REPORTING == 1
00359     renorm_sqrt_count++;
00360 #endif
00361   } else {
00362     problem = TRUE;
00363 #if PERFORMANCE_REPORTING == 1
00364     renorm_blowup_count++;
00365 #endif
00366   }
00367   Vector_Scale(&DCM_Matrix[0][0], &temporary[0][0], renorm);
00368 
00369   // Normalize lenght of Y
00370   renorm= Vector_Dot_Product(&temporary[1][0],&temporary[1][0]);
00371   if (renorm < 1.5625f && renorm > 0.64f) {
00372     renorm= .5 * (3-renorm);                                                 //eq.21
00373   } else if (renorm < 100.0f && renorm > 0.01f) {
00374     renorm= 1. / sqrt(renorm);
00375 #if PERFORMANCE_REPORTING == 1
00376     renorm_sqrt_count++;
00377 #endif
00378   } else {
00379     problem = TRUE;
00380 #if PERFORMANCE_REPORTING == 1
00381     renorm_blowup_count++;
00382 #endif
00383   }
00384   Vector_Scale(&DCM_Matrix[1][0], &temporary[1][0], renorm);
00385 
00386   // Normalize lenght of Z
00387   renorm= Vector_Dot_Product(&temporary[2][0],&temporary[2][0]);
00388   if (renorm < 1.5625f && renorm > 0.64f) {
00389     renorm= .5 * (3-renorm);                                                 //eq.21
00390   } else if (renorm < 100.0f && renorm > 0.01f) {
00391     renorm= 1. / sqrt(renorm);
00392 #if PERFORMANCE_REPORTING == 1
00393     renorm_sqrt_count++;
00394 #endif
00395   } else {
00396     problem = TRUE;
00397 #if PERFORMANCE_REPORTING == 1
00398     renorm_blowup_count++;
00399 #endif
00400   }
00401   Vector_Scale(&DCM_Matrix[2][0], &temporary[2][0], renorm);
00402 
00403   // Reset on trouble
00404   if (problem) {                // Our solution is blowing up and we will force back to initial condition.  Hope we are not upside down!
00405     set_dcm_matrix_from_rmat(&ahrs_impl.body_to_imu_rmat);
00406     problem = FALSE;
00407   }
00408 }

00411 void Drift_correction(void)
00412 {
00413   //Compensation the Roll, Pitch and Yaw drift.
00414   static float Scaled_Omega_P[3];
00415   static float Scaled_Omega_I[3];
00416   float Accel_magnitude;
00417   float Accel_weight;
00418   float Integrator_magnitude;
00419 
00420   // Local Working Variables
00421   float errorRollPitch[3];
00422   float errorYaw[3];
00423   float errorCourse;
00424 
00425   //*****Roll and Pitch***************
00426 
00427   // Calculate the magnitude of the accelerometer vector
00428   Accel_magnitude = sqrt(accel_float.x*accel_float.x + accel_float.y*accel_float.y + accel_float.z*accel_float.z);
00429   Accel_magnitude = Accel_magnitude / GRAVITY; // Scale to gravity.
00430   // Dynamic weighting of accelerometer info (reliability filter)
00431   // Weight for accelerometer info (<0.5G = 0.0, 1G = 1.0 , >1.5G = 0.0)
00432   Accel_weight = Chop(1 - 2*fabs(1 - Accel_magnitude),0,1);  //
00433 
00434 #if USE_HIGH_ACCEL_FLAG
00435   // Test for high acceleration:
00436   //  - low speed
00437   //  - high thrust
00438   float speed = *stateGetHorizontalSpeedNorm_f();
00439   if (speed < HIGH_ACCEL_LOW_SPEED && ap_state->commands[COMMAND_THROTTLE] > HIGH_ACCEL_HIGH_THRUST && !high_accel_done) {
00440     high_accel_flag = TRUE;
00441   } else {
00442     high_accel_flag = FALSE;
00443     if (speed > HIGH_ACCEL_LOW_SPEED && !high_accel_done) {
00444       high_accel_done = TRUE; // After takeoff, don't use high accel before landing (GS small, Throttle small)
00445     }
00446     if (speed < HIGH_ACCEL_HIGH_THRUST_RESUME && ap_state->commands[COMMAND_THROTTLE] < HIGH_ACCEL_HIGH_THRUST_RESUME) {
00447       high_accel_done = FALSE; // Activate high accel after landing
00448     }
00449   }
00450   if (high_accel_flag) { Accel_weight = 0.; }
00451 #endif
00452 
00453 
00454   #if PERFORMANCE_REPORTING == 1
00455   {
00456 
00457     float tempfloat = ((Accel_weight - 0.5) * 256.0f);    //amount added was determined to give imu_health a time constant about twice the time constant of the roll/pitch drift correction
00458     imu_health += tempfloat;
00459     Bound(imu_health,129,65405);
00460   }
00461   #endif
00462 
00463   Vector_Cross_Product(&errorRollPitch[0],&accel_float.x,&DCM_Matrix[2][0]); //adjust the ground of reference
00464   Vector_Scale(&Omega_P[0],&errorRollPitch[0],Kp_ROLLPITCH*Accel_weight);
00465 
00466   Vector_Scale(&Scaled_Omega_I[0],&errorRollPitch[0],Ki_ROLLPITCH*Accel_weight);
00467   Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);
00468 
00469   //*****YAW***************
00470 
00471 #if USE_MAGNETOMETER
00472   // We make the gyro YAW drift correction based on compass magnetic heading
00473 //  float mag_heading_x = cos(MAG_Heading);
00474 //  float mag_heading_y = sin(MAG_Heading);
00475   // 2D dot product
00476   errorCourse=(DCM_Matrix[0][0]*MAG_Heading_Y) + (DCM_Matrix[1][0]*MAG_Heading_X);  //Calculating YAW error
00477   Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00478 
00479   Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW);
00480   Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.
00481 
00482   Vector_Scale(&Scaled_Omega_I[0],&errorYaw[0],Ki_YAW);
00483   Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);//adding integrator to the Omega_I
00484 
00485 #else // Use GPS Ground course to correct yaw gyro drift
00486 
00487   if (ahrs_impl.gps_course_valid) {
00488     float course = ahrs_impl.gps_course - M_PI; //This is the runaway direction of you "plane" in rad
00489     float COGX = cosf(course); //Course overground X axis
00490     float COGY = sinf(course); //Course overground Y axis
00491 
00492     errorCourse=(DCM_Matrix[0][0]*COGY) - (DCM_Matrix[1][0]*COGX);  //Calculating YAW error
00493     Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00494 
00495     Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW);
00496     Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.
00497 
00498     Vector_Scale(&Scaled_Omega_I[0],&errorYaw[0],Ki_YAW);
00499     Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);//adding integrator to the Omega_I
00500   }
00501 #if USE_MAGNETOMETER_ONGROUND == 1
00502 #pragma message AHRS_FLOAT_DCM uses magnetometer prior to takeoff and GPS during flight
00503   else if (launch == FALSE)
00504   {
00505     float COGX = imu.mag.x; // Non-Tilt-Compensated (for filter stability reasons)
00506     float COGY = imu.mag.y; // Non-Tilt-Compensated (for filter stability reasons)
00507 
00508     errorCourse=(DCM_Matrix[0][0]*COGY) - (DCM_Matrix[1][0]*COGX);  //Calculating YAW error
00509     Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00510 
00511     // P only
00512     Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW / 10.0);
00513     Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.fi
00514   }
00515 #endif // USE_MAGNETOMETER_ONGROUND
00516 #endif
00517 
00518   //  Here we will place a limit on the integrator so that the integrator cannot ever exceed half the saturation limit of the gyros
00519   Integrator_magnitude = sqrt(Vector_Dot_Product(Omega_I,Omega_I));
00520   if (Integrator_magnitude > RadOfDeg(300)) {
00521     Vector_Scale(Omega_I,Omega_I,0.5f*RadOfDeg(300)/Integrator_magnitude);
00522   }
00523 
00524 
00525 }

00531   Vector_Add(&Omega_Vector[0], &Omega[0], &Omega_P[0]); //adding Integrator term
