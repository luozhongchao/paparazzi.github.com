00154 void dma1_c4_irq_handler(void) {
00155 
00156   ADS8344_values[channel] = (buf_in[1] << 8 | buf_in[2]) << 1 | buf_in[3] >> 7;
00157   channel++;
00158   if (channel > 6) {
00159     ADS8344_available = TRUE;
00160     ADS8344Unselect();
00161     DMA_ITConfig(DMA1_Channel4, DMA_IT_TC, DISABLE);
00162     /* Disable SPI_2 Rx and TX request */
00163     SPI_I2S_DMACmd(SPI2, SPI_I2S_DMAReq_Rx, DISABLE);
00164     SPI_I2S_DMACmd(SPI2, SPI_I2S_DMAReq_Tx, DISABLE);
00165     /* Disable DMA1 Channel4 and 5 */
00166     DMA_Cmd(DMA1_Channel4, DISABLE);
00167     DMA_Cmd(DMA1_Channel5, DISABLE);
00168   }
00169   else {
00170     ADS8344_read_channel();
00171   }
