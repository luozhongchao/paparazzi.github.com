443 static void spi_rw(struct spi_periph* p, struct spi_transaction * _trans)
444 {
445  struct spi_periph_dma *dma;
446  uint8_t config = 0x00;
447 
448  // Store local copy to notify of the results
449  _trans->status = SPITransRunning;
450  p->status = SPIRunning;
451 
452  // Select the slave if required
453  if ( _trans->select == SPISelectUnselect || _trans->select == SPISelect ) {
454  SpiSlaveSelect( _trans->slave_idx );
455  }
456 
457  // Run the callback AFTER selecting the slave
458  if (_trans->before_cb != 0) {
459  _trans->before_cb( _trans );
460  }
461 
462  dma = p->init_struct;
463 
464  config = (_trans->dss << 6) | (_trans->cdiv << 3) | (_trans->bitorder << 2) | (_trans->cpha << 1) | (_trans->cpol);
465  if ( config != dma->config ) {
466  dma->config = config;
467 
468  // A different config is required in this transaction...
469  if ( _trans->dss == SPIDss8bit ) {
470  dma->dss = SPI_CR1_DFF_8BIT;
471  } else {
472  dma->dss = SPI_CR1_DFF_16BIT;
473  }
474  if ( _trans->bitorder == SPIMSBFirst ) {
475  dma->bo = SPI_CR1_MSBFIRST;
476  } else {
477  dma->bo = SPI_CR1_LSBFIRST;
478  }
479  if ( _trans->cpha == SPICphaEdge1 ) {
480  dma->cpha = SPI_CR1_CPHA_CLK_TRANSITION_1;
481  } else {
482  dma->cpha = SPI_CR1_CPHA_CLK_TRANSITION_2;
483  }
484  if ( _trans->cpol == SPICpolIdleLow ) {
485  dma->cpol = SPI_CR1_CPOL_CLK_TO_0_WHEN_IDLE;
486  } else {
487  dma->cpol = SPI_CR1_CPOL_CLK_TO_1_WHEN_IDLE;
488  }
489 
490  switch( _trans->cdiv ) {
491  case SPIDiv2:
492  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_2;
493  break;
494  case SPIDiv4:
495  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_4;
496  break;
497  case SPIDiv8:
498  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_8;
499  break;
500  case SPIDiv16:
501  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_16;
502  break;
503  case SPIDiv32:
504  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_32;
505  break;
506  case SPIDiv64:
507  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_64;
508  break;
509  case SPIDiv128:
510  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_128;
511  break;
512  case SPIDiv256:
513  dma->cdiv = SPI_CR1_BAUDRATE_FPCLK_DIV_256;
514  break;
515  default:
516  break;
517  }
518 
519  spi_disable( dma->spi );
520  spi_init_master( dma->spi, dma->cdiv, dma->cpol, dma->cpha, dma->dss, dma->bo );
521  spi_enable_software_slave_management( dma->spi );
522  spi_set_nss_high( dma->spi );
523  spi_enable( dma->spi );
524  // FIXME this is also called immediately after spi_rw in spi_submit is this needed?
525  spi_arch_int_enable( p );
526  }
527 
528  /*
529  * Clear flag for interrupt order handling
530  *
531  * Note: If one of the transaction lengths is 0, it won't trigger an interrupt.
532  * This is like the interrupt has already finished, so you specify that the other
533  * dma has already finished, and everything is cleaned up after the one interrupt
534  * that actually runs. The case that both lengths are zero is guarded against in
535  * spi_submit.
536  */
537  dma->other_dma_finished = 0;
538 
539  if ( _trans->input_length > 0 ) {
540  // Rx_DMA_Channel configuration ------------------------------------
541  dma_channel_reset( dma->dma, dma->rx_chan );
542  dma_set_peripheral_address(dma->dma, dma->rx_chan, (u32)dma->spidr);
543  dma_set_memory_address(dma->dma, dma->rx_chan, (uint32_t)_trans->input_buf);
544  dma_set_number_of_data(dma->dma, dma->rx_chan, _trans->input_length);
545  dma_set_read_from_peripheral(dma->dma, dma->rx_chan);
546  //dma_disable_peripheral_increment_mode(dma->dma, dma->rx_chan);
547  dma_enable_memory_increment_mode(dma->dma, dma->rx_chan);
548  dma_set_peripheral_size(dma->dma, dma->rx_chan, DMA_CCR_PSIZE_8BIT);
549  dma_set_memory_size(dma->dma, dma->rx_chan, DMA_CCR_MSIZE_8BIT);
550  //dma_set_mode(dma->dma, dma->rx_chan, DMA_???_NORMAL);
551  dma_set_priority(dma->dma, dma->rx_chan, DMA_CCR_PL_VERY_HIGH);
552  } else {
553  // There will be no interrupt in this case, i.e. like the interrupt already finished
554  dma->other_dma_finished = 1;
555  }
556 
557  if ( _trans->output_length > 0 ) {
558  // SPI Tx_DMA_Channel configuration ------------------------------------
559  dma_channel_reset(dma->dma, dma->tx_chan);
560  dma_set_peripheral_address(dma->dma, dma->tx_chan, (u32)dma->spidr);
561  dma_set_memory_address(dma->dma, dma->tx_chan, (uint32_t)_trans->output_buf);
562  dma_set_number_of_data(dma->dma, dma->tx_chan, _trans->output_length);
563  dma_set_read_from_memory(dma->dma, dma->tx_chan);
564  //dma_disable_peripheral_increment_mode(dma->dma, dma->tx_chan);
565  dma_enable_memory_increment_mode(dma->dma, dma->tx_chan);
566  dma_set_peripheral_size(dma->dma, dma->tx_chan, DMA_CCR_PSIZE_8BIT);
567  dma_set_memory_size(dma->dma, dma->tx_chan, DMA_CCR_MSIZE_8BIT);
568  //dma_set_mode(dma->dma, dma->tx_chan, DMA_???_NORMAL);
569  dma_set_priority(dma->dma, dma->tx_chan, DMA_CCR_PL_MEDIUM);
570  } else {
571  // There will be no interrupt in this case, i.e. like the interrupt already finished
572  dma->other_dma_finished = 1;
573  }
574 
575  if ( _trans->input_length > 0 ) {
576  // Enable SPI Rx request
577  spi_enable_rx_dma(dma->spi);
578  // Enable dma->dma rx channel
579  dma_enable_channel(dma->dma, dma->rx_chan);
580  }
581 
582  if ( _trans->output_length > 0 ) {
583  // Enable SPI Tx request
584  spi_enable_tx_dma(dma->spi);
585  // Enable dma->dma tx Channel
586  dma_enable_channel(dma->dma, dma->tx_chan);
587  }
588 
589  // FIXME do we need to explicitly disable the half transfer interrupt?
590  if ( _trans->input_length > 0 ) {
591  // Enable dma->dma rx Channel Transfer Complete interrupt

601 {

603  idx = p->trans_insert_idx + 1;
604  if (idx >= SPI_TRANSACTION_QUEUE_LEN) idx = 0;
605  if ((idx == p->trans_extract_idx) || ((t->input_length == 0) && (t->output_length == 0))) {
606  t->status = SPITransFailed;
607  return FALSE; /* queue full or input_length and output_length both 0 */

614  //FIXME

616 
617  // GT: no copy? There's a queue implying a copy here...
618  p->trans[p->trans_insert_idx] = t;
619  p->trans_insert_idx = idx;

803 void process_rx_dma_interrupt( struct spi_periph *spi ) {
804  struct spi_periph_dma *dma = spi->init_struct;
805  struct spi_transaction *trans = spi->trans[spi->trans_extract_idx];
806 
807  // disable DMA Channel
808  dma_disable_transfer_complete_interrupt( dma->dma, dma->rx_chan );
809 
810  // Disable SPI Rx request
811  spi_disable_rx_dma( dma->spi );
812 
813  // Disable DMA rx channel
814  dma_disable_channel( dma->dma, dma->rx_chan );
815 
816  if ( dma->other_dma_finished != 0 ) {
817  // this transaction is finished
818  // run the callback
819  trans->status = SPITransSuccess;
820  if (trans->after_cb != 0) {
821  trans->after_cb( trans );
822  }
823 
824  // AFTER the callback, then unselect the slave if required
825  if ( trans->select == SPISelectUnselect || trans->select == SPIUnselect ) {
826  SpiSlaveUnselect( trans->slave_idx );
827  }
828 
829  // increment the transaction to handle
830  spi->trans_extract_idx++;
831 
832  // Check if there is another pending SPI transaction
833  if (spi->trans_extract_idx >= SPI_TRANSACTION_QUEUE_LEN)
834  spi->trans_extract_idx = 0;
835  if (spi->trans_extract_idx == spi->trans_insert_idx || spi->suspend)
836  spi->status = SPIIdle;
837  else
838  spi_rw(spi, spi->trans[spi->trans_extract_idx]);
839  } else {
840  // if this is not the last part of the transaction, set finished flag
841  dma->other_dma_finished = 1;
842  }
843 }

846 void process_tx_dma_interrupt( struct spi_periph *spi ) {
847  struct spi_periph_dma *dma = spi->init_struct;
848  struct spi_transaction *trans = spi->trans[spi->trans_extract_idx];
849

854  spi_disable_tx_dma( dma->spi );
855 
856  // Disable DMA tx channel
857  dma_disable_channel( dma->dma, dma->tx_chan );
858 
859  if ( dma->other_dma_finished != 0 ) {
860  // this transaction is finished
861  // run the callback
862  trans->status = SPITransSuccess;
863  if (trans->after_cb != 0) {
864  trans->after_cb( trans );
865  }
866

868  if ( trans->select == SPISelectUnselect || trans->select == SPIUnselect ) {
