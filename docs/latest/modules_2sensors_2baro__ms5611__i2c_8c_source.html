63 static int8_t baro_ms5611_crc(uint16_t* prom) {
64  int32_t i, j;
65  uint32_t res = 0;
66  uint8_t crc = prom[7] & 0xF;
67  prom[7] &= 0xFF00;
68  for (i = 0; i < 16; i++) {
69  if (i & 1) res ^= ((prom[i>>1]) & 0x00FF);
70  else res ^= (prom[i>>1]>>8);
71  for (j = 8; j > 0; j--) {
72  if (res & 0x8000) res ^= 0x1800;
73  res <<= 1;
74  }
75  }
76  prom[7] |= crc;
77  if (crc == ((res >> 12) & 0xF)) return 0;
78  else return -1;
79 }

90 void baro_ms5611_periodic( void ) {
91  if (sys_time.nb_sec > 1) {
92  if (ms5611_status >= MS5611_IDLE) {
93  /* start D1 conversion */
94  ms5611_status = MS5611_CONV_D1;
95  ms5611_trans.buf[0] = MS5611_START_CONV_D1;
96  I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
97  RunOnceEvery((4*30), DOWNLINK_SEND_MS5611_COEFF(DefaultChannel, DefaultDevice,
98  &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
99  &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]));
100  }
101  else if (ms5611_status == MS5611_UNINIT) {
102  /* reset sensor */
103  ms5611_status = MS5611_RESET;
104  ms5611_trans.buf[0] = MS5611_SOFT_RESET;
105  I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
106  }
107  else if (ms5611_status == MS5611_RESET_OK) {
108  /* start getting prom data */
109  ms5611_status = MS5611_PROM;
110  ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
111  I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
112  }
113  }
114 }

138 void baro_ms5611_event( void ) {
139  if (ms5611_trans.status == I2CTransSuccess) {
140  switch (ms5611_status) {
141 
142  case MS5611_RESET:
143  ms5611_status = MS5611_RESET_OK;
144  ms5611_trans.status = I2CTransDone;
145  break;
146 
147  case MS5611_PROM:
148  /* read prom data */
149  ms5611_c[prom_cnt++] = (ms5611_trans.buf[0] << 8) | ms5611_trans.buf[1];
150  if (prom_cnt < PROM_NB) {
151  /* get next prom data */
152  ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
153  I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
154  }
155  else {
156  /* done reading prom */
157  ms5611_trans.status = I2CTransDone;
158  /* check prom crc */
159  if (baro_ms5611_crc(ms5611_c) == 0) {
160  DOWNLINK_SEND_MS5611_COEFF(DefaultChannel, DefaultDevice,
161  &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
162  &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]);
163  ms5611_status = MS5611_IDLE;
164  }
165  else {
166  /* checksum error, try again */
167  baro_ms5611_init();
168  }
169  }
170  break;
171 
172  case MS5611_CONV_D1:
173  ms5611_status = MS5611_CONV_D1_OK;
174  ms5611_trans.status = I2CTransDone;
175  break;
176 
177  case MS5611_ADC_D1:
178  /* read D1 (pressure) */
179  ms5611_d1 = (ms5611_trans.buf[0] << 16) |
180  (ms5611_trans.buf[1] << 8) |
181  ms5611_trans.buf[2];
182  /* start D2 conversion */
183  ms5611_status = MS5611_CONV_D2;
184  ms5611_trans.buf[0] = MS5611_START_CONV_D2;
185  I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
186  break;
187 
188  case MS5611_CONV_D2:
189  ms5611_status = MS5611_CONV_D2_OK;
190  ms5611_trans.status = I2CTransDone;
191  break;
192 
193  case MS5611_ADC_D2: {
194  int64_t dt, baroms, tempms, off, sens, t2, off2, sens2;
195  /* read D2 (temperature) */
196  ms5611_d2 = (ms5611_trans.buf[0] << 16) |
197  (ms5611_trans.buf[1] << 8) |
198  ms5611_trans.buf[2];
199  ms5611_status = MS5611_IDLE;
200  ms5611_trans.status = I2CTransDone;
201 
202  /* difference between actual and ref temperature */
203  dt = ms5611_d2 - (int64_t)ms5611_c[5] * (1<<8);
204  /* actual temperature */
205  tempms = 2000 + ((int64_t)dt * ms5611_c[6]) / (1<<23);
206  /* offset at actual temperature */
207  off = ((int64_t)ms5611_c[2] * (1<<16)) + ((int64_t)ms5611_c[4] * dt) / (1<<7);
208  /* sensitivity at actual temperature */
209  sens = ((int64_t)ms5611_c[1] * (1<<15)) + ((int64_t)ms5611_c[3] * dt) / (1<<8);
210  /* second order temperature compensation */
211  if (tempms < 2000) {
212  t2 = (dt*dt) / (1<<31);
213  off2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<1);
214  sens2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<2);
215  if (tempms < -1500) {
216  off2 = off2 + 7 * (int64_t)(tempms+1500)*(tempms+1500);
217  sens2 = sens2 + 11 * ((int64_t)(tempms+1500)*(tempms+1500)) / (1<<1);
218  }
219  tempms = tempms - t2;
220  off = off - off2;
221  sens = sens - sens2;
222  }
223  /* temperature compensated pressure */
224  baroms = (((int64_t)ms5611_d1 * sens) / (1<<21) - off) / (1<<15);
225 
226  float tmp_float = baroms/101325.0; //pressure at sea level
227  tmp_float = pow(tmp_float,0.190295); //eleva pressao ao expoente
228  baro_ms5611_alt = 44330*(1.0-tmp_float); //altitude above MSL
229 
230  baro_ms5611_valid = TRUE;
231 
232 #ifdef SENSOR_SYNC_SEND
233  ftempms = tempms / 100.;
234  fbaroms = baroms / 100.;
235  DOWNLINK_SEND_BARO_MS5611(DefaultChannel, DefaultDevice,
236  &ms5611_d1, &ms5611_d2, &fbaroms, &ftempms);
237 #endif
238 
239  break;
240  }
241 
242  default:
243  ms5611_trans.status = I2CTransDone;
244  break;
245  }
246  }
