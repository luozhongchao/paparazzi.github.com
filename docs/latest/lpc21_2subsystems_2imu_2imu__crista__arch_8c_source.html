00058 void imu_crista_arch_init(void) {
00059   channel = 0;
00060 
00061   /* setup pins for SSP (SCK, MISO, MOSI) */
00062   PINSEL1 |= 2 << 2 | 2 << 4 | 2 << 6;
00063 
00064   /* setup SSP */
00065   SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
00066   SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
00067   SSPCPSR = 2; /* -> 50kHz */
00068 
00069   /* initialize interrupt vector */
00070   VICIntSelect &= ~VIC_BIT(VIC_SPI1);   // SPI1 selected as IRQ
00071   VICIntEnable = VIC_BIT(VIC_SPI1);     // SPI1 interrupt enabled
00072   VICVectCntl7 = VIC_ENABLE | VIC_SPI1;
00073   VICVectAddr7 = (uint32_t)SPI1_ISR;    // address of the ISR
00074

00076   /* configure SS pin */
00077   SetBit( ADS8344_SS_IODIR,  ADS8344_SS_PIN);  /* pin is output  */
