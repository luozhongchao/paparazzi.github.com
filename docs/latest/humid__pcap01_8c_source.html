54 void writePCAP01_SRAM(uint8_t data, uint16_t s_add)
55 {
56  while (pcap01_trans.status == I2CTransPending);
57 
58  pcap01_trans.buf[0] = 0x90+(unsigned char)(s_add>>8);
59  pcap01_trans.buf[1] = (unsigned char)(s_add);
60  pcap01_trans.buf[2] = data;
61  I2CTransmit(PCAP01_I2C_DEV, pcap01_trans, PCAP01_ADDR, 3);

221 void pcap01_event(void)
222 {
223  float humidity;
224  float temperature;
225 
226  if (pcap01_trans.status == I2CTransSuccess) {
227  switch (pcap01Value.status) {
228 
229  case PCAP01_GET_HUMID:
230  pcap01Value.C_ratio = pcap01_trans.buf[0] << 16;
231  pcap01Value.C_ratio |= (pcap01_trans.buf[1] << 8);
232  pcap01Value.C_ratio |= pcap01_trans.buf[2];
233  pcap01Value.status = PCAP01_GET_TEMP;
234 #ifdef PCAP01_STANDARD
235  pcap01readRegister(PCAP01_REG13);
236 #endif
237 #ifdef PCAP01_200HZ
238  pcap01readRegister(PCAP01_REG3);
239 #endif
240  break;
241 
242  case PCAP01_GET_TEMP:
243  pcap01Value.R_ratio = pcap01_trans.buf[0] << 16;
244  pcap01Value.R_ratio |= (pcap01_trans.buf[1] << 8);
245  pcap01Value.R_ratio |= pcap01_trans.buf[2];
246  humidity = pcap01Value.C_ratio * (-0.0023959245437) + 516.4124438673063;
247  temperature = pcap01Value.R_ratio * 61.927 - 259.74;
248  DOWNLINK_SEND_PCAP01_STATUS(DefaultChannel, DefaultDevice,
249  &pcap01Value.C_ratio,
250  &pcap01Value.R_ratio,
251  &humidity,
252  &temperature);
253  pcap01_trans.status = I2CTransDone;
254  pcap01Value.status = PCAP01_IDLE;
255  break;
256 
257  default:
258  pcap01_trans.status = I2CTransDone;
259  break;
260  }
261  }
