107 void nav_circle_XY(float x, float y, float radius) {
108  struct EnuCoor_f* pos = stateGetPositionEnu_f();
109  float last_trigo_qdr = nav_circle_trigo_qdr;
110  nav_circle_trigo_qdr = atan2(pos->y - y, pos->x - x);
111 
112  if (nav_in_circle) {
113  float trigo_diff = nav_circle_trigo_qdr - last_trigo_qdr;
114  NormRadAngle(trigo_diff);
115  nav_circle_radians += trigo_diff;
116  }
117 
118  float dist2_center = DistanceSquare(pos->x, pos->y, x, y);
119  float dist_carrot = CARROT*NOMINAL_AIRSPEED;
120  float sign_radius = radius > 0 ? 1 : -1;
121 
122  radius += -nav_shift;
123 
124  float abs_radius = fabs(radius);
125 
127  circle_bank =
128  (dist2_center > Square(abs_radius + dist_carrot)
129  || dist2_center < Square(abs_radius - dist_carrot)) ?
130  0 :
131  atan((*stateGetHorizontalSpeedNorm_f())*(*stateGetHorizontalSpeedNorm_f()) / (G*radius));
132 
133  float carrot_angle = dist_carrot / abs_radius;
134  carrot_angle = Min(carrot_angle, M_PI/4);
135  carrot_angle = Max(carrot_angle, M_PI/16);
136  float alpha_carrot = nav_circle_trigo_qdr - sign_radius * carrot_angle;
137  horizontal_mode = HORIZONTAL_MODE_CIRCLE;
138  float radius_carrot = abs_radius;
139  if (nav_mode == NAV_MODE_COURSE) {
140  radius_carrot += (abs_radius / cos(carrot_angle) - abs_radius);
141  }
142  fly_to_xy(x+cos(alpha_carrot)*radius_carrot,
143  y+sin(alpha_carrot)*radius_carrot);
144  nav_in_circle = TRUE;
145  nav_circle_x = x;
146  nav_circle_y = y;
147  nav_circle_radius = radius;
148 }

217 static inline bool_t nav_compute_baseleg(uint8_t wp_af, uint8_t wp_td, uint8_t wp_baseleg, float radius ) {
218  nav_radius = radius;
219 
220  float x_0 = waypoints[wp_td].x - waypoints[wp_af].x;
221  float y_0 = waypoints[wp_td].y - waypoints[wp_af].y;
222 
223  /* Unit vector from AF to TD */
224  float d = sqrt(x_0*x_0+y_0*y_0);
225  float x_1 = x_0 / d;
226  float y_1 = y_0 / d;
227 
228  waypoints[wp_baseleg].x = waypoints[wp_af].x + y_1 * nav_radius;
229  waypoints[wp_baseleg].y = waypoints[wp_af].y - x_1 * nav_radius;
230  waypoints[wp_baseleg].a = waypoints[wp_af].a;
231  baseleg_out_qdr = M_PI - atan2(-y_1, -x_1);
232  if (nav_radius < 0)
233  baseleg_out_qdr += M_PI;
234 
235  return FALSE;
236 }

238 static inline bool_t nav_compute_final_from_glide(uint8_t wp_af, uint8_t wp_td, float glide ) {
239 
240  float x_0 = waypoints[wp_td].x - waypoints[wp_af].x;
241  float y_0 = waypoints[wp_td].y - waypoints[wp_af].y;
242  float h_0 = waypoints[wp_td].a - waypoints[wp_af].a;
243 
244  /* Unit vector from AF to TD */
245  float d = sqrt(x_0*x_0+y_0*y_0);
246  float x_1 = x_0 / d;
247  float y_1 = y_0 / d;
248 
249  waypoints[wp_af].x = waypoints[wp_td].x + x_1 * h_0 * glide;
250  waypoints[wp_af].y = waypoints[wp_td].y + y_1 * h_0 * glide;
251  waypoints[wp_af].a = waypoints[wp_af].a;
252 
253  return FALSE;
254 }

257 /* For a landing UPWIND.
258  Computes Top Of Descent waypoint from Touch Down and Approach Fix
259  waypoints, using glide airspeed, glide vertical speed and wind */
260 static inline bool_t compute_TOD(uint8_t _af, uint8_t _td, uint8_t _tod, float glide_airspeed, float glide_vspeed) {
261  struct FloatVect2* wind = stateGetHorizontalWindspeed_f();
262  float td_af_x = WaypointX(_af) - WaypointX(_td);
263  float td_af_y = WaypointY(_af) - WaypointY(_td);
264  float td_af = sqrt( td_af_x*td_af_x + td_af_y*td_af_y);
265  float td_tod = (WaypointAlt(_af) - WaypointAlt(_td)) / glide_vspeed * (glide_airspeed - sqrt(wind->x*wind->x + wind->y*wind->y));
266  WaypointX(_tod) = WaypointX(_td) + td_af_x / td_af * td_tod;
267  WaypointY(_tod) = WaypointY(_td) + td_af_y / td_af * td_tod;
268  WaypointAlt(_tod) = WaypointAlt(_af);
269  return FALSE;
270 }

314 bool_t nav_approaching_xy(float x, float y, float from_x, float from_y, float approaching_time) {
316  float pw_x = x - stateGetPositionEnu_f()->x;
318  float pw_y = y - stateGetPositionEnu_f()->y;
319 
320  dist2_to_wp = pw_x*pw_x + pw_y *pw_y;
321  float min_dist = approaching_time * (*stateGetHorizontalSpeedNorm_f());
322  if (dist2_to_wp < min_dist*min_dist)
323  return TRUE;
324 
325  float scal_prod = (x - from_x) * pw_x + (y - from_y) * pw_y;
326 
327  return (scal_prod < 0.);
328 }

335 void fly_to_xy(float x, float y) {
336  struct EnuCoor_f* pos = stateGetPositionEnu_f();
337  desired_x = x;
338  desired_y = y;
339  if (nav_mode == NAV_MODE_COURSE) {
340  h_ctl_course_setpoint = atan2(x - pos->x, y - pos->y);
341  if (h_ctl_course_setpoint < 0.)
342  h_ctl_course_setpoint += 2 * M_PI;
343  lateral_mode = LATERAL_MODE_COURSE;
344  } else {
345  float diff = atan2(x - pos->x, y - pos->y) - (*stateGetHorizontalSpeedDir_f());
346  NormRadAngle(diff);
347  BoundAbs(diff,M_PI/2.);
348  float s = sin(diff);
349  float speed = *stateGetHorizontalSpeedNorm_f();
350  h_ctl_roll_setpoint = atan(2 * speed * speed * s * h_ctl_course_pgain / (CARROT * NOMINAL_AIRSPEED * 9.81) );
351  BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
352  lateral_mode = LATERAL_MODE_ROLL;
353  }
354 }

359 void nav_route_xy(float last_wp_x, float last_wp_y, float wp_x, float wp_y) {
360  float leg_x = wp_x - last_wp_x;
361  float leg_y = wp_y - last_wp_y;
362  float leg2 = Max(leg_x * leg_x + leg_y * leg_y, 1.);
363  nav_leg_progress = ((stateGetPositionEnu_f()->x - last_wp_x) * leg_x + (stateGetPositionEnu_f()->y - last_wp_y) * leg_y) / leg2;
364  nav_leg_length = sqrt(leg2);
365 
367  float carrot = CARROT * NOMINAL_AIRSPEED;
368 
369  nav_leg_progress += Max(carrot / nav_leg_length, 0.);
370  nav_in_segment = TRUE;
371  nav_segment_x_1 = last_wp_x;
372  nav_segment_y_1 = last_wp_y;
373  nav_segment_x_2 = wp_x;
374  nav_segment_y_2 = wp_y;
375  horizontal_mode = HORIZONTAL_MODE_ROUTE;
376 
377  fly_to_xy(last_wp_x + nav_leg_progress*leg_x +nav_shift*leg_y/nav_leg_length, last_wp_y + nav_leg_progress*leg_y-nav_shift*leg_x/nav_leg_length);
378 }

488 void nav_eight(uint8_t target, uint8_t c1, float radius) {
489  float aradius = fabs(radius);
490  float alt = waypoints[target].a;
491  waypoints[c1].a = alt;
492 
493  float target_c1_x = waypoints[c1].x - waypoints[target].x;
494  float target_c1_y = waypoints[c1].y - waypoints[target].y;
495  float d = sqrt(target_c1_x*target_c1_x+target_c1_y*target_c1_y);
496  d = Max(d, 1.); /* To prevent a division by zero */
497 
498  /* Unit vector from target to c1 */
499  float u_x = target_c1_x / d;
500  float u_y = target_c1_y / d;
501 
502  /* Move [c1] closer if needed */
503  if (d > 2 * aradius) {
504  d = 2*aradius;
505  waypoints[c1].x = waypoints[target].x + d*u_x;
506  waypoints[c1].y = waypoints[target].y + d*u_y;
507  }
508 
509  /* The other center */
510  struct point c2 = {
511  waypoints[target].x - d*u_x,
512  waypoints[target].y - d*u_y,
513  alt };
514 
515  struct point c1_in = {
516  waypoints[c1].x + radius * -u_y,
517  waypoints[c1].y + radius * u_x,
518  alt };
519  struct point c1_out = {
520  waypoints[c1].x - radius * -u_y,
521  waypoints[c1].y - radius * u_x,
522  alt };
523 
524  struct point c2_in = {
525  c2.x + radius * -u_y,
526  c2.y + radius * u_x,
527  alt };
528  struct point c2_out = {
529  c2.x - radius * -u_y,
530  c2.y - radius * u_x,
531  alt };
532 
533  float qdr_out = M_PI - atan2(u_y, u_x);
534  if (radius < 0)
535  qdr_out += M_PI;
536 
537  switch (eight_status) {
538  case C1 :
539  NavCircleWaypoint(c1, radius);
540  if (NavQdrCloseTo(DegOfRad(qdr_out)-10)) {
541  eight_status = R1T;
542  InitStage();
543  }
544  return;
545 
546  case R1T:
547  nav_route_xy(c1_out.x, c1_out.y, c2_in.x, c2_in.y);
548  if (nav_approaching_xy(waypoints[target].x, waypoints[target].y, c1_out.x, c1_out.y, 0)) {
549  eight_status = RT2;
550  InitStage();
551  }
552  return;
553 
554  case RT2:
555  nav_route_xy(c1_out.x, c1_out.y, c2_in.x, c2_in.y);
556  if (nav_approaching_xy(c2_in.x, c2_in.y, c1_out.x, c1_out.y, CARROT)) {
557  eight_status = C2;
558  InitStage();
559  }
560  return;
561 
562  case C2 :
563  nav_circle_XY(c2.x, c2.y, -radius);
564  if (NavQdrCloseTo(DegOfRad(qdr_out)+10)) {
565  eight_status = R2T;
566  InitStage();
567  }
568  return;
569 
570  case R2T:
571  nav_route_xy(c2_out.x, c2_out.y, c1_in.x, c1_in.y);
572  if (nav_approaching_xy(waypoints[target].x, waypoints[target].y, c2_out.x, c2_out.y, 0)) {
573  eight_status = RT1;
574  InitStage();
575  }
576  return;
577 
578  case RT1:
579  nav_route_xy(c2_out.x, c2_out.y, c1_in.x, c1_in.y);
580  if (nav_approaching_xy(c1_in.x, c1_in.y, c2_out.x, c2_out.y, CARROT)) {
581  eight_status = C1;
582  InitStage();
583  }
584  return;
585 
586  default:/* Should not occur !!! Doing nothing */
587  return;
588  } /* switch */
589 }

620  float u_x = p2_p1_x / d;
621  float u_y = p2_p1_y / d;
622

640  if (radius < 0) {
