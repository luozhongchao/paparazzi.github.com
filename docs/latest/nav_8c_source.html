00108 void nav_circle_XY(float x, float y, float radius) {
00109   float last_trigo_qdr = nav_circle_trigo_qdr;
00110   nav_circle_trigo_qdr = atan2(estimator_y - y, estimator_x - x);
00111 
00112   if (nav_in_circle) {
00113     float trigo_diff = nav_circle_trigo_qdr - last_trigo_qdr;
00114     NormRadAngle(trigo_diff);
00115     nav_circle_radians += trigo_diff;
00116   }
00117 
00118   float dist2_center = DistanceSquare(estimator_x, estimator_y, x, y);
00119   float dist_carrot = CARROT*NOMINAL_AIRSPEED;
00120   float sign_radius = radius > 0 ? 1 : -1;
00121 
00122   radius += -nav_shift;
00123 
00124   float abs_radius = fabs(radius);
00125 
00127   circle_bank =
00128     (dist2_center > Square(abs_radius + dist_carrot)
00129       || dist2_center < Square(abs_radius - dist_carrot)) ?
00130     0 :
00131     atan(estimator_hspeed_mod*estimator_hspeed_mod / (G*radius));
00132 
00133   float carrot_angle = dist_carrot / abs_radius;
00134   carrot_angle = Min(carrot_angle, M_PI/4);
00135   carrot_angle = Max(carrot_angle, M_PI/16);
00136   float alpha_carrot = nav_circle_trigo_qdr - sign_radius * carrot_angle;
00137   horizontal_mode = HORIZONTAL_MODE_CIRCLE;
00138   float radius_carrot = abs_radius;
00139   if (nav_mode == NAV_MODE_COURSE)
00140     radius_carrot += (abs_radius / cos(carrot_angle) - abs_radius);
00141   fly_to_xy(x+cos(alpha_carrot)*radius_carrot,
00142       y+sin(alpha_carrot)*radius_carrot);
00143   nav_in_circle = TRUE;
00144   nav_circle_x = x;
00145   nav_circle_y = y;
00146   nav_circle_radius = radius;
00147 }

00256 /* For a landing UPWIND.
00257    Computes Top Of Descent waypoint from Touch Down and Approach Fix
00258    waypoints, using glide airspeed, glide vertical speed and wind */
00259 static inline bool_t compute_TOD(uint8_t _af, uint8_t _td, uint8_t _tod, float glide_airspeed, float glide_vspeed) {
00260   float td_af_x = WaypointX(_af) - WaypointX(_td);
00261   float td_af_y = WaypointY(_af) - WaypointY(_td);
00262   float td_af = sqrt( td_af_x*td_af_x + td_af_y*td_af_y);
00263   float td_tod = (WaypointAlt(_af) - WaypointAlt(_td)) / glide_vspeed * (glide_airspeed - sqrt(wind_east*wind_east + wind_north*wind_north));
00264   WaypointX(_tod) = WaypointX(_td) + td_af_x / td_af * td_tod;
00265   WaypointY(_tod) = WaypointY(_td) + td_af_y / td_af * td_tod;
00266   WaypointAlt(_tod) = WaypointAlt(_af);
00267   return FALSE;
00268 }

00312 bool_t nav_approaching_xy(float x, float y, float from_x, float from_y, float approaching_time) {
00314   float pw_x = x - estimator_x;
00316   float pw_y = y - estimator_y;
00317 
00318   dist2_to_wp = pw_x*pw_x + pw_y *pw_y;
00319   float min_dist = approaching_time * estimator_hspeed_mod;
00320   if (dist2_to_wp < min_dist*min_dist)
00321     return TRUE;
00322

00324 
00325   return (scal_prod < 0.);

00484 void nav_eight(uint8_t target, uint8_t c1, float radius) {

00486   float alt = waypoints[target].a;
00487   waypoints[c1].a = alt;
00488 
00489   float target_c1_x = waypoints[c1].x - waypoints[target].x;
00490   float target_c1_y = waypoints[c1].y - waypoints[target].y;
00491   float d = sqrt(target_c1_x*target_c1_x+target_c1_y*target_c1_y);
00492   d = Max(d, 1.); /* To prevent a division by zero */
00493 
00494   /* Unit vector from target to c1 */
00495   float u_x = target_c1_x / d;
00496   float u_y = target_c1_y / d;
00497 
00498   /* Move [c1] closer if needed */
00499   if (d > 2 * aradius) {
00500     d = 2*aradius;
00501     waypoints[c1].x = waypoints[target].x + d*u_x;
00502     waypoints[c1].y = waypoints[target].y + d*u_y;
00503   }
00504 
00505   /* The other center */
00506   struct point c2 = {
00507     waypoints[target].x - d*u_x,
00508     waypoints[target].y - d*u_y,
00509     alt };

00512     waypoints[c1].x + radius * -u_y,
00513     waypoints[c1].y + radius * u_x,
00514     alt };

00517     waypoints[c1].y - radius * u_x,
00518     alt };

00521     c2.x + radius * -u_y,
00522     c2.y + radius * u_x,

00526     c2.y - radius * u_x,
