106 void nav_circle_XY(float x, float y, float radius) {
107  struct EnuCoor_f* pos = stateGetPositionEnu_f();
108  float last_trigo_qdr = nav_circle_trigo_qdr;
109  nav_circle_trigo_qdr = atan2(pos->y - y, pos->x - x);
110 
111  if (nav_in_circle) {
112  float trigo_diff = nav_circle_trigo_qdr - last_trigo_qdr;
113  NormRadAngle(trigo_diff);
114  nav_circle_radians += trigo_diff;
115  }
116 
117  float dist2_center = DistanceSquare(pos->x, pos->y, x, y);
118  float dist_carrot = CARROT*NOMINAL_AIRSPEED;
119  float sign_radius = radius > 0 ? 1 : -1;
120 
121  radius += -nav_shift;
122 
123  float abs_radius = fabs(radius);
124 
126  circle_bank =
127  (dist2_center > Square(abs_radius + dist_carrot)
128  || dist2_center < Square(abs_radius - dist_carrot)) ?
129  0 :
130  atan((*stateGetHorizontalSpeedNorm_f())*(*stateGetHorizontalSpeedNorm_f()) / (G*radius));
131 
132  float carrot_angle = dist_carrot / abs_radius;
133  carrot_angle = Min(carrot_angle, M_PI/4);
134  carrot_angle = Max(carrot_angle, M_PI/16);
135  float alpha_carrot = nav_circle_trigo_qdr - sign_radius * carrot_angle;
136  horizontal_mode = HORIZONTAL_MODE_CIRCLE;
137  float radius_carrot = abs_radius;
138  if (nav_mode == NAV_MODE_COURSE)
139  radius_carrot += (abs_radius / cos(carrot_angle) - abs_radius);
140  fly_to_xy(x+cos(alpha_carrot)*radius_carrot,
141  y+sin(alpha_carrot)*radius_carrot);
142  nav_in_circle = TRUE;
143  nav_circle_x = x;
144  nav_circle_y = y;
145  nav_circle_radius = radius;
146 }

215 static inline bool_t nav_compute_baseleg(uint8_t wp_af, uint8_t wp_td, uint8_t wp_baseleg, float radius ) {
216  nav_radius = radius;
217 
218  float x_0 = waypoints[wp_td].x - waypoints[wp_af].x;
219  float y_0 = waypoints[wp_td].y - waypoints[wp_af].y;
220 
221  /* Unit vector from AF to TD */
222  float d = sqrt(x_0*x_0+y_0*y_0);
223  float x_1 = x_0 / d;
224  float y_1 = y_0 / d;
225 
226  waypoints[wp_baseleg].x = waypoints[wp_af].x + y_1 * nav_radius;
227  waypoints[wp_baseleg].y = waypoints[wp_af].y - x_1 * nav_radius;
228  waypoints[wp_baseleg].a = waypoints[wp_af].a;
229  baseleg_out_qdr = M_PI - atan2(-y_1, -x_1);
230  if (nav_radius < 0)
231  baseleg_out_qdr += M_PI;
232 
233  return FALSE;
234 }

236 static inline bool_t nav_compute_final_from_glide(uint8_t wp_af, uint8_t wp_td, float glide ) {
237 
238  float x_0 = waypoints[wp_td].x - waypoints[wp_af].x;
239  float y_0 = waypoints[wp_td].y - waypoints[wp_af].y;
240  float h_0 = waypoints[wp_td].a - waypoints[wp_af].a;
241 
242  /* Unit vector from AF to TD */
243  float d = sqrt(x_0*x_0+y_0*y_0);
244  float x_1 = x_0 / d;
245  float y_1 = y_0 / d;

247  waypoints[wp_af].x = waypoints[wp_td].x + x_1 * h_0 * glide;
248  waypoints[wp_af].y = waypoints[wp_td].y + y_1 * h_0 * glide;
249  waypoints[wp_af].a = waypoints[wp_af].a;
250 
251  return FALSE;
252 }

255 /* For a landing UPWIND.
256  Computes Top Of Descent waypoint from Touch Down and Approach Fix
257  waypoints, using glide airspeed, glide vertical speed and wind */
258 static inline bool_t compute_TOD(uint8_t _af, uint8_t _td, uint8_t _tod, float glide_airspeed, float glide_vspeed) {
259  struct FloatVect2* wind = stateGetHorizontalWindspeed_f();
260  float td_af_x = WaypointX(_af) - WaypointX(_td);
261  float td_af_y = WaypointY(_af) - WaypointY(_td);
262  float td_af = sqrt( td_af_x*td_af_x + td_af_y*td_af_y);
263  float td_tod = (WaypointAlt(_af) - WaypointAlt(_td)) / glide_vspeed * (glide_airspeed - sqrt(wind->x*wind->x + wind->y*wind->y));
264  WaypointX(_tod) = WaypointX(_td) + td_af_x / td_af * td_tod;
265  WaypointY(_tod) = WaypointY(_td) + td_af_y / td_af * td_tod;
266  WaypointAlt(_tod) = WaypointAlt(_af);
267  return FALSE;
268 }

312 bool_t nav_approaching_xy(float x, float y, float from_x, float from_y, float approaching_time) {
314  float pw_x = x - stateGetPositionEnu_f()->x;
316  float pw_y = y - stateGetPositionEnu_f()->y;
317 
318  dist2_to_wp = pw_x*pw_x + pw_y *pw_y;
319  float min_dist = approaching_time * (*stateGetHorizontalSpeedNorm_f());
320  if (dist2_to_wp < min_dist*min_dist)
321  return TRUE;
322 
323  float scal_prod = (x - from_x) * pw_x + (y - from_y) * pw_y;
324 
325  return (scal_prod < 0.);
326 }

333 void fly_to_xy(float x, float y) {
334  struct EnuCoor_f* pos = stateGetPositionEnu_f();
335  desired_x = x;
336  desired_y = y;
337  if (nav_mode == NAV_MODE_COURSE) {
338  h_ctl_course_setpoint = atan2(x - pos->x, y - pos->y);
339  if (h_ctl_course_setpoint < 0.)
340  h_ctl_course_setpoint += 2 * M_PI;
341  lateral_mode = LATERAL_MODE_COURSE;
342  } else {
343  float diff = atan2(x - pos->x, y - pos->y) - (*stateGetHorizontalSpeedDir_f());
344  NormRadAngle(diff);
345  BoundAbs(diff,M_PI/2.);
346  float s = sin(diff);
347  float speed = *stateGetHorizontalSpeedNorm_f();
348  h_ctl_roll_setpoint = atan(2 * speed * speed * s * h_ctl_course_pgain / (CARROT * NOMINAL_AIRSPEED * 9.81) );
349  BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
350  lateral_mode = LATERAL_MODE_ROLL;
351  }

357 void nav_route_xy(float last_wp_x, float last_wp_y, float wp_x, float wp_y) {
358  float leg_x = wp_x - last_wp_x;
359  float leg_y = wp_y - last_wp_y;
360  float leg2 = Max(leg_x * leg_x + leg_y * leg_y, 1.);
361  nav_leg_progress = ((stateGetPositionEnu_f()->x - last_wp_x) * leg_x + (stateGetPositionEnu_f()->y - last_wp_y) * leg_y) / leg2;
362  nav_leg_length = sqrt(leg2);
363 
365  float carrot = CARROT * NOMINAL_AIRSPEED;
366 
367  nav_leg_progress += Max(carrot / nav_leg_length, 0.);
368  nav_in_segment = TRUE;
369  nav_segment_x_1 = last_wp_x;
370  nav_segment_y_1 = last_wp_y;
371  nav_segment_x_2 = wp_x;
372  nav_segment_y_2 = wp_y;
373  horizontal_mode = HORIZONTAL_MODE_ROUTE;
374 
375  fly_to_xy(last_wp_x + nav_leg_progress*leg_x +nav_shift*leg_y/nav_leg_length, last_wp_y + nav_leg_progress*leg_y-nav_shift*leg_x/nav_leg_length);
376 }

486 void nav_eight(uint8_t target, uint8_t c1, float radius) {
487  float aradius = fabs(radius);
488  float alt = waypoints[target].a;
489  waypoints[c1].a = alt;
490 
491  float target_c1_x = waypoints[c1].x - waypoints[target].x;
492  float target_c1_y = waypoints[c1].y - waypoints[target].y;
493  float d = sqrt(target_c1_x*target_c1_x+target_c1_y*target_c1_y);
494  d = Max(d, 1.); /* To prevent a division by zero */
495 
496  /* Unit vector from target to c1 */
497  float u_x = target_c1_x / d;
498  float u_y = target_c1_y / d;
499 
500  /* Move [c1] closer if needed */
501  if (d > 2 * aradius) {
502  d = 2*aradius;
503  waypoints[c1].x = waypoints[target].x + d*u_x;
504  waypoints[c1].y = waypoints[target].y + d*u_y;
505  }
506 
507  /* The other center */
508  struct point c2 = {
509  waypoints[target].x - d*u_x,
510  waypoints[target].y - d*u_y,
511  alt };
512 
513  struct point c1_in = {
514  waypoints[c1].x + radius * -u_y,
515  waypoints[c1].y + radius * u_x,
516  alt };
517  struct point c1_out = {
518  waypoints[c1].x - radius * -u_y,
519  waypoints[c1].y - radius * u_x,
520  alt };
521 
522  struct point c2_in = {
523  c2.x + radius * -u_y,
524  c2.y + radius * u_x,
525  alt };
526  struct point c2_out = {
527  c2.x - radius * -u_y,
528  c2.y - radius * u_x,
529  alt };
530 
531  float qdr_out = M_PI - atan2(u_y, u_x);
532  if (radius < 0)
533  qdr_out += M_PI;
534 
535  switch (eight_status) {
536  case C1 :
537  NavCircleWaypoint(c1, radius);
538  if (NavQdrCloseTo(DegOfRad(qdr_out)-10)) {
539  eight_status = R1T;
540  InitStage();
541  }
542  return;
543 
544  case R1T:
545  nav_route_xy(c1_out.x, c1_out.y, c2_in.x, c2_in.y);
546  if (nav_approaching_xy(waypoints[target].x, waypoints[target].y, c1_out.x, c1_out.y, 0)) {
547  eight_status = RT2;
548  InitStage();
549  }
550  return;
551 
552  case RT2:
553  nav_route_xy(c1_out.x, c1_out.y, c2_in.x, c2_in.y);
554  if (nav_approaching_xy(c2_in.x, c2_in.y, c1_out.x, c1_out.y, CARROT)) {
555  eight_status = C2;
556  InitStage();
557  }
558  return;
559 
560  case C2 :
561  nav_circle_XY(c2.x, c2.y, -radius);
562  if (NavQdrCloseTo(DegOfRad(qdr_out)+10)) {
563  eight_status = R2T;
564  InitStage();
565  }
566  return;
567 
568  case R2T:
569  nav_route_xy(c2_out.x, c2_out.y, c1_in.x, c1_in.y);
570  if (nav_approaching_xy(waypoints[target].x, waypoints[target].y, c2_out.x, c2_out.y, 0)) {
571  eight_status = RT1;
572  InitStage();
573  }
574  return;
575 
576  case RT1:
577  nav_route_xy(c2_out.x, c2_out.y, c1_in.x, c1_in.y);
578  if (nav_approaching_xy(c1_in.x, c1_in.y, c2_out.x, c2_out.y, CARROT)) {
579  eight_status = C1;
580  InitStage();
581  }
582  return;
583 
584  default:/* Should not occur !!! Doing nothing */
585  return;
586  } /* switch */
587 }
