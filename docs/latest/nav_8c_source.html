00106 void nav_circle_XY(float x, float y, float radius) {
00107   struct EnuCoor_f* pos = stateGetPositionEnu_f();
00108   float last_trigo_qdr = nav_circle_trigo_qdr;
00109   nav_circle_trigo_qdr = atan2(pos->y - y, pos->x - x);
00110 
00111   if (nav_in_circle) {
00112     float trigo_diff = nav_circle_trigo_qdr - last_trigo_qdr;
00113     NormRadAngle(trigo_diff);
00114     nav_circle_radians += trigo_diff;
00115   }
00116 
00117   float dist2_center = DistanceSquare(pos->x, pos->y, x, y);
00118   float dist_carrot = CARROT*NOMINAL_AIRSPEED;
00119   float sign_radius = radius > 0 ? 1 : -1;
00120 
00121   radius += -nav_shift;
00122 
00123   float abs_radius = fabs(radius);
00124 
00126   circle_bank =
00127     (dist2_center > Square(abs_radius + dist_carrot)
00128       || dist2_center < Square(abs_radius - dist_carrot)) ?
00129     0 :
00130     atan((*stateGetHorizontalSpeedNorm_f())*(*stateGetHorizontalSpeedNorm_f()) / (G*radius));
00131 
00132   float carrot_angle = dist_carrot / abs_radius;
00133   carrot_angle = Min(carrot_angle, M_PI/4);
00134   carrot_angle = Max(carrot_angle, M_PI/16);
00135   float alpha_carrot = nav_circle_trigo_qdr - sign_radius * carrot_angle;
00136   horizontal_mode = HORIZONTAL_MODE_CIRCLE;
00137   float radius_carrot = abs_radius;
00138   if (nav_mode == NAV_MODE_COURSE)
00139     radius_carrot += (abs_radius / cos(carrot_angle) - abs_radius);
00140   fly_to_xy(x+cos(alpha_carrot)*radius_carrot,
00141       y+sin(alpha_carrot)*radius_carrot);
00142   nav_in_circle = TRUE;
00143   nav_circle_x = x;
00144   nav_circle_y = y;
00145   nav_circle_radius = radius;

00255 /* For a landing UPWIND.
00256    Computes Top Of Descent waypoint from Touch Down and Approach Fix
00257    waypoints, using glide airspeed, glide vertical speed and wind */
00258 static inline bool_t compute_TOD(uint8_t _af, uint8_t _td, uint8_t _tod, float glide_airspeed, float glide_vspeed) {
00259   struct FloatVect2* wind = stateGetHorizontalWindspeed_f();
00260   float td_af_x = WaypointX(_af) - WaypointX(_td);
00261   float td_af_y = WaypointY(_af) - WaypointY(_td);
00262   float td_af = sqrt( td_af_x*td_af_x + td_af_y*td_af_y);
00263   float td_tod = (WaypointAlt(_af) - WaypointAlt(_td)) / glide_vspeed * (glide_airspeed - sqrt(wind->x*wind->x + wind->y*wind->y));
00264   WaypointX(_tod) = WaypointX(_td) + td_af_x / td_af * td_tod;
00265   WaypointY(_tod) = WaypointY(_td) + td_af_y / td_af * td_tod;
00266   WaypointAlt(_tod) = WaypointAlt(_af);
00267   return FALSE;
00268 }

00312 bool_t nav_approaching_xy(float x, float y, float from_x, float from_y, float approaching_time) {
