00204 static inline void autopilot_check_in_flight( bool_t motors_on ) {
00205   if (autopilot_in_flight) {
00206     if (autopilot_in_flight_counter > 0) {
00207       if (THROTTLE_STICK_DOWN()) {
00208         autopilot_in_flight_counter--;
00209         if (autopilot_in_flight_counter == 0) {
00210           autopilot_in_flight = FALSE;
00211         }
00212       }
00213       else {  /* !THROTTLE_STICK_DOWN */
00214         autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
00215       }
00216     }
00217   }
00218   else { /* not in flight */
00219     if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
00220         motors_on) {
00221       if (!THROTTLE_STICK_DOWN()) {
00222         autopilot_in_flight_counter++;
00223         if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME)
00224           autopilot_in_flight = TRUE;
00225       }
00226       else { /*  THROTTLE_STICK_DOWN */
00227         autopilot_in_flight_counter = 0;
00228       }
00229     }
00230   }
00231 }

00245 
00246   if (kill_switch_is_on())

00248   else {
00249     uint8_t new_autopilot_mode = 0;

00251     /* don't enter NAV mode if GPS is lost (this also prevents mode oscillations) */
00252     if (!(new_autopilot_mode == AP_MODE_NAV && GpsIsLost()))
00253       autopilot_set_mode(new_autopilot_mode);
00254   }
00255 
00256   /* if not in FAILSAFE mode check motor and in_flight status, read RC */
00257   if (autopilot_mode > AP_MODE_FAILSAFE) {
00258 
00259     /* an arming sequence is used to start/stop motors */
00260     autopilot_arming_check_motors_on();
