124 void autopilot_set_mode(uint8_t new_autopilot_mode) {
125 
126  /* force kill mode as long as AHRS is not aligned */
127  if (!ahrs_is_aligned())
128  new_autopilot_mode = AP_MODE_KILL;
129 
130  if (new_autopilot_mode != autopilot_mode) {
131  /* horizontal mode */
132  switch (new_autopilot_mode) {
133  case AP_MODE_FAILSAFE:
134 #ifndef KILL_AS_FAILSAFE
135  stab_att_sp_euler.phi = 0;
136  stab_att_sp_euler.theta = 0;
137  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
138  break;
139 #endif
140  case AP_MODE_KILL:
141  autopilot_set_motors_on(FALSE);
142  autopilot_in_flight = FALSE;
143  autopilot_in_flight_counter = 0;
144  guidance_h_mode_changed(GUIDANCE_H_MODE_KILL);
145  break;
146  case AP_MODE_RC_DIRECT:
147  guidance_h_mode_changed(GUIDANCE_H_MODE_RC_DIRECT);
148  break;
149  case AP_MODE_RATE_DIRECT:
150  case AP_MODE_RATE_Z_HOLD:
151  guidance_h_mode_changed(GUIDANCE_H_MODE_RATE);
152  break;
153  case AP_MODE_ATTITUDE_RC_CLIMB:
154  case AP_MODE_ATTITUDE_DIRECT:
155  case AP_MODE_ATTITUDE_CLIMB:
156  case AP_MODE_ATTITUDE_Z_HOLD:
157  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
158  break;
159  case AP_MODE_CARE_FREE_DIRECT:
160  guidance_h_mode_changed(GUIDANCE_H_MODE_CARE_FREE);
161  break;
162  case AP_MODE_HOVER_DIRECT:
163  case AP_MODE_HOVER_CLIMB:
164  case AP_MODE_HOVER_Z_HOLD:
165  guidance_h_mode_changed(GUIDANCE_H_MODE_HOVER);
166  break;
167  case AP_MODE_NAV:
168  guidance_h_mode_changed(GUIDANCE_H_MODE_NAV);
169  break;
170  default:
171  break;
172  }
173  /* vertical mode */
174  switch (new_autopilot_mode) {
175  case AP_MODE_FAILSAFE:
176 #ifndef KILL_AS_FAILSAFE
177  guidance_v_zd_sp = SPEED_BFP_OF_REAL(0.5);
178  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
179  break;
180 #endif
181  case AP_MODE_KILL:
182  guidance_v_mode_changed(GUIDANCE_V_MODE_KILL);
183  break;
184  case AP_MODE_RC_DIRECT:
185  case AP_MODE_RATE_DIRECT:
186  case AP_MODE_ATTITUDE_DIRECT:
187  case AP_MODE_HOVER_DIRECT:
188  case AP_MODE_CARE_FREE_DIRECT:
189  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_DIRECT);
190  break;
191  case AP_MODE_RATE_RC_CLIMB:
192  case AP_MODE_ATTITUDE_RC_CLIMB:
193  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_CLIMB);
194  break;
195  case AP_MODE_ATTITUDE_CLIMB:
196  case AP_MODE_HOVER_CLIMB:
197  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
198  break;
199  case AP_MODE_RATE_Z_HOLD:
200  case AP_MODE_ATTITUDE_Z_HOLD:
201  case AP_MODE_HOVER_Z_HOLD:
202  guidance_v_mode_changed(GUIDANCE_V_MODE_HOVER);
203  break;
204  case AP_MODE_NAV:
205  guidance_v_mode_changed(GUIDANCE_V_MODE_NAV);
206  break;
207  default:
208  break;
209  }
210  autopilot_mode = new_autopilot_mode;
211  }

216 static inline void autopilot_check_in_flight( bool_t motors_on ) {
217  if (autopilot_in_flight) {
218  if (autopilot_in_flight_counter > 0) {
219  if (THROTTLE_STICK_DOWN()) {
220  autopilot_in_flight_counter--;
221  if (autopilot_in_flight_counter == 0) {
222  autopilot_in_flight = FALSE;
223  }
224  }
225  else { /* !THROTTLE_STICK_DOWN */
226  autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
227  }
228  }
229  }
230  else { /* not in flight */
231  if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
232  motors_on) {
233  if (!THROTTLE_STICK_DOWN()) {
234  autopilot_in_flight_counter++;
235  if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME)
236  autopilot_in_flight = TRUE;
237  }
238  else { /* THROTTLE_STICK_DOWN */
239  autopilot_in_flight_counter = 0;
240  }
241  }
242  }
243 }

256 void autopilot_on_rc_frame(void) {
257 
258  if (kill_switch_is_on())
259  autopilot_set_mode(AP_MODE_KILL);
260  else {
261  uint8_t new_autopilot_mode = 0;
262  AP_MODE_OF_PPRZ(radio_control.values[RADIO_MODE], new_autopilot_mode);
263  /* don't enter NAV mode if GPS is lost (this also prevents mode oscillations) */
264  if (!(new_autopilot_mode == AP_MODE_NAV
265 #if USE_GPS
266  && GpsIsLost()
267 #endif
268  ))
269  autopilot_set_mode(new_autopilot_mode);
270  }
271 
272  /* if not in FAILSAFE mode check motor and in_flight status, read RC */
273  if (autopilot_mode > AP_MODE_FAILSAFE) {
274 
275  /* if there are some commands that should always be set from RC, do it */
276 #ifdef SetAutoCommandsFromRC
277  SetAutoCommandsFromRC(commands, radio_control.values);
278 #endif
279 
280  /* if not in NAV_MODE set commands from the rc */
281 #ifdef SetCommandsFromRC
282  if (autopilot_mode != AP_MODE_NAV) {
283  SetCommandsFromRC(commands, radio_control.values);
284  }
285 #endif
286 
287  /* an arming sequence is used to start/stop motors */
288  autopilot_arming_check_motors_on();
289  kill_throttle = ! autopilot_motors_on;
290 
291  autopilot_check_in_flight(autopilot_motors_on);
292 
293  guidance_v_read_rc();
294  guidance_h_read_rc(autopilot_in_flight);
295  }
