00104 void sys_time_arch_init( void ) {
00105   /* setup Timer 0 to count forever  */
00106   /* reset & disable timer 0         */
00107   T0TCR = TCR_RESET;
00108   /* set the prescale divider        */
00109   T0PR = T0_PCLK_DIV - 1;
00110   /* enable interrupt on match0  for sys_ticks */
00111   T0MCR = TMCR_MR0_I;
00112   /* disable capture registers       */
00113   T0CCR = 0;
00114   /* disable external match register */
00115   T0EMR = 0;
00116 
00117   /* set first sys tick interrupt    */
00118   T0MR0 = SYS_TIME_RESOLUTION_CPU_TICKS;
00119 
00120   /* enable timer 0                  */
00121   T0TCR = TCR_ENABLE;
00122 
00123   /* select TIMER0 as IRQ    */
00124   VICIntSelect &= ~VIC_BIT(VIC_TIMER0);
00125   /* enable TIMER0 interrupt */
00126   VICIntEnable = VIC_BIT(VIC_TIMER0);
00127   /* on slot vic slot 1      */
00128   _VIC_CNTL(TIMER0_VIC_SLOT) = VIC_ENABLE | VIC_TIMER0;
00129   /* address of the ISR      */
00130   _VIC_ADDR(TIMER0_VIC_SLOT) = (uint32_t)TIMER0_ISR;
00131 }

00139 static inline void sys_tick_irq_handler(void) {
00140   /* set match register for next interrupt */
00141   T0MR0 += SYS_TIME_RESOLUTION_CPU_TICKS - 1;
00142 
00143   sys_time.nb_tick++;
00144   sys_time.nb_sec_rem += SYS_TIME_RESOLUTION_CPU_TICKS;
00145   if (sys_time.nb_sec_rem >= CPU_TICKS_PER_SEC) {
00146     sys_time.nb_sec_rem -= CPU_TICKS_PER_SEC;
00147     sys_time.nb_sec++;
00148 #ifdef SYS_TIME_LED
00149     LED_TOGGLE(SYS_TIME_LED);
00150 #endif
00151   }
00152   for (unsigned int i=0; i<SYS_TIME_NB_TIMER; i++) {
00153     if (sys_time.timer[i].in_use &&
00154         sys_time.nb_tick >= sys_time.timer[i].end_time) {
00155       sys_time.timer[i].end_time += sys_time.timer[i].duration;
00156       sys_time.timer[i].elapsed = TRUE;
00157       if (sys_time.timer[i].cb) sys_time.timer[i].cb(i);
00158     }
00159   }
00160 }
