105 void sys_time_arch_init( void ) {
106  /* setup Timer 0 to count forever */
107  /* reset & disable timer 0 */
108  T0TCR = TCR_RESET;
109  /* set the prescale divider */
110  T0PR = T0_PCLK_DIV - 1;
111  /* enable interrupt on match0 for sys_ticks */
112  T0MCR = TMCR_MR0_I;
113  /* disable capture registers */
114  T0CCR = 0;
115  /* disable external match register */
116  T0EMR = 0;
117 
118  /* set first sys tick interrupt */
119  T0MR0 = SYS_TIME_RESOLUTION_CPU_TICKS;
120 
121  /* enable timer 0 */
122  T0TCR = TCR_ENABLE;
123 
124  /* select TIMER0 as IRQ */
125  VICIntSelect &= ~VIC_BIT(VIC_TIMER0);
126  /* enable TIMER0 interrupt */
127  VICIntEnable = VIC_BIT(VIC_TIMER0);
128  /* on slot vic slot 1 */
129  _VIC_CNTL(TIMER0_VIC_SLOT) = VIC_ENABLE | VIC_TIMER0;
130  /* address of the ISR */
131  _VIC_ADDR(TIMER0_VIC_SLOT) = (uint32_t)TIMER0_ISR;
132 }

136 //
137 // 97 days at 512hz
138 // 12 hours at 100khz
139 //
140 static inline void sys_tick_irq_handler(void) {
141  /* set match register for next interrupt */
142  T0MR0 += SYS_TIME_RESOLUTION_CPU_TICKS - 1;
143 
144  sys_time.nb_tick++;
145  sys_time.nb_sec_rem += SYS_TIME_RESOLUTION_CPU_TICKS;
146  if (sys_time.nb_sec_rem >= CPU_TICKS_PER_SEC) {
147  sys_time.nb_sec_rem -= CPU_TICKS_PER_SEC;
148  sys_time.nb_sec++;
149 #ifdef SYS_TIME_LED
150  LED_TOGGLE(SYS_TIME_LED);
151 #endif
152  }
153  for (unsigned int i=0; i<SYS_TIME_NB_TIMER; i++) {
154  if (sys_time.timer[i].in_use &&
155  sys_time.nb_tick >= sys_time.timer[i].end_time) {
156  sys_time.timer[i].end_time += sys_time.timer[i].duration;
157  sys_time.timer[i].elapsed = TRUE;
158  if (sys_time.timer[i].cb) sys_time.timer[i].cb(i);
159  }
160  }
161 }

163 void TIMER0_ISR ( void ) {
164  ISR_ENTRY();
165 
166  while (T0IR & TIMER0_IT_MASK) {
167 
168  if (T0IR&SYS_TICK_IT) {
169  sys_tick_irq_handler();
170  T0IR = SYS_TICK_IT;
171  }
172 
173 #if defined ACTUATORS && ( defined SERVOS_4017 || defined SERVOS_4015_MAT || defined SERVOS_PPM_MAT)
174  if (T0IR&ACTUATORS_IT) {
175 #ifdef SERVOS_4017
176  SERVOS_4017_ISR();
177 #endif
178 #ifdef SERVOS_4015_MAT
179  Servos4015Mat_ISR();
180 #endif
181 #ifdef SERVOS_PPM_MAT
182  ServosPPMMat_ISR();
183 #endif
184  T0IR = ACTUATORS_IT;
185  }
186 #endif /* ACTUATORS && (SERVOS_4017 || SERVOS_4015_MAT || SERVOS_PPM_MAT) */
187 
188 #if defined RADIO_CONTROL && defined RADIO_CONTROL_TYPE_PPM
189  if (T0IR&PPM_IT) {
190  PPM_ISR();
191  T0IR = PPM_IT;
192  }
193 #endif
194 #ifdef TRIGGER_EXT
195  if (T0IR&TRIGGER_IT) {
196  TRIG_ISR();
197  T0IR = TRIGGER_IT;
198  LED_TOGGLE(3);
199  }
200 #endif
201 #ifdef MB_SCALE
202  if (T0IR&MB_SCALE_IT) {
203  MB_SCALE_ICP_ISR();
204  T0IR = MB_SCALE_IT;
205  }
206 #endif
207 #ifdef MB_TACHO
208  if (T0IR&MB_TACHO_IT) {
209  MB_TACHO_ISR();
210  T0IR = MB_TACHO_IT;
211  }
212 #endif
213 #ifdef USE_PWM_INPUT1
214  if (T0IR&PWM_INPUT_IT1) {
215  PWM_INPUT_ISR_1();
216  T0IR = PWM_INPUT_IT1;
217  }
218 #endif
219 #ifdef USE_PWM_INPUT2
220  if (T0IR&PWM_INPUT_IT2) {
221  PWM_INPUT_ISR_2();
222  T0IR = PWM_INPUT_IT2;
223  }
224 #endif
225 #ifdef USE_AMI601

227  AMI601_ISR();
228  T0IR = AMI601_IT;
229  }
230 #endif
231  }
232  VICVectAddr = 0x00000000;
233  ISR_EXIT();
234 }
