00096 
00097   /* setup pins for SSP (SCK, MISO, MOSI) */
00098   PINSEL1 |= SSP_PINSEL1_SCK  | SSP_PINSEL1_MISO | SSP_PINSEL1_MOSI | SSP_PINSEL1_SSEL;
00099 
00100   /* setup SSP */
00101   // Control Registers
00102   SSPCR0 = SSPCR0_VAL;
00103   SSPCR1 = SSPCR1_VAL;
00104   // Clock Prescale Registers
00105   SSPCPSR = CPSDVSR;
00106 
00107   /* initialize interrupt vector */
00108   VICIntSelect &= ~VIC_BIT( VIC_SPI1 );             /* SPI1 selected as IRQ */
