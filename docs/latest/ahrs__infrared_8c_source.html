65 void ahrs_update_gps(void) {
66 
67  float hspeed_mod_f = gps.gspeed / 100.;
68  float course_f = gps.course / 1e7;
69 
70  // Heading estimator from wind-information, usually computed with -DWIND_INFO
71  // wind_north and wind_east initialized to 0, so still correct if not updated
72  float w_vn = cosf(course_f) * hspeed_mod_f - stateGetHorizontalWindspeed_f()->x;
73  float w_ve = sinf(course_f) * hspeed_mod_f - stateGetHorizontalWindspeed_f()->y;
74  heading = atan2f(w_ve, w_vn);
75  if (heading < 0.)
76  heading += 2 * M_PI;
77 
78 }

80 void ahrs_update_infrared(void) {
81  float phi = atan2(infrared.roll, infrared.top) - infrared.roll_neutral;
82  float theta = atan2(infrared.pitch, infrared.top) - infrared.pitch_neutral;
83 
84  if (theta < -M_PI_2) theta += M_PI;
85  else if (theta > M_PI_2) theta -= M_PI;
86 
87  if (phi >= 0) phi *= infrared.correction_right;
88  else phi *= infrared.correction_left;
89 
90  if (theta >= 0) theta *= infrared.correction_up;
91  else theta *= infrared.correction_down;
92 
93  struct FloatEulers att = { phi, theta, heading };
94  stateSetNedToBodyEulers_f(&att);
95 
96 }
