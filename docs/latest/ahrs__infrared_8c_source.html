00065 void ahrs_update_gps(void) {
00066 
00067   float hspeed_mod_f = gps.gspeed / 100.;
00068   float course_f = gps.course / 1e7;
00069 
00070   // Heading estimator from wind-information, usually computed with -DWIND_INFO
00071   // wind_north and wind_east initialized to 0, so still correct if not updated
00072   float w_vn = cosf(course_f) * hspeed_mod_f - stateGetHorizontalWindspeed_f()->x;
00073   float w_ve = sinf(course_f) * hspeed_mod_f - stateGetHorizontalWindspeed_f()->y;
00074   heading = atan2f(w_ve, w_vn);
00075   if (heading < 0.)
00076     heading += 2 * M_PI;
00077 
00078 }

00080 void ahrs_update_infrared(void) {
00081   float phi  = atan2(infrared.roll, infrared.top) - infrared.roll_neutral;
00082   float theta  = atan2(infrared.pitch, infrared.top) - infrared.pitch_neutral;
00083 
00084   if (theta < -M_PI_2) theta += M_PI;
00085   else if (theta > M_PI_2) theta -= M_PI;
00086 
00087   if (phi >= 0) phi *= infrared.correction_right;
00088   else phi *= infrared.correction_left;
00089 
00090   if (theta >= 0) theta *= infrared.correction_up;
00091   else theta *= infrared.correction_down;

00093   struct FloatEulers att = { phi, theta, heading };
