00061 void lcd_dogm_init_hw( void ) {
00062   /* setup pins for SSP (SCK, MOSI) */
00063   PINSEL1 |= PINSEL1_SCK | PINSEL1_MOSI;
00064 
00065   /* setup SSP */
00066   SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
00067   SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
00068   SSPCPSR = SSPCPSR_VAL; /* Prescaler */
00069 
00070   /* SS, RS pin is output */

00072   SetBit(LCDDOGM_RS_IODIR, LCDDOGM_RS_PIN);
00073   /* unselected lcd */

00076   /* Configure interrupt vector for SPI */
