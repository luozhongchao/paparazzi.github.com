00059 void lcd_dogm_init_hw( void ) {
00060   /* setup pins for SSP (SCK, MOSI) */
00061   PINSEL1 |= PINSEL1_SCK | PINSEL1_MOSI;
00062 
00063   /* setup SSP */
00064   SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
00065   SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
00066   SSPCPSR = SSPCPSR_VAL; /* Prescaler */
00067 
00068   /* SS, RS pin is output */

00070   SetBit(LCDDOGM_RS_IODIR, LCDDOGM_RS_PIN);
00071   /* unselected lcd */

00073 
00074   /* Configure interrupt vector for SPI */
00075   VICIntSelect &= ~VIC_BIT(VIC_SPI1);   /* SPI1 selected as IRQ */

00077   VICVectCntl7 = VIC_ENABLE | VIC_SPI1;
00078   VICVectAddr7 = (uint32_t)SPI1_ISR;    /* address of the ISR */
