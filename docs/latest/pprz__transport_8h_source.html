00157 static inline void parse_pprz(struct pprz_transport * t, uint8_t c ) {
00158   switch (t->status) {
00159   case UNINIT:
00160     if (c == STX)
00161       t->status++;
00162     break;
00163   case GOT_STX:
00164     if (t->trans.msg_received) {
00165       t->trans.ovrn++;
00166       goto error;
00167     }
00168     t->trans.payload_len = c-4; /* Counting STX, LENGTH and CRC1 and CRC2 */
00169     t->ck_a = t->ck_b = c;
00170     t->status++;
00171     t->payload_idx = 0;
00172     break;
00173   case GOT_LENGTH:
00174     t->trans.payload[t->payload_idx] = c;
00175     t->ck_a += c; t->ck_b += t->ck_a;
00176     t->payload_idx++;
00177     if (t->payload_idx == t->trans.payload_len)
00178       t->status++;
00179     break;
00180   case GOT_PAYLOAD:
00181     if (c != t->ck_a)
00182       goto error;
00183     t->status++;
00184     break;
00185   case GOT_CRC1:
00186     if (c != t->ck_b)
00187       goto error;
00188     t->trans.msg_received = TRUE;
00189     goto restart;
00190   default:
00191     goto error;
00192   }
00193   return;
00194  error:
00195   t->trans.error++;
00196  restart:
00197   t->status = UNINIT;
00198   return;
00199 }
