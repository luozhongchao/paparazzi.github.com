00050 static unsigned long UpdateCRC(unsigned long CRC_acc, void *data, unsigned long data_len)
00051 {
00052   unsigned long i; // loop counter
00053 #define POLY 0xEDB88320 // bit-reversed version of the poly 0x04C11DB7
00054   // Create the CRC "dividend" for polynomial arithmetic (binary arithmetic
00055   // with no carries)
00056 
00057   unsigned char *CRC_input = (unsigned char*)data;
00058   for (unsigned long j = data_len; j; --j) {
00059 
00060     CRC_acc = CRC_acc ^ *CRC_input++;
00061     // "Divide" the poly into the dividend using CRC XOR subtraction
00062     // CRC_acc holds the "remainder" of each divide
00063     //
00064     // Only complete this division for 8 bits since input is 1 byte
00065     for (i = 0; i < 8; i++)  {
00066       // Check if the MSB is set (if MSB is 1, then the POLY can "divide"
00067       // into the "dividend")
00068       if ((CRC_acc & 0x00000001) == 0x00000001) {
00069         // if so, shift the CRC value, and XOR "subtract" the poly
00070         CRC_acc = CRC_acc >> 1;
00071         CRC_acc ^= POLY;
00072       }
00073       else {
00074         // if not, just shift the CRC value
00075         CRC_acc = CRC_acc >> 1;
00076       }
00077     }
00078   }
00079   // Return the final remainder (CRC value)
00080   return CRC_acc;
00081 }

00182     } else {
00183       pstData->m_State = CHIMU_STATE_MACHINE_START; //Length byte exceeds buffer.  Signal a fail and restart

00185     }

00200       pstData->m_State = CHIMU_STATE_MACHINE_START;

00203       pstData->m_FullMessage[pstData->m_Index++]=btData;
00204       pstData->m_PayloadIndex = 0;
00205       pstData->m_State = CHIMU_STATE_MACHINE_PAYLOAD; //Finally....Good to go...
00206     }
00207     break;
00208   case CHIMU_STATE_MACHINE_PAYLOAD:  // Waiting for number of bytes in payload
00209     pstData->m_Payload[pstData->m_PayloadIndex++]= btData;
00210     pstData->m_FullMessage[pstData->m_Index++]=btData;
00211     if ((pstData->m_Index) >= (pstData->m_MsgLen + 5)) {
00212       //Now we have the payload.  Verify XSUM and then parse it next
00213       pstData->m_Checksum = (unsigned char) ((UpdateCRC(0xFFFFFFFF , pstData->m_FullMessage , (unsigned long) (pstData->m_MsgLen)+5)) & 0xFF);
00214       pstData->m_State = CHIMU_STATE_MACHINE_XSUM;
00215     } else {
00216       return FALSE;
00217     }
00218     break;
00219   case CHIMU_STATE_MACHINE_XSUM:  // Verify
00220     pstData->m_ReceivedChecksum = btData;
00221     pstData->m_FullMessage[pstData->m_Index++]=btData;
00222     if (pstData->m_Checksum!=pstData->m_ReceivedChecksum) {
00223       bUpdate = FALSE;
00224       //BuiltInTest(BIT_COM_UART_RECEIPTFAIL, BIT_FAIL);
00225     } else {
00226       //Xsum passed, go parse it.
00227       // We have pstData->m_MsgID to parse off of, pstData->m_pstData->m_Payload as the data.
00228       bUpdate = CHIMU_ProcessMessage(&pstData->m_MsgID, pstData->m_Payload, pstData);
00229     }
00230     pstData->m_State = CHIMU_STATE_MACHINE_START;
00231     break;
00232   default:
00233     pstData->m_State = CHIMU_STATE_MACHINE_START;
00234   } /* End of SWITCH */
00235   return (bUpdate);
00236 }

00244 static CHIMU_attitude_data GetEulersFromQuat(CHIMU_attitude_data attitude)
00245 {
00246   CHIMU_attitude_data ps;
00247   ps = attitude;
00248   float x, sqw,sqx,sqy,sqz,norm;
00249   sqw = ps.q.s * ps.q.s;
00250   sqx = ps.q.v.x * ps.q.v.x;
00251   sqy = ps.q.v.y * ps.q.v.y;
00252   sqz = ps.q.v.z * ps.q.v.z;
00253   norm = sqrt(sqw + sqx + sqy + sqz);
00254   //Normalize the quat
00255   ps.q.s = ps.q.s / norm;
00256   ps.q.v.x = ps.q.v.x / norm;
00257   ps.q.v.y = ps.q.v.y / norm;
00258   ps.q.v.z = ps.q.v.z / norm;
00259   ps.euler.phi =atan2(2.0 * (ps.q.s * ps.q.v.x + ps.q.v.y * ps.q.v.z), (1 - 2 * (sqx + sqy)));
00260   if (ps.euler.phi < 0)  ps.euler.phi = ps.euler.phi + 2 *M_PI;
00261   x = ((2.0 * (ps.q.s * ps.q.v.y - ps.q.v.z * ps.q.v.x)));
00262   //Below needed in event normalization not done
00263   if (x > 1.0) x = 1.0;
00264   if (x < -1.0) x = -1.0;
00265   //
00266   if ((ps.q.v.x * ps.q.v.y + ps.q.v.z * ps.q.s) == 0.5)  {
00267     ps.euler.theta = 2 *atan2(ps.q.v.x, ps.q.s);
00268   }
00269   else
00270     if ((ps.q.v.x * ps.q.v.y + ps.q.v.z * ps.q.s) == -0.5)  {
00271       ps.euler.theta = -2 *atan2(ps.q.v.x, ps.q.s);
00272     }
00273     else{
00274       ps.euler.theta = asin(x);
00275     }
00276   ps.euler.psi = atan2(2.0 * (ps.q.s * ps.q.v.z + ps.q.v.x * ps.q.v.y), (1 - 2 * (sqy + sqz)));
00277   if (ps.euler.psi < 0) {
00278     ps.euler.psi = ps.euler.psi + (2 * M_PI);
00279   }
00280 
00281   return(ps);
00282 
00283 }
