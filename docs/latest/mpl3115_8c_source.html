62 // Configuration function called once before normal use
63 static void mpl3115_send_config(void)
64 {
65  switch (mpl3115_init_status) {
66  case MPL_CONF_PT_DATA:
67  mpl3115_trans.buf[0] = MPL3115_REG_PT_DATA_CFG;
68  mpl3115_trans.buf[1] = MPL3115_PT_DATA_CFG;
69  I2CTransmit(MPL3115_I2C_DEV, mpl3115_trans, MPL3115_I2C_ADDR, 2);
70  mpl3115_init_status++;
71  break;
72  case MPL_CONF_CTRL1:
73  mpl3115_trans.buf[0] = MPL3115_REG_CTRL_REG1;
74  mpl3115_trans.buf[1] = MPL3115_CTRL_REG1;
75  I2CTransmit(MPL3115_I2C_DEV, mpl3115_trans, MPL3115_I2C_ADDR, 2);
76  mpl3115_init_status++;
77  break;
78  case MPL_CONF_DONE:
79  mpl3115_initialized = TRUE;
80  mpl3115_trans.status = I2CTransDone;
81  break;
82  default:
83  break;
84  }
85 }

87 // Configure
88 void mpl3115_configure(void)
89 {
90  if (mpl3115_init_status == MPL_CONF_UNINIT) {
91  mpl3115_init_status++;
92  if (mpl3115_trans.status == I2CTransSuccess || mpl3115_trans.status == I2CTransDone) {
93  mpl3115_send_config();
94  }

98 // Normal reading
99 void mpl3115_read(void)
100 {
101  // ask for a reading and then prepare next conversion
102  if (mpl3115_initialized && mpl3115_trans.status == I2CTransDone) {
103  mpl3115_trans.buf[0] = MPL3115_REG_STATUS;
104  I2CTransceive(MPL3115_I2C_DEV, mpl3115_trans, MPL3115_I2C_ADDR, 1, 6);
105  if (mpl3115_req_trans.status == I2CTransDone) {
106  mpl3115_req_trans.buf[0] = MPL3115_REG_CTRL_REG1;
107  mpl3115_req_trans.buf[1] = MPL3115_CTRL_REG1 | MPL3115_OST_BIT;
108  I2CTransmit(MPL3115_I2C_DEV, mpl3115_req_trans, MPL3115_I2C_ADDR, 2);
109  }
110  }
111 }

113 void mpl3115_event(void)
114 {
115  if (mpl3115_initialized) {
116  if (mpl3115_trans.status == I2CTransFailed) {
117  mpl3115_trans.status = I2CTransDone;
118  }
119  else if (mpl3115_trans.status == I2CTransSuccess) {
120  // Successfull reading and new pressure data available
121  if (mpl3115_trans.buf[0] & (1<<2)) {
122 #if MPL3115_RAW_OUTPUT
123  // New data available
124  mpl3115_pressure = ((uint32_t)mpl3115_trans.buf[1]<<16)|((uint16_t)mpl3115_trans.buf[2]<<8)|mpl3115_trans.buf[3];
125  mpl3115_temperature = ((int16_t)mpl3115_trans.buf[4]<<8)|mpl3115_trans.buf[5];
126  mpl3115_data_available = TRUE;
127 #else // Not in raw mode
128 #if MPL3115_ALT_MODE
129  uint32_t tmp = ((uint32_t)mpl3115_trans.buf[1]<<16)|((uint16_t)mpl3115_trans.buf[2]<<8)|mpl3115_trans.buf[3];
130  mpl3115_alt = (float)(tmp>>4)/(1<<4);
131  tmp = ((int16_t)mpl3115_trans.buf[4]<<8)|mpl3115_trans.buf[5];
132  mpl3115_temperature = (tmp>>4);
133  mpl3115_data_available = TRUE;
134 #else // Pressure mode
135  uint32_t tmp = ((uint32_t)mpl3115_trans.buf[1]<<16)|((uint16_t)mpl3115_trans.buf[2]<<8)|mpl3115_trans.buf[3];
136  mpl3115_pressure = (tmp>>4);
137  tmp = ((int16_t)mpl3115_trans.buf[4]<<8)|mpl3115_trans.buf[5];
138  mpl3115_temperature = (tmp>>4);
139  mpl3115_data_available = TRUE;
140 #endif // end alt mode
141 #endif // end raw mode
142  }
143  mpl3115_trans.status = I2CTransDone;
144  }
145  }
146  else if (!mpl3115_initialized && mpl3115_init_status != MPL_CONF_UNINIT) { // Configuring
147  if (mpl3115_trans.status == I2CTransSuccess || mpl3115_trans.status == I2CTransDone) {
148  mpl3115_trans.status = I2CTransDone;
149  mpl3115_send_config();
150  }
151  if (mpl3115_trans.status == I2CTransFailed) {
152  mpl3115_init_status--;
153  mpl3115_trans.status = I2CTransDone;
154  mpl3115_send_config(); // Retry config (TODO max retry)
155  }
156  }
157  if (mpl3115_req_trans.status == I2CTransSuccess || mpl3115_req_trans.status == I2CTransFailed) {
158  mpl3115_req_trans.status = I2CTransDone;
159  }
160 }
