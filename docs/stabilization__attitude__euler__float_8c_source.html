00086 void stabilization_attitude_run(bool_t  in_flight) {
00087 
00088   stabilization_attitude_ref_update();
00089 
00090   /* Compute feedforward */
00091   stabilization_att_ff_cmd[COMMAND_ROLL] =
00092     stabilization_gains.dd.x * stab_att_ref_accel.p / 32.;
00093   stabilization_att_ff_cmd[COMMAND_PITCH] =
00094     stabilization_gains.dd.y * stab_att_ref_accel.q / 32.;
00095   stabilization_att_ff_cmd[COMMAND_YAW] =
00096     stabilization_gains.dd.z * stab_att_ref_accel.r / 32.;
00097 
00098   /* Compute feedback                  */
00099   /* attitude error            */
00100   struct FloatEulers att_float;
00101   EULERS_FLOAT_OF_BFP(att_float, ahrs.ltp_to_body_euler);
00102   struct FloatEulers att_err;
00103   EULERS_DIFF(att_err, att_float, stab_att_ref_euler);
00104   FLOAT_ANGLE_NORMALIZE(att_err.psi);
00105 
00106   if (in_flight) {
00107     /* update integrator */
00108     EULERS_ADD(stabilization_att_sum_err, att_err);
00109     EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
00110   }
00111   else {
00112     FLOAT_EULERS_ZERO(stabilization_att_sum_err);
00113   }
00114 
00115   /*  rate error                */
00116   struct FloatRates rate_float;
00117   RATES_FLOAT_OF_BFP(rate_float, ahrs.body_rate);
00118   struct FloatRates rate_err;
00119   RATES_DIFF(rate_err, rate_float, stab_att_ref_rate);
00120 
00121   /*  PID                  */
00122 
00123   stabilization_att_fb_cmd[COMMAND_ROLL] =
00124     stabilization_gains.p.x  * att_err.phi +
00125     stabilization_gains.d.x  * rate_err.p +
00126     stabilization_gains.i.x  * stabilization_att_sum_err.phi / 1024.;
00127 
00128   stabilization_att_fb_cmd[COMMAND_PITCH] =
00129     stabilization_gains.p.y  * att_err.theta +
00130     stabilization_gains.d.y  * rate_err.q +
00131     stabilization_gains.i.y  * stabilization_att_sum_err.theta / 1024.;
00132 
00133   stabilization_att_fb_cmd[COMMAND_YAW] =
00134     stabilization_gains.p.z  * att_err.psi +
00135     stabilization_gains.d.z  * rate_err.r +
00136     stabilization_gains.i.z  * stabilization_att_sum_err.psi / 1024.;
00137 
00138 
00139   stabilization_cmd[COMMAND_ROLL] =
00140     (stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL])/16.;
00141   stabilization_cmd[COMMAND_PITCH] =
00142     (stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH])/16.;
00143   stabilization_cmd[COMMAND_YAW] =
00144     (stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW])/16.;
