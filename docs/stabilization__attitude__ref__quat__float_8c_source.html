00113 void stabilization_attitude_ref_update() {
00114 
00115   /* integrate reference attitude            */
00116   struct FloatQuat qdot;
00117   FLOAT_QUAT_DERIVATIVE(qdot, stab_att_ref_rate, stab_att_ref_quat);
00118   QUAT_SMUL(qdot, qdot, DT_UPDATE);
00119   QUAT_ADD(stab_att_ref_quat, qdot);
00120   FLOAT_QUAT_NORMALIZE(stab_att_ref_quat);
00121 
00122   /* integrate reference rotational speeds   */
00123   struct FloatRates delta_rate;
00124   RATES_SMUL(delta_rate, stab_att_ref_accel, DT_UPDATE);
00125   RATES_ADD(stab_att_ref_rate, delta_rate);
00126 
00127   /* compute reference angular accelerations */
00128   struct FloatQuat err;
00129   /* compute reference attitude error        */
00130   FLOAT_QUAT_INV_COMP(err, stab_att_sp_quat, stab_att_ref_quat);
00131   /* wrap it in the shortest direction       */
00132   FLOAT_QUAT_WRAP_SHORTEST(err);
00133   /* propagate the 2nd order linear model    */
00134   stab_att_ref_accel.p = -2.*stab_att_ref_model[ref_idx].zeta.p*stab_att_ref_model[ref_idx].omega.p*stab_att_ref_rate.p
00135     - stab_att_ref_model[ref_idx].omega.p*stab_att_ref_model[ref_idx].omega.p*err.qx;
00136   stab_att_ref_accel.q = -2.*stab_att_ref_model[ref_idx].zeta.q*stab_att_ref_model[ref_idx].omega.q*stab_att_ref_rate.q
00137     - stab_att_ref_model[ref_idx].omega.q*stab_att_ref_model[ref_idx].omega.q*err.qy;
00138   stab_att_ref_accel.r = -2.*stab_att_ref_model[ref_idx].zeta.r*stab_att_ref_model[ref_idx].omega.r*stab_att_ref_rate.r
00139     - stab_att_ref_model[ref_idx].omega.r*stab_att_ref_model[ref_idx].omega.r*err.qz;
00140 
00141   /*  saturate acceleration */
00142   const struct FloatRates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
00143   const struct FloatRates MAX_ACCEL = {  REF_ACCEL_MAX_P,  REF_ACCEL_MAX_Q,  REF_ACCEL_MAX_R };
00144   RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
00145 
00146   /* compute ref_euler */
00147   FLOAT_EULERS_OF_QUAT(stab_att_ref_euler, stab_att_ref_quat);
00148 }
