00363   static float roll_rate_sum_err = 0.;
00364   static uint8_t roll_rate_sum_idx = 0;
00365   static float roll_rate_sum_values[H_CTL_ROLL_RATE_SUM_NB_SAMPLES];
00366 
00367   roll_rate_sum_err -= roll_rate_sum_values[roll_rate_sum_idx];
00368   roll_rate_sum_values[roll_rate_sum_idx] = err;
00369   roll_rate_sum_err += err;
00370   roll_rate_sum_idx++;
00371   if (roll_rate_sum_idx >= H_CTL_ROLL_RATE_SUM_NB_SAMPLES) roll_rate_sum_idx = 0;
00372 
00373   /* D term calculations */
00374   static float last_err = 0;
00375   float d_err = err - last_err;
00376   last_err = err;

00379     Blend(h_ctl_hi_throttle_roll_rate_pgain, h_ctl_lo_throttle_roll_rate_pgain, v_ctl_throttle_setpoint/((float)MAX_PPRZ));
00380   float cmd = throttle_dep_pgain * ( err + h_ctl_roll_rate_igain * roll_rate_sum_err / H_CTL_ROLL_RATE_SUM_NB_SAMPLES + h_ctl_roll_rate_dgain * d_err);

00382   h_ctl_aileron_setpoint = TRIM_PPRZ(cmd);
