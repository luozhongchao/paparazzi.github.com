00043 void mcu_arch_init(void) {
00044 #ifdef USE_OPENCM3
00045   rcc_clock_setup_in_hse_12mhz_out_72mhz();
00046   NVIC_SetVectorTable(NVIC_VectTab_FLASH, 0x0);

00049 #ifdef HSE_TYPE_EXT_CLK
00050 #pragma message "Using external clock."
00051   /* Setup the microcontroller system.
00052    *  Initialize the Embedded Flash Interface,
00053    *  initialize the PLL and update the SystemFrequency variable.
00054    */
00055   /* RCC system reset(for debug purpose) */
00056   RCC_DeInit();
00057   /* Enable HSE with external clock ( HSE_Bypass ) */
00058   RCC_HSEConfig( STM32_RCC_MODE );
00059   /* Wait till HSE is ready */
00060   ErrorStatus HSEStartUpStatus = RCC_WaitForHSEStartUp();
00061   if (HSEStartUpStatus != SUCCESS) {
00062     /* block if something went wrong */
00063     while(1) {}
00064   }
00065   else {
00066     /* Enable Prefetch Buffer */
00067     FLASH_PrefetchBufferCmd(FLASH_PrefetchBuffer_Enable);
00068     /* Flash 2 wait state */
00069     FLASH_SetLatency(FLASH_Latency_2);
00070     /* HCLK = SYSCLK */
00071     RCC_HCLKConfig(RCC_SYSCLK_Div1);
00072     /* PCLK2 = HCLK */
00073     RCC_PCLK2Config(RCC_HCLK_Div1);
00074     /* PCLK1 = HCLK/2 */
00075     RCC_PCLK1Config(RCC_HCLK_Div2);
00076     /* PLLCLK = 8MHz * 9 = 72 MHz */
00077     RCC_PLLConfig(RCC_PLLSource_HSE_Div1, STM32_PLL_MULT);
00078     /* Enable PLL */
00079     RCC_PLLCmd(ENABLE);
00080     /* Wait till PLL is ready */
00081     while (RCC_GetFlagStatus(RCC_FLAG_PLLRDY) == RESET) {}
00082     /* Select PLL as system clock source */
00083     RCC_SYSCLKConfig(RCC_SYSCLKSource_PLLCLK);
00084     /* Wait till PLL is used as system clock source */
00085     while(RCC_GetSYSCLKSource() != 0x08) {}
00086   }
00087 #else  /* HSE_TYPE_EXT_CLK */
00088 #pragma message "Using normal system clock setup."
00089   SystemInit();
00090 #endif /* HSE_TYPE_EXT_CLK */
00091    /* Set the Vector Table base location at 0x08000000 */
