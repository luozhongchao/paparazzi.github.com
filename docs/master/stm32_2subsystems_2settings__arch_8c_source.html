00069 static uint32_t pflash_checksum(uint32_t ptr, uint32_t size) {
00070   uint32_t i;
00071 
00072   /* reset crc */
00073   CRC_CR = CRC_CR_RESET;
00074 
00075   if (ptr % 4) {
00076     /* calc in 8bit chunks */
00077     for (i=0; i<(size & ~3); i+=4) {
00078       CRC_DR = (*(uint8_t*) (ptr+i)) |
00079                (*(uint8_t*) (ptr+i+1)) << 8 |
00080                (*(uint8_t*) (ptr+i+2)) << 16 |
00081                (*(uint8_t*) (ptr+i+3)) << 24;
00082     }
00083   } else {
00084     /* calc in 32bit */
00085     for (i=0; i<(size & ~3); i+=4) {
00086       CRC_DR = *(uint32_t*) (ptr+i);
00087     }
00088   }
00089 
00090   /* remaining bytes */
00091   switch (size % 4) {
00092     case 1:
00093       CRC_DR = *(uint8_t*) (ptr+i);
00094       break;
00095     case 2:
00096       CRC_DR = (*(uint8_t*) (ptr+i)) |
00097                (*(uint8_t*) (ptr+i+1)) << 8;
00098       break;
00099     case 3:
00100       CRC_DR = (*(uint8_t*) (ptr+i)) |
00101                (*(uint8_t*) (ptr+i+1)) << 8 |
00102                (*(uint8_t*) (ptr+i+2)) << 16;
00103       break;
00104     default:
00105       break;
00106   }
00107 
00108   return CRC_DR;
00109 }

00195 static int32_t pflash_program_bytes(struct FlashInfo* flash,
00196                     uint32_t   src,
00197                     uint32_t   size,
00198                     uint32_t   chksum) {
00199   uint32_t i;
00200 
00201   /* erase */
00202   flash_unlock();
00203   flash_erase_page(flash->addr);
00204   flash_lock();
00205 
00206   /* verify erase */
00207   for (i=0; i<flash->page_size; i+=4) {
00208     if ((*(uint32_t*) (flash->addr + i)) != 0xFFFFFFFF) return -1;
00209   }
00210 
00211   flash_unlock();
00212   /* write full 16 bit words */
00213   for (i=0; i<(size & ~1); i+=2) {
00214     flash_program_half_word(flash->addr+i,
00215         (uint16_t)(*(uint8_t*)(src+i) | (*(uint8_t*)(src+i+1)) << 8));
00216   }
00217   /* fill bytes with a zero */
00218   if (size & 1) {
00219     flash_program_half_word(flash->addr+i, (uint16_t)(*(uint8_t*)(src+i)));
00220   }
00221   /* write size */
00222   flash_program_half_word(flash->addr+flash->page_size-FSIZ,
00223                           (uint16_t)(size & 0xFFFF));
00224   flash_program_half_word(flash->addr+flash->page_size-FSIZ+2,
00225                           (uint16_t)((size >> 16) & 0xFFFF));
00226   /* write checksum */
00227   flash_program_half_word(flash->addr+flash->page_size-FCHK,
00228                           (uint16_t)(chksum & 0xFFFF));
00229   flash_program_half_word(flash->addr+flash->page_size-FCHK+2,
00230                           (uint16_t)((chksum >> 16) & 0xFFFF));
00231   flash_lock();
00232 
00233   /* verify data */
00234   for (i=0; i<size; i++) {
00235     if ((*(uint8_t*) (flash->addr+i)) != (*(uint8_t*) (src+i))) return -2;
00236   }
00237   if (*(uint32_t*) (flash->addr+flash->page_size-FSIZ) != size) return -3;
00238   if (*(uint32_t*) (flash->addr+flash->page_size-FCHK) != chksum) return -4;
00239 
00240   return 0;
