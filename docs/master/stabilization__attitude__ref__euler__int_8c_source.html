00076 void stabilization_attitude_ref_update() {
00077 
00078 #if USE_ATTITUDE_REF
00079 
00080     /* dumb integrate reference attitude        */
00081     const struct Int32Eulers d_angle = {
00082       stab_att_ref_rate.p >> ( F_UPDATE_RES + REF_RATE_FRAC - REF_ANGLE_FRAC),
00083       stab_att_ref_rate.q >> ( F_UPDATE_RES + REF_RATE_FRAC - REF_ANGLE_FRAC),
00084       stab_att_ref_rate.r >> ( F_UPDATE_RES + REF_RATE_FRAC - REF_ANGLE_FRAC)};
00085     EULERS_ADD(stab_att_ref_euler, d_angle );
00086     ANGLE_REF_NORMALIZE(stab_att_ref_euler.psi);
00087 
00088     /* integrate reference rotational speeds   */
00089     const struct Int32Rates d_rate = {
00090       stab_att_ref_accel.p >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
00091       stab_att_ref_accel.q >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC),
00092       stab_att_ref_accel.r >> ( F_UPDATE_RES + REF_ACCEL_FRAC - REF_RATE_FRAC)};
00093     RATES_ADD(stab_att_ref_rate, d_rate);
00094 
00095     /* compute reference attitude error        */
00096     struct Int32Eulers ref_err;
00097     EULERS_DIFF(ref_err, stab_att_ref_euler, stab_att_sp_euler);
00098     /* wrap it in the shortest direction       */
00099     ANGLE_REF_NORMALIZE(ref_err.psi);
00100 
00101     /* compute reference angular accelerations */
00102     const struct Int32Rates accel_rate = {
00103       ((int32_t)(-2.*ZETA_OMEGA_P) * (stab_att_ref_rate.p >> (REF_RATE_FRAC - REF_ACCEL_FRAC)))
00104       >> (ZETA_OMEGA_P_RES),
00105       ((int32_t)(-2.*ZETA_OMEGA_Q) * (stab_att_ref_rate.q >> (REF_RATE_FRAC - REF_ACCEL_FRAC)))
00106       >> (ZETA_OMEGA_Q_RES),
00107       ((int32_t)(-2.*ZETA_OMEGA_R) * (stab_att_ref_rate.r >> (REF_RATE_FRAC - REF_ACCEL_FRAC)))
00108       >> (ZETA_OMEGA_R_RES) };
00109 
00110     const struct Int32Rates accel_angle = {
00111       ((int32_t)(-OMEGA_2_P)* (ref_err.phi   >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_P_RES),
00112       ((int32_t)(-OMEGA_2_Q)* (ref_err.theta >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_Q_RES),
00113       ((int32_t)(-OMEGA_2_R)* (ref_err.psi   >> (REF_ANGLE_FRAC - REF_ACCEL_FRAC))) >> (OMEGA_2_R_RES) };
00114 
00115     RATES_SUM(stab_att_ref_accel, accel_rate, accel_angle);
00116 
00117     /*  saturate acceleration */
00118     const struct Int32Rates MIN_ACCEL = { -REF_ACCEL_MAX_P, -REF_ACCEL_MAX_Q, -REF_ACCEL_MAX_R };
00119     const struct Int32Rates MAX_ACCEL = {  REF_ACCEL_MAX_P,  REF_ACCEL_MAX_Q,  REF_ACCEL_MAX_R };
00120     RATES_BOUND_BOX(stab_att_ref_accel, MIN_ACCEL, MAX_ACCEL);
00121 
00122     /* saturate speed and trim accel accordingly */
00123     SATURATE_SPEED_TRIM_ACCEL();
00124 
00125 #else  /* !USE_ATTITUDE_REF  */
00126     EULERS_COPY(stab_att_ref_euler, stab_att_sp_euler);
00127     INT_RATES_ZERO(stab_att_ref_rate);
00128     INT_RATES_ZERO(stab_att_ref_accel);
00129 #endif /* USE_ATTITUDE_REF   */
00130 
00131 }
