00200 static void calibration( void ) {
00201   /* End of init, configuration (page 11) */
00202   c1 = words[0] >> 1;
00203   c2 = ((words[2] & 0x3f) << 6) | (words[3] & 0x3f);
00204   c3 = words[3] >> 6;
00205   c4 = words[2] >> 6;

00218 /* Handle the SPI message, i.e. store the received values in variables */
00219 void baro_MS5534A_event_task( void ) {
00220   if (status_read_data) {
00221     switch (status) {
00222     case STATUS_MEASURE_PRESSURE:
00223       d1 = Uint16(buf_input);
00224       break;
00225     case STATUS_MEASURE_TEMPERATURE:
00226       d2 = Uint16(buf_input);
00227       /* Compute pressure and temp (page 10) */
00228       int16_t dT = d2 - ut1;
00229       baro_MS5534A_temp = 200 + (dT*(c6+50)) / (1 << 10);
00230       int16_t off = c2*4 + ((c4-512)*dT)/(1 << 12);
00231       uint32_t sens = c1 + (c3*dT)/(1<<10) + 24576;
00232       uint16_t x = (sens*(d1-7168))/(1<<14) - off;
00233       // baro_MS5534A = ((x*10)>>5) + 250*10;
00234       baro_MS5534A_pressure = ((x*100)>>5) + 250*100;
00235       baro_MS5534A_available = TRUE;
00236 
00237       break;
00238     case STATUS_RESET:
00239       break;
00240     default: /* Init status */
00241       words[status] = Uint16(buf_input);
00242       if (status == STATUS_INIT4) {
00243         calibration();
00244       }
00245     }
00246   } /* else nothing to read */
00247 
00248   NextStatus();
00249   if (!status_read_data) {
00250     /* Ask next conversion now */
00251     baro_MS5534A_send();
00252   }
00253 }
