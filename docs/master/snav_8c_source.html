00130 /* Compute the ground speed for courses 0, 360/NB_ANGLES, ...
00131    (NB_ANGLES-1)360/NB_ANGLES */
00132 static void compute_ground_speed(float airspeed,
00133          float wind_x,
00134          float wind_y) {
00135   uint8_t i;
00136   float alpha = 0;
00137   float c = wind_x*wind_x+wind_y*wind_y-airspeed*airspeed;
00138   for( i=0; i < NB_ANGLES; i++, alpha+=ANGLE_STEP) {
00139     /* g^2 -2 scal g + c = 0 */
00140     float scal = wind_x*cos(alpha) + wind_y*sin(alpha);
00141     float delta = 4 * (scal*scal - c);
00142     ground_speeds[i] = scal + sqrt(delta)/2.;
00143     Bound(ground_speeds[i], NOMINAL_AIRSPEED/4, 2*NOMINAL_AIRSPEED);
00144   }
00145 }
