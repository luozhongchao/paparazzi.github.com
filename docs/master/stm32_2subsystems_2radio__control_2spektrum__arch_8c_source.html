00146 /*****************************************************************************
00147  * The bind function means that the satellite receivers believe they are
00148  * connected to a 9 channel JR-R921 24 receiver thus during the bind process
00149  * they try to get the transmitter to transmit at the highest resolution that
00150  * it can manage. The data is contained in 16 byte packets transmitted at
00151  * 115200 baud. Depending on the transmitter either 1 or 2 frames are required
00152  * to contain the data for all channels. These frames are either 11ms or 22ms
00153  * apart.
00154  *
00155  * The format of each frame for the main receiver is as follows
00156  *
00157  *  byte1:  frame loss data
00158  *  byte2:  transmitter information
00159  *  byte3:  and byte4:  channel data
00160  *  byte5:  and byte6:  channel data
00161  *  byte7:  and byte8:  channel data
00162  *  byte9:  and byte10: channel data
00163  *  byte11: and byte12: channel data
00164  *  byte13: and byte14: channel data
00165  *  byte15: and byte16: channel data
00166  *
00167  *
00168  * The format of each frame for the secondary receiver is as follows
00169  *
00170  *  byte1:  frame loss data
00171  *  byte2:  frame loss data
00172  *  byte3:  and byte4:  channel data
00173  *  byte5:  and byte6:  channel data
00174  *  byte7:  and byte8:  channel data
00175  *  byte9:  and byte10: channel data
00176  *  byte11: and byte12: channel data
00177  *  byte13: and byte14: channel data
00178  *  byte15: and byte16: channel data
00179  *
00180  * The frame loss data bytes starts out containing 0 as long as the
00181  * transmitter is switched on before the receivers. It then increments
00182  * whenever frames are dropped.
00183  *
00184  * Three values for the transmitter information byte have been seen thus far
00185  *
00186  * 0x01 From a Spektrum DX7eu which transmits a single frame containing all
00187  * channel data every 22ms with 10bit resolution.
00188  *
00189  * 0x02 From a Spektrum DM9 module which transmits two frames to carry the
00190  * data for all channels 11ms apart with 10bit resolution.
00191  *
00192  * 0x12 From a Spektrum DX7se which transmits two frames to carry the
00193  * data for all channels 11ms apart with 11bit resolution.
00194  *
00195  * 0x12 From a JR X9503 which transmits two frames to carry the
00196  * data for all channels 11ms apart with 11bit resolution.
00197  *
00198  * 0x01 From a Spektrum DX7 which transmits a single frame containing all
00199  * channel data every 22ms with 10bit resolution.
00200  *
00201  * 0x12 From a JR DSX12 which transmits two frames to carry the
00202  * data for all channels 11ms apart with 11bit resolution.
00203  *
00204  * 0x1 From a Spektru DX5e which transmits a single frame containing all
00205  * channel data every 22ms with 10bit resolution.
00206  *
00207  * 0x01 From a Spektrum DX6i which transmits a single frame containing all
00208  * channel data every 22ms with 10bit resolution.
00209  *
00210  * Currently the assumption is that the data has the form :
00211  *
00212  * [0 0 0 R 0 0 N1 N0]
00213  *
00214  * where :
00215  *
00216  * 0 means a '0' bit
00217  * R: 0 for 10 bit resolution 1 for 11 bit resolution channel data
00218  * N1 to N0 is the number of frames required to receive all channel
00219  * data.
00220  *
00221  * Channels can have either 10bit or 11bit resolution. Data from a tranmitter
00222  * with 10 bit resolution has the form:
00223  *
00224  * [F 0 C3 C2 C1 C0 D9 D8 D7 D6 D5 D4 D3 D2 D1 D0]
00225  *
00226  * Data from a tranmitter with 11 bit resolution has the form
00227  *
00228  * [F C3 C2 C1 C0 D10 D9 D8 D7 D6 D5 D4 D3 D2 D1 D0]
00229  *
00230  * where :
00231  *
00232  * 0 means a '0' bit
00233  * F: Normally 0 but set to 1 for the first channel of the 2nd frame if a
00234  * second frame is transmitted.
00235  *
00236  * C3 to C0 is the channel number, 4 bit, matching the numbers allocated in
00237  * the transmitter.
00238  *
00239  * D9 to D0 is the channel data (10 bit) 0xaa..0x200..0x356 for
00240  * 100% transmitter-travel
00241  *
00242  *
00243  * D10 to D0 is the channel data (11 bit) 0x154..0x400..0x6AC for
00244  * 100% transmitter-travel
00245  *****************************************************************************/

00253 #define SpektrumParser(_c, _SpektrumState, _receiver)  {                      \
00254                                                                               \
00255   uint16_t ChannelData;                                                       \
00256   uint8_t TimedOut;                                                           \
00257   static uint8_t TmpEncType = 0;        /* 0 = 10bit, 1 = 11 bit        */    \
00258   static uint8_t TmpExpFrames = 0;      /* # of frames for channel data */    \
00259                                                                               \
00260    TimedOut = (!_SpektrumState.SpektrumTimer) ? 1 : 0;                        \
00261                                                                               \
00262   /* If we have just started the resync process or */                         \
00263   /* if we have recieved a character before our    */                         \
00264   /* 7ms wait has finished                         */                         \
00265   if ((_SpektrumState.ReSync == 1) ||                                         \
00266       ((_SpektrumState.Sync == 0) && (!TimedOut))) {                          \
00267                                                                               \
00268     _SpektrumState.ReSync = 0;                                                \
00269     _SpektrumState.SpektrumTimer = MIN_FRAME_SPACE;                           \
00270     _SpektrumState.Sync = 0;                                                  \
00271     _SpektrumState.ChannelCnt = 0;                                            \
00272     _SpektrumState.FrameCnt = 0;                                              \
00273     _SpektrumState.SecondFrame = 0;                                           \
00274     return;                                                                   \
00275   }                                                                           \
00276                                                                               \
00277   /* the first byte of a new frame. It was received */                        \
00278   /* more than 7ms after the last received byte.    */                        \
00279   /* It represents the number of lost frames so far.*/                        \
00280   if (_SpektrumState.Sync == 0) {                                             \
00281       _SpektrumState.LostFrameCnt = _c;                                       \
00282       if(_receiver) /* secondary receiver */                                  \
00283         _SpektrumState.LostFrameCnt = _SpektrumState.LostFrameCnt << 8;       \
00284       _SpektrumState.Sync = 1;                                                \
00285       _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                          \
00286       return;                                                                 \
00287   }                                                                           \
00288                                                                               \
00289   /* all other bytes should be recieved within     */                         \
00290   /* MAX_BYTE_SPACE time of the last byte received */                         \
00291   /* otherwise something went wrong resynchronise  */                         \
00292   if(TimedOut) {                                                              \
00293     _SpektrumState.ReSync = 1;                                                \
00294     /* next frame not expected sooner than 7ms     */                         \
00295     _SpektrumState.SpektrumTimer = MIN_FRAME_SPACE;                           \
00296     return;                                                                   \
00297   }                                                                           \
00298                                                                               \
00299   /* second character determines resolution and frame rate for main */        \
00300   /* receiver or low byte of LostFrameCount for secondary receiver  */        \
00301   if(_SpektrumState.Sync == 1) {                                              \
00302     if(_receiver) {                                                           \
00303       _SpektrumState.LostFrameCnt +=_c;                                       \
00304       TmpExpFrames = ExpectedFrames;                                          \
00305     } else {                                                                  \
00306       /* TODO: collect more data. I suspect that there is a low res         */\
00307       /* protocol that is still 10 bit but without using the full range.    */\
00308       TmpEncType =(_c & 0x10)>>4;      /* 0 = 10bit, 1 = 11 bit             */\
00309       TmpExpFrames = _c & 0x03;        /* 1 = 1 frame contains all channels */\
00310                                        /* 2 = 2 channel data in 2 frames    */\
00311     }                                                                         \
00312     _SpektrumState.Sync = 2;                                                  \
00313     _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                            \
00314     return;                                                                   \
00315   }                                                                           \
00316                                                                               \
00317   /* high byte of channel data if this is the first byte */                   \
00318   /* of channel data and the most significant bit is set */                   \
00319   /* then this is the second frame of channel data.      */                   \
00320   if(_SpektrumState.Sync == 2) {                                              \
00321     _SpektrumState.HighByte = _c;                                             \
00322     if (_SpektrumState.ChannelCnt == 0) {                                     \
00323       _SpektrumState.SecondFrame = (_SpektrumState.HighByte & 0x80) ? 1 : 0;  \
00324     }                                                                         \
00325     _SpektrumState.Sync = 3;                                                  \
00326     _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                            \
00327     return;                                                                   \
00328   }                                                                           \
00329                                                                               \
00330   /* low byte of channel data */                                              \
00331   if(_SpektrumState.Sync == 3) {                                              \
00332     _SpektrumState.Sync = 2;                                                  \
00333     _SpektrumState.SpektrumTimer = MAX_BYTE_SPACE;                            \
00334     /* we overwrite the buffer now so rc data is not available now */         \
00335     _SpektrumState.RcAvailable = 0;                                           \
00336     ChannelData = ((uint16_t)_SpektrumState.HighByte << 8) | _c;              \
00337     _SpektrumState.values[_SpektrumState.ChannelCnt                           \
00338                           + (_SpektrumState.SecondFrame * 7)] = ChannelData;  \
00339     _SpektrumState.ChannelCnt ++;                                             \
00340   }                                                                           \
00341                                                                               \
00342   /* If we have a whole frame */                                              \
00343   if(_SpektrumState.ChannelCnt >= SPEKTRUM_CHANNELS_PER_FRAME) {              \
00344     /* how many frames did we expect ? */                                     \
00345     ++_SpektrumState.FrameCnt;                                                \
00346     if (_SpektrumState.FrameCnt == TmpExpFrames)                              \
00347     {                                                                         \
00348       /* set the rc_available_flag */                                         \
00349       _SpektrumState.RcAvailable = 1;                                         \
00350       _SpektrumState.FrameCnt = 0;                                            \
00351     }                                                                         \
00352     if(!_receiver) { /* main receiver */                                      \
00353       EncodingType = TmpEncType;         /* only update on a good */          \
00354       ExpectedFrames = TmpExpFrames;     /* main receiver frame   */          \
00355     }                                                                         \
00356     _SpektrumState.Sync = 0;                                                  \
00357     _SpektrumState.ChannelCnt = 0;                                            \
00358     _SpektrumState.SecondFrame = 0;                                           \
00359     _SpektrumState.SpektrumTimer = MIN_FRAME_SPACE;                           \
00360   }                                                                           \
00361 }                                                                             \

00370 void RadioControlEventImp(void (*frame_handler)(void)) {
00371   uint8_t ChannelCnt;
00372   uint8_t ChannelNum;
00373   uint16_t ChannelData;
00374   uint8_t MaxChannelNum = 0;
00375 
00376 #ifdef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00377   /* If we have two receivers and at least one of them has new data */
00378   uint8_t BestReceiver;
00379   if ((PrimarySpektrumState.RcAvailable) ||
00380       (SecondarySpektrumState.RcAvailable)) {
00381     /* if both receivers have new data select the one  */
00382     /* that has had the least number of frames lost    */
00383     if ((PrimarySpektrumState.RcAvailable) &&
00384         (SecondarySpektrumState.RcAvailable)) {
00385       BestReceiver  = (PrimarySpektrumState.LostFrameCnt
00386                        <= SecondarySpektrumState.LostFrameCnt) ? 0 : 1;
00387     } else {
00388       /* if only one of the receivers have new data use it */
00389       BestReceiver  = (PrimarySpektrumState.RcAvailable) ? 0 : 1;
00390     }
00391     /* clear the data ready flags */
00392     PrimarySpektrumState.RcAvailable = 0;
00393     SecondarySpektrumState.RcAvailable = 0;
00394 
00395 #else
00396   /* if we have one receiver and it has new data */
00397   if(PrimarySpektrumState.RcAvailable) {
00398     PrimarySpektrumState.RcAvailable = 0;
00399 #endif
00400     ChannelCnt = 0;
00401     /* for every piece of channel data we have received */
00402     for(int i = 0; (i < SPEKTRUM_CHANNELS_PER_FRAME * ExpectedFrames); i++) {
00403 #ifndef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00404       ChannelData = PrimarySpektrumState.values[i];
00405 #else
00406       ChannelData = (!BestReceiver) ? PrimarySpektrumState.values[i] :
00407                                  SecondarySpektrumState.values[i];
00408 #endif
00409       /* find out the channel number and its value by  */
00410       /* using the EncodingType which is only received */
00411       /* from the main receiver                        */
00412       switch(EncodingType) {
00413         case(0) : /* 10 bit */
00414           ChannelNum = (ChannelData >> 10) & 0x0f;
00415           /* don't bother decoding unused channels */
00416           if (ChannelNum < RADIO_CONTROL_NB_CHANNEL) {
00417            SpektrumBuf[ChannelNum] = ChannelData & 0x3ff;
00418            SpektrumBuf[ChannelNum] -= 0x200;
00419            SpektrumBuf[ChannelNum] *= MAX_PPRZ/0x156;
00420            ChannelCnt++;
00421           }
00422           break;
00423 
00424         case(1) : /* 11 bit */
00425           ChannelNum = (ChannelData >> 11) & 0x0f;
00426           /* don't bother decoding unused channels */
00427           if (ChannelNum < RADIO_CONTROL_NB_CHANNEL) {
00428             SpektrumBuf[ChannelNum] = ChannelData & 0x7ff;
00429             SpektrumBuf[ChannelNum] -= 0x400;
00430             SpektrumBuf[ChannelNum] *= MAX_PPRZ/0x2AC;
00431             ChannelCnt++;
00432           }
00433           break;
00434 
00435         default : ChannelNum = 0x0F; break;  /* never going to get here */
00436       }
00437       /* store the value of the highest valid channel */
00438       if ((ChannelNum != 0x0F) && (ChannelNum > MaxChannelNum))
00439         MaxChannelNum = ChannelNum;
00440 
00441     }
00442 
00443     /* if we have a valid frame the pass it to the frame handler */
00444     if (ChannelCnt >= (MaxChannelNum + 1)) {
00445       radio_control.frame_cpt++;
00446       radio_control.time_since_last_frame = 0;
00447       radio_control.status = RC_OK;
00448       for (int i = 0; i < (MaxChannelNum + 1); i++) {
00449         radio_control.values[i] = SpektrumBuf[i];
00450         if (i == RADIO_THROTTLE ) {
00451           radio_control.values[i] += MAX_PPRZ;
00452           radio_control.values[i] /= 2;
00453         }
00454         radio_control.values[i] *= SpektrumSigns[i];
00455       }
00456       (*frame_handler)();
00457     }
00458   }
00459 }

00650 /*****************************************************************************
00651  *
00652  * The following functions provide functionality to allow binding of
00653  * spektrum satellite receivers. The pulse train sent to them means
00654  * that Lisa is emulating a 9 channel JR-R921 24.
00655  *
00656  *****************************************************************************/
00657 /*****************************************************************************
00658  *
00659  * radio_control_spektrum_try_bind(void) must called on powerup as spektrum
00660  * satellites can only bind immediately after power up also it must be called
00661  * before the call to SpektrumUartInit as we leave them with their Rx pins set
00662  * as outputs.
00663  *
00664  *****************************************************************************/
00665 void radio_control_spektrum_try_bind(void) {
00666 
00667   /* init RCC */
00668   RCC_APB2PeriphClockCmd(BIND_PIN_PERIPH , ENABLE);
00669 
00670   /* Init GPIO for the bind pin */
00671   GPIO_InitTypeDef GPIO_InitStructure;
00672   GPIO_InitStructure.GPIO_Pin = BIND_PIN;
00673   GPIO_InitStructure.GPIO_Mode = GPIO_Mode_IPU;
00674   GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
00675   GPIO_Init(BIND_PIN_PORT, &GPIO_InitStructure);
00676   /* exit if the BIND_PIN is high, it needs to
00677      be pulled low at startup to initiate bind */
00678   if (GPIO_ReadInputDataBit(BIND_PIN_PORT, BIND_PIN))
00679     return;
00680 
00681   /* bind initiated, initialise the delay timer */
00682   SpektrumDelayInit();
00683 
00684   /* initialise the uarts rx pins as  GPIOS */
00685   RCC_APB2PeriphClockCmd(PrimaryUart(_Periph) , ENABLE);
00686   /* Master receiver Rx push-pull */
00687   GPIO_InitStructure.GPIO_Pin = PrimaryUart(_RxPin);
00688   GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
00689   GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
00690   GPIO_Init(PrimaryUart(_RxPort), &GPIO_InitStructure);
00691   /* Master receiver RX line, drive high */
00692   GPIO_WriteBit(PrimaryUart(_RxPort), PrimaryUart(_RxPin) , Bit_SET );
00693 
00694 #ifdef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00695    RCC_APB2PeriphClockCmd(SecondaryUart(_Periph) , ENABLE);
00696   /* Slave receiver Rx push-pull */
00697   GPIO_InitStructure.GPIO_Pin = SecondaryUart(_RxPin);
00698   GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
00699   GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
00700   GPIO_Init(SecondaryUart(_RxPort), &GPIO_InitStructure);
00701   /* Slave receiver RX line, drive high */
00702   GPIO_WriteBit(SecondaryUart(_RxPort), SecondaryUart(_RxPin) , Bit_SET );
00703 #endif
00704 
00705   /* We have no idea how long the window for allowing binding after
00706      power up is. This works for the moment but will need revisiting */
00707   DelayMs(61);
00708 
00709   for (int i = 0; i < MASTER_RECEIVER_PULSES ; i++)
00710   {
00711     GPIO_WriteBit(PrimaryUart(_RxPort), PrimaryUart(_RxPin), Bit_RESET );
00712     DelayUs(118);
00713     GPIO_WriteBit(PrimaryUart(_RxPort), PrimaryUart(_RxPin), Bit_SET );
00714     DelayUs(122);
00715   }
00716 
00717 #ifdef RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT
00718   for (int i = 0; i < SLAVE_RECEIVER_PULSES; i++)
00719   {
00720     GPIO_WriteBit(SecondaryUart(_RxPort), SecondaryUart(_RxPin), Bit_RESET );
00721     DelayUs(120);
00722     GPIO_WriteBit(SecondaryUart(_RxPort), SecondaryUart(_RxPin), Bit_SET );
00723     DelayUs(120);
00724   }
00725 #endif /* RADIO_CONTROL_SPEKTRUM_SECONDARY_PORT */
00726 }
