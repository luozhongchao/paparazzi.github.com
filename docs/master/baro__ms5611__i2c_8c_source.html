00062 static int8_t baro_ms5611_crc(uint16_t* prom) {
00063   int32_t i, j;
00064   uint32_t res = 0;
00065   uint8_t crc = prom[7] & 0xF;
00066   prom[7] &= 0xFF00;
00067   for (i = 0; i < 16; i++) {
00068     if (i & 1) res ^= ((prom[i>>1]) & 0x00FF);
00069     else res ^= (prom[i>>1]>>8);
00070     for (j = 8; j > 0; j--) {
00071       if (res & 0x8000) res ^= 0x1800;
00072       res <<= 1;
00073     }
00074   }
00075   prom[7] |= crc;
00076   if (crc == ((res >> 12) & 0xF)) return 0;
00077   else return -1;
00078 }

00085 void baro_ms5611_periodic( void ) {
00086   if (cpu_time_sec > 1) {
00087     if (ms5611_status >= MS5611_IDLE) {

00096     else if (ms5611_status == MS5611_UNINIT) {
00097       /* reset sensor */
00098       ms5611_status = MS5611_RESET;

00102     else if (ms5611_status == MS5611_RESET_OK) {
00103       /* start getting prom data */
00104       ms5611_status = MS5611_PROM;

00133 void baro_ms5611_event( void ) {
00134   if (ms5611_trans.status == I2CTransSuccess) {
00135     switch (ms5611_status) {
00136 
00137     case MS5611_RESET:
00138       ms5611_status = MS5611_RESET_OK;
00139       ms5611_trans.status = I2CTransDone;
00140       break;
00141 
00142     case MS5611_PROM:
00143       /* read prom data */
00144       ms5611_c[prom_cnt++] = (ms5611_trans.buf[0] << 8) | ms5611_trans.buf[1];
00145       if (prom_cnt < PROM_NB) {
00146         /* get next prom data */
00147         ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
00148         I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
00149       }
00150       else {
00151         /* done reading prom */
00152         ms5611_trans.status = I2CTransDone;
00153         /* check prom crc */
00154         if (baro_ms5611_crc(ms5611_c) == 0) {
00155           DOWNLINK_SEND_MS5611_COEFF(DefaultChannel,
00156               &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
00157               &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]);
00158           ms5611_status = MS5611_IDLE;
00159         }
00160         else {
00161           /* checksum error, try again */
00162           baro_ms5611_init();
00163         }
00164       }
00165       break;
00166 
00167     case MS5611_CONV_D1:
00168       ms5611_status = MS5611_CONV_D1_OK;
00169       ms5611_trans.status = I2CTransDone;
00170       break;
00171 
00172     case  MS5611_ADC_D1:
00173       /* read D1 (pressure) */
00174       ms5611_d1 = (ms5611_trans.buf[0] << 16) |
00175                   (ms5611_trans.buf[1] << 8) |
00176                   ms5611_trans.buf[2];
00177       /* start D2 conversion */
00178       ms5611_status = MS5611_CONV_D2;
00179       ms5611_trans.buf[0] = MS5611_START_CONV_D2;
00180       I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
00181       break;
00182 
00183     case MS5611_CONV_D2:
00184       ms5611_status = MS5611_CONV_D2_OK;
00185       ms5611_trans.status = I2CTransDone;
00186       break;
00187 
00188     case  MS5611_ADC_D2: {
00189       int64_t dt, baroms, tempms, off, sens, t2, off2, sens2;
00190       /* read D2 (temperature) */
00191       ms5611_d2 = (ms5611_trans.buf[0] << 16) |
00192                   (ms5611_trans.buf[1] << 8) |
00193                   ms5611_trans.buf[2];
00194       ms5611_status = MS5611_IDLE;
00195       ms5611_trans.status = I2CTransDone;
00196 
00197       /* difference between actual and ref temperature */
00198       dt = ms5611_d2 - (int64_t)ms5611_c[5] * (1<<8);
00199       /* actual temperature */
00200       tempms = 2000 + ((int64_t)dt * ms5611_c[6]) / (1<<23);
00201       /* offset at actual temperature */
00202       off = ((int64_t)ms5611_c[2] * (1<<16)) + ((int64_t)ms5611_c[4] * dt) / (1<<7);
00203       /* sensitivity at actual temperature */
00204       sens = ((int64_t)ms5611_c[1] * (1<<15)) + ((int64_t)ms5611_c[3] * dt) / (1<<8);
00205       /* second order temperature compensation */
00206       if (tempms < 2000) {
00207         t2 = (dt*dt) / (1<<31);
00208         off2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<1);
00209         sens2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<2);
00210         if (tempms < -1500) {
00211           off2 = off2 + 7 * (int64_t)(tempms+1500)*(tempms+1500);
00212           sens2 = sens2 + 11 * ((int64_t)(tempms+1500)*(tempms+1500)) / (1<<1);
00213         }
00214         tempms = tempms - t2;
00215         off = off - off2;
00216         sens = sens - sens2;
00217       }
00218       /* temperature compensated pressure */
00219       baroms = (((int64_t)ms5611_d1 * sens) / (1<<21) - off) / (1<<15);
00220 #ifdef SENSOR_SYNC_SEND
00221       ftempms = tempms / 100.;
00222       fbaroms = baroms / 100.;
00223       DOWNLINK_SEND_BARO_MS5611(DefaultChannel,
00224                                 &ms5611_d1, &ms5611_d2, &fbaroms, &ftempms);
00225 #endif
00226       break;
00227     }
00228 
00229     default:

00231       break;
00232     }
