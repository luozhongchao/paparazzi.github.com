00060   switch (itg3200_init_status) {

00098     itg3200_init_status++;

00116 void itg3200_event(void)
00117 {
00118   if (itg3200_initialized) {
00119     if (itg3200_i2c_trans.status == I2CTransFailed) {
00120       itg3200_i2c_trans.status = I2CTransDone;
00121     }
00122     else if (itg3200_i2c_trans.status == I2CTransSuccess) {
00123       // Successfull reading and new data available
00124       if (itg3200_i2c_trans.buf[0] & 0x01) {
00125         // New data available
00126         itg3200_data.p = Int16FromBuf(itg3200_i2c_trans.buf,3);
00127         itg3200_data.q = Int16FromBuf(itg3200_i2c_trans.buf,5);
00128         itg3200_data.r = Int16FromBuf(itg3200_i2c_trans.buf,7);
00129         itg3200_data_available = TRUE;
00130       }
00131       itg3200_i2c_trans.status = I2CTransDone;
00132     }
00133   }
00134   else if (!itg3200_initialized && itg3200_init_status != ITG_CONF_UNINIT) { // Configuring
00135     if (itg3200_i2c_trans.status == I2CTransSuccess || itg3200_i2c_trans.status == I2CTransDone) {
00136       itg3200_i2c_trans.status = I2CTransDone;
00137       itg3200_send_config();
00138     }
00139     if (itg3200_i2c_trans.status == I2CTransFailed) {
00140       itg3200_init_status--;
00141       itg3200_i2c_trans.status = I2CTransDone;
00142       itg3200_send_config(); // Retry config (TODO max retry)
00143     }
00144   }
00145 }
