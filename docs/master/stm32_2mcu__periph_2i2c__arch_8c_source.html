00592     // The first data byte will be acked in read many so the slave knows it should send more
00593     I2C_CR1(i2c) &= ~ I2C_CR1_POS;
00594     I2C_CR1(i2c) |= I2C_CR1_ACK;
00595     // Clear the SB flag
00596     I2C_DR(i2c) = trans->slave_addr | 0x01;
00597 
00598     // Document the current Status
00599     periph->status = I2CAddrRdSent;
00600   }
00601   // Address Was Sent
00602   else if (BIT_X_IS_SET_IN_REG(I2C_SR1_ADDR, SR1) )
00603   {
00604     periph->idx_buf = 0;
00605 
00606     // Enable RXNE: receive an interrupt any time a byte is available
00607     // only enable if MORE than 3 bytes need to be read
00608     if (periph->idx_buf < (trans->len_r - 3))
00609     {
00610       I2C_CR2(i2c) |= I2C_CR2_ITBUFEN;
00611     }
00612 
00613     // ACK is still on to get more DATA
00614     // Read SR2 to clear the ADDR (next byte will start arriving)
00615     uint16_t SR2 __attribute__ ((unused)) = I2C_SR2(i2c);
00616 
00617     // Document the current Status
00618     periph->status = I2CReadingByte;
00619   }
00620   // one or more bytes are available AND we were interested in Buffer interrupts
00621   else if ( (BIT_X_IS_SET_IN_REG(I2C_SR1_RxNE, SR1) ) && (BIT_X_IS_SET_IN_REG(I2C_CR2_ITBUFEN, I2C_CR2(i2c)))  )
00622   {
00623     // read byte until 3 bytes remain to be read (e.g. len_r = 6, -> idx=3 means idx 3,4,5 = 3 remain to be read
00624     if (periph->idx_buf < (trans->len_r - 3))
00625     {
00626       trans->buf[periph->idx_buf] = I2C_DR(i2c);
00627       periph->idx_buf ++;
00628     }
00629     // from : 3bytes -> last byte: do nothing
00630     //
00631     // finally: this was the last byte
00632     else if (periph->idx_buf >= (trans->len_r - 1))
00633     {
00634       I2C_CR2(i2c) &= ~ I2C_CR2_ITBUFEN;
00635 
00636       // Last Value
00637       trans->buf[periph->idx_buf] = I2C_DR(i2c);
00638       periph->idx_buf ++;
00639 
00640       // We got all the results
00641       trans->status = I2CTransSuccess;
00642 
00643       return STMI2C_SubTra_Ready_StopRequested;
00644     }
00645 
00646     // Check for end of transaction: start waiting for BTF instead of RXNE
00647     if (periph->idx_buf < (trans->len_r - 3))
00648     {
00649       I2C_CR2(i2c) |= I2C_CR2_ITBUFEN;
00650     }
00651     else // idx >= len-3: there are 3 bytes to be read
00652     {
00653       // We want to halt I2C to have sufficient time to clear ACK, so:
00654       // Stop listening to RXNE as it will be triggered infinitely since we did not empty the buffer
00655       // on the next (second in buffer) received byte BTF will be set (buffer full and I2C halted)
00656       I2C_CR2(i2c) &= ~ I2C_CR2_ITBUFEN;
00657     }
00658   }
00659   // Buffer is full while this was not a RXNE interrupt
00660   else if (BIT_X_IS_SET_IN_REG(I2C_SR1_BTF, SR1) )
00661   {
00662     // Now the shift register and data register contain data(n-2) and data(n-1)
00663     // And I2C is halted so we have time
00664 
00665     // --- Make absolutely sure the next 2 I2C actions are performed with no delay
00666     __I2C_REG_CRITICAL_ZONE_START;
00667 
00668     // First we clear the ACK while the SCL is held low by BTF
00669     I2C_CR1(i2c) &= ~ I2C_CR1_ACK;
00670 
00671     // Now that ACK is cleared we read one byte: instantly the last byte is being clocked in...
00672     trans->buf[periph->idx_buf] = I2C_DR(i2c);
00673     periph->idx_buf ++;
00674 
00675     // Now the last byte is being clocked. Stop in MUST be set BEFORE the transfer of the last byte is complete
00676     PPRZ_I2C_SEND_STOP(i2c);
00677 
00678     __I2C_REG_CRITICAL_ZONE_STOP;
00679     // --- end of critical zone -----------
00680 
00681     // Document the current Status
00682     periph->status = I2CStopRequested;
00683 
00684     // read the byte2 we had in the buffer (BTF means 2 bytes available)
00685     trans->buf[periph->idx_buf] = I2C_DR(i2c);
00686     periph->idx_buf ++;
00687 
00688     // Ask for an interrupt to read the last byte (which is normally still busy now)
00689     // The last byte will be received with RXNE
00690     I2C_CR2(i2c) |= I2C_CR2_ITBUFEN;
00691   }
00692   else // Event Logic Error
00693   {
00694     return STMI2C_SubTra_Error;
00695   }
00696 
00697   return STMI2C_SubTra_Busy;
00698 }

00703 static inline void i2c_error(struct i2c_periph *periph)
00704 {
00705 #ifdef I2C_DEBUG_LED
00706   uint8_t err_nr = 0;
00707 #endif
00708   periph->errors->er_irq_cnt;
00709   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_AF) != 0) { /* Acknowledge failure */
00710     periph->errors->ack_fail_cnt++;
00711     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_AF;
00712 #ifdef I2C_DEBUG_LED
00713     err_nr = 1;
00714 #endif
00715   }
00716   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_BERR) != 0) {     /* Misplaced Start or Stop condition */
00717     periph->errors->miss_start_stop_cnt++;
00718     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_BERR;
00719 #ifdef I2C_DEBUG_LED
00720     err_nr = 2;
00721 #endif
00722   }
00723   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_ARLO) != 0) {     /* Arbitration lost */
00724     periph->errors->arb_lost_cnt++;
00725     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_ARLO;
00726 #ifdef I2C_DEBUG_LED
00727     err_nr = 3;
00728 #endif
00729   }
00730   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_OVR) != 0) {      /* Overrun/Underrun */
00731     periph->errors->over_under_cnt++;
00732     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_OVR;
00733 #ifdef I2C_DEBUG_LED
00734     err_nr = 4;
00735 #endif
00736   }
00737   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_PECERR) != 0) {   /* PEC Error in reception */
00738     periph->errors->pec_recep_cnt++;
00739     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_PECERR;
00740 #ifdef I2C_DEBUG_LED
00741     err_nr = 5;
00742 #endif
00743   }
00744   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_TIMEOUT) != 0) {  /* Timeout or Tlow error */
00745     periph->errors->timeout_tlow_cnt++;
00746     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_TIMEOUT;
00747 #ifdef I2C_DEBUG_LED
00748     err_nr = 6;
00749 #endif
00750   }
00751   if ((I2C_SR1((u32)periph->reg_addr) & I2C_SR1_SMBALERT) != 0) { /* SMBus alert */
00752     periph->errors->smbus_alert_cnt++;
00753     I2C_SR1((u32)periph->reg_addr) &= ~I2C_SR1_SMBALERT;
00754 #ifdef I2C_DEBUG_LED
00755     err_nr = 7;
00756 #endif
00757   }
00758 
00759 #ifdef I2C_DEBUG_LED
00760   LED_ERROR(20, err_nr);
00761 #endif
00762 
00763   return;
00764 }

00768 {
00769   uint16_t SR1 = I2C_SR1(i2c);
00770 
00771   // Certainly do not wait for buffer interrupts:
00772   // -------------------------------------------
00773   I2C_CR2(i2c) &= ~ I2C_CR2_ITBUFEN;      // Disable TXE, RXNE
00774 
00775   // Error interrupts are handled separately:
00776   // ---------------------------------------
00777 
00778   // Clear Event interrupt conditions:
00779   // --------------------------------
00780 
00781   // Start Condition Was Generated
00782   if (BIT_X_IS_SET_IN_REG( I2C_SR1_SB, SR1 ) )
00783   {
00784     // SB: cleared by software when reading SR1 and writing to DR
00785     I2C_DR(i2c) = 0x00;
00786   }
00787   // Address Was Sent
00788   if (BIT_X_IS_SET_IN_REG(I2C_SR1_ADDR, SR1) )

00791     uint16_t SR2 __attribute__ ((unused)) = I2C_SR2(i2c);
00792   }
00793   // Byte Transfer Finished
00794   if (BIT_X_IS_SET_IN_REG(I2C_SR1_BTF, SR1) )
00795   {
00796     // SB: cleared by software when reading SR1 and reading/writing to DR
00797     uint8_t dummy __attribute__ ((unused)) = I2C_DR(i2c);
00798     I2C_DR(i2c) = 0x00;
00799   }

00805 // Restore bus conditions to normal after errors
00806 
00807 static inline void i2c_irq(struct i2c_periph *periph)
00808 {
00809 
00810   /*
00811     There are 7 possible event reasons to get here + all errors
00812 
00813     If IT_EV_FEN
00814     -------------------------
00815 
00816     We are always interested in all IT_EV_FEV: all are required.
00817 
00818     1) SB   // Start Condition Success in Master mode
00819     2) ADDR   // Address sent received Acknoledge
00820     [ADDR10]  // -- 10bit address stuff: not used
00821     [STOPF]   // -- only for slaves: master has no stop interrupt: not used
00822     3) BTF    // I2C has stopped working (it is waiting for new data, all buffers are tx_empty/rx_full)
00823 
00824     // Beware: using the buffered I2C has some interesting properties:
00825       -in master receive mode: BTF only occurs after the 2nd received byte: after the first byte is received it is
00826            in RD but the I2C can still receive a second byte. Only when the 2nd byte is received while the RxNE is 1
00827        then a BTF occurs (I2C can not continue receiving bytes or they will get lost). During BTF I2C is halted (SCL held low)
00828       -in master transmit mode: when writing a byte to WD, you instantly get a new TxE interrupt while the first is not
00829        transmitted yet. The byte was pushed to the I2C shift register and the buffer is ready for more. You can already
00830        fill new data in the buffer while the first is still being transmitted for max performance transmission.
00831 
00832         // Beware: besides data buffering you can/must plan several consecutive actions. You can send 2 bytes to the buffer, ask for a stop and
00833            a new start in one go.
00834 
00835           -thanks to / because of this buffering and event sheduling there is not 1 interrupt per start / byte / stop
00836            This also means you must think more in advance and a transaction could be popped from the transaction stack even before it's
00837            stop condition is actually generated.
00838 
00839     // Beware: the order in which Status (and other register) is read determines how flags are cleared.
00840            You should NOT simply read SR1 & SR2 every time
00841 
00842     If IT_EV_FEN AND IT_EV_BUF
00843     --------------------------
00844 
00845     Buffer event are not always wanted and are typically switched on during longer data transfers. Make sure to turn off in time.
00846 
00847     4) RxNE
00848     5) TxE
00849 
00850     --------------------------------------------------------------------------------------------------
00851 
00852     The STM waits indefinately (holding SCL low) for user interaction:
00853     a) after a master-start (waiting for address)
00854     b) after an address (waiting for data)
00855        not during data sending when using buffered
00856     c) after the last byte is transmitted (waiting for either stop or restart)
00857        not during data receiving when using buffered
00858        not after the last byte is received
00859 
00860     -The STM I2C stalls indefinately when a stop condition was attempted that
00861     did not succeed. The BUSY flag remains on.
00862         -There is no STOP interrupt.
00863 
00864     Caution Reading the status:
00865     - Caution: this clears several flags and can start transmissions etc...
00866     - Certain flags like STOP / (N)ACK need to be guaranteed to be set before
00867       the transmission of the byte is finished. At higher clock rates that can be
00868       quite fast: so we allow no other interrupt to be triggered in between
00869       reading the status and setting all needed flags
00870 
00871    */
00872 
00873   // Here we go ...
00874 
00875   // Apparently we got an I2C interrupt: EVT BUF or ERR
00876 
00877 #ifdef I2C_DEBUG_LED
00878   // Notify ISR is triggered
00879   LED1_ON();
00880   LED1_OFF();
00881 #endif
00882 
00883   // Save Some Direct Access to the I2C Registers ...
00884   u32 i2c = (u32) periph->reg_addr;
00885 
00887   // Check if we were ready ...
00888   if (periph->trans_extract_idx == periph->trans_insert_idx)
00889   {
00890     // Nothing Left To Do
00891 
00892 #ifdef I2C_DEBUG_LED
00893         LED2_ON();
00894         LED1_ON();
00895     LED2_OFF();
00896     LED1_OFF();
00897 
00898     // no transaction and also an error?
00899     LED_SHOW_ACTIVE_BITS(regs);
00900 #endif
00901 
00902     // If we still get an interrupt but there are no more things to do
00903     // (which can happen if an event was sheduled just before a bus error occurs)
00904     // (or can happen if both error and event interrupts were called together [the 2nd will then get this error])
00905 
00906     // since there is nothing more to do: its easy: just stop: clear all interrupt generating bits
00907 
00908     // Count The Errors
00909     i2c_error(periph);
00910 
00911     // Clear Running Events
00912     stmi2c_clear_pending_interrupts(i2c);
00913 
00914     // Mark this as a special error
00915     periph->errors->last_unexpected_event++;
00916 
00917     // Document the current Status
00918     periph->status = I2CIdle;
00919 
00920     // There are no transactions anymore: return
00921     // further-on in this routine we need a transaction pointer: so we are not allowed to continue
00922     return;
00923   }
00924 
00925   // get the I2C transaction we were working on ...
00926 
00927   enum STMI2CSubTransactionStatus ret = 0;
00928   struct i2c_transaction* trans = periph->trans[periph->trans_extract_idx];
00929 
00931   // If there was an error:
00932   if (( I2C_SR1(i2c) & I2C_SR1_ERR_MASK ) != 0x0000)
00933   {
00934 
00935 #ifdef I2C_DEBUG_LED
00936         LED1_ON();
00937         LED2_ON();
00938     LED1_OFF();
00939     LED2_OFF();
00940 
00941         LED_SHOW_ACTIVE_BITS(regs);
00942 #endif
00943 
00944     // Notify everyone about the error ...
00945 
00946     // Set result in transaction
00947     trans->status = I2CTransFailed;
00948 
00949     // Document the current Status
00950     periph->status = I2CFailed;
00951 
00952     // Make sure a TxRx does not Restart
00953     trans->type = I2CTransRx;
00954 
00955     // Count The Errors
00956     i2c_error(periph);
00957 
00958     // Clear Running Events
00959     stmi2c_clear_pending_interrupts(i2c);
00960 
00961     // Now continue as if everything was normal from now on
00962     ret = STMI2C_SubTra_Ready;
00963 
00964   }
00965 
00967   // Normal Event:
00968   else
00969   {
00970 
00973     //
00974     //  SUB-TRANSACTION HANDLER
00975 
00976     if (trans->type == I2CTransRx) // TxRx are converted to Rx after the Tx Part
00977     {
00978       switch (trans->len_r)
00979       {
00980         case 1:
00981           ret = stmi2c_read1(i2c,periph,trans);
00982           break;
00983         case 2:
00984           ret = stmi2c_read2(i2c,periph,trans);
00985           break;
00986         default:
00987           ret = stmi2c_readmany(i2c,periph,trans);
00988           break;
00989       }
00990     }
00991     else // TxRx or Tx
00992     {
00993       ret = stmi2c_send(i2c,periph,trans);
00994     }
00995   }
00996 
00998   // Sub-transaction has finished
00999   if (ret != STMI2C_SubTra_Busy)
01000   {
01001     // Ready or SubTraError
01002     // -ready: with or without stop already asked
01003 
01004     // In case of unexpected event condition during subtransaction handling:
01005     if (ret == STMI2C_SubTra_Error)
01006     {
01007       // Tell everyone about the subtransaction error:
01008       // this is the previously called SPURRIOUS INTERRUPT
01009       periph->status = I2CFailed;
01010       trans->type = I2CTransRx;   // Avoid possible restart
01011       trans->status = I2CTransFailed; // Notify Ready
01012       periph->errors->unexpected_event_cnt++;
01013 
01014     // Error
01015 #ifdef I2C_DEBUG_LED
01016         LED2_ON();
01017         LED1_ON();
01018     LED2_OFF();
01019     LED1_OFF();
01020 
01021         LED_SHOW_ACTIVE_BITS(regs);
01022 #endif
01023 
01024       // Clear Running Events
01025       stmi2c_clear_pending_interrupts(i2c);
01026     }
01027 
01028     // RxTx -> Restart and do Rx part
01029     if (trans->type == I2CTransTxRx)
01030     {
01031       trans->type = I2CTransRx;
01032       periph->status = I2CStartRequested;
01033       I2C_CR1(i2c) |= I2C_CR1_START;
01034 
01035       // Silent any BTF that would occur before SB
01036       I2C_DR(i2c) = 0x00;
01037     }
01038     // If a restart is not needed: Rx part or Tx-only
01039     else
01040     {
01041       // Ready, no stop condition set yet
01042       if (ret == STMI2C_SubTra_Ready)
01043       {
01044 
01045         // Program a stop
01046         PPRZ_I2C_SEND_STOP(i2c);
01047 
01048         // Silent any BTF that would occur before STOP is executed
01049         I2C_DR(i2c) = 0x00;
01050       }
01051 
01052       // Jump to the next transaction
01053       periph->trans_extract_idx++;
01054       if (periph->trans_extract_idx >= I2C_TRANSACTION_QUEUE_LEN)
01055         periph->trans_extract_idx = 0;
01056 
01057       // Tell everyone we are ready
01058       periph->status = I2CIdle;
01059 
01060 
01061       // if we have no more transaction to process, stop here
01062       if (periph->trans_extract_idx == periph->trans_insert_idx)
01063       {
01064 
01065 #ifdef I2C_DEBUG_LED
01066         LED2_ON();
01067         LED1_ON();
01068     LED1_OFF();
01069         LED1_ON();
01070     LED1_OFF();
01071     LED2_OFF();
01072 #endif
01073       }
01074       // if not, start next transaction
01075       else
01076       {
01077         // Restart transaction doing the Rx part now
01078 // --- moved to idle function
01079         PPRZ_I2C_SEND_START(periph);
01080 // ------
01081      }
01082     }
01083   }
01084 
01085   return;
01086 }

01260 // Set Bitrate to Match your application:
01261 // -short wires, low capacitance bus: IMU: high speed
01262 // -long wires with a lot of capacitance: motor controller: put speed as low as possible
01263 
01264 void i2c_setbitrate(struct i2c_periph *periph, int bitrate)
01265 {
01266   // If NOT Busy
01267   if (i2c_idle(periph))
01268   {
01269     volatile int devider;
01270     volatile int risetime;
01271 
01272     u32 i2c = (u32) periph->reg_addr;
01273 
01274 /*****************************************************
01275     Bitrate:
01276 
01277     -CR2 + CCR + TRISE registers
01278     -only change when PE=0
01279 
01280     e.g.
01281 
01282     10kHz:  36MHz + Standard 0x708 + 0x25
01283     70kHz:  36MHz + Standard 0x101 +
01284     400kHz: 36MHz + Fast 0x1E      + 0xb
01285 
01286       // 1) Program peripheral input clock CR2: to get correct timings
01287       // 2) Configure clock control registers
01288       // 3) Configure rise time register
01289 ******************************************************/
01290 
01291     if (bitrate < 3000)
01292       bitrate = 3000;
01293 
01294     // 36MHz, fast scl: 2counts low 1 count high -> / 3:
01295     devider = 18000 / (bitrate/1000);
01296 
01297     // never allow faster than 600kbps
01298     if (devider < 20)
01299       devider = 20;
01300 
01301     // no overflow either
01302     if (devider >=4095)
01303       devider = 4095;
01304 
01305     // risetime can be up to 1/6th of the period
01306     risetime = 1000000 / (bitrate/1000) / 6 / 28;
01307 
01308     if (risetime < 10)
01309       risetime = 10;
01310 
01311     // more will overflow the register: for more you should lower the FREQ
01312     if (risetime >=31)
01313       risetime = 31;
01314 
01315     // we do not expect an interrupt as the interface should have been idle, but just in case...
01316     __disable_irq(); // this code is in user space:
01317 
01318     // CCR can only be written when PE is disabled
01319      // p731 note 5
01320     I2C_CR1(i2c) &= ~ I2C_CR1_PE;
01321 
01322     // 1)
01323     I2C_CR2(i2c) = 0x0324;
01324     // 2)
01325     //I2C_CCR(i2c) = 0x8000 + devider;
01326     I2C_CCR(i2c) = 0x0000 + devider;
01327     // 3)
01328     I2C_TRISE(i2c) = risetime;
01329 
01330     // Re-Enable
01331     I2C_CR1(i2c) |=   I2C_CR1_PE;
01332 
01333     __enable_irq();
01334 
01335 #ifdef I2C_DEBUG_LED
01336         __disable_irq(); // this code is in user space:
01337 
01338         LED2_ON();
01339         LED1_ON();
01340         LED2_OFF();
01341         LED1_OFF();
01342         LED2_ON();
01343         LED1_ON();
01344         LED2_OFF();
01345         LED1_OFF();
01346 
01347         __enable_irq();
01348 #endif
01349 
01350    }
01351 }
