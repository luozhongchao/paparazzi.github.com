00084 void ADS8344_init( void ) {
00085   channel = 0;
00086   ADS8344_available = FALSE;
00087 
00088   /* setup pins for SSP (SCK, MISO, MOSI) */
00089   PINSEL1 |= 2 << 2 | 2 << 4 | 2 << 6;
00090 
00091   /* setup SSP */
00092   SSPCR0 = SSP_DSS | SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR;
00093   SSPCR1 = SSP_LBM | SSP_MS | SSP_SOD;
00094   SSPCPSR = CPSDVSR; /* -> 50kHz */
00095 
00096   /* initialize interrupt vector */
00097   VICIntSelect &= ~VIC_BIT(VIC_SPI1);   // SPI1 selected as IRQ
00098   VICIntEnable = VIC_BIT(VIC_SPI1);     // SPI1 interrupt enabled
00099   VICVectCntl7 = VIC_ENABLE | VIC_SPI1;
00100   VICVectAddr7 = (uint32_t)SPI1_ISR;    // address of the ISR
00101 
00102   /* setup slave select */
00103   /* configure SS pin */
00104   SetBit( ADS8344_SS_IODIR,  ADS8344_SS_PIN);  /* pin is output  */
00105   ADS8344Unselect();                           /* pin low        */
00106 }
