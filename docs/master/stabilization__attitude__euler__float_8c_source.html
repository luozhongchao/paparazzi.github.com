00100 void stabilization_attitude_run(bool_t  in_flight) {
00101 
00102   stabilization_attitude_ref_update();
00103 
00104   /* Compute feedforward */
00105   stabilization_att_ff_cmd[COMMAND_ROLL] =
00106     stabilization_gains.dd.x * stab_att_ref_accel.p / 32.;
00107   stabilization_att_ff_cmd[COMMAND_PITCH] =
00108     stabilization_gains.dd.y * stab_att_ref_accel.q / 32.;
00109   stabilization_att_ff_cmd[COMMAND_YAW] =
00110     stabilization_gains.dd.z * stab_att_ref_accel.r / 32.;
00111 
00112   /* Compute feedback                  */
00113   /* attitude error            */
00114   struct FloatEulers att_float;
00115   EULERS_FLOAT_OF_BFP(att_float, ahrs.ltp_to_body_euler);
00116   struct FloatEulers att_err;
00117   EULERS_DIFF(att_err, stab_att_ref_euler, att_float);
00118   FLOAT_ANGLE_NORMALIZE(att_err.psi);
00119 
00120   if (in_flight) {
00121     /* update integrator */
00122     EULERS_ADD(stabilization_att_sum_err, att_err);
00123     EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
00124   }
00125   else {
00126     FLOAT_EULERS_ZERO(stabilization_att_sum_err);
00127   }
00128 
00129   /*  rate error                */
00130   struct FloatRates rate_float;
00131   RATES_FLOAT_OF_BFP(rate_float, ahrs.body_rate);
00132   struct FloatRates rate_err;
00133   RATES_DIFF(rate_err, stab_att_ref_rate, rate_float);
00134 
00135   /*  PID                  */
00136 
00137   stabilization_att_fb_cmd[COMMAND_ROLL] =
00138     stabilization_gains.p.x  * att_err.phi +
00139     stabilization_gains.d.x  * rate_err.p +
00140     stabilization_gains.i.x  * stabilization_att_sum_err.phi / 1024.;
00141 
00142   stabilization_att_fb_cmd[COMMAND_PITCH] =
00143     stabilization_gains.p.y  * att_err.theta +
00144     stabilization_gains.d.y  * rate_err.q +
00145     stabilization_gains.i.y  * stabilization_att_sum_err.theta / 1024.;
00146 
00147   stabilization_att_fb_cmd[COMMAND_YAW] =
00148     stabilization_gains.p.z  * att_err.psi +
00149     stabilization_gains.d.z  * rate_err.r +
00150     stabilization_gains.i.z  * stabilization_att_sum_err.psi / 1024.;
00151 
00152 
00153   stabilization_cmd[COMMAND_ROLL] =
00154     (stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL])/16.;
00155   stabilization_cmd[COMMAND_PITCH] =
00156     (stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH])/16.;
00157   stabilization_cmd[COMMAND_YAW] =
00158     (stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW])/16.;
00159 
00160 }
