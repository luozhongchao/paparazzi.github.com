00058 void ahrs_aligner_run(void) {
00059 
00060   RATES_ADD(gyro_sum,  imu.gyro);
00061   VECT3_ADD(accel_sum, imu.accel);

00063 
00064   ref_sensor_samples[samples_idx] = imu.accel.z;
00065   samples_idx++;
00066 
00067 #ifdef AHRS_ALIGNER_LED
00068   RunOnceEvery(50, {LED_TOGGLE(AHRS_ALIGNER_LED);});
00069 #endif
00070 
00071   if (samples_idx >= SAMPLES_NB) {
00072     int32_t avg_ref_sensor = accel_sum.z;
00073     if ( avg_ref_sensor >= 0)
00074       avg_ref_sensor += SAMPLES_NB / 2;
00075     else
00076       avg_ref_sensor -= SAMPLES_NB / 2;
00077     avg_ref_sensor /= SAMPLES_NB;
00078 
00079     ahrs_aligner.noise = 0;
00080     int i;
00081     for (i=0; i<SAMPLES_NB; i++) {
00082       int32_t diff = ref_sensor_samples[i] - avg_ref_sensor;
00083       ahrs_aligner.noise += abs(diff);
00084     }
00085 
00086     RATES_SDIV(ahrs_aligner.lp_gyro,  gyro_sum,  SAMPLES_NB);
00087     VECT3_SDIV(ahrs_aligner.lp_accel, accel_sum, SAMPLES_NB);

00089 
00090     INT_RATES_ZERO(gyro_sum);
00091     INT_VECT3_ZERO(accel_sum);

00093     samples_idx = 0;
00094 
00095     if (ahrs_aligner.noise < LOW_NOISE_THRESHOLD)
00096       ahrs_aligner.low_noise_cnt++;

00098       if ( ahrs_aligner.low_noise_cnt > 0)
00099         ahrs_aligner.low_noise_cnt--;
00100 
00101     if (ahrs_aligner.low_noise_cnt > LOW_NOISE_TIME) {
00102       ahrs_aligner.status = AHRS_ALIGNER_LOCKED;
00103 #ifdef AHRS_ALIGNER_LED
