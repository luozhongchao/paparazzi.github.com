00186 void stabilization_attitude_run(bool_t enable_integrator) {
00187 
00188   /*
00189    * Update reference
00190    */
00191   stabilization_attitude_ref_update();
00192 
00193   /*
00194    * Compute errors for feedback
00195    */
00196 
00197   /* attitude error                          */
00198   struct FloatQuat att_err;
00199   FLOAT_QUAT_INV_COMP(att_err, ahrs_float.ltp_to_body_quat, stab_att_ref_quat);
00200   /* wrap it in the shortest direction       */
00201   FLOAT_QUAT_WRAP_SHORTEST(att_err);
00202 
00203   /*  rate error                */
00204   struct FloatRates rate_err;
00205   RATES_DIFF(rate_err, stab_att_ref_rate, ahrs_float.body_rate);
00206 
00207   /* integrated error */
00208   if (enable_integrator) {
00209     struct FloatQuat new_sum_err, scaled_att_err;
00210     /* update accumulator */
00211     scaled_att_err.qi = att_err.qi;
00212     scaled_att_err.qx = att_err.qx / IERROR_SCALE;
00213     scaled_att_err.qy = att_err.qy / IERROR_SCALE;
00214     scaled_att_err.qz = att_err.qz / IERROR_SCALE;
00215     FLOAT_QUAT_COMP(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
00216     FLOAT_QUAT_NORMALIZE(new_sum_err);
00217     FLOAT_QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
00218     FLOAT_EULERS_OF_QUAT(stabilization_att_sum_err_eulers, stabilization_att_sum_err_quat);
00219   } else {
00220     /* reset accumulator */
00221     FLOAT_QUAT_ZERO( stabilization_att_sum_err_quat );
00222     FLOAT_EULERS_ZERO( stabilization_att_sum_err_eulers );
00223   }
00224 
00225   attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains[gain_idx], &stab_att_ref_accel);
00226 
00227   attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains[gain_idx], &att_err, &rate_err, &ahrs_float.body_rate_d, &stabilization_att_sum_err_quat);
00228 
00229   // FIXME: this is very dangerous! only works if this really includes all commands
00230   for (int i = COMMAND_ROLL; i <= COMMAND_YAW_SURFACE; i++) {
00231     stabilization_cmd[i] = stabilization_att_fb_cmd[i]+stabilization_att_ff_cmd[i];
00232   }
