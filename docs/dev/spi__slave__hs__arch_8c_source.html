00097 void spi_init(void) {
00098 
00099   /* setup pins for SSP (SCK, MISO, MOSI) */
00100   PINSEL1 |= SSP_PINSEL1_SCK  | SSP_PINSEL1_MISO | SSP_PINSEL1_MOSI | SSP_PINSEL1_SSEL;
00101 
00102   /* setup SSP */
00103   // Control Registers
00104   SSPCR0 = SSPCR0_VAL;
00105   SSPCR1 = SSPCR1_VAL;
00106   // Clock Prescale Registers
00107   SSPCPSR = CPSDVSR;
00108 
00109   /* initialize interrupt vector */
00110   VICIntSelect &= ~VIC_BIT( VIC_SPI1 );             /* SPI1 selected as IRQ */
00111   VICIntEnable = VIC_BIT( VIC_SPI1 );               /* enable it            */
00112   _VIC_CNTL(SSP_VIC_SLOT) = VIC_ENABLE | VIC_SPI1;
00113   _VIC_ADDR(SSP_VIC_SLOT) = (uint32_t)SSP_ISR;      /* address of the ISR   */
00114 
00115 
00116   // Enable SPI Slave
00117   SetBit(SSPCR1, SSE);
00118 
00119   // Enable Receive interrupt
00120   SetBit(SSPIMSC, RXIM);
00121 
00122 }
