00332 void Normalize(void)
00333 {
00334   float error=0;
00335   float temporary[3][3];
00336   float renorm=0;
00337   uint8_t problem=FALSE;
00338 
00339   // Find the non-orthogonality of X wrt Y
00340   error= -Vector_Dot_Product(&DCM_Matrix[0][0],&DCM_Matrix[1][0])*.5; //eq.19
00341 
00342   // Add half the XY error to X, and half to Y
00343   Vector_Scale(&temporary[0][0], &DCM_Matrix[1][0], error);           //eq.19
00344   Vector_Scale(&temporary[1][0], &DCM_Matrix[0][0], error);           //eq.19
00345   Vector_Add(&temporary[0][0], &temporary[0][0], &DCM_Matrix[0][0]);  //eq.19
00346   Vector_Add(&temporary[1][0], &temporary[1][0], &DCM_Matrix[1][0]);  //eq.19
00347 
00348   // The third axis is simply set perpendicular to the first 2. (there is not correction of XY based on Z)
00349   Vector_Cross_Product(&temporary[2][0],&temporary[0][0],&temporary[1][0]); // c= a x b //eq.20
00350 
00351   // Normalize lenght of X
00352   renorm= Vector_Dot_Product(&temporary[0][0],&temporary[0][0]);
00353   // a) if norm is close to 1, use the fast 1st element from the tailer expansion of SQRT
00354   // b) if the norm is further from 1, use a real sqrt
00355   // c) norm is huge: disaster! reset! mayday!
00356   if (renorm < 1.5625f && renorm > 0.64f) {
00357     renorm= .5 * (3-renorm);                                          //eq.21
00358   } else if (renorm < 100.0f && renorm > 0.01f) {
00359     renorm= 1. / sqrt(renorm);
00360 #if PERFORMANCE_REPORTING == 1
00361     renorm_sqrt_count++;
00362 #endif
00363   } else {
00364     problem = TRUE;
00365 #if PERFORMANCE_REPORTING == 1
00366     renorm_blowup_count++;
00367 #endif
00368   }
00369   Vector_Scale(&DCM_Matrix[0][0], &temporary[0][0], renorm);
00370 
00371   // Normalize lenght of Y
00372   renorm= Vector_Dot_Product(&temporary[1][0],&temporary[1][0]);
00373   if (renorm < 1.5625f && renorm > 0.64f) {
00374     renorm= .5 * (3-renorm);                                                 //eq.21
00375   } else if (renorm < 100.0f && renorm > 0.01f) {
00376     renorm= 1. / sqrt(renorm);
00377 #if PERFORMANCE_REPORTING == 1
00378     renorm_sqrt_count++;
00379 #endif
00380   } else {
00381     problem = TRUE;
00382 #if PERFORMANCE_REPORTING == 1
00383     renorm_blowup_count++;
00384 #endif
00385   }
00386   Vector_Scale(&DCM_Matrix[1][0], &temporary[1][0], renorm);
00387 
00388   // Normalize lenght of Z
00389   renorm= Vector_Dot_Product(&temporary[2][0],&temporary[2][0]);
00390   if (renorm < 1.5625f && renorm > 0.64f) {
00391     renorm= .5 * (3-renorm);                                                 //eq.21
00392   } else if (renorm < 100.0f && renorm > 0.01f) {
00393     renorm= 1. / sqrt(renorm);
00394 #if PERFORMANCE_REPORTING == 1
00395     renorm_sqrt_count++;
00396 #endif
00397   } else {
00398     problem = TRUE;
00399 #if PERFORMANCE_REPORTING == 1
00400     renorm_blowup_count++;
00401 #endif
00402   }
00403   Vector_Scale(&DCM_Matrix[2][0], &temporary[2][0], renorm);
00404 
00405   // Reset on trouble
00406   if (problem) {                // Our solution is blowing up and we will force back to initial condition.  Hope we are not upside down!
00407     set_dcm_matrix_from_rmat(&ahrs_impl.body_to_imu_rmat);
00408     problem = FALSE;
00409   }
00410 }

00413 void Drift_correction(void)
00414 {
00415   //Compensation the Roll, Pitch and Yaw drift.
00416   static float Scaled_Omega_P[3];
00417   static float Scaled_Omega_I[3];
00418   float Accel_magnitude;
00419   float Accel_weight;
00420   float Integrator_magnitude;
00421 
00422   // Local Working Variables
00423   float errorRollPitch[3];
00424   float errorYaw[3];
00425   float errorCourse;
00426 
00427   //*****Roll and Pitch***************
00428 
00429   // Calculate the magnitude of the accelerometer vector
00430   Accel_magnitude = sqrt(accel_float.x*accel_float.x + accel_float.y*accel_float.y + accel_float.z*accel_float.z);
00431   Accel_magnitude = Accel_magnitude / GRAVITY; // Scale to gravity.
00432   // Dynamic weighting of accelerometer info (reliability filter)
00433   // Weight for accelerometer info (<0.5G = 0.0, 1G = 1.0 , >1.5G = 0.0)
00434   Accel_weight = Chop(1 - 2*fabs(1 - Accel_magnitude),0,1);  //
00435 
00436 #if USE_HIGH_ACCEL_FLAG
00437   // Test for high acceleration:
00438   //  - low speed
00439   //  - high thrust
00440   if (estimator_hspeed_mod < HIGH_ACCEL_LOW_SPEED && ap_state->commands[COMMAND_THROTTLE] > HIGH_ACCEL_HIGH_THRUST && !high_accel_done) {
00441     high_accel_flag = TRUE;
00442   } else {
00443     high_accel_flag = FALSE;
00444     if (estimator_hspeed_mod > HIGH_ACCEL_LOW_SPEED && !high_accel_done) {
00445       high_accel_done = TRUE; // After takeoff, don't use high accel before landing (GS small, Throttle small)
00446     }
00447     if (estimator_hspeed_mod < HIGH_ACCEL_HIGH_THRUST_RESUME && ap_state->commands[COMMAND_THROTTLE] < HIGH_ACCEL_HIGH_THRUST_RESUME) {
00448       high_accel_done = FALSE; // Activate high accel after landing
00449     }
00450   }
00451   if (high_accel_flag) { Accel_weight = 0.; }
00452 #endif
00453 
00454 
00455   #if PERFORMANCE_REPORTING == 1
00456   {
00457 
00458     float tempfloat = ((Accel_weight - 0.5) * 256.0f);    //amount added was determined to give imu_health a time constant about twice the time constant of the roll/pitch drift correction
00459     imu_health += tempfloat;
00460     Bound(imu_health,129,65405);
00461   }
00462   #endif
00463 
00464   Vector_Cross_Product(&errorRollPitch[0],&accel_float.x,&DCM_Matrix[2][0]); //adjust the ground of reference
00465   Vector_Scale(&Omega_P[0],&errorRollPitch[0],Kp_ROLLPITCH*Accel_weight);
00466 
00467   Vector_Scale(&Scaled_Omega_I[0],&errorRollPitch[0],Ki_ROLLPITCH*Accel_weight);
00468   Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);
00469 
00470   //*****YAW***************
00471 
00472 #if USE_MAGNETOMETER
00473   // We make the gyro YAW drift correction based on compass magnetic heading
00474 //  float mag_heading_x = cos(MAG_Heading);
00475 //  float mag_heading_y = sin(MAG_Heading);
00476   // 2D dot product
00477   errorCourse=(DCM_Matrix[0][0]*MAG_Heading_Y) + (DCM_Matrix[1][0]*MAG_Heading_X);  //Calculating YAW error
00478   Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00479 
00480   Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW);
00481   Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.
00482 
00483   Vector_Scale(&Scaled_Omega_I[0],&errorYaw[0],Ki_YAW);
00484   Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);//adding integrator to the Omega_I
00485 
00486 #elif USE_GPS // Use GPS Ground course to correct yaw gyro drift
00487 
00488   if(gps.fix == GPS_FIX_3D && gps.gspeed>= 500) { //got a 3d fix and ground speed is more than 0.5 m/s
00489     float ground_course = ((float)gps.course)/1.e7 - M_PI; //This is the runaway direction of you "plane" in rad
00490     float COGX = cosf(ground_course); //Course overground X axis
00491     float COGY = sinf(ground_course); //Course overground Y axis
00492 
00493     errorCourse=(DCM_Matrix[0][0]*COGY) - (DCM_Matrix[1][0]*COGX);  //Calculating YAW error
00494     Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00495 
00496     Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW);
00497     Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.
00498 
00499     Vector_Scale(&Scaled_Omega_I[0],&errorYaw[0],Ki_YAW);
00500     Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);//adding integrator to the Omega_I
00501   }
00502 #endif
00503 
00504   //  Here we will place a limit on the integrator so that the integrator cannot ever exceed half the saturation limit of the gyros
