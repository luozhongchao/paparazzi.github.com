00108     /* LPC2141, 32k Flash, 8k RAM */

00110     /* LPC2142, 64k Flash, 16k RAM */
00111     case 0x0402FF11:

00113     case 0x0402FF12:

00200 static int32_t pflash_program_bytes(FlashInfo* flash,
00201                                     uint32_t src,
00202                                     uint32_t size,
00203                                     uint32_t chksum) {
00204   uint32_t data[BOUND/4], i, j, ret;
00205   uint32_t ptr = (uint32_t) &data;
00206 
00207   /* erase */
00208   if ((ret = pflash_erase_page(flash))) return ret;
00209 
00210   /* write data in arrays */
00211   for (i=0; i<size; i+=BOUND) {
00212     /* copy data to aligned memory */
00213     for (j=0; j<BOUND; j++) {
00214       *(uint8_t*) (ptr+j) = *(uint8_t*) (src+i+j);
00215     }
00216     if (i == flash->page_size - BOUND) {
00217       data[(BOUND-FSIZ)/4] = size;
00218       data[(BOUND-FCHK)/4] = chksum;
00219     }
00220     if ((ret = pflash_program_array(flash, flash->addr+i, ptr))) return ret;
00221   }
00222 
00223   /* last array */
00224   if (i <= flash->page_size - BOUND) {
00225     data[(BOUND-FSIZ)/4] = size;
00226     data[(BOUND-FCHK)/4] = chksum;
00227     if ((ret = pflash_program_array(flash,
00228                                     flash->addr+flash->page_size-BOUND,
00229                                     ptr)))
00230       return ret;
00231   }
00232 
00233   /* verify data */
00234   for (i=0; i<size; i++) {
00235     if ((*(uint8_t*) (flash->addr+i)) != (*(uint8_t*) (src+i))) return -2;
00236   }
00237   if (*(uint32_t*) (flash->addr+flash->page_size-FSIZ) != size) return -3;
00238   if (*(uint32_t*) (flash->addr+flash->page_size-FCHK) != chksum) return -4;
00239 
00240   return 0;
