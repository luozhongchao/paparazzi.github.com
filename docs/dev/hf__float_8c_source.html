00126 /* compute the mean of the last n accel measurements */
00127 static inline void b2_hff_compute_accel_body_mean(uint8_t n) {
00128   struct Int32Vect3 sum;
00129   int i, j;
00130 
00131   INT_VECT3_ZERO(sum);
00132 
00133   if (n > 1) {
00134     if (n > acc_body.n) {
00135       n = acc_body.n;
00136     }
00137     for (i = 1; i <= n; i++) {
00138       j = (acc_body.w - i) > 0 ? (acc_body.w - i) : (acc_body.w - i + acc_body.size);
00139       VECT3_ADD(sum, acc_body.buf[j]);
00140     }
00141   VECT3_SDIV(acc_body_mean, sum, n);
00142   } else {
00143   VECT3_COPY(acc_body_mean, sum);
00144   }

00421 void b2_hff_propagate(void) {
00422   if (b2_hff_lost_counter < b2_hff_lost_limit)
00423     b2_hff_lost_counter++;
00424 
00425 #ifdef GPS_LAG
00426   /* continue re-propagating to catch up with the present */
00427   if (b2_hff_rb_last->rollback) {
00428   b2_hff_propagate_past(b2_hff_rb_last);
00429   }
00430 #endif
00431 
00432   /* store body accelerations for mean computation */
00433   b2_hff_store_accel_body();
00434 
00435   /* propagate current state if it is time */
00436   if (b2_hff_ps_counter == HFF_PRESCALER) {
00437   b2_hff_ps_counter = 1;
00438 
00439     if (b2_hff_lost_counter < b2_hff_lost_limit) {
00440       /* compute float ltp mean acceleration */
00441       b2_hff_compute_accel_body_mean(HFF_PRESCALER);
00442       struct Int32Vect3 mean_accel_ltp;
00443       INT32_RMAT_TRANSP_VMULT(mean_accel_ltp, ahrs.ltp_to_body_rmat, acc_body_mean);
00444       b2_hff_xdd_meas = ACCEL_FLOAT_OF_BFP(mean_accel_ltp.x);
00445       b2_hff_ydd_meas = ACCEL_FLOAT_OF_BFP(mean_accel_ltp.y);
00446 #ifdef GPS_LAG
00447       b2_hff_store_accel_ltp(b2_hff_xdd_meas, b2_hff_ydd_meas);
00448 #endif
00449 
00450       /*
00451        * propagate current state
00452        */
00453       b2_hff_propagate_x(&b2_hff_state);
00454       b2_hff_propagate_y(&b2_hff_state);
00455 
00456       /* update ins state from horizontal filter */
00457       ins_ltp_accel.x = ACCEL_BFP_OF_REAL(b2_hff_state.xdotdot);
00458       ins_ltp_accel.y = ACCEL_BFP_OF_REAL(b2_hff_state.ydotdot);
00459       ins_ltp_speed.x = SPEED_BFP_OF_REAL(b2_hff_state.xdot);

00463 
00464 #ifdef GPS_LAG

00467         b2_hff_rb_last->lag_counter++;

00471         PRINT_DBG(1, ("save current state\n"));
00472         b2_hff_rb_put_state(&b2_hff_state);

00476       }
00477 #endif
00478     }
00479   } else {

00481   }

00533       past_save_counter = GPS_DT_N-1;// + lag_counter_err;

00535       b2_hff_propagate_past(b2_hff_rb_last);
00536     } else if (lag_counter_err >= GPS_DT_N - (GPS_LAG_TOL_N+1)) {

00538       PRINT_DBG(2, ("try next saved state\n"));
00539       b2_hff_rb_drop_last();
00540       b2_hff_update_gps();
00541     }
00542   } else if (save_counter < 0) {
00543     /* ringbuffer empty -> save output filter state at next GPS validity point in time */
00544     save_counter = GPS_DT_N-1 - (GPS_LAG_N % GPS_DT_N);
00545     PRINT_DBG(2, ("rb empty, save counter set: %d\n", save_counter));
00546   }

00548 #endif /* GPS_LAG */

00568  *
00569  * Propagation
00570  *
00571  *
00572 
00573   F = [ 1 dt
00574         0  1 ];
00575 
00576   B = [ dt^2/2 dt]';
00577 
00578   Q = [ 0.01  0
00579         0     0.01];
00580 
00581   Xk1 = F * Xk0 + B * accel;
00582 
00583   Pk1 = F * Pk0 * F' + Q;
00584 
00585 */
00586 static inline void b2_hff_propagate_x(struct HfilterFloat* hff_work) {
00587   /* update state */
00588   hff_work->xdotdot = b2_hff_xdd_meas;
00589   hff_work->x = hff_work->x + DT_HFILTER * hff_work->xdot;
00590   hff_work->xdot = hff_work->xdot + DT_HFILTER * hff_work->xdotdot;
00591   /* update covariance */
00592   const float FPF00 = hff_work->xP[0][0] + DT_HFILTER * ( hff_work->xP[1][0] + hff_work->xP[0][1] + DT_HFILTER * hff_work->xP[1][1] );
00593   const float FPF01 = hff_work->xP[0][1] + DT_HFILTER * hff_work->xP[1][1];
00594   const float FPF10 = hff_work->xP[1][0] + DT_HFILTER * hff_work->xP[1][1];
00595   const float FPF11 = hff_work->xP[1][1];

00597   hff_work->xP[0][0] = FPF00 + Q;
00598   hff_work->xP[0][1] = FPF01;
00599   hff_work->xP[1][0] = FPF10;
00600   hff_work->xP[1][1] = FPF11 + Qdotdot;
