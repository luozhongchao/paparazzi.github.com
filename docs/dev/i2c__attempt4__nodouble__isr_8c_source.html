00350   }
00351   // Address Was Sent
00352   else if (BIT_X_IS_SET_IN_REG(I2C_SR1_BIT_ADDR, SR1) )

00369       periph->idx_buf = 1;
00370     }
00371

00600     // Now the last byte is being clocked. Stop in MUST be set BEFORE the transfer of the last byte is complete
00601     PPRZ_I2C_SEND_STOP(regs);
00602 
00603     __I2C_REG_CRITICAL_ZONE_STOP;
00604     // --- end of critical zone -----------
00605 
00606     // read the byte2 we had in the buffer (BTF means 2 bytes available)
00607     trans->buf[periph->idx_buf] = regs->DR;
00608     periph->idx_buf ++;
00609 
00610     // Ask for an interrupt to read the last byte (which is normally still busy now)
00611     // The last byte will be received with RXNE
00612     regs->CR2 |= I2C_CR2_BIT_ITBUFEN;
00613   }
00614   else // Hardware error
00615   {
00616     // Error
00617 #ifdef I2C_DEBUG_LED
00618         LED2_ON();
00619         LED1_ON();
00620   LED2_OFF();
00621   LED1_OFF();
00622 #endif
00623     return STMI2C_SubTra_Error;
00624   }
00625 
00626   return STMI2C_SubTra_Busy;
00627 }

00672 static inline void i2c_irq(struct i2c_periph *periph)
00673 {
00674 
00675   /*
00676   There are 7 possible reasons to get here:
00677 
00678   If IT_EV_FEN
00679   -------------------------
00680 
00681   We are always interested in all IT_EV_FEV: all are required.
00682 
00683   1) SB   // Start Condition Success in Master mode
00684   2) ADDR   // Address sent received Acknoledge
00685   [3 ADDR10]  // -- 10bit address stuff
00686   [4 STOPF] // -- only for slaves: master has no stop interrupt
00687   5) BTF    // I2C has stopped working (it is waiting for new data, all buffers are tx_empty/rx_full)
00688 
00689   // Beware: using the buffered I2C has some interesting properties:
00690     -in master receive mode: BTF only occurs after the 2nd received byte: after the first byte is received it is
00691            in RD but the I2C can still receive a second byte. Only when the 2nd byte is received while the RxNE is 1
00692      then a BTF occurs (I2C can not continue receiving bytes or they will get lost). During BTF I2C is halted (SCL held low)
00693     -in master transmitmode: when writing a byte to WD, you instantly get a new TxE interrupt while the first is not
00694      transmitted yet. The byte was pushed to the I2C shift register and the buffer is ready for more. You can already
00695      fill new data in the buffer while the first is still being transmitted for max performance transmission.
00696 
00697         // Beware: besides data buffering you can/must plan several consecutive actions. You can send 2 bytes to the buffer, ask for a stop and
00698            a new start in one go.
00699 
00700           -thanks to / because of this buffering and event sheduling there is not 1 interrupt per start / byte / stop
00701            This also means you must think more in advance and a transaction could be popped from the stack even before it is
00702            actually completely transmitted. But then you would not know the result yet so you have to keep it until the result
00703            is known.
00704 
00705   // Beware: the order in which Status is read determines how flags are cleared. You should not just read SR1 & SR2 every time
00706 
00707   If IT_EV_FEN AND IT_EV_BUF
00708   --------------------------
00709 
00710   Buffer event are not always wanted and are tipically switched on during longer data transfers. Make sure to turn off in time.
00711 
00712   6) RxNE
00713   7) TxE
00714 
00715   --------------------------------------------------------------------------------------------------
00716   // This driver uses only a subset of the pprz_i2c_states for several reasons:
00717   // -we have less interrupts than the I2CStatus states (for efficiency)
00718   // -STM32 has such a powerfull I2C engine with plenty of status register flags that
00719             only little extra status information needs to be stored.
00720 
00721        // Status is re-used (abused) to remember the last COMMAND THAT WAS SENT to the STM I2C hardware.
00722 
00723 // TODO: check which are used
00724   enum I2CStatus {
00725     I2CIdle,      // No more last command
00726 
00727     I2CStartRequested,    // Last command was start
00728     I2CRestartRequested,    // Last command was restart
00729     I2CStopRequested,   // Very important to not send double stop conditions
00730 
00731     I2CSendingByte,   // Some address/data operation
00732 
00733     // Following are not used
00734     I2CReadingByte,
00735     I2CAddrWrSent,
00736     I2CAddrRdSent,
00737     I2CSendingLastByte,
00738     I2CReadingLastByte,
00739     I2CComplete,
00740     I2CFailed
00741   };
00742 
00743   ---------
00744 
00745   The STM waits indefinately (holding SCL low) for user interaction:
00746   a) after a master-start (waiting for address)
00747   b) after an address (waiting for data)
00748      not during data sending when using buffered
00749   c) after the last byte is transmitted (waiting for either stop or restart)
00750      not during data receiving when using buffered
00751      not after the last byte is received
00752 
00753   -The STM I2C stalls indefinately when a stop condition was attempted that
00754   did not succeed. The BUSY flag remains on.
00755         -There is no STOP interrupt: use needs another way to finish.
00756 
00757    */
00758 
00759 
00761   // Reading the status:
00762   // - Caution: this clears several flags and can start transmissions etc...
00763   // - Certain flags like STOP / (N)ACK need to be guaranteed to be set before
00764   //   the transmission of the byte is finished. At higher clock rates that can be
00765   //   quite fast: so we allow no other interrupt to be triggered in between
00766   //   reading the status and setting all needed flags
00767 
00768   // Direct Access to the I2C Registers
00769   // Do not read SR2 as it might start the reading while an (n)ack bit might be needed first
00770   I2C_TypeDef *regs = (I2C_TypeDef *) periph->reg_addr;
00771 
00772 #ifdef I2C_DEBUG_LED
00773   LED1_ON();
00774   LED1_OFF();
00775 #endif
00776 
00777 
00780   //
00781   //  TRANSACTION HANDLER
00782 
00783   enum STMI2CSubTransactionStatus ret = 0;
00784 
00786   // Nothing Left To Do
00787   if (periph->trans_extract_idx == periph->trans_insert_idx)
00788   {
00789 #ifdef I2C_DEBUG_LED
00790         LED2_ON();
00791         LED1_ON();
00792   LED2_OFF();
00793   LED1_OFF();
00794 
00795     // no transaction and also an error?
00796     LED_SHOW_ACTIVE_BITS(regs);
00797 #endif
00798 
00799     // If we still get an interrupt but there are no more things to do
00800     // (which can happen if an event was sheduled just before a bus error occurs)
00801     // then its easy: just stop: clear all interrupt generating bits
00802 
00803     // Count The Errors
00804     i2c_error(periph);
00805 
00806     // Clear Running Events
00807     stmi2c_clear_pending_interrupts(regs);
00808 
00809     // Mark this as a special error
00810     periph->errors->unexpected_event_cnt++;
00811 
00812     periph->status = I2CIdle;
00813 
00814     // There are no transactions anymore:
00815     // furtheron we need a transaction pointer: so we are not allowed to continue
00816     return;
00817   }
00818 
00819   struct i2c_transaction* trans = periph->trans[periph->trans_extract_idx];
00820 
00822   // If there was an error:
00823   if (( regs->SR1 & I2C_SR1_BITS_ERR ) != 0x0000)
00824   {
00825 
00826 #ifdef I2C_DEBUG_LED
00827         LED1_ON();
00828         LED2_ON();
00829   LED1_OFF();
00830   LED2_OFF();
00831 
00832         LED_SHOW_ACTIVE_BITS(regs);
00833 #endif
00834 
00835     // Set result in transaction
00836     trans->status = I2CTransFailed;
00837 
00838     // Prepare for next
00839     ret = STMI2C_SubTra_Ready;
00840 
00841     // Make sure a TxRx does not Restart
00842     trans->type = I2CTransRx;
00843 
00844 /*
00845     // There are 2 types of errors: some need a STOP, some better do without: Following will not get an extra stop
00846     if (
00847            // Lost Arbitration
00848            (BIT_X_IS_SET_IN_REG( I2C_SR1_BIT_ERR_ARLO, regs->SR1 ) )
00849            // Buss Error When Master Only
00850         || ((BIT_X_IS_SET_IN_REG( I2C_SR1_BIT_ERR_BUS,  regs->SR1 ) )  &&  (!BIT_X_IS_SET_IN_REG( I2C_SR2_BIT_MSL,  regs->SR2 ) ))
00851         || (BIT_X_IS_SET_IN_REG( I2C_SR1_BIT_ERR_OVR,  regs->SR1 ) )
00852        )
00853     {
00854       ret = STMI2C_SubTra_Error;
00855     }
00856 */
00857 
00858     // Count The Errors
00859     i2c_error(periph);
00860 
00861     // Clear Running Events
00862     stmi2c_clear_pending_interrupts(regs);
00863 
00864   }
00865 
00866 
00868   // Normal Event:
00869   else
00870   {
00871 
00872     if (trans->type == I2CTransRx) // TxRx are converted to Rx after the Tx Part
00873     {
00874       switch (trans->len_r)
00875       {
00876         case 1:
00877           ret = stmi2c_read1(regs,trans);
00878           break;
00879         case 2:
00880           ret = stmi2c_read2(regs,trans);
00881           break;
00882         default:
00883           ret = stmi2c_readmany(regs,periph, trans);
00884           break;
00885       }
00886     }
00887     else // TxRx or Tx
00888     {
00889       ret = stmi2c_send(regs,periph,trans);
00890     }
00891   }
00892 
00894   // Sub-transaction has finished
00895   if (ret != STMI2C_SubTra_Busy)
00896   {
00897     // If a restart is not needed
00898     if (trans->type != I2CTransTxRx)
00899     {
00900       // Ready, no stop condition set yet
00901       if (ret == STMI2C_SubTra_Ready)
00902       {
00903 
00904         // Program a stop
00905         PPRZ_I2C_SEND_STOP(regs);
00906 
00907         // Silent any BTF that would occur before STOP is executed
00908         regs->DR = 0x00;
00909       }
00910 
00911       // In case of unexpected condition: e.g. not slave, no event
00912       if (ret == STMI2C_SubTra_Error)
00913       {
00914 
00915         trans->status = I2CTransFailed;
00916 
00917     // Error
00918 #ifdef I2C_DEBUG_LED
00919         LED2_ON();
00920         LED1_ON();
00921   LED2_OFF();
00922   LED1_OFF();
00923 #endif
00924 
00925         LED_SHOW_ACTIVE_BITS(regs);
00926 
00927         // Clear Running Events
00928         stmi2c_clear_pending_interrupts(regs);
00929 
00930       }
00931 
00932 
00933       // Jump to the next transaction
00934       periph->trans_extract_idx++;
00935       if (periph->trans_extract_idx >= I2C_TRANSACTION_QUEUE_LEN)
00936         periph->trans_extract_idx = 0;
00937 
00938       // Tell everyone we are ready
00939       periph->status = I2CIdle;
00940 
00941 
00942       // if we have no more transaction to process, stop here
00943       if (periph->trans_extract_idx == periph->trans_insert_idx)
00944       {
00945 
00946 #ifdef I2C_DEBUG_LED
00947         LED2_ON();
00948         LED1_ON();
00949   LED1_OFF();
00950         LED1_ON();
00951   LED1_OFF();
00952   LED2_OFF();
00953 #endif
00954       }
00955       // if not, start next transaction
00956       else
00957       {
00958         // Restart transaction doing the Rx part now
00959 // --- moved to idle function
00960         PPRZ_I2C_SEND_START(periph);
00961 // ------
00962      }
00963 
00964     }
00965     // RxTx -> Restart and do Rx part
00966     else
00967     {
00968       trans->type = I2CTransRx;
00969       periph->status = I2CStartRequested;
00970       regs->CR1 |= I2C_CR1_BIT_START;
00971 
00972       // Silent any BTF that would occur before SB
00973       regs->DR = 0x00;
00974     }
00975   }
00976 
00977   return;
00978 }

00996     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_ARLO);
