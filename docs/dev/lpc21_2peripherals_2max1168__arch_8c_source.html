00028 void max1168_arch_init( void ) {
00029 
00030   /* unselected max1168 */
00031   Max1168Unselect();
00032   /* SS pin is output */
00033   SetBit(MAX1168_SS_IODIR, MAX1168_SS_PIN);
00034 
00035   /* connect P0.16 to extint0 (EOC) */
00036   MAX1168_EOC_PINSEL |= MAX1168_EOC_PINSEL_VAL << MAX1168_EOC_PINSEL_BIT;
00037   /* extint0 is edge trigered */
00038   SetBit(EXTMODE, MAX1168_EOC_EINT);
00039   /* extint0 is trigered on falling edge */
00040   ClearBit(EXTPOLAR, MAX1168_EOC_EINT);
00041   /* clear pending extint0 before enabling interrupts */
00042   SetBit(EXTINT, MAX1168_EOC_EINT);
00043 
00044    /* initialize interrupt vector */
00045   VICIntSelect &= ~VIC_BIT( VIC_EINT0 );                     // EXTINT0 selected as IRQ
00046   VICIntEnable = VIC_BIT( VIC_EINT0 );                       // EXTINT0 interrupt enabled
00047   _VIC_CNTL(MAX1168_EOC_VIC_SLOT) = VIC_ENABLE | VIC_EINT0;
00048   _VIC_ADDR(MAX1168_EOC_VIC_SLOT) = (uint32_t)EXTINT0_ISR;   // address of the ISR
00049 }
