00148 void ahrs_propagate(void) {
00149 
00150   /* unbias gyro             */
00151   struct Int32Rates uf_rate;
00152   RATES_DIFF(uf_rate, imu.gyro, ahrs_impl.gyro_bias);
00153 #if USE_NOISE_CUT
00154   static struct Int32Rates last_uf_rate = { 0, 0, 0 };
00155   if (!cut_rates(uf_rate, last_uf_rate, RATE_CUT_THRESHOLD)) {
00156 #endif
00157     /* low pass rate */
00158 #if USE_NOISE_FILTER
00159     RATES_SUM_SCALED(ahrs.imu_rate, ahrs.imu_rate, uf_rate, NOISE_FILTER_GAIN);
00160     RATES_SDIV(ahrs.imu_rate, ahrs.imu_rate, NOISE_FILTER_GAIN+1);
00161 #else
00162     RATES_ADD(ahrs.imu_rate, uf_rate);
00163     RATES_SDIV(ahrs.imu_rate, ahrs.imu_rate, 2);
00164 #endif
00165 #if USE_NOISE_CUT
00166   }
00167   RATES_COPY(last_uf_rate, uf_rate);
00168 #endif
00169 
00170   /* integrate eulers */
00171   struct Int32Eulers euler_dot;
00172   INT32_EULERS_DOT_OF_RATES(euler_dot, ahrs.ltp_to_imu_euler, ahrs.imu_rate);
00173   EULERS_ADD(ahrs_impl.hi_res_euler, euler_dot);
00174 
00175   /* low pass measurement */
00176   EULERS_ADD(ahrs_impl.measure, ahrs_impl.measurement);
00177   EULERS_SDIV(ahrs_impl.measure, ahrs_impl.measure, 2);
00178 
00179   /* compute residual */
00180   EULERS_DIFF(ahrs_impl.residual, ahrs_impl.measure, ahrs_impl.hi_res_euler);
00181   INTEG_EULER_NORMALIZE(ahrs_impl.residual.psi);
00182 
00183   struct Int32Eulers correction;
00184   /* compute a correction */
00185   EULERS_SDIV(correction, ahrs_impl.residual, ahrs_impl.reinj_1);
00186   /* correct estimation */
00187   EULERS_ADD(ahrs_impl.hi_res_euler, correction);
00188   INTEG_EULER_NORMALIZE(ahrs_impl.hi_res_euler.psi);
00189 
00190 
00191   /* Compute LTP to IMU eulers      */
00192   EULERS_SDIV(ahrs.ltp_to_imu_euler, ahrs_impl.hi_res_euler, F_UPDATE);
00193 
00194   compute_imu_quat_and_rmat_from_euler();
00195 
00196   compute_body_orientation();
00197 
00198 }

00233 
00234   INT32_ATAN2(*phi_meas, -accel.y, -accel.z);
00235   int32_t cphi;
00236   PPRZ_ITRIG_COS(cphi, *phi_meas);

00238   INT32_ATAN2(*theta_meas, -cphi_ax, -accel.z);
00239   *phi_meas   *= F_UPDATE;

00244 /* measure psi by projecting magnetic vector in local tangeant plan */
00245 __attribute__ ((always_inline)) static inline void get_psi_measurement_from_mag(int32_t* psi_meas, int32_t phi_est, int32_t theta_est, struct Int32Vect3 mag) {
00246 
00247   int32_t sphi;
00248   PPRZ_ITRIG_SIN(sphi, phi_est);
00249   int32_t cphi;
00250   PPRZ_ITRIG_COS(cphi, phi_est);
00251   int32_t stheta;
00252   PPRZ_ITRIG_SIN(stheta, theta_est);
00253   int32_t ctheta;
00254   PPRZ_ITRIG_COS(ctheta, theta_est);
00255 
00256   int32_t sphi_stheta = (sphi*stheta)>>INT32_TRIG_FRAC;
00257   int32_t cphi_stheta = (cphi*stheta)>>INT32_TRIG_FRAC;
00258   //int32_t sphi_ctheta = (sphi*ctheta)>>INT32_TRIG_FRAC;
00259   //int32_t cphi_ctheta = (cphi*ctheta)>>INT32_TRIG_FRAC;
00260 
00261   const int32_t mn = ctheta * mag.x + sphi_stheta * mag.y + cphi_stheta * mag.z;
00262   const int32_t me = 0      * mag.x + cphi        * mag.y - sphi        * mag.z;
00263   //const int32_t md =
00264   //  -stheta     * imu.mag.x +
00265   //  sphi_ctheta * imu.mag.y +
00266   //  cphi_ctheta * imu.mag.z;
00267   float m_psi = -atan2(me, mn);
00268   *psi_meas = ((m_psi - ahrs_mag_offset)*(float)(1<<(INT32_ANGLE_FRAC))*F_UPDATE);
