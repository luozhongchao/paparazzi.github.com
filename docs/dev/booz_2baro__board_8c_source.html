00056 void baro_periodic(void) {
00057 
00058   baro.absolute = baro_board.buf.sum/baro_board.buf.av_nb_sample;
00059   baro_board.value_filtered = (3*baro_board.value_filtered + baro.absolute)/4;
00060   if (baro.status == BS_UNINITIALIZED) {
00061     RunOnceEvery(10, { baro_board_calibrate();});
00062   }
00063   /*  else */
00064   baro_board.data_available = TRUE;
00065 }

00067 /* decrement offset until adc reading is over a threshold */
00068 void baro_board_calibrate(void) {
00069   if (baro_board.value_filtered < BOOZ_ANALOG_BARO_THRESHOLD && baro_board.offset >= 1) {
00070     if (baro_board.value_filtered == 0 && baro_board.offset > 15)
00071       baro_board.offset -= 15;
00072     else
00073       baro_board.offset--;
00074     DACSet(baro_board.offset);
00075 #ifdef ROTORCRAFT_BARO_LED
00076     LED_TOGGLE(ROTORCRAFT_BARO_LED);
00077 #endif
00078   }
00079   else {
00080     baro.status = BS_RUNNING;
00081 #ifdef ROTORCRAFT_BARO_LED
00082     LED_ON(ROTORCRAFT_BARO_LED);
00083 #endif
00084   }
00085 }
