00100 #define FillBufWith32bit(_buf, _index, _value) {  \
00101   _buf[_index] = (uint8_t) (_value);              \
00102   _buf[_index+1] = (uint8_t) ((_value) >> 8);     \

00107 void ArduIMU_periodicGPS( void ) {
00108 
00109   if (ardu_gps_trans.status != I2CTransDone) { return; }
00110 
00111 #if USE_HIGH_ACCEL_FLAG
00112   // Test for high acceleration:
00113   //  - low speed
00114   //  - high thrust
00115   if (estimator_hspeed_mod < HIGH_ACCEL_LOW_SPEED && ap_state->commands[COMMAND_THROTTLE] > HIGH_ACCEL_HIGH_THRUST && !high_accel_done) {
00116     high_accel_flag = TRUE;
00117   } else {
00118     high_accel_flag = FALSE;
00119     if (estimator_hspeed_mod > HIGH_ACCEL_LOW_SPEED && !high_accel_done) {
00120       high_accel_done = TRUE; // After takeoff, don't use high accel before landing (GS small, Throttle small)
00121     }
00122     if (estimator_hspeed_mod < HIGH_ACCEL_HIGH_THRUST_RESUME && ap_state->commands[COMMAND_THROTTLE] < HIGH_ACCEL_HIGH_THRUST_RESUME) {
00123       high_accel_done = FALSE; // Activate high accel after landing
00124     }
00125   }
00126 #endif
00127 
00128   FillBufWith32bit(ardu_gps_trans.buf, 0, (int32_t)gps.speed_3d); // speed 3D
00129   FillBufWith32bit(ardu_gps_trans.buf, 4, (int32_t)gps.gspeed);   // ground speed
00130   FillBufWith32bit(ardu_gps_trans.buf, 8, (int32_t)gps.course);   // course
00131   ardu_gps_trans.buf[12] = gps.fix;                               // status gps fix
00132   ardu_gps_trans.buf[13] = (uint8_t)arduimu_calibrate_neutrals;   // calibration flag
00133   ardu_gps_trans.buf[14] = (uint8_t)high_accel_flag;              // high acceleration flag (disable accelerometers in the arduimu filter)
00134   I2CTransmit(ARDUIMU_I2C_DEV, ardu_gps_trans, ArduIMU_SLAVE_ADDR, 15);
00135 
00136   // Reset calibration flag
00137   if (arduimu_calibrate_neutrals) arduimu_calibrate_neutrals = FALSE;
00138 }

00162 void ArduIMU_event( void ) {
00163   // Handle INS I2C event
00164   if (ardu_ins_trans.status == I2CTransSuccess) {
00165     // received data from I2C transaction
00166     recievedData[0] = (ardu_ins_trans.buf[1]<<8) | ardu_ins_trans.buf[0];
00167     recievedData[1] = (ardu_ins_trans.buf[3]<<8) | ardu_ins_trans.buf[2];
00168     recievedData[2] = (ardu_ins_trans.buf[5]<<8) | ardu_ins_trans.buf[4];
00169     recievedData[3] = (ardu_ins_trans.buf[7]<<8) | ardu_ins_trans.buf[6];
00170     recievedData[4] = (ardu_ins_trans.buf[9]<<8) | ardu_ins_trans.buf[8];
00171     recievedData[5] = (ardu_ins_trans.buf[11]<<8) | ardu_ins_trans.buf[10];
00172     recievedData[6] = (ardu_ins_trans.buf[13]<<8) | ardu_ins_trans.buf[12];
00173     recievedData[7] = (ardu_ins_trans.buf[15]<<8) | ardu_ins_trans.buf[14];
00174     recievedData[8] = (ardu_ins_trans.buf[17]<<8) | ardu_ins_trans.buf[16];
00175 
00176     // Update ArduIMU data
00177     arduimu_eulers.phi = ANGLE_FLOAT_OF_BFP(recievedData[0]);
00178     arduimu_eulers.theta = ANGLE_FLOAT_OF_BFP(recievedData[1]);
00179     arduimu_eulers.psi = ANGLE_FLOAT_OF_BFP(recievedData[2]);
00180     arduimu_rates.p = RATE_FLOAT_OF_BFP(recievedData[3]);
00181     arduimu_rates.q = RATE_FLOAT_OF_BFP(recievedData[4]);
00182     arduimu_rates.r = RATE_FLOAT_OF_BFP(recievedData[5]);
00183     arduimu_accel.x = ACCEL_FLOAT_OF_BFP(recievedData[6]);
00184     arduimu_accel.y = ACCEL_FLOAT_OF_BFP(recievedData[7]);
00185     arduimu_accel.z = ACCEL_FLOAT_OF_BFP(recievedData[8]);
00186 
00187     // Update estimator
00188     estimator_phi = arduimu_eulers.phi - ins_roll_neutral;
00189     estimator_theta = arduimu_eulers.theta - ins_pitch_neutral;
00190     estimator_p = arduimu_rates.p;
00191     estimator_q = arduimu_rates.q;
00192     estimator_r = arduimu_rates.r;
00193     ardu_ins_trans.status = I2CTransDone;
00194 
00195 #ifdef ARDUIMU_SYNC_SEND
00196     //RunOnceEvery(15, DOWNLINK_SEND_AHRS_EULER(DefaultChannel, DefaultDevice, &arduimu_eulers.phi, &arduimu_eulers.theta, &arduimu_eulers.psi));
00197     RunOnceEvery(15, DOWNLINK_SEND_IMU_GYRO(DefaultChannel, DefaultDevice, &arduimu_rates.p, &arduimu_rates.q, &arduimu_rates.r));
00198     RunOnceEvery(15, DOWNLINK_SEND_IMU_ACCEL(DefaultChannel, DefaultDevice, &arduimu_accel.x, &arduimu_accel.y, &arduimu_accel.z));
00199 #endif
00200   }
00201   else if (ardu_ins_trans.status == I2CTransFailed) {
00202     ardu_ins_trans.status = I2CTransDone;
00203   }
00204   // Handle GPS I2C event
00205   if (ardu_gps_trans.status == I2CTransSuccess || ardu_gps_trans.status == I2CTransFailed) {
00206     ardu_gps_trans.status = I2CTransDone;
00207   }
00208 }
