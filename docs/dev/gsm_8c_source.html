00204 // A line of length gsm_buf_len is available in the gsm_buf buffer
00205 static void gsm_got_line(void)
00206 {
00207   if (gsm_status == STATUS_WAITING_DATA) { // Currently receiving a SMS
00208     gsm_receive_content();
00209     Suppr_SMS(index_msg);
00210     gsm_status = STATUS_DELETE_SMS;
00211   } else if (gsm_status == STATUS_IDLE
00212       && strncmp(CMTI, gsm_buf, strlen(CMTI)) == 0) {
00213     /* A SMS is available */
00214     /* Extracting the index of the message */
00215     char * first_comma = indexn(gsm_buf, ',',MAXLEN_CMTI_ANSWER);
00216     if (first_comma) {
00217       index_msg = atoi(first_comma+1);
00218       request_for_msg();
00219       gsm_status = STATUS_REQUESTING_MESSAGE;
00220     }
00221   } else if (waiting_for_reply) { // Other cases
00222     // Do we get what we were expecting
00223 
00224     bool gsm_answer = strncmp(expected_ack, gsm_buf, strlen(expected_ack)) == 0;
00225     if (gsm_answer) {
00226       waiting_for_reply = false;
00227 
00228       switch(gsm_status) {
00229         case STATUS_CSQ :
00230           gsm_send_report_continue();
00231           gsm_status = STATUS_WAITING_PROMPT;
00232           break;
00233 
00234         case STATUS_REQUESTING_MESSAGE:
00235           parse_msg_header();
00236           gsm_status = STATUS_WAITING_DATA;
00237           break;
00238 
00239         case STATUS_SEND_AT :
00240           gsm_answer = false;
00241           Send_CMGF();
00242           gsm_status = STATUS_SEND_CMGF;
00243           break;
00244 
00245         case STATUS_SEND_CMGF :
00246           gsm_answer = false;
00247           Send_CNMI();
00248           gsm_status = STATUS_SEND_CNMI;
00249           break;
00250 
00251         case STATUS_SEND_CNMI :
00252           gsm_answer = false;
00253           Send_CPMS();
00254           gsm_status = STATUS_SEND_CPMS;
00255           break;
00256 
00257         case STATUS_SEND_CPMS :
00258           gsm_answer = false;
00259           gsm_status = STATUS_IDLE;
00260           gsm_gsm_send_report_status = MODULES_START; 
00261           break;
00262 
00263         case STATUS_DELETE_SMS :
00264           gsm_status = STATUS_IDLE;
00265           break;
00266 
00267         default:
00268           break;
00269       }
00270     } else { 
00271       /* Let's wait for the next line */
00272     }

00480 static void gsm_parse(uint8_t c) {
00481   switch(c) {
00482     case GSM_CMD_LINE_TERMINATION:
00483       break;
00484     case '>':
00485       prompt_received = true;
00486       break;
00487     case GSM_RESPONSE_FORMATING:
00488       gsm_buf[gsm_buf_idx] = '\0';
00489       gsm_line_received = true;
00490       gsm_buf_len = gsm_buf_idx;
00491       gsm_buf_idx=0;
00492       break;
00493     default:
00494       if (gsm_buf_idx < GSM_MAX_PAYLOAD) {
00495         gsm_buf[gsm_buf_idx] = c;
00496         gsm_buf_idx++;
00497       } /* else extra characters are ignored */
00498       break;
00499   }
00500 }
