00163 static inline void parse_xbee( struct xbee_transport * t, uint8_t c ) {
00164   switch (t->status) {
00165   case XBEE_UNINIT:
00166     if (c == XBEE_START)
00167       t->status++;
00168     break;
00169   case XBEE_GOT_START:
00170     if (t->trans.msg_received) {
00171       t->trans.ovrn++;
00172       goto error;
00173     }
00174     t->trans.payload_len = c<<8;
00175     t->status++;
00176     break;
00177   case XBEE_GOT_LENGTH_MSB:
00178     t->trans.payload_len |= c;
00179     t->status++;
00180     t->payload_idx = 0;
00181     t->cs=0;
00182     break;
00183   case XBEE_GOT_LENGTH_LSB:
00184     t->trans.payload[t->payload_idx] = c;
00185     t->cs += c;
00186     t->payload_idx++;
00187     if (t->payload_idx == t->trans.payload_len)
00188       t->status++;
00189     break;
00190   case XBEE_GOT_PAYLOAD:
00191     if (c + t->cs != 0xff)
00192       goto error;
00193     t->trans.msg_received = TRUE;
00194     goto restart;
00195     break;
00196   default:
00197     goto error;
00198   }
00199   return;
00200  error:
00201   t->trans.error++;
00202  restart:
00203   t->status = XBEE_UNINIT;
00204   return;
00205 }

00209   switch (t->trans.payload[0]) {
00210   case XBEE_RX_ID:
00211   case XBEE_TX_ID: /* Useful if A/C is connected to the PC with a cable */
00212     XbeeGetRSSI(t->trans.payload);
00213     uint8_t i;

00216     dl_msg_available = TRUE;
00217     break;
00218   default:

00220   }
