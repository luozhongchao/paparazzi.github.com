00101 /* Prepare data to be sent to mcu0 */
00102 static inline void inter_mcu_fill_fbw_state (void) {
00103   uint8_t status = 0;
00104 
00105 #ifdef RADIO_CONTROL
00106   uint8_t i;
00107   for(i = 0; i < RADIO_CONTROL_NB_CHANNEL; i++)
00108     fbw_state->channels[i] = radio_control.values[i];
00109 
00110   fbw_state->ppm_cpt = radio_control.frame_rate;
00111 
00112   status = (radio_control.status == RC_OK ? _BV(STATUS_RADIO_OK) : 0);
00113   status |= (radio_control.status == RC_REALLY_LOST ? _BV(STATUS_RADIO_REALLY_LOST) : 0);
00114   status |= (radio_control.status == RC_OK ? _BV(AVERAGED_CHANNELS_SENT) : 0); // Any valid frame contains averaged channels
00115 #endif // RADIO_CONTROL
00116 
00117   status |= (fbw_mode == FBW_MODE_AUTO ? _BV(STATUS_MODE_AUTO) : 0);
00118   status |= (fbw_mode == FBW_MODE_FAILSAFE ? _BV(STATUS_MODE_FAILSAFE) : 0);
00119   fbw_state->status  = status;
00120 
00121   fbw_state->vsupply = electrical.vsupply;
00122   fbw_state->current = electrical.current;
00123 #if defined SINGLE_MCU
00124 
00125   inter_mcu_received_fbw = TRUE;
00126 #endif
00127 }
