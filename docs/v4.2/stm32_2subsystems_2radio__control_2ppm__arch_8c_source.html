00164 void tim1_cc_irq_handler(void) {
00165   if(TIM_GetITStatus(TIM1, TIM_IT_CC3) == SET) {
00166     TIM_ClearITPendingBit(TIM1, TIM_IT_CC3);
00167 
00168     uint32_t now = TIM_GetCapture3(TIM1) + timer_rollover_cnt;

00171 }

00175      TIM_ClearITPendingBit(TIM2, TIM_IT_CC2);
00176      uint32_t now = TIM_GetCapture2(TIM2) + timer_rollover_cnt;
00177      DecodePpmFrame(now);
00178   }
00179   else if(TIM_GetITStatus(TIM2, TIM_IT_Update) == SET) {
