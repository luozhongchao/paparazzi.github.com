57 void atmega_i2c_cam_ctrl_send(uint8_t cmd)
58 {
59  static uint8_t zoom = 0;
60  static uint8_t mode = 0;
61  unsigned char cam_ret[1];
62 
63  if (cmd == DC_SHOOT)
64  {
65  dc_send_shot_position();
66  }
67  else if (cmd == DC_TALLER)
68  {
69  zoom = 1;
70  }
71  else if (cmd == DC_WIDER)
72  {
73  zoom = 0;
74  }
75  else if (cmd == DC_GET_STATUS)
76  {
77  mode++;
78  if (mode > 15)
79  mode = 0;
80  }
81 
82  cam_ret[0] = mode + zoom * 0x20;
83  RunOnceEvery(6,DOWNLINK_SEND_PAYLOAD(DefaultChannel, DefaultDevice, 1, cam_ret ));
84 
85 }
