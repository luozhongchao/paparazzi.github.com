56 // Configuration function called once before normal use
57 static void hmc58xx_send_config(void)
58 {
59  switch (hmc58xx_init_status) {
60  case HMC_CONF_CRA:
61  hmc58xx_i2c_trans.buf[0] = HMC58XX_REG_CFGA;
62  hmc58xx_i2c_trans.buf[1] = HMC58XX_CRA;
63  I2CTransmit(HMC58XX_I2C_DEVICE, hmc58xx_i2c_trans, HMC58XX_ADDR, 2);
64  hmc58xx_init_status++;
65  break;
66  case HMC_CONF_CRB:
67  hmc58xx_i2c_trans.buf[0] = HMC58XX_REG_CFGB;
68  hmc58xx_i2c_trans.buf[1] = HMC58XX_CRB;
69  I2CTransmit(HMC58XX_I2C_DEVICE, hmc58xx_i2c_trans, HMC58XX_ADDR, 2);
70  hmc58xx_init_status++;
71  break;
72  case HMC_CONF_MODE:
73  hmc58xx_i2c_trans.buf[0] = HMC58XX_REG_MODE;
74  hmc58xx_i2c_trans.buf[1] = HMC58XX_MD;
75  I2CTransmit(HMC58XX_I2C_DEVICE, hmc58xx_i2c_trans, HMC58XX_ADDR, 2);
76  hmc58xx_init_status++;
77  break;
78  case HMC_CONF_DONE:
79  hmc58xx_initialized = TRUE;
80  hmc58xx_i2c_trans.status = I2CTransDone;
81  break;
82  default:
83  break;
84  }
85 }

87 // Configure
88 void hmc58xx_configure(void)
89 {
90  if (hmc58xx_init_status == HMC_CONF_UNINIT) {
91  hmc58xx_init_status++;
92  if (hmc58xx_i2c_trans.status == I2CTransSuccess || hmc58xx_i2c_trans.status == I2CTransDone) {
93  hmc58xx_send_config();
94  }
95  }
96 }

109 void hmc58xx_event(void)
110 {
111  if (hmc58xx_initialized) {
112  if (hmc58xx_i2c_trans.status == I2CTransFailed) {
113  hmc58xx_i2c_trans.status = I2CTransDone;
114  }
115  else if (hmc58xx_i2c_trans.status == I2CTransSuccess) {
116  hmc58xx_data.x = Int16FromBuf(hmc58xx_i2c_trans.buf,0);
117  hmc58xx_data.y = Int16FromBuf(hmc58xx_i2c_trans.buf,4);
118  hmc58xx_data.z = Int16FromBuf(hmc58xx_i2c_trans.buf,2);
119  hmc58xx_data_available = TRUE;
120  hmc58xx_i2c_trans.status = I2CTransDone;
121  }
122  }
123  else if (!hmc58xx_initialized && hmc58xx_init_status != HMC_CONF_UNINIT) { // Configuring
124  if (hmc58xx_i2c_trans.status == I2CTransSuccess || hmc58xx_i2c_trans.status == I2CTransDone) {
125  hmc58xx_i2c_trans.status = I2CTransDone;
126  hmc58xx_send_config();
127  }
128  if (hmc58xx_i2c_trans.status == I2CTransFailed) {
129  hmc58xx_init_status--;
130  hmc58xx_i2c_trans.status = I2CTransDone;
131  hmc58xx_send_config(); // Retry config (TODO max retry)
132  }
133  }
134 }
