57 static inline bool_t ins_configure( void ) {
58  switch (ins_init_status) {
59  case INS_VN100_SET_BAUD :
60  last_send_packet.RegID = VN100_REG_SBAUD;
61  spi_buffer_length = 4+VN100_REG_SBAUD_SIZE;
62  ins_init_status++;
63  break;
64  case INS_VN100_SET_ADOR :
65  last_send_packet.RegID = VN100_REG_ADOR;
66  spi_buffer_length = 4+VN100_REG_ADOR_SIZE;
67  ins_init_status++;
68  break;
69  case INS_VN100_SET_ADOF :
70  last_send_packet.RegID = VN100_REG_ADOF;
71  spi_buffer_length = 4+VN100_REG_ADOF_SIZE;
72  ins_init_status++;
73  break;
74  case INS_VN100_READY :
75  return TRUE;
76  }
77  last_send_packet.CmdID = VN100_CmdID_WriteRegister;
78  spi_buffer_input = (uint8_t*)&last_received_packet;
79  spi_buffer_output = (uint8_t*)&last_send_packet;
80  SpiSelectSlave0();
81  SpiStart();
82  return FALSE;
83 }
