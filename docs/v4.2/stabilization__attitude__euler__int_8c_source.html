00102 void stabilization_attitude_run(bool_t  in_flight) {
00103 
00104 
00105   /* update reference */
00106   stabilization_attitude_ref_update();
00107 
00108   /* compute feedforward command */
00109   stabilization_att_ff_cmd[COMMAND_ROLL] =
00110     OFFSET_AND_ROUND(stabilization_gains.dd.x * stab_att_ref_accel.p, 5);
00111   stabilization_att_ff_cmd[COMMAND_PITCH] =
00112     OFFSET_AND_ROUND(stabilization_gains.dd.y * stab_att_ref_accel.q, 5);
00113   stabilization_att_ff_cmd[COMMAND_YAW] =
00114     OFFSET_AND_ROUND(stabilization_gains.dd.z * stab_att_ref_accel.r, 5);
00115 
00116   /* compute feedback command */
00117   /* attitude error            */
00118   const struct Int32Eulers att_ref_scaled = {
00119     OFFSET_AND_ROUND(stab_att_ref_euler.phi,   (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
00120     OFFSET_AND_ROUND(stab_att_ref_euler.theta, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
00121     OFFSET_AND_ROUND(stab_att_ref_euler.psi,   (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)) };
00122   struct Int32Eulers att_err;
00123   EULERS_DIFF(att_err, att_ref_scaled, ahrs.ltp_to_body_euler);
00124   INT32_ANGLE_NORMALIZE(att_err.psi);
00125 
00126   if (in_flight) {
00127     /* update integrator */
00128     EULERS_ADD(stabilization_att_sum_err, att_err);
00129     EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
00130   }
00131   else {
00132     INT_EULERS_ZERO(stabilization_att_sum_err);
00133   }
00134 
00135   /* rate error                */
00136   const struct Int32Rates rate_ref_scaled = {
00137     OFFSET_AND_ROUND(stab_att_ref_rate.p, (REF_RATE_FRAC - INT32_RATE_FRAC)),
00138     OFFSET_AND_ROUND(stab_att_ref_rate.q, (REF_RATE_FRAC - INT32_RATE_FRAC)),
00139     OFFSET_AND_ROUND(stab_att_ref_rate.r, (REF_RATE_FRAC - INT32_RATE_FRAC)) };
00140   struct Int32Rates rate_err;
00141   RATES_DIFF(rate_err, rate_ref_scaled, ahrs.body_rate);
00142 
00143   /* PID                  */
00144   stabilization_att_fb_cmd[COMMAND_ROLL] =
00145     stabilization_gains.p.x    * att_err.phi +
00146     stabilization_gains.d.x    * rate_err.p +
00147     OFFSET_AND_ROUND2((stabilization_gains.i.x  * stabilization_att_sum_err.phi), 10);
00148 
00149   stabilization_att_fb_cmd[COMMAND_PITCH] =
00150     stabilization_gains.p.y    * att_err.theta +
00151     stabilization_gains.d.y    * rate_err.q +
00152     OFFSET_AND_ROUND2((stabilization_gains.i.y  * stabilization_att_sum_err.theta), 10);
00153 
00154   stabilization_att_fb_cmd[COMMAND_YAW] =
00155     stabilization_gains.p.z    * att_err.psi +
00156     stabilization_gains.d.z    * rate_err.r +
00157     OFFSET_AND_ROUND2((stabilization_gains.i.z  * stabilization_att_sum_err.psi), 10);
00158 
00159 
00160   /* with P gain of 100, att_err of 180deg (3.14 rad)
00161    * fb cmd: 100 * 3.14 * 2^12 / 2^CMD_SHIFT = 628
00162    * max possible command is 9600
00163    */
00164 #define CMD_SHIFT 11
00165 
00166   /* sum feedforward and feedback */
00167   stabilization_cmd[COMMAND_ROLL] =
00168     OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_ROLL]+stabilization_att_ff_cmd[COMMAND_ROLL]), CMD_SHIFT);
00169 
00170   stabilization_cmd[COMMAND_PITCH] =
00171     OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_PITCH]+stabilization_att_ff_cmd[COMMAND_PITCH]), CMD_SHIFT);
00172 
00173   stabilization_cmd[COMMAND_YAW] =
00174     OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_YAW]+stabilization_att_ff_cmd[COMMAND_YAW]), CMD_SHIFT);
00175 
00176   /* bound the result */
00177   BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
00178   BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
00179   BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
00180 
00181 }
