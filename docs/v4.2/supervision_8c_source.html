00142 void supervision_run(bool_t motors_on, bool_t override_on, int32_t in_cmd[] ) {
00143   uint8_t i;
00144   if (motors_on) {
00145     int32_t min_cmd = INT32_MAX;
00146     int32_t max_cmd = INT32_MIN;
00147     /* do the mixing in float to avoid overflows, implicitly casted back to int32_t */
00148     for (i=0; i<SUPERVISION_NB_MOTOR; i++) {
00149       supervision.commands[i] = SUPERVISION_MIN_MOTOR +
00150         (thrust_coef[i] * in_cmd[COMMAND_THRUST] +
00151          roll_coef[i]   * in_cmd[COMMAND_ROLL]   +
00152          pitch_coef[i]  * in_cmd[COMMAND_PITCH]  +
00153          yaw_coef[i]    * in_cmd[COMMAND_YAW]    +
00154          supervision.trim[i]) / SUPERVISION_SCALE *
00155         (SUPERVISION_MAX_MOTOR - SUPERVISION_MIN_MOTOR) / MAX_PPRZ;
00156       if (supervision.commands[i] < min_cmd)
00157         min_cmd = supervision.commands[i];
00158       if (supervision.commands[i] > max_cmd)
00159         max_cmd = supervision.commands[i];
00160     }
00161     if (min_cmd < SUPERVISION_MIN_MOTOR && max_cmd > SUPERVISION_MAX_MOTOR)
00162       supervision.nb_failure++;
00163     if (min_cmd < SUPERVISION_MIN_MOTOR)
00164       offset_commands(-(min_cmd - SUPERVISION_MIN_MOTOR));
00165     if (max_cmd > SUPERVISION_MAX_MOTOR)
00166       offset_commands(-(max_cmd - SUPERVISION_MAX_MOTOR));
00167 
00168     /* For testing motor failure */
00169     if (motors_on && override_on) {
00170       for (i = 0; i < SUPERVISION_NB_MOTOR; i++) {
00171         if (supervision.override_enabled[i])
00172           supervision.commands[i] = supervision.override_value[i];
00173       }
00174     }
00175     bound_commands();
00176     bound_commands_step();
00177   }
00178   else {
00179     for (i=0; i<SUPERVISION_NB_MOTOR; i++) {
00180       supervision.commands[i] = SUPERVISION_STOP_MOTOR;
00181     }
00182   }
