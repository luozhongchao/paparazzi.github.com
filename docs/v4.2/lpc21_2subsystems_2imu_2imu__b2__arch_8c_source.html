00030 /* SSPCR0 settings */
00031 #define SSP_DDS8  0x07 << 0  /* data size         : 8 bits                    */
00032 #define SSP_DDS16 0x0F << 0  /* data size         : 16 bits                   */
00033 #define SSP_FRF   0x00 << 4  /* frame format      : SPI                       */
00034 #define SSP_CPOL  0x00 << 6  /* clock polarity    : data captured on first clock transition */
00035 #define SSP_CPHA  0x00 << 7  /* clock phase       : SCK idles low             */
00036 #define SSP_SCR   0x0F << 8  /* serial clock rate : divide by 16              */
00037 
00038 /* SSPCR1 settings */
00039 #define SSP_LBM   0x00 << 0  /* loopback mode     : disabled                  */
00040 #define SSP_SSE   0x00 << 1  /* SSP enable        : disabled                  */
00041 #define SSP_MS    0x00 << 2  /* master slave mode : master                    */
00042 #define SSP_SOD   0x00 << 3  /* slave output disable : don't care when master */

00044 #define SSPCR0_VAL8  (SSP_DDS8  |  SSP_FRF | SSP_CPOL | SSP_CPHA | SSP_SCR )

00062 void imu_b2_arch_init(void) {
00063 
00064   imu_ssp_status = IMU_SSP_STA_IDLE;
00065 
00066   /* setup pins for SSP (SCK, MISO, MOSI) */
00067   PINSEL1 |= SSP_PINSEL1_SCK  | SSP_PINSEL1_MISO | SSP_PINSEL1_MOSI;
00068 
00069   /* setup SSP */
00070   SSPCR0 = SSPCR0_VAL16;
00071   SSPCR1 = SSPCR1_VAL;
00072   SSPCPSR = 0x02;
00073 
00074   /* initialize interrupt vector */
00075   VICIntSelect &= ~VIC_BIT( VIC_SPI1 );             /* SPI1 selected as IRQ */
00076   VICIntEnable = VIC_BIT( VIC_SPI1 );               /* enable it            */
00077   _VIC_CNTL(SSP_VIC_SLOT) = VIC_ENABLE | VIC_SPI1;
00078   _VIC_ADDR(SSP_VIC_SLOT) = (uint32_t)SSP_ISR;      /* address of the ISR   */
00079 
00080 }
