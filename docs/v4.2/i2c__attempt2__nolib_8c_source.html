00834 static inline void i2c_error(struct i2c_periph *periph)
00835 {
00836   uint8_t err_nr = 0;
00837   periph->errors->er_irq_cnt;
00838   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_AF)) {       /* Acknowledge failure */
00839     periph->errors->ack_fail_cnt++;
00840     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_AF);
00841     err_nr = 1;
00842   }
00843   if (I2C_GetITStatus(periph->reg_addr, I2C_IT_BERR)) {     /* Misplaced Start or Stop condition */
00844     periph->errors->miss_start_stop_cnt++;
00845     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_BERR);
00846     err_nr = 2;
00847   }

00849     periph->errors->arb_lost_cnt++;
00850     I2C_ClearITPendingBit(periph->reg_addr, I2C_IT_ARLO);
00851     err_nr = 3;
