00303     unsigned long *pul_src, *pul_dest;
00304 
00305     /* Copy the data segment initializers from flash to SRAM */
00306     pul_src = &_sidata;
00307     for(pul_dest = &_sdata; pul_dest < &_edata; ){
00308         *(pul_dest++) = *(pul_src++);
00309     }
00310     /* Zero fill the bss segment.  */
00311     for(pul_dest = &_sbss; pul_dest < &_ebss; ){
00312         *(pul_dest++) = 0;
