00218   ahrs_float.imu_rate.q += dq;

00276     ahrs_impl.gps_speed = 0;
00277     ahrs_impl.gps_acceleration = 0;

00340 void Normalize(void)
00341 {
00342   float error=0;
00343   float temporary[3][3];
00344   float renorm=0;
00345   uint8_t problem=FALSE;
00346 
00347   // Find the non-orthogonality of X wrt Y
00348   error= -Vector_Dot_Product(&DCM_Matrix[0][0],&DCM_Matrix[1][0])*.5; //eq.19
00349 
00350   // Add half the XY error to X, and half to Y
00351   Vector_Scale(&temporary[0][0], &DCM_Matrix[1][0], error);           //eq.19
00352   Vector_Scale(&temporary[1][0], &DCM_Matrix[0][0], error);           //eq.19
00353   Vector_Add(&temporary[0][0], &temporary[0][0], &DCM_Matrix[0][0]);  //eq.19
00354   Vector_Add(&temporary[1][0], &temporary[1][0], &DCM_Matrix[1][0]);  //eq.19
00355 
00356   // The third axis is simply set perpendicular to the first 2. (there is not correction of XY based on Z)
00357   Vector_Cross_Product(&temporary[2][0],&temporary[0][0],&temporary[1][0]); // c= a x b //eq.20
00358 
00359   // Normalize lenght of X
00360   renorm= Vector_Dot_Product(&temporary[0][0],&temporary[0][0]);
00361   // a) if norm is close to 1, use the fast 1st element from the tailer expansion of SQRT
00362   // b) if the norm is further from 1, use a real sqrt
00363   // c) norm is huge: disaster! reset! mayday!
00364   if (renorm < 1.5625f && renorm > 0.64f) {
00365     renorm= .5 * (3-renorm);                                          //eq.21
00366   } else if (renorm < 100.0f && renorm > 0.01f) {
00367     renorm= 1. / sqrt(renorm);
00368 #if PERFORMANCE_REPORTING == 1
00369     renorm_sqrt_count++;
00370 #endif
00371   } else {
00372     problem = TRUE;
00373 #if PERFORMANCE_REPORTING == 1
00374     renorm_blowup_count++;
00375 #endif
00376   }
00377   Vector_Scale(&DCM_Matrix[0][0], &temporary[0][0], renorm);
00378 
00379   // Normalize lenght of Y
00380   renorm= Vector_Dot_Product(&temporary[1][0],&temporary[1][0]);
00381   if (renorm < 1.5625f && renorm > 0.64f) {
00382     renorm= .5 * (3-renorm);                                                 //eq.21
00383   } else if (renorm < 100.0f && renorm > 0.01f) {
00384     renorm= 1. / sqrt(renorm);
00385 #if PERFORMANCE_REPORTING == 1
00386     renorm_sqrt_count++;
00387 #endif
00388   } else {
00389     problem = TRUE;
00390 #if PERFORMANCE_REPORTING == 1
00391     renorm_blowup_count++;
00392 #endif
00393   }
00394   Vector_Scale(&DCM_Matrix[1][0], &temporary[1][0], renorm);
00395 
00396   // Normalize lenght of Z
00397   renorm= Vector_Dot_Product(&temporary[2][0],&temporary[2][0]);
00398   if (renorm < 1.5625f && renorm > 0.64f) {
00399     renorm= .5 * (3-renorm);                                                 //eq.21
00400   } else if (renorm < 100.0f && renorm > 0.01f) {
00401     renorm= 1. / sqrt(renorm);
00402 #if PERFORMANCE_REPORTING == 1
00403     renorm_sqrt_count++;
00404 #endif
00405   } else {
00406     problem = TRUE;
00407 #if PERFORMANCE_REPORTING == 1
00408     renorm_blowup_count++;
00409 #endif
00410   }
00411   Vector_Scale(&DCM_Matrix[2][0], &temporary[2][0], renorm);
00412 
00413   // Reset on trouble
00414   if (problem) {                // Our solution is blowing up and we will force back to initial condition.  Hope we are not upside down!
00415     set_dcm_matrix_from_rmat(&ahrs_impl.body_to_imu_rmat);

00417   }
00418 }

00421 void Drift_correction(void)
00422 {
00423   //Compensation the Roll, Pitch and Yaw drift.
00424   static float Scaled_Omega_P[3];
00425   static float Scaled_Omega_I[3];
00426   float Accel_magnitude;
00427   float Accel_weight;
00428   float Integrator_magnitude;
00429 
00430   // Local Working Variables
00431   float errorRollPitch[3];
00432   float errorYaw[3];
00433   float errorCourse;
00434 
00435   //*****Roll and Pitch***************
00436 
00437   // Calculate the magnitude of the accelerometer vector
00438   Accel_magnitude = sqrt(accel_float.x*accel_float.x + accel_float.y*accel_float.y + accel_float.z*accel_float.z);
00439   Accel_magnitude = Accel_magnitude / GRAVITY; // Scale to gravity.
00440   // Dynamic weighting of accelerometer info (reliability filter)
00441   // Weight for accelerometer info (<0.5G = 0.0, 1G = 1.0 , >1.5G = 0.0)
00442   Accel_weight = Chop(1 - 2*fabs(1 - Accel_magnitude),0,1);  //
00443 
00444 #if USE_HIGH_ACCEL_FLAG
00445   // Test for high acceleration:
00446   //  - low speed
00447   //  - high thrust
00448   if (estimator_hspeed_mod < HIGH_ACCEL_LOW_SPEED && ap_state->commands[COMMAND_THROTTLE] > HIGH_ACCEL_HIGH_THRUST && !high_accel_done) {
00449     high_accel_flag = TRUE;
00450   } else {
00451     high_accel_flag = FALSE;
00452     if (estimator_hspeed_mod > HIGH_ACCEL_LOW_SPEED && !high_accel_done) {
00453       high_accel_done = TRUE; // After takeoff, don't use high accel before landing (GS small, Throttle small)
00454     }
00455     if (estimator_hspeed_mod < HIGH_ACCEL_HIGH_THRUST_RESUME && ap_state->commands[COMMAND_THROTTLE] < HIGH_ACCEL_HIGH_THRUST_RESUME) {
00456       high_accel_done = FALSE; // Activate high accel after landing
00457     }
00458   }
00459   if (high_accel_flag) { Accel_weight = 0.; }
00460 #endif
00461 
00462 
00463   #if PERFORMANCE_REPORTING == 1
00464   {
00465 
00466     float tempfloat = ((Accel_weight - 0.5) * 256.0f);    //amount added was determined to give imu_health a time constant about twice the time constant of the roll/pitch drift correction
00467     imu_health += tempfloat;
00468     Bound(imu_health,129,65405);
00469   }
00470   #endif
00471 
00472   Vector_Cross_Product(&errorRollPitch[0],&accel_float.x,&DCM_Matrix[2][0]); //adjust the ground of reference
00473   Vector_Scale(&Omega_P[0],&errorRollPitch[0],Kp_ROLLPITCH*Accel_weight);
00474 
00475   Vector_Scale(&Scaled_Omega_I[0],&errorRollPitch[0],Ki_ROLLPITCH*Accel_weight);
00476   Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);
00477 
00478   //*****YAW***************
00479 
00480 #if USE_MAGNETOMETER
00481   // We make the gyro YAW drift correction based on compass magnetic heading
00482 //  float mag_heading_x = cos(MAG_Heading);
00483 //  float mag_heading_y = sin(MAG_Heading);
00484   // 2D dot product
00485   errorCourse=(DCM_Matrix[0][0]*MAG_Heading_Y) + (DCM_Matrix[1][0]*MAG_Heading_X);  //Calculating YAW error
00486   Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00487 
00488   Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW);
00489   Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.
00490 
00491   Vector_Scale(&Scaled_Omega_I[0],&errorYaw[0],Ki_YAW);
00492   Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);//adding integrator to the Omega_I
00493 
00494 #else // Use GPS Ground course to correct yaw gyro drift
00495 
00496   if (ahrs_impl.gps_course_valid) {
00497     float course = ahrs_impl.gps_course - M_PI; //This is the runaway direction of you "plane" in rad
00498     float COGX = cosf(course); //Course overground X axis
00499     float COGY = sinf(course); //Course overground Y axis
00500 
00501     errorCourse=(DCM_Matrix[0][0]*COGY) - (DCM_Matrix[1][0]*COGX);  //Calculating YAW error
00502     Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
00503 
00504     Vector_Scale(&Scaled_Omega_P[0],&errorYaw[0],Kp_YAW);
00505     Vector_Add(Omega_P,Omega_P,Scaled_Omega_P);//Adding  Proportional.
00506 
00507     Vector_Scale(&Scaled_Omega_I[0],&errorYaw[0],Ki_YAW);
00508     Vector_Add(Omega_I,Omega_I,Scaled_Omega_I);//adding integrator to the Omega_I
00509   }
00510 #if USE_MAGNETOMETER_ONGROUND == 1
00511 #pragma message AHRS_FLOAT_DCM uses magnetometer prior to takeoff and GPS during flight
00512   else if (launch == FALSE)
00513   {
00514     float COGX = imu.mag.x; // Non-Tilt-Compensated (for filter stability reasons)
00515     float COGY = imu.mag.y; // Non-Tilt-Compensated (for filter stability reasons)
00516 
00517     errorCourse=(DCM_Matrix[0][0]*COGY) - (DCM_Matrix[1][0]*COGX);  //Calculating YAW error
00518     Vector_Scale(errorYaw,&DCM_Matrix[2][0],errorCourse); //Applys the yaw correction to the XYZ rotation of the aircraft, depeding the position.
