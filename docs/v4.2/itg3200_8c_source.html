50 {

52  itg3200_i2c_trans.slave_addr = ITG3200_ADDR;

57 // Configuration function called once before normal use
58 static void itg3200_send_config(void)
59 {
60  switch (itg3200_init_status) {
61  case ITG_CONF_SD:
62  itg3200_i2c_trans.buf[0] = ITG3200_REG_SMPLRT_DIV;
63  itg3200_i2c_trans.buf[1] = ITG3200_SMPLRT_DIV;
64  I2CTransmit(ITG3200_I2C_DEVICE, itg3200_i2c_trans, ITG3200_I2C_ADDR, 2);
65  itg3200_init_status++;
66  break;
67  case ITG_CONF_DF:
68  itg3200_i2c_trans.buf[0] = ITG3200_REG_DLPF_FS;
69  itg3200_i2c_trans.buf[1] = ITG3200_DLPF_FS;
70  I2CTransmit(ITG3200_I2C_DEVICE, itg3200_i2c_trans, ITG3200_I2C_ADDR, 2);
71  itg3200_init_status++;
72  break;
73  case ITG_CONF_INT:
74  itg3200_i2c_trans.buf[0] = ITG3200_REG_INT_CFG;
75  itg3200_i2c_trans.buf[1] = ITG3200_INT_CFG;
76  I2CTransmit(ITG3200_I2C_DEVICE, itg3200_i2c_trans, ITG3200_I2C_ADDR, 2);
77  itg3200_init_status++;
78  break;
79  case ITG_CONF_PWR:
80  itg3200_i2c_trans.buf[0] = ITG3200_REG_PWR_MGM;
81  itg3200_i2c_trans.buf[1] = ITG3200_PWR_MGM;
82  I2CTransmit(ITG3200_I2C_DEVICE, itg3200_i2c_trans, ITG3200_I2C_ADDR, 2);
83  itg3200_init_status++;
84  break;
85  case ITG_CONF_DONE:
86  itg3200_initialized = TRUE;
87  itg3200_i2c_trans.status = I2CTransDone;
88  break;
89  default:
90  break;
91  }
92 }

94 // Configure
95 void itg3200_configure(void)
96 {
97  if (itg3200_init_status == ITG_CONF_UNINIT) {
98  itg3200_init_status++;
99  if (itg3200_i2c_trans.status == I2CTransSuccess || itg3200_i2c_trans.status == I2CTransDone) {
100  itg3200_send_config();
101  }
102  }
103 }

106 void itg3200_read(void)

114 #define Int16FromBuf(_buf,_idx) ((int16_t)((_buf[_idx]<<8) | _buf[_idx+1]))

116 void itg3200_event(void)
117 {
118  if (itg3200_initialized) {
119  if (itg3200_i2c_trans.status == I2CTransFailed) {
120  itg3200_i2c_trans.status = I2CTransDone;
121  }
122  else if (itg3200_i2c_trans.status == I2CTransSuccess) {
123  // Successfull reading and new data available
124  if (itg3200_i2c_trans.buf[0] & 0x01) {
125  // New data available
126  itg3200_data.p = Int16FromBuf(itg3200_i2c_trans.buf,3);
127  itg3200_data.q = Int16FromBuf(itg3200_i2c_trans.buf,5);
128  itg3200_data.r = Int16FromBuf(itg3200_i2c_trans.buf,7);
129  itg3200_data_available = TRUE;
130  }
131  itg3200_i2c_trans.status = I2CTransDone;
132  }
133  }
134  else if (!itg3200_initialized && itg3200_init_status != ITG_CONF_UNINIT) { // Configuring
135  if (itg3200_i2c_trans.status == I2CTransSuccess || itg3200_i2c_trans.status == I2CTransDone) {
136  itg3200_i2c_trans.status = I2CTransDone;
137  itg3200_send_config();
138  }
139  if (itg3200_i2c_trans.status == I2CTransFailed) {
140  itg3200_init_status--;
141  itg3200_i2c_trans.status = I2CTransDone;
142  itg3200_send_config(); // Retry config (TODO max retry)
143  }
144  }
