59 void ahrs_init(void) {
60  //ahrs_float.status = AHRS_UNINIT;
61  // set to running for now and only use ahrs.status, not ahrs_float.status
62  ahrs.status = AHRS_RUNNING;
63 
64  ahrs_sim_available = FALSE;
65 
66  /* set ltp_to_body to zero */
67  FLOAT_QUAT_ZERO(ahrs_float.ltp_to_body_quat);
68  FLOAT_EULERS_ZERO(ahrs_float.ltp_to_body_euler);
69  FLOAT_RMAT_ZERO(ahrs_float.ltp_to_body_rmat);
70  FLOAT_RATES_ZERO(ahrs_float.body_rate);
71 
72  /* set ltp_to_imu to same as ltp_to_body, currently no difference simulated */
73  QUAT_COPY(ahrs_float.ltp_to_imu_quat, ahrs_float.ltp_to_body_quat);

75  RMAT_COPY(ahrs_float.ltp_to_imu_rmat, ahrs_float.ltp_to_body_rmat);
76  RATES_COPY(ahrs_float.imu_rate, ahrs_float.body_rate);
77 }
