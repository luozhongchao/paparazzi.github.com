88 void imu_umarim_event( void )
89 {
90 
91  // If the itg3200 I2C transaction has succeeded: convert the data
92  itg3200_event();
93  if (itg3200_data_available) {
94  RATES_ASSIGN(imu.gyro_unscaled, itg3200_data.p, itg3200_data.q, itg3200_data.r);
95  itg3200_data_available = FALSE;
96  gyr_valid = TRUE;
97  }
98 
99  // If the adxl345 I2C transaction has succeeded: convert the data
100  adxl345_event();
101  if (adxl345_data_available) {
102  // Be careful with orientation of the ADXL (ITG axes are taken as default reference)
103  VECT3_ASSIGN(imu.accel_unscaled, adxl345_data.y, -adxl345_data.x, adxl345_data.z);
104  adxl345_data_available = FALSE;
105  acc_valid = TRUE;
106  }
