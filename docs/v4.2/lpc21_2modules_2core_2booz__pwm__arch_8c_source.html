30 void booz_pwm_init_arch( void ) {
31 
32  /* start PWM5 */
33  /* select P0.21 as PWM5 */
34  PWM0_PINSEL |= PWM0_PINSEL_VAL << PWM0_PINSEL_BIT;
35  /* select pwm period */
36  PWMMR0 = PWM_PERIOD;
37  /* select pwm value to 50% at init (1500 us) */
38  PWMMR5 = PWM_DUTY;
39  /* commit values */
40  PWMLER = PWMLER_LATCH0 | PWMLER_LATCH5;
41  /* prescle timer to match TIMER 0 (15MHz) */
42  PWMPR = T0_PCLK_DIV - 1;
43  /* enable counter and pwm mode */
44  PWMTCR = PWMTCR_COUNTER_ENABLE | PWMTCR_PWM_ENABLE;
45  /* enable PWM5 */
46  PWMPCR = PWMPCR_ENA5;
