46 static inline void autopilot_arming_set(bool_t motors_on) {
47  if (motors_on) {
48  autopilot_arming_state = STATE_MOTORS_ON;
49  }
50  else {
51  if (autopilot_arming_state == STATE_MOTORS_ON) {
52  autopilot_arming_state = STATE_STARTABLE;
53  /* if turned off in an AUTO mode, remember it so it can be turned on again in AUTO */
54  if (autopilot_mode != MODE_MANUAL) {
55  autopilot_unarmed_in_auto = TRUE;
56  }
57  else {
58  autopilot_unarmed_in_auto = FALSE;
59  }
60  }
61  }
62 }

71 static inline void autopilot_arming_check_motors_on( void ) {
72  switch(autopilot_arming_state) {
73  case STATE_UNINIT:
74  autopilot_motors_on = FALSE;
75  if (kill_switch_is_on()) {
76  autopilot_arming_state = STATE_STARTABLE;
77  }
78  else {
79  autopilot_arming_state = STATE_WAITING;
80  }
81  break;
82  case STATE_WAITING:
83  autopilot_motors_on = FALSE;
84  if (kill_switch_is_on()) {
85  autopilot_arming_state = STATE_STARTABLE;
86  }
87  break;
88  case STATE_STARTABLE:
89  autopilot_motors_on = FALSE;
90  if (!kill_switch_is_on() &&
91  THROTTLE_STICK_DOWN() &&
92  rc_attitude_sticks_centered() &&
93  (autopilot_mode == MODE_MANUAL || autopilot_unarmed_in_auto)) {
94  autopilot_arming_state = STATE_MOTORS_ON;
95  }
96  break;
97  case STATE_MOTORS_ON:
98  autopilot_motors_on = TRUE;
99  if (kill_switch_is_on()) {
100  /* if killed, go to STATE_STARTABLE where motors will be turned off */
101  autopilot_arming_state = STATE_STARTABLE;
102  /* if turned off in an AUTO mode, remember it so it can be turned on again in AUTO */
103  if (autopilot_mode != MODE_MANUAL) {
104  autopilot_unarmed_in_auto = TRUE;
105  }
106  else {
107  autopilot_unarmed_in_auto = FALSE;
108  }
109  }
110  break;
111  default:
112  break;
113  }
