00073 void ahrs_update_gps(void) {
00074 
00075   float hspeed_mod_f = gps.gspeed / 100.;
00076   float course_f = gps.course / 1e7;
00077 
00078   // Heading estimator from wind-information, usually computed with -DWIND_INFO
00079   // wind_north and wind_east initialized to 0, so still correct if not updated
00080   float w_vn = cosf(course_f) * hspeed_mod_f - wind_north;
00081   float w_ve = sinf(course_f) * hspeed_mod_f - wind_east;
00082   ahrs_float.ltp_to_body_euler.psi = atan2f(w_ve, w_vn);
00083   if (ahrs_float.ltp_to_body_euler.psi < 0.)
00084     ahrs_float.ltp_to_body_euler.psi += 2 * M_PI;
00085 
00086   ahrs_update_fw_estimator();
00087 }

00089 void ahrs_update_infrared(void) {
00090   ahrs_float.ltp_to_body_euler.phi  = atan2(infrared.roll, infrared.top) - infrared.roll_neutral;
00091 
00092   ahrs_float.ltp_to_body_euler.theta  = atan2(infrared.pitch, infrared.top) - infrared.pitch_neutral;
00093 
00094   if (ahrs_float.ltp_to_body_euler.theta < -M_PI_2)
00095     ahrs_float.ltp_to_body_euler.theta += M_PI;
00096   else if (ahrs_float.ltp_to_body_euler.theta > M_PI_2)
00097     ahrs_float.ltp_to_body_euler.theta -= M_PI;
00098 
00099   if (ahrs_float.ltp_to_body_euler.phi >= 0)
00100     ahrs_float.ltp_to_body_euler.phi *= infrared.correction_right;
00101   else
00102     ahrs_float.ltp_to_body_euler.phi *= infrared.correction_left;
00103 
00104   if (ahrs_float.ltp_to_body_euler.theta >= 0)
00105     ahrs_float.ltp_to_body_euler.theta *= infrared.correction_up;
00106   else
00107     ahrs_float.ltp_to_body_euler.theta *= infrared.correction_down;
00108 
00109   ahrs_update_fw_estimator();
00110 }
