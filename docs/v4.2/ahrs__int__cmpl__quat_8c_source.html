00131 void ahrs_propagate(void) {
00132 
00133   /* unbias gyro             */
00134   struct Int32Rates omega;
00135   RATES_DIFF(omega, imu.gyro_prev, ahrs_impl.gyro_bias);
00136 
00137   /* low pass rate */
00138 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
00139   RATES_SMUL(ahrs.imu_rate, ahrs.imu_rate,2);
00140   RATES_ADD(ahrs.imu_rate, omega);
00141   RATES_SDIV(ahrs.imu_rate, ahrs.imu_rate, 3);
00142 #else
00143   RATES_COPY(ahrs.imu_rate, omega);
00144 #endif
00145 
00146   /* add correction     */
00147   RATES_ADD(omega, ahrs_impl.rate_correction);
00148   /* and zeros it */
00149   INT_RATES_ZERO(ahrs_impl.rate_correction);
00150 
00151   /* integrate quaternion */
00152   INT32_QUAT_INTEGRATE_FI(ahrs.ltp_to_imu_quat, ahrs_impl.high_rez_quat, omega, AHRS_PROPAGATE_FREQUENCY);
00153   INT32_QUAT_NORMALIZE(ahrs.ltp_to_imu_quat);
00154 
00155   compute_imu_euler_and_rmat_from_quat();
00156 
00157   compute_body_orientation();
00158 
00159 }

00164 void ahrs_update_accel(void) {
00165 
00166   // c2 = ltp z-axis in imu-frame
00167   struct Int32Vect3 c2 = { RMAT_ELMT(ahrs.ltp_to_imu_rmat, 0,2),
00168                            RMAT_ELMT(ahrs.ltp_to_imu_rmat, 1,2),
00169                            RMAT_ELMT(ahrs.ltp_to_imu_rmat, 2,2)};
00170   struct Int32Vect3 residual;
00171 
00172   struct Int32Vect3 pseudo_gravity_measurement;
00173 
00174   if (ahrs_impl.correct_gravity && ahrs_impl.ltp_vel_norm_valid) {
00175     /*
00176      * centrifugal acceleration in body frame
00177      * a_c_body = omega x (omega x r)
00178      * (omega x r) = tangential velocity in body frame
00179      * a_c_body = omega x vel_tangential_body
00180      * assumption: tangential velocity only along body x-axis
00181      */
00182 
00183     // FIXME: check overflows !
00184 #define COMPUTATION_FRAC 16
00185 
00186     const struct Int32Vect3 vel_tangential_body = {ahrs_impl.ltp_vel_norm >> COMPUTATION_FRAC, 0, 0};
00187     struct Int32Vect3 acc_c_body;
00188     VECT3_RATES_CROSS_VECT3(acc_c_body, ahrs.body_rate, vel_tangential_body);
00189     INT32_VECT3_RSHIFT(acc_c_body, acc_c_body, INT32_SPEED_FRAC+INT32_RATE_FRAC-INT32_ACCEL_FRAC-COMPUTATION_FRAC);
00190 
00191     /* convert centrifucal acceleration from body to imu frame */
00192     struct Int32Vect3 acc_c_imu;
00193     INT32_RMAT_VMULT(acc_c_imu, imu.body_to_imu_rmat, acc_c_body);
00194 
00195     /* and subtract it from imu measurement to get a corrected measurement of the gravitiy vector */
00196     INT32_VECT3_DIFF(pseudo_gravity_measurement, imu.accel, acc_c_imu);
00197   } else {
00198     VECT3_COPY(pseudo_gravity_measurement, imu.accel);
00199   }
00200 
00201   /* compute the residual of the pseudo gravity vector in imu frame */
00202   INT32_VECT3_CROSS_PRODUCT(residual, pseudo_gravity_measurement, c2);
00203 
00204 
00205   int32_t inv_weight;
00206   if (ahrs_impl.use_gravity_heuristic) {
00207     /* heuristic on acceleration norm */
00208 
00209     /* FIR filtered pseudo_gravity_measurement */
00210     static struct Int32Vect3 filtered_gravity_measurement = {0, 0, 0};
00211     VECT3_SMUL(filtered_gravity_measurement, filtered_gravity_measurement, 7);
00212     VECT3_ADD(filtered_gravity_measurement, pseudo_gravity_measurement);
00213     VECT3_SDIV(filtered_gravity_measurement, filtered_gravity_measurement, 8);
00214 
00215     int32_t acc_norm;
00216     INT32_VECT3_NORM(acc_norm, filtered_gravity_measurement);
00217     const int32_t acc_norm_d = ABS(ACCEL_BFP_OF_REAL(9.81)-acc_norm);
00218     inv_weight = Chop(50*acc_norm_d/ACCEL_BFP_OF_REAL(9.81), 1, 50);
00219   }
00220   else {
00221     inv_weight = 1;
00222   }
00223 
00224   // residual FRAC : ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
00225   // rate_correction FRAC = RATE_FRAC = 12
00226   // 2^12 / 2^24 * 5e-2 = 1/81920
00227   ahrs_impl.rate_correction.p += -residual.x/82000/inv_weight;
00228   ahrs_impl.rate_correction.q += -residual.y/82000/inv_weight;
00229   ahrs_impl.rate_correction.r += -residual.z/82000/inv_weight;
00230 
00231   // residual FRAC = ACCEL_FRAC + TRIG_FRAC = 10 + 14 = 24
00232   // high_rez_bias = RATE_FRAC+28 = 40
00233   // 2^40 / 2^24 * 5e-6 = 1/3.05
00234 
00235   //  ahrs_impl.high_rez_bias.p += residual.x*3;
00236   //  ahrs_impl.high_rez_bias.q += residual.y*3;
00237   //  ahrs_impl.high_rez_bias.r += residual.z*3;
00238 
00239   ahrs_impl.high_rez_bias.p += residual.x/(2*inv_weight);
00240   ahrs_impl.high_rez_bias.q += residual.y/(2*inv_weight);
00241   ahrs_impl.high_rez_bias.r += residual.z/(2*inv_weight);
00242 
00243 
00244   /*                        */
00245   INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
00246 
00247 }

00357 void ahrs_update_heading(int32_t heading) {
00358 
00359   INT32_ANGLE_NORMALIZE(heading);
00360 
00361   // row 0 of ltp_to_body_rmat = body x-axis in ltp frame
00362   // we only consider x and y
00363   struct Int32Vect2 expected_ltp =
00364     { RMAT_ELMT(ahrs.ltp_to_body_rmat, 0, 0),
00365       RMAT_ELMT(ahrs.ltp_to_body_rmat, 0, 1) };
00366 
00367   int32_t heading_x, heading_y;
00368   PPRZ_ITRIG_COS(heading_x, heading); // measured course in x-direction
00369   PPRZ_ITRIG_SIN(heading_y, heading); // measured course in y-direction
00370 
00371   // expected_heading cross measured_heading ??
00372   struct Int32Vect3 residual_ltp =
00373     { 0,
00374       0,
00375       (expected_ltp.x * heading_y - expected_ltp.y * heading_x)/(1<<INT32_ANGLE_FRAC)};
00376 
00377   struct Int32Vect3 residual_imu;
00378   INT32_RMAT_VMULT(residual_imu, ahrs.ltp_to_imu_rmat, residual_ltp);
00379 
00380   // residual FRAC = TRIG_FRAC + TRIG_FRAC = 14 + 14 = 28
00381   // rate_correction FRAC = RATE_FRAC = 12
00382   // 2^12 / 2^28 * 4.0 = 1/2^14
00383   // (1<<INT32_ANGLE_FRAC)/2^14 = 1/4
00384   ahrs_impl.rate_correction.p += residual_imu.x/4;
00385   ahrs_impl.rate_correction.q += residual_imu.y/4;
00386   ahrs_impl.rate_correction.r += residual_imu.z/4;
00387 
00388 
00389   /* crude attempt to only update bias if deviation is small
00390    * e.g. needed when you only have gps providing heading
00391    * and the inital heading is totally different from
00392    * the gps course information you get once you have a gps fix.
00393    * Otherwise the bias will be falsely "corrected".
00394    */
00395   int32_t sin_max_angle_deviation;
00396   PPRZ_ITRIG_SIN(sin_max_angle_deviation, TRIG_BFP_OF_REAL(RadOfDeg(5.)));
00397   if (ABS(residual_ltp.z) < sin_max_angle_deviation)
00398   {
00399     // residual_ltp FRAC = 2 * TRIG_FRAC = 28
00400     // high_rez_bias = RATE_FRAC+28 = 40
00401     // 2^40 / 2^28 * 2.5e-4 = 1
00402     ahrs_impl.high_rez_bias.p -= residual_imu.x*(1<<INT32_ANGLE_FRAC);
00403     ahrs_impl.high_rez_bias.q -= residual_imu.y*(1<<INT32_ANGLE_FRAC);
00404     ahrs_impl.high_rez_bias.r -= residual_imu.z*(1<<INT32_ANGLE_FRAC);
00405 
00406     INT_RATES_RSHIFT(ahrs_impl.gyro_bias, ahrs_impl.high_rez_bias, 28);
00407   }
00408 }

00410 void ahrs_realign_heading(int32_t heading) {
00411 
00412   /* quaternion representing only the heading rotation from ltp to body */
00413   struct Int32Quat q_h_new;
00414   q_h_new.qx = 0;
00415   q_h_new.qy = 0;
00416   PPRZ_ITRIG_SIN(q_h_new.qz, heading/2);
00417   PPRZ_ITRIG_COS(q_h_new.qi, heading/2);
00418 
00419   /* quaternion representing current heading only */
00420   struct Int32Quat q_h;
00421   QUAT_COPY(q_h, ahrs.ltp_to_body_quat);
00422   q_h.qx = 0;
00423   q_h.qy = 0;
00424   INT32_QUAT_NORMALIZE(q_h);
00425 
00426   /* quaternion representing rotation from current to new heading */
00427   struct Int32Quat q_c;
00428   INT32_QUAT_INV_COMP_NORM_SHORTEST(q_c, q_h, q_h_new);
00429 
00430   /* correct current heading in body frame */
00431   struct Int32Quat q;
00432   INT32_QUAT_COMP_NORM_SHORTEST(q, q_c, ahrs.ltp_to_body_quat);
00433   QUAT_COPY(ahrs.ltp_to_body_quat, q);
00434 
00435   /* compute other representations in body frame */
00436   compute_body_euler_and_rmat_from_quat();
00437 
00438   /* compute ltp to imu rotations */
00439   compute_imu_orientation();
00440 
00441   ahrs_impl.heading_aligned = TRUE;
00442 }
