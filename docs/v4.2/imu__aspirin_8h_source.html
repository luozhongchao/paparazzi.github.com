169 static inline void gyro_read_i2c(void)
170 {
171  imu_aspirin.i2c_trans_gyro.type = I2CTransTxRx;
172  imu_aspirin.i2c_trans_gyro.len_w = 1;
173  imu_aspirin.i2c_trans_gyro.len_r = 6;
174  imu_aspirin.i2c_trans_gyro.buf[0] = ITG3200_REG_GYRO_XOUT_H;

179 static inline void gyro_copy_i2c(void)
180 {
181  int16_t gp = imu_aspirin.i2c_trans_gyro.buf[0]<<8 | imu_aspirin.i2c_trans_gyro.buf[1];
182  int16_t gq = imu_aspirin.i2c_trans_gyro.buf[2]<<8 | imu_aspirin.i2c_trans_gyro.buf[3];
183  int16_t gr = imu_aspirin.i2c_trans_gyro.buf[4]<<8 | imu_aspirin.i2c_trans_gyro.buf[5];
184  RATES_ASSIGN(imu.gyro_unscaled, gp, gq, gr);
185 }

187 static inline void accel_copy_spi(void)
188 {
189  const int16_t ax = imu_aspirin.accel_rx_buf[1] | (imu_aspirin.accel_rx_buf[2]<<8);
190  const int16_t ay = imu_aspirin.accel_rx_buf[3] | (imu_aspirin.accel_rx_buf[4]<<8);
191  const int16_t az = imu_aspirin.accel_rx_buf[5] | (imu_aspirin.accel_rx_buf[6]<<8);
192  VECT3_ASSIGN(imu.accel_unscaled, ax, ay, az);
193 }

196 static inline void imu_aspirin_event(void (* _gyro_handler)(void), void (* _accel_handler)(void), void (* _mag_handler)(void))
197 {
198  if (imu_aspirin.status == AspirinStatusUninit) return;
199 
200  imu_aspirin_arch_int_disable();
201  if (imu_aspirin.accel_available) {
202  imu_aspirin.time_since_last_accel_reading = 0;
203  imu_aspirin.accel_available = FALSE;
204  accel_copy_spi();
205  _accel_handler();
206  }
207  imu_aspirin_arch_int_enable();
208 
209  // Reset everything if we've been waiting too long
210  if (imu_aspirin.time_since_last_reading > ASPIRIN_GYRO_TIMEOUT) {
211  // FIXME: there should be no arch specific code like that here!
212  i2c2_er_irq_handler();
213  gyro_read_i2c();
214  imu_aspirin.time_since_last_reading = 0;
215  return;
216  }
217 
218  // Try again later if transaction is in progress
219  if (imu_aspirin.i2c_trans_gyro.status == I2CTransPending || imu_aspirin.i2c_trans_gyro.status == I2CTransRunning)
220  {
221  return;
222  }
223 
224  ImuMagEvent(_mag_handler);
225 
226  // Try back later if things are not idle
227  if ((i2c2.status != I2CIdle) || !i2c_idle(&i2c2)) {
228  return;
229  }
230 
231  if (imu_aspirin.reading_gyro) {
232  if (imu_aspirin.i2c_trans_gyro.status == I2CTransSuccess) {
233  gyro_copy_i2c();
234  if (imu_aspirin.gyro_available_blaaa) {
235  imu_aspirin.gyro_available_blaaa = FALSE;
236  _gyro_handler();
237  }
238  }
239  imu_aspirin.reading_gyro = 0;
240  return;
241  }
242 
243  // If we're not already waiting for read, and conversion is complete, schedule a read
244  if (!imu_aspirin.reading_gyro && imu_aspirin_eoc() && i2c2.status == I2CIdle && i2c_idle(&i2c2)) {
245  if (imu_aspirin.i2c_trans_gyro.status == I2CTransSuccess) {
246  imu_aspirin.time_since_last_reading = 0;
247  }
248  gyro_read_i2c();
249  return;
250  }
251 
252 }
