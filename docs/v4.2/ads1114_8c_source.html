37 void ads1114_init( void ) {
38  /* configure the ads1114_1 */
39 #if USE_ADS1114_1
40  ads1114_1.i2c_addr = ADS1114_1_I2C_ADDR;
41  ads1114_1.trans.buf[0] = ADS1114_POINTER_CONFIG_REG;
42  ads1114_1.trans.buf[1] = ADS1114_1_CONFIG_MSB;
43  ads1114_1.trans.buf[2] = ADS1114_1_CONFIG_LSB;

62 void ads1114_read( struct ads1114_periph * p ) {
63  // Config done with success
64  // start new reading when previous is done (and read if success)
65  if (p->config_done && p->trans.status == I2CTransDone) {
66  p->trans.buf[0] = ADS1114_POINTER_CONV_REG;
67  I2CTransceive(ADS1114_I2C_DEVICE, p->trans, p->i2c_addr, 1, 2);
68  }
