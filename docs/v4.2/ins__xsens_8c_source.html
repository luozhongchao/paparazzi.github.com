00153  * -----------

00156  *  bit 23  0=quaternion, 1=euler, 2=DCM

00246 void ins_periodic_task( void ) {
00247   if (xsens_configured > 0)
00248   {
00249      switch (xsens_configured)
00250      {
00251   case 20:
00252     /* send mode and settings to MT */
00253     XSENS_GoToConfig();
00254     XSENS_SetOutputMode(xsens_output_mode);
00255     XSENS_SetOutputSettings(xsens_output_settings);
00256     break;
00257   case 18:
00258     // Give pulses on SyncOut
00259     XSENS_SetSyncOutSettings(0,0x0002);
00260     break;
00261   case 17:
00262     // 1 pulse every 100 samples
00263     XSENS_SetSyncOutSettings(1,100);
00264     break;
00265   case 2:
00266     XSENS_ReqLeverArmGps();
00267     break;
00268 
00269   case 6:
00270     XSENS_ReqMagneticDeclination();
00271     break;
00272 
00273   case 13:
00274     #ifdef AHRS_H_X
00275     #pragma message "Sending XSens Magnetic Declination."
00276     xsens_declination = atan2(AHRS_H_Y, AHRS_H_X);
00277     XSENS_SetMagneticDeclination(xsens_declination);
00278     #endif
00279     break;
00280   case 12:
00281     #ifdef GPS_IMU_LEVER_ARM_X
00282     #pragma message "Sending XSens GPS Arm."
00283     XSENS_SetLeverArmGps(GPS_IMU_LEVER_ARM_X,GPS_IMU_LEVER_ARM_Y,GPS_IMU_LEVER_ARM_Z);
00284     #endif
00285     break;
00286   case 10:
00287     {
00288     uint8_t baud = 1;
00289     XSENS_SetBaudrate(baud);
00290     }
00291     break;
00292 
00293   case 1:
00294     XSENS_GoToMeasurment();
00295     break;
00296      }
00297      xsens_configured--;
00298      return;
00299   }
00300 
00301   RunOnceEvery(100,XSENS_ReqGPSStatus());
00302 }

00306 void handle_ins_msg( void) {
00307 
00308 
00309   // Send to Estimator (Control)
00310 #ifdef XSENS_BACKWARDS
00311   EstimatorSetAtt((-ins_phi+ins_roll_neutral), (ins_psi + RadOfDeg(180)), (-ins_theta+ins_pitch_neutral));
00312   EstimatorSetRate(-ins_p,-ins_q, ins_r);
00313 #else
00314   EstimatorSetAtt(ins_phi+ins_roll_neutral, ins_psi, ins_theta+ins_pitch_neutral);
00315   EstimatorSetRate(ins_p, ins_q, ins_r);
00316 #endif
00317 
00318   // Position
00319   float gps_east = gps.utm_pos.east / 100.;
00320   float gps_north = gps.utm_pos.north / 100.;
00321   gps_east -= nav_utm_east0;
00322   gps_north -= nav_utm_north0;
00323   EstimatorSetPosXY(gps_east, gps_north);
00324 
00325   // Altitude and vertical speed
00326   float hmsl = gps.hmsl;
00327   hmsl /= 1000.0f;
00328   EstimatorSetAlt(hmsl);
00329 
00330   #ifndef ALT_KALMAN
00331   #warning NO_VZ
00332   #endif
00333 
00334   // Horizontal speed
00335   float fspeed = sqrt(ins_vx*ins_vx + ins_vy*ins_vy);
00336   if (gps.fix != GPS_FIX_3D)
00337   {
00338     fspeed = 0;
00339   }
00340   float fclimb = -ins_vz;
00341   float fcourse = atan2f((float)ins_vy, (float)ins_vx);
00342   EstimatorSetSpeedPol(fspeed, fcourse, fclimb);
00343 
00344   // Now also finish filling the gps struct for telemetry purposes
00345   gps.gspeed = fspeed * 100.;
00346   gps.speed_3d = (uint16_t)(sqrt(ins_vx*ins_vx + ins_vy*ins_vy + ins_vz*ins_vz) * 100);
00347   gps.course = fcourse * 1e7;
00348 
00349 }

00370   }
00371   else if (xsens_id == XSENS_Error_ID) {
00372     xsens_errorcode = XSENS_Error_errorcode(xsens_msg_buf);
00373   }
00374 #if USE_GPS_XSENS
00375   else if (xsens_id == XSENS_GPSStatus_ID) {

00392         gps.num_sv++;

00394     }
00395   }
00396 #endif

00398     /* test RAW modes else calibrated modes */
00399     //if ((XSENS_MASK_RAWInertial(xsens_output_mode)) || (XSENS_MASK_RAWGPS(xsens_output_mode))) {
00400       if (XSENS_MASK_RAWInertial(xsens_output_mode)) {
00401         ins_p = XSENS_DATA_RAWInertial_gyrX(xsens_msg_buf,offset);
00402         ins_q = XSENS_DATA_RAWInertial_gyrY(xsens_msg_buf,offset);
