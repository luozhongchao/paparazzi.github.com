00121 void baro_ets_read_event( void ) {
00122   // Get raw altimeter from buffer
00123   baro_ets_adc = ((uint16_t)(baro_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(baro_ets_i2c_trans.buf[0]);
00124   // Check if this is valid altimeter
00125   if (baro_ets_adc == 0)
00126     baro_ets_valid = FALSE;
00127   else
00128     baro_ets_valid = TRUE;
00129 
00130   // Continue only if a new altimeter value was received
00131   if (baro_ets_valid) {
00132     // Calculate offset average if not done already
00133     if (!baro_ets_offset_init) {
00134       --baro_ets_cnt;
00135       // Check if averaging completed
00136       if (baro_ets_cnt == 0) {
00137         // Calculate average
00138         baro_ets_offset = (uint16_t)(baro_ets_offset_tmp / BARO_ETS_OFFSET_NBSAMPLES_AVRG);
00139         // Limit offset
00140         if (baro_ets_offset < BARO_ETS_OFFSET_MIN)
00141           baro_ets_offset = BARO_ETS_OFFSET_MIN;
00142         if (baro_ets_offset > BARO_ETS_OFFSET_MAX)
00143           baro_ets_offset = BARO_ETS_OFFSET_MAX;
00144         baro_ets_offset_init = TRUE;
00145       }
00146       // Check if averaging needs to continue
00147       else if (baro_ets_cnt <= BARO_ETS_OFFSET_NBSAMPLES_AVRG)
00148         baro_ets_offset_tmp += baro_ets_adc;
00149     }
00150     // Convert raw to m/s
00151     if (baro_ets_offset_init) {
00152       baro_ets_altitude = ground_alt + BARO_ETS_SCALE * (float)(baro_ets_offset-baro_ets_adc);
00153       // New value available
00154       EstimatorSetAlt(baro_ets_altitude);
00155 #ifdef BARO_ETS_TELEMETRY
00156       DOWNLINK_SEND_BARO_ETS(DefaultChannel, DefaultDevice, &baro_ets_adc, &baro_ets_offset, &baro_ets_altitude);
00157 #endif
00158     } else {
00159       baro_ets_altitude = 0.0;
00160     }
00161   } else {
00162     baro_ets_altitude = 0.0;
00163   }
00164 
00165   // Transaction has been read
00166   baro_ets_i2c_trans.status = I2CTransDone;
00167 }
