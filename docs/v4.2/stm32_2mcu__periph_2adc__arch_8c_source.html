00450 static inline void adc_push_sample(struct adc_buf * buf, uint16_t value) {
00451     uint8_t new_head = buf->head + 1;
00452 
00453     if (new_head >= buf->av_nb_sample) { new_head = 0; }
00454     buf->sum -= buf->values[new_head];
00455     buf->values[new_head] = value;
00456     buf->sum += value;
00457     buf->head = new_head;
00458 }
