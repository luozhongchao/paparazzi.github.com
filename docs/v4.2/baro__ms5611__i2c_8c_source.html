00067 static int8_t baro_ms5611_crc(uint16_t* prom) {
00068   int32_t i, j;
00069   uint32_t res = 0;
00070   uint8_t crc = prom[7] & 0xF;
00071   prom[7] &= 0xFF00;
00072   for (i = 0; i < 16; i++) {
00073     if (i & 1) res ^= ((prom[i>>1]) & 0x00FF);
00074     else res ^= (prom[i>>1]>>8);
00075     for (j = 8; j > 0; j--) {
00076       if (res & 0x8000) res ^= 0x1800;
00077       res <<= 1;
00078     }
00079   }
00080   prom[7] |= crc;
00081   if (crc == ((res >> 12) & 0xF)) return 0;
00082   else return -1;
00083 }

00142 void baro_ms5611_event( void ) {
00143   if (ms5611_trans.status == I2CTransSuccess) {
00144     switch (ms5611_status) {
00145 
00146     case MS5611_RESET:
00147       ms5611_status = MS5611_RESET_OK;
00148       ms5611_trans.status = I2CTransDone;
00149       break;
00150 
00151     case MS5611_PROM:
00152       /* read prom data */
00153       ms5611_c[prom_cnt++] = (ms5611_trans.buf[0] << 8) | ms5611_trans.buf[1];
00154       if (prom_cnt < PROM_NB) {
00155         /* get next prom data */
00156         ms5611_trans.buf[0] = MS5611_PROM_READ | (prom_cnt << 1);
00157         I2CTransceive(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1, 2);
00158       }
00159       else {
00160         /* done reading prom */
00161         ms5611_trans.status = I2CTransDone;
00162         /* check prom crc */
00163         if (baro_ms5611_crc(ms5611_c) == 0) {
00164           DOWNLINK_SEND_MS5611_COEFF(DefaultChannel, DefaultDevice,
00165               &ms5611_c[0], &ms5611_c[1], &ms5611_c[2], &ms5611_c[3],
00166               &ms5611_c[4], &ms5611_c[5], &ms5611_c[6], &ms5611_c[7]);
00167           ms5611_status = MS5611_IDLE;
00168         }
00169         else {
00170           /* checksum error, try again */
00171           baro_ms5611_init();
00172         }
00173       }
00174       break;
00175 
00176     case MS5611_CONV_D1:
00177       ms5611_status = MS5611_CONV_D1_OK;
00178       ms5611_trans.status = I2CTransDone;
00179       break;
00180 
00181     case  MS5611_ADC_D1:
00182       /* read D1 (pressure) */
00183       ms5611_d1 = (ms5611_trans.buf[0] << 16) |
00184                   (ms5611_trans.buf[1] << 8) |
00185                   ms5611_trans.buf[2];
00186       /* start D2 conversion */
00187       ms5611_status = MS5611_CONV_D2;
00188       ms5611_trans.buf[0] = MS5611_START_CONV_D2;
00189       I2CTransmit(MS5611_I2C_DEV, ms5611_trans, MS5611_SLAVE_ADDR, 1);
00190       break;
00191 
00192     case MS5611_CONV_D2:
00193       ms5611_status = MS5611_CONV_D2_OK;
00194       ms5611_trans.status = I2CTransDone;
00195       break;
00196 
00197     case  MS5611_ADC_D2: {
00198       int64_t dt, baroms, tempms, off, sens, t2, off2, sens2;
00199       /* read D2 (temperature) */
00200       ms5611_d2 = (ms5611_trans.buf[0] << 16) |
00201                   (ms5611_trans.buf[1] << 8) |
00202                   ms5611_trans.buf[2];
00203       ms5611_status = MS5611_IDLE;
00204       ms5611_trans.status = I2CTransDone;
00205 
00206       /* difference between actual and ref temperature */
00207       dt = ms5611_d2 - (int64_t)ms5611_c[5] * (1<<8);
00208       /* actual temperature */
00209       tempms = 2000 + ((int64_t)dt * ms5611_c[6]) / (1<<23);
00210       /* offset at actual temperature */
00211       off = ((int64_t)ms5611_c[2] * (1<<16)) + ((int64_t)ms5611_c[4] * dt) / (1<<7);
00212       /* sensitivity at actual temperature */
00213       sens = ((int64_t)ms5611_c[1] * (1<<15)) + ((int64_t)ms5611_c[3] * dt) / (1<<8);
00214       /* second order temperature compensation */
00215       if (tempms < 2000) {
00216         t2 = (dt*dt) / (1<<31);
00217         off2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<1);
00218         sens2 = 5 * ((int64_t)(tempms-2000)*(tempms-2000)) / (1<<2);
00219         if (tempms < -1500) {
00220           off2 = off2 + 7 * (int64_t)(tempms+1500)*(tempms+1500);
00221           sens2 = sens2 + 11 * ((int64_t)(tempms+1500)*(tempms+1500)) / (1<<1);
00222         }
00223         tempms = tempms - t2;
00224         off = off - off2;
00225         sens = sens - sens2;
00226       }
00227       /* temperature compensated pressure */
00228       baroms = (((int64_t)ms5611_d1 * sens) / (1<<21) - off) / (1<<15);
00229 
00230       tmp_float = baroms/101325.0; //pressao nivel mar
00231       tmp_float = pow(tmp_float,0.190295); //eleva pressao ao expoente
00232       baro_alt = 44330*(1.0-tmp_float); //altitude relativa nivel do mar
00233 
00234       if (!baro_offset_init) {
00235         baro_offset = baro_alt;
00236         baro_offset_init = TRUE;
00237       } //baro offset init
00238 
00239       baro_temp = (baro_alt - baro_offset);
00240 
00241       if (baro_offset_init) {        // New value available
00242         baro_altitude = ground_alt + baro_temp;
00243 
00244 #if USE_BARO_MS5611
00245 #pragma message "USING BARO MS5611"

00247 #endif
00248 
00249       } else {
00250         baro_altitude = 0.0;
00251       }
00252 #ifdef SENSOR_SYNC_SEND

00256                                 &ms5611_d1, &ms5611_d2, &fbaroms, &ftempms);
00257 #endif
00258 
00259       break;
00260     }
00261 
00262     default:

00264       break;
00265     }
