00135 void alt_kalman(float gps_z) {
00136   float DT;
00137   float R;
00138   float SIGMA2;
00139 
00140 #if USE_BARO_MS5534A
00141   if (alt_baro_enabled) {
00142     DT = BARO_DT;
00143     R = baro_MS5534A_r;
00144     SIGMA2 = baro_MS5534A_sigma2;
00145   } else
00146 #elif USE_BARO_ETS
00147   if (baro_ets_enabled) {
00148     DT = BARO_ETS_DT;
00149     R = baro_ets_r;
00150     SIGMA2 = baro_ets_sigma2;
00151   } else
00152 #elif USE_BARO_MS5611
00153   if (baro_ms5611_enabled) {
00154     DT = BARO_MS5611_DT;
00155     R = baro_ms5611_r;
00156     SIGMA2 = baro_ms5611_sigma2;
00157   } else
00158 #elif USE_BARO_BMP
00159   if (baro_bmp_enabled) {
00160     DT = BARO_BMP_DT;
00161     R = baro_bmp_r;
00162     SIGMA2 = baro_bmp_sigma2;
00163   } else
00164 #endif
00165   {
00166     DT = GPS_DT;
00167     R = GPS_R;
00168     SIGMA2 = GPS_SIGMA2;
00169   }
00170 
00171   float q[2][2];
00172   q[0][0] = DT*DT*DT*DT/4.;
00173   q[0][1] = DT*DT*DT/2.;
00174   q[1][0] = DT*DT*DT/2.;
00175   q[1][1] = DT*DT;
00176 
00177 
00178   /* predict */
00179   estimator_z += estimator_z_dot * DT;
00180   p[0][0] = p[0][0]+p[1][0]*DT+DT*(p[0][1]+p[1][1]*DT) + SIGMA2*q[0][0];
00181   p[0][1] = p[0][1]+p[1][1]*DT + SIGMA2*q[0][1];
00182   p[1][0] = p[1][0]+p[1][1]*DT + SIGMA2*q[1][0];
00183   p[1][1] = p[1][1] + SIGMA2*q[1][1];
00184 
00185   /* error estimate */
00186   float e = p[0][0] + R;
00187 
00188   if (fabs(e) > 1e-5) {
00189     float k_0 = p[0][0] / e;
00190     float k_1 =  p[1][0] / e;
00191     e = gps_z - estimator_z;
00192 
00193     /* correction */
00194     estimator_z += k_0 * e;
00195     estimator_z_dot += k_1 * e;
00196 
00197     p[1][0] = -p[0][0]*k_1+p[1][0];
00198     p[1][1] = -p[0][1]*k_1+p[1][1];
00199     p[0][0] = p[0][0] * (1-k_0);
00200     p[0][1] = p[0][1] * (1-k_0);
00201   }
00202 
00203 #ifdef DEBUG_ALT_KALMAN
00204   DOWNLINK_SEND_ALT_KALMAN(DefaultChannel,DefaultDevice,&(p[0][0]),&(p[0][1]),&(p[1][0]), &(p[1][1]));
00205 #endif
00206 }
