00194 void parse_cam_buffer( uint8_t c ) {
00195   char bla[1];
00196   bla[1] = c;
00197   //DOWNLINK_SEND_DEBUG(DefaultChannel, DefaultDevice,1,bla);
00198   switch (cam_status) {
00199   case UNINIT:
00200     if (c != CAM_START_1)
00201       goto error;
00202     cam_status++;
00203     break;
00204   case GOT_START_1:
00205     if (c != CAM_START_2)
00206       goto error;
00207     cam_status++;
00208     break;
00209   case GOT_START_2:
00210     cam_data_len = c;
00211     if (cam_data_len > CAM_MAX_PAYLOAD)
00212       goto error;
00213     cam_data_idx = 0;
00214     cam_status++;
00215     break;
00216   case GOT_LEN:
00217     cam_data_buf[cam_data_idx] = c;
00218     cam_data_idx++;
00219     if (cam_data_idx >= cam_data_len)
00220       cam_status++;
00221     break;
00222   case GOT_DATA:
00223     if (c != CAM_END)
00224       goto error;
00225     cam_msg_received = TRUE;
00226     goto restart;
00227     break;
00228   }
00229   return;
00230  error:
00231  restart:
00232   cam_status = UNINIT;
00233   return;
00234 }
