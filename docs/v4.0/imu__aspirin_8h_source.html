00179 static inline void gyro_copy_i2c(void)
00180 {
00181   int16_t gp = imu_aspirin.i2c_trans_gyro.buf[0]<<8 | imu_aspirin.i2c_trans_gyro.buf[1];
00182   int16_t gq = imu_aspirin.i2c_trans_gyro.buf[2]<<8 | imu_aspirin.i2c_trans_gyro.buf[3];
00183   int16_t gr = imu_aspirin.i2c_trans_gyro.buf[4]<<8 | imu_aspirin.i2c_trans_gyro.buf[5];
00184   RATES_ASSIGN(imu.gyro_unscaled, gp, gq, gr);

00196 static inline void imu_aspirin_event(void (* _gyro_handler)(void), void (* _accel_handler)(void), void (* _mag_handler)(void))
00197 {
00198   if (imu_aspirin.status == AspirinStatusUninit) return;
00199 
00200   imu_aspirin_arch_int_disable();
00201   if (imu_aspirin.accel_available) {
00202     imu_aspirin.time_since_last_accel_reading = 0;
00203     imu_aspirin.accel_available = FALSE;
00204     accel_copy_spi();
00205     _accel_handler();
00206   }
00207   imu_aspirin_arch_int_enable();
00208 
00209   // Reset everything if we've been waiting too long
00210   if (imu_aspirin.time_since_last_reading > ASPIRIN_GYRO_TIMEOUT) {
00211     // FIXME: there should be no arch specific code like that here!
00212     i2c2_er_irq_handler();
00213     gyro_read_i2c();
00214     imu_aspirin.time_since_last_reading = 0;
00215     return;
00216   }
00217 
00218   // Try again later if transaction is in progress
00219   if (imu_aspirin.i2c_trans_gyro.status == I2CTransPending || imu_aspirin.i2c_trans_gyro.status == I2CTransRunning)
00220   {
00221     return;
00222   }
00223 
00224   ImuMagEvent(_mag_handler);
00225 
00226   // Try back later if things are not idle
00227   if ((i2c2.status != I2CIdle) || !i2c_idle(&i2c2)) {
00228     return;
00229   }
00230 
00231   if (imu_aspirin.reading_gyro) {
00232     if (imu_aspirin.i2c_trans_gyro.status == I2CTransSuccess) {
00233       gyro_copy_i2c();
00234       if (imu_aspirin.gyro_available_blaaa) {
00235         imu_aspirin.gyro_available_blaaa = FALSE;
00236         _gyro_handler();
00237       }
00238     }
00239     imu_aspirin.reading_gyro = 0;
00240     return;
00241   }
00242 
00243   // If we're not already waiting for read, and conversion is complete, schedule a read
00244   if (!imu_aspirin.reading_gyro && imu_aspirin_eoc() && i2c2.status == I2CIdle && i2c_idle(&i2c2)) {
00245     if (imu_aspirin.i2c_trans_gyro.status == I2CTransSuccess) {
00246       imu_aspirin.time_since_last_reading = 0;
00247     }
00248     gyro_read_i2c();
00249     return;
00250   }
00251 
00252 }
