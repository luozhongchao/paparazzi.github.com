00256 __attribute__ ((always_inline)) static inline void run_hover_loop(bool_t in_flight) {
00257 
00258   /* convert our reference to generic representation */
00259   int64_t tmp  = gv_z_ref>>(GV_Z_REF_FRAC - INT32_POS_FRAC);
00260   guidance_v_z_ref = (int32_t)tmp;
00261   guidance_v_zd_ref = gv_zd_ref<<(INT32_SPEED_FRAC - GV_ZD_REF_FRAC);
00262   guidance_v_zdd_ref = gv_zdd_ref<<(INT32_ACCEL_FRAC - GV_ZDD_REF_FRAC);
00263   /* compute the error to our reference */
00264   int32_t err_z  = guidance_v_z_ref - ins_ltp_pos.z;
00265   Bound(err_z, GUIDANCE_V_MIN_ERR_Z, GUIDANCE_V_MAX_ERR_Z);
00266   int32_t err_zd = guidance_v_zd_ref - ins_ltp_speed.z;
00267   Bound(err_zd, GUIDANCE_V_MIN_ERR_ZD, GUIDANCE_V_MAX_ERR_ZD);
00268 
00269   if (in_flight) {
00270     guidance_v_z_sum_err += err_z;
00271     Bound(guidance_v_z_sum_err, -GUIDANCE_V_MAX_SUM_ERR, GUIDANCE_V_MAX_SUM_ERR);
00272   }
00273   else
00274     guidance_v_z_sum_err = 0;
00275 
00276   /* our nominal command : (g + zdd)*m   */
00277 #ifdef GUIDANCE_V_NOMINAL_HOVER_THROTTLE
00278   const int32_t inv_m = BFP_OF_REAL(9.81/(guidance_v_nominal_throttle*MAX_PPRZ), FF_CMD_FRAC);
00279 #else
00280   const int32_t inv_m =  gv_adapt_X>>(GV_ADAPT_X_FRAC - FF_CMD_FRAC);
00281 #endif
00282   const int32_t g_m_zdd = (int32_t)BFP_OF_REAL(9.81, FF_CMD_FRAC) -
00283                           (guidance_v_zdd_ref<<(FF_CMD_FRAC - INT32_ACCEL_FRAC));
00284 
00285   guidance_v_ff_cmd = g_m_zdd / inv_m;
00286   int32_t cphi,ctheta,cphitheta;
00287   PPRZ_ITRIG_COS(cphi, ahrs.ltp_to_body_euler.phi);
00288   PPRZ_ITRIG_COS(ctheta, ahrs.ltp_to_body_euler.theta);
00289   cphitheta = (cphi * ctheta) >> INT32_TRIG_FRAC;
00290   if (cphitheta < MAX_BANK_COEF) cphitheta = MAX_BANK_COEF;
00291   /* feed forward command */
00292   guidance_v_ff_cmd = (guidance_v_ff_cmd << INT32_TRIG_FRAC) / cphitheta;
00293 
00294   /* bound the nominal command to 0.9*MAX_PPRZ */
00295   Bound(guidance_v_ff_cmd, 0, 8640);
00296 
00297 
00298   /* our error feed back command                   */
00299   /* z-axis pointing down -> positive error means we need less thrust */
00300   guidance_v_fb_cmd = ((-guidance_v_kp * err_z)  >> 12) +
00301                       ((-guidance_v_kd * err_zd) >> 21) +
00302                       ((-guidance_v_ki * guidance_v_z_sum_err) >> 21);
00303 
00304   guidance_v_delta_t = guidance_v_ff_cmd + (guidance_v_fb_cmd * GUIDANCE_V_GAIN_SCALER);
00305 
00306   /* bound the result */
00307   Bound(guidance_v_delta_t, 0, MAX_PPRZ);
00308 
00309 }
