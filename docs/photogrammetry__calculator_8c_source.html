00077 void photogrammetry_calculator_update(void)
00078 {
00079 
00080   // Photogrammetry Goals
00081   float photogrammetry_sidelap_f = ((float) photogrammetry_sidelap) / 100.0f;

00084   // Linear Projection Camera Model

00087   float pixel_projection_width = viewing_ratio_width / ((float)PHOTOGRAMMETRY_PIXELS_WIDTH);
00088 
00089   // Flightplan Variables
00090   photogrammetry_height = ((float) photogrammetry_resolution) / pixel_projection_width / 1000.0f;
00091 
00092   if (photogrammetry_height > photogrammetry_height_max)
00093     photogrammetry_height = photogrammetry_height_max;
00094   else if (photogrammetry_height < photogrammetry_height_min)
00095     photogrammetry_height = photogrammetry_height_min;
00096 
00097   photogrammetry_sidestep = viewing_ratio_width * photogrammetry_height * (1.0f - photogrammetry_sidelap_f);
